#include <stdio.h>
int main()
{
int i;
  int j;
  int a[10];
  for(i=0;i<10;i++)
    scanf("%d",&a[i]);
  for(i=0;i<10;i+=3)
  {
    for(j=i;j<i+3;j++)
    {
      if(i==9)
      {
        printf("%d",a[i]);
        break;
      }
      else if(j<i+2)
        printf("%d,",a[j]);
      else if(j>=i+2)
        printf("%d",a[j]);
    }
    printf("\n");
  }
	return 0;
  
}
