#include <stdio.h>
int main()
{
  int e1,e2,s,a1,a2,a3;
  float e,sp,a;
  scanf("%d %d",&e1,&e2);
  scanf("%d",&s);
  scanf("%d %d %d",&a1,&a2,&a3);
  e = ((e1+e2)*50)/200.0;
  sp = (s*20)/50.0;
  a = ((a1+a2+a3)*30)/60.0;
  printf("Result\n");
  printf("Total Percent in examination:%.2f\n",e);
  printf("Total Percent in activities:%.2f\n",a);
  printf("Total Percent in sports:%.2f\n",sp);
  printf("Total Percentage:%.2f",e+a+sp);
	return 0;
}
