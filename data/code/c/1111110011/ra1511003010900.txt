#include <stdio.h>
int main()
{
int e1,e2,s,a1,a2,a3;
  scanf("%d",&e1);
  if(e1==0)
  {
    printf("0");
  }
  else
  {
    scanf("%d",&e2);
    scanf("%d",&s);
    scanf("%d %d %d",&a1,&a2,&a3);
    float ep,sp,ap,tot;
    ep=((e1+e2)*50)/200.0;
    sp=(s*20)/50.0;
    ap=((a1+a2+a3)*30)/60.0;
    tot=ap+sp+ep;
    printf("Result");
    printf("\nTotal Percent in examination:%.2f",ep);
     printf("\nTotal Percent in activities:%.2f",ap);
     printf("\nTotal Percent in sports:%.2f",sp);
     printf("\nTotal Percentage:%.2f",tot);
  }
	return 0;
}
