#include <stdio.h>
int main()
{
int a;
  int Preadd,Presub,Postadd,Postsub;
  scanf("%d",&a);
  if(a!=0)
  {
    Preadd=++a;
    printf("%d\n",Preadd);
    Presub=--a;
        printf("%d\n",Presub);

    Postadd=a++;
        printf("%d\n",Postadd);

    Postsub=a--;
        printf("%d\n",Postsub);

	return 0;
}
}
