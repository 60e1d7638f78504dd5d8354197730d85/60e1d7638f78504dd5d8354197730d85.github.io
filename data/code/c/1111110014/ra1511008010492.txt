#include <stdio.h>
int main()
{
  int a,Preadd,Presub,Postadd,Postsub;
  scanf("%d",&a);
  if(a!=0)
  {
  Preadd=++a;
  printf("%d\n",Preadd);  
  Presub=--a;
  printf("%d\n",Presub);
    Postadd=a++;
    printf("%d\n",Postadd);
    Postsub=a--;
  printf("%d",Postsub);
  }
  else
    printf("0");
	return 0;
}
