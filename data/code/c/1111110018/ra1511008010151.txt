#include <stdio.h>
#include <math.h>

 
int main(){
    float sideOne, sideTwo, sideThree, s, area;
    scanf("%f %f %f", &sideOne, &sideTwo, &sideThree);
    s = (sideOne + sideTwo + sideThree)/2;
    area = sqrt(s*(s-sideOne)*(s-sideTwo)*(s-sideThree));
    printf("perimeter of triangle is=%0.6f\n", area);
     

    return 0;
}
