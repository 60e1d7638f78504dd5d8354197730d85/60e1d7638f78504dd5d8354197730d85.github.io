#include<stdio.h>
#include<math.h>
int main()
{
   int startA, endA, startB, endB;
   float distance;
   scanf ("%d %d", &startA, &startB) ;
   scanf("%d %d", &endA, &endB);
   distance= sqrt(pow((endA-startA), 2)+pow((endB-startB), 2));
   printf("\n The distance between two points is=%.2f units", distance);
   return 0;
}
