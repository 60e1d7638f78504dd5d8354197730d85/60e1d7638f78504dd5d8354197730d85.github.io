#include <stdio.h>
int main()
{
 float celsius, fahrenheit;
    scanf("%f", &fahrenheit);
 
    // Convert th temperature from fahrenheit to celsius

    //celsius = 5 * (fahrenheit - 32) / 9;
    //celsius = (fahrenheit - 32) * 0.55556; 
 
    printf("\n%.2fC",celsius);

}
