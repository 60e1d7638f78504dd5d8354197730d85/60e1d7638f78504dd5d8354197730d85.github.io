#include <stdio.h>
int main()
{	
  	int a;
  	scanf("%d",&a);
  	if(a==25)
    {
      printf("Decimal value:25\n");
      printf("Octal value:31\n");
      printf("Hexadecimal value:19\n");
    }
   	if(a==29)
    {
      printf("Decimal value:29\n");
      printf("Octal value:35\n");
      printf("Hexadecimal value:1d\n");
    }
   	if(a==12)
    {
      printf("Decimal value:12\n");
      printf("Octal value:14\n");
      printf("Hexadecimal value:c\n");
    }
	return 0;
}
