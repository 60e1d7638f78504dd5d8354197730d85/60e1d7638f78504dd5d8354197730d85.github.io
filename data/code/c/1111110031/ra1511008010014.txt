#include<stdio.h>
void convert(int, int);
int main()
{
	int num;
	scanf("%d", &num);
	printf("Decimal value:");
	convert(num, 10);
	printf("\n");
	printf("Octal value:");
	convert(num, 8);
	printf("\n");
	printf("Hexadecimal value:");
	convert(num, 16);
	printf("\n");
	return 0;
}
 void convert (int num, int base)
{
	int rem = num%base;
 	if(num==0)
		return;
	convert(num/base,base);
 	if(rem < 10)
		printf("%d",rem);
	else
		printf("%c",rem-10+'a');
}
