#include<stdio.h>
int main()
{
  int s=1, n, d=1, i=1, ans=0;
  //printf("Enter  number of  terms\n");
  scanf ("%d",&n);
  //printf("\n The series is: \n");
  do
  {
    printf("%d \t",s);
    ans = ans+s;
    s = s + d;
    d = d + 2;
    i = i + 1;
  } while (i<=n);
  //printf ("\n The sum of the series is %d", ans);
  return 0;
}
