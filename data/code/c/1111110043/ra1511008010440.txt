 #include <stdio.h>
  int main() {
        int hour, minute, second, timeinsec;

        
        /* get hour value from user*/
        scanf("%d", &hour);

        
        /* get minute value from user */
        scanf("%d", &minute);

        
        /* get sec value from the user */
        scanf("%d", &second);

        /* calculate total seconds */
        timeinsec = second + (minute * 60) + (hour * 60 * 60);
        printf("%d",
                timeinsec);
        return 0;
  }
