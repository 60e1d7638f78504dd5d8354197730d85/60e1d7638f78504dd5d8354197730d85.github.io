#include <stdio.h>

int main()
{
    int n;
    long long product=1ll;

    /* Input number from user */
    scanf("%d", &n);

    /* Repeat the steps till n becomes 0 */
    while(n != 0)
    {
        /* Get the last digit from n and multiplies to product */
        product = product * (n % 10);

        /* Remove the last digit from n */
        n = n / 10;
    }

    printf("%lld", product);

    return 0;
}
