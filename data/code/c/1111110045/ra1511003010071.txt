#include <stdio.h>
#include<math.h>
int main()
{
  float x[10];
    int  i, n;
    float average, variance, std_deviation, sum = 0, sum1 = 0;
 
    
    scanf("%d", &n);
    ;
    for (i = 0; i < n; i++)
    {
        scanf("%f", &x[i]);
    }
   
    for (i = 0; i < n; i++)
    {
        sum = sum + x[i];
    }
    average = sum / (float)n;
    
    for (i = 0; i < n; i++)
    {
        sum1 = sum1 + pow((x[i] - average), 2);
    }
    variance = sum1 / (float)n;
    std_deviation = sqrt(variance);
    printf("Mean=%.2f\n",average);
   
    printf("SD=%.2f\n",std_deviation);
  return 0;
}
