#include <stdio.h>

#include <math.h>

int main()

{

       float x [50];           

       int n;                     

       float sum, mean, sd;  

       int i;

       scanf("%d", &n);

       for(i = 0; i<n; i++)

           scanf("%f", &x[i]);

           

           sum= 0.0;

      for(i = 0; i < n; i++)

           sum+= x[i];

           mean = sum / n;

 

           sum= 0.0;

      for(i = 0; i < n; i++)

           sum+= (x[i] - mean) * (x[i] - mean);

           sd = sqrt(sum / n);

           printf("Mean=%0.2f\n", mean);

           printf("SD=%0.2f\n", sd);

           return 0;

}
