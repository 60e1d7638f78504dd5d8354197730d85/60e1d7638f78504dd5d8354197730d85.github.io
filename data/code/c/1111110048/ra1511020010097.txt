#include <stdio.h>
  #define HOURTOSEC 3600
  #define MINTOSEC 60

  int main() {
        int seconds, hour = 0, min = 0;

        scanf("%d", &seconds);

        

      
        if (seconds >= HOURTOSEC) {
                while (1) {
                        hour++;
                        seconds = seconds - HOURTOSEC;
                        if (seconds < HOURTOSEC)
                                break;
                }
        }

        
        if (seconds >= MINTOSEC) {
                while (1) {
                        min++;
                        seconds = seconds - MINTOSEC;
                        if (seconds < MINTOSEC)
                                break;
                }
        }

        
        printf("%d %d %d ", hour, min, seconds);
        return 0;
  }
