#include <stdio.h>

int main()
{
    int num1, num2;

    /* Input two numbers from user */

    scanf("%d%d", &num1, &num2);

    /* If num1 is maximum */
    if(num1 > num2)
    {
        printf("%d", num1);        
    }

    /* If num2 is maximum */
    if(num2 > num1)
    {
        printf("%d ", num2);
    }

    /* Additional condition check for equality */
    if(num1 == num2)
    {
        printf("Both are equal");
    }

    return 0;
}
