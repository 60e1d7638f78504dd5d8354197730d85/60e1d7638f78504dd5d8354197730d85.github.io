#include <stdio.h>
 
int main (int argc, char *argv [])
{
    int finalScore,i;
    //printf ("Enter final score: ");
    scanf ("%d", &finalScore);
     
    int pushUps = 0;
    for (i = 7; i <= finalScore; i = i + 7 )
        pushUps = pushUps + i;
     
    printf ( "%d", pushUps );
     
    return 0;
}
