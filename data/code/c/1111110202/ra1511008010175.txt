#include <stdio.h>
#include <string.h>
int main ()
{
  	int i,length;
    char string [11];    
    scanf ("%s", string);     
    length = strlen(string) - 1;
    int letter = 0;
    if ( string [0] >= 'A' && string [0] <= 'Z' )
        letter = 1;
    if ( string [0] >= 'a' && string [0] <= 'z' )
        letter = 1;
    int underScore = 1;
    if ( string [length] == '_' )
        underScore = 0;
    int flag = 0;
    for (i = 1; i <= length; i++) 
    {
		if (string [i] == '_')
            flag++;
        else
            flag = 0;
        if ( flag == 2 ) 
        {
            underScore = 0;
            break;
        }
    }
	if (letter==1 && underScore==1)
        printf ("valid identifier");
    else
        printf ("not a valid identifier");     
    return 0;
}
