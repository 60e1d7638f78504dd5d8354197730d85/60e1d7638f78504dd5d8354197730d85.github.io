#include <stdio.h>
#include <math.h>
int main()  
{  
    int n,i;  
    while(~scanf("%d",&n))  
    {  
        for(i=sqrt(n); i<=n; i++)  
        {  
            if(n%i==0)  
            {  
                int a=i,b=n/i;  
                printf("%d %d\n",a>b?b:a,a<b?b:a);  
                break;  
            }  
        }  
    }  
    return 0;  
}
