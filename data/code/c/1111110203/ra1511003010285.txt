#include <stdio.h>
#include<math.h>
#include<string.h>
int main()
{
  int n;
  while(~scanf("%d",&n))
  {int a,b,i;
   int minn=0x3f3f3f;
   for(i=1;i<=n;i++)
   {
     if(n%i==0)
     {
       int j=n/i;
       if(j>=i)
       {
         if(j-i<minn)
         {
           minn=j-i;
           a=i;
           b=j;
         }
       }
     }
   }
  printf("%d %d\n",a,b);
  }
	return 0;
}
