#include<stdio.h>
#include<math.h>
#define  LL long long
#define  m(a) memset(a,0,sizeof(a))
int main()
{
    int n,i;
    while(~scanf("%d",&n))
    {
        for(i=(int)sqrt(n*1.0);i>=1;i--)
        {
            if(n%i==0)
            {
                printf("%d %d\n",i,n/i);
                break;
            }
        }
    }
    return 0;
}
