#include <stdio.h>  
#include <string.h>  
#include <math.h>  
#define mem(a,b) memset(a,b,sizeof(a))  
int main()  
{  
    int n,i;  
    while(~scanf("%d",&n))  
    {  
        for( i=sqrt(n); i<=n; i++)  
        {  
            if(n%i==0)  
            {  
                int a=i,b=n/i;  
                printf("%d %d\n",a>b?b:a,a<b?b:a);  
                break;  
            }  
        }  
    }  
    return 0;  
}
