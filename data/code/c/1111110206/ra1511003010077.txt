#include <stdio.h>
int main()
{int upper = 0, lower = 0;
char ch[80];
int i;

scanf("%[^\n]s",ch);

i = 0;
while (ch[i] != '$') {
  if (ch[i] >= 'A' && ch[i] <= 'Z')
     upper++;
  if (ch[i] >= 'a' && ch[i] <= 'z')
     lower++;
  i++;
  }

  printf("Uppercase Letters=%d\nLowercase Letters=%d", upper, lower);


	return 0;
}
