#include<stdio.h>
#include<string.h>
int main() {
   int upper = 0, lower = 0;
   char ch[80];
   int i;
     scanf("%s",&ch[i++]);
  i = 0;
   while (ch[i] != '\n') {
      if (ch[i] >= 'A' && ch[i] <= 'Z')
         upper++;
      if (ch[i] >= 'a' && ch[i] <= 'z')
         lower++;
      i++;
   }
   printf("\nUppercase Letters=%d", upper);
   printf("\nLowercase Letters=%d", lower);
 
   return (0);
}
