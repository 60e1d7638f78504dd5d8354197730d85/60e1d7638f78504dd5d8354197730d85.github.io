#include<stdio.h>
 
int main() {
   int upper = 0, lower = 0;
   char ch[80];
   int i;
 	fgets(ch,80,stdin);
   i = 0;
   while (ch[i] != '$') {
      if ((ch[i] >= 'A') && (ch[i] <= 'Z'))
      {upper++;}
      if ((ch[i] >= 'a') && (ch[i] <= 'z'))
      {lower++;}
      i++;
   }
 
   printf("Uppercase Letters=%d\n", upper);
   printf("Lowercase Letters=%d", lower);
 
   return (0);
}
