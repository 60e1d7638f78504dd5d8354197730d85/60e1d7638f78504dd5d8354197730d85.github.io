#include <stdio.h>
int main()
{
	int upper=0,lower=0,i;
  	char ch[80];
  	fgets(ch, sizeof(ch), stdin);
  	i=0;
  	while(ch[i]!='\0')
    {
      if(ch[i]>='A'&&ch[i]<='Z')
        upper++;
      if(ch[i]>='a'&&ch[i]<='z')
        lower++;
      i++;
    }
  	printf("\nUppercase Letters=%d",upper);
  	printf("\nLowercase Letters=%d",lower);
	return 0;
}
