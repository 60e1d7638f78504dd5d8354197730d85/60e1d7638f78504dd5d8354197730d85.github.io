#include <stdio.h>
#include<string.h>
int main()
{
  int upper=0;
  int lower=0;
  char ch[50];
  scanf("%[^\n]s",ch);
  int i=0;
  while (ch[i]!='\0')
  {
    if(ch[i]>='A' && ch[i]<='Z')
    {
      ++upper;
    }
      if(ch[i]>='a' && ch[i]<='z')
      {
        ++lower;
      }
    i++;
  }
  printf("Uppercase Letters=%d\n",upper);
  printf("Lowercase Letters=%d",lower);
	return 0;
}
