int main()
{
	long int n,n1,n2,m1,m2,r;

	{
		r=n % 10;
		n=n / 10;
		if(r % 2 ==0) 
		{
			r=r*m2;
			n2=n2 + r;
			m2=m2 * 10;
		}
		else
		{
			r=r*m1;
			n1=n1 + r;
			m1=m1 * 10;
		}
	 }
  
 printf("%ld %ld",n1,n2); 
  
  return 0;
}
