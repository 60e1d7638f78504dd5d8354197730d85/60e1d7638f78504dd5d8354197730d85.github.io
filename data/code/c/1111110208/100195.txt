#include <stdio.h>
#include <string.h>
#define ull unsigned long long
ull t,i,d,n,x;
ull sum(ull n)
{
	if(n%2==1)
	return (n+1)/2*n;
	else return n/2*(n+1);
}	
int main() 
{
	
  
	scanf("%llu",&t);
	while(t--)
	{
		scanf("%llu %llu",&d,&n);
		x=sum(n);
		for(i=2;i<=d;i++) x=sum(x);
		printf("%llu\n",x);
	}
	return 0;
}
