#include <stdio.h>
#include<stdlib.h>
int main()
{
	int t,i,j, d, n, k;
  scanf("%d",&t);
  while(t--)
  {
    scanf("%d%d", &d, &n);
    int p=n;
    for(i=0;i<d;i++)
    {
      int sum=0;
      for(k=1;k<=p;k++)
      {
        sum+=k;
      }
        p=sum;
    }
    printf("%d\n",p);
  }
  return 0;
}
