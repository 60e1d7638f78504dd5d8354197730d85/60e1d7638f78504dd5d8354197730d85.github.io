#include <stdio.h>
int calcSum(int n) {
	int i,sum = 0;
	for( i=1; i<=n; i++)
		sum += i;
	return sum;
}
int main() {
	int i,test, d, n, result;
	scanf("%d", &test);
	while(test--) {
		scanf("%d%d", &d, &n);
		result = n;
		for( i=0; i<d; i++) {
			result = calcSum(result);
		}
		printf("%d\n", result);
    }


	return 0;
}
