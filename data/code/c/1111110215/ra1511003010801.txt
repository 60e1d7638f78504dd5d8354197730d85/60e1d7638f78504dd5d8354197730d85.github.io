#include <stdio.h>
int main()
{
int n,i,a[20],count=0,sum=0;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
  	scanf("%d",&a[i]);
  }
  
  for(i=0;i<n;i++)
  {
  		if(a[i]>0)
        {
        	count++;
        }
    	else
        {
        	sum=sum+a[i];
        }
  }
  
  printf("%d %d",count,sum);
  
	return 0;
}
