#include <stdio.h>
int main()
{
  int a[100];
  int i;
  int n;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  int count=0;
  int sum=0;
  for(i=0;i<n;i++)
  {
    if(a[i]>0)
    {
      count++;
    }
    else if(a[i]<0)
    {
     sum=sum+a[i]; 
    }
  }
  printf("%d ",count);
  printf("%d",sum);

	return 0;
}
