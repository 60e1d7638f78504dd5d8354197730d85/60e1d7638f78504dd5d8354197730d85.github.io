#include <stdio.h>
int main()
{
int n;
  scanf("%d",&n);
  int a[n];
  int i;
  int pc=0;
  int sn=0;
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  for(i=0;i<n;i++)
  {
    if(a[i]>0)
    {
      pc++;
    }
    if(a[i]<0)
    {
      sn=sn+a[i];
    }
  }
 printf("%d %d",pc,sn);
 
	return 0;
}
