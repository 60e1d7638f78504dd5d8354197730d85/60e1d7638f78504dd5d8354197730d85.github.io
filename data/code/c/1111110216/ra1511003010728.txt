#include <stdio.h>
int main()
{
  int a,b,i,gcd,min;
  scanf("%d",&a);
  scanf("%d",&b);
  for(i=1;i<=a && i<=b;++i)
  {
    if(a%i==0 && b%i==0)
    gcd=i;
  }
printf("GCD=%d\n",gcd);
  min=(a>b)?a:b;
  while(1)
  {
    if (min%a==0 && min%b==0)
    {
      printf("LCM=%d",min);
      break;
    }
    ++min;
  }
  
	return 0;
}
