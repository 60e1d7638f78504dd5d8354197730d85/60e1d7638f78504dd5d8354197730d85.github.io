#include <stdio.h>
int main()
{
  int num1, num2, gcd, lcm, r, n, d;
  scanf("%d",&num1);
  scanf("%d",&num2);
  if (num1 > num2)
    {
        n = num1;
        d = num2;
    }
    else
    {
        n = num2;
        d = num1;
    }
    r = n % d;
    while (r != 0)
    {
        n = d;
        d = r;
        r   = n % d;
    }
    gcd = d;
    lcm = num1 * num2 / gcd;
  printf("GCD=%d\n",gcd);
  printf("LCM=%d",lcm);

	return 0;
}
