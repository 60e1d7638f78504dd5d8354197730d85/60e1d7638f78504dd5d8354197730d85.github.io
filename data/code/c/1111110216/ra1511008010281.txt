#include <stdio.h>
int main()
{
   int n1, n2, prod, gcd, lcm ; 
 scanf("%d %d", &n1, &n2) ; 
 prod = n1 * n2 ; 
 while(n1 != n2) 
 { 
  if(n1 > n2) 
   n1 = n1 - n2 ; 
  if(n2 > n1) 
   n2 = n2 - n1 ; 
 } 
 gcd = n1 ; 
 lcm = prod / gcd ; 
 printf("\nGCD=%d", gcd) ; 
 printf("\nLCM=%d", lcm); 

	return 0;
}
