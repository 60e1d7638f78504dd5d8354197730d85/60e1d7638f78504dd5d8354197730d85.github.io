#include <stdio.h>
int main()
{
int i=1,l,x,h,n1,n2;
  scanf("%d",&n1);
  scanf("%d",&n2);
  for(i=1;i<=n1*n2;i++)
  {
    if(i%n1==0&&i%n2==0)
    {
      l=i;
      break;
    }
  }
  
  if(n1<n2)
  {
    x=n1;
  }
  else
    x=n2;
  
   for(i=x;i>=1;i--)
  {
    if(n1%i==0&&n2%i==0)
    {
      h=i;
      break;
    }
   }
  printf("GCD=%d\nLCM=%d",h,l);
	return 0;
}
