#include<stdio.h>
int main()
{
    int t;
    char id;
    scanf("%d",&t);
    while(t--)
    {
     scanf(" %c",&id);
       switch(id)
       {
         case 'B':
         case 'b':
            printf("BattleShip\n");
            break;
         case 'c':
         case 'C':
             printf("Cruiser\n");
             break;
         case 'd':
         case 'D':
             printf("Destroyer\n");
             break;
         case 'F':
         case 'f':
             printf("Frigate\n");
        }
    }
  return 0;
}
