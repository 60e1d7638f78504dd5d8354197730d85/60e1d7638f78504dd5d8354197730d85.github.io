#include <stdio.h>
 
int main() {
	int t;
	char c,ch;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%c",&c);
		scanf("%c",&ch);
		if(ch=='B' || ch=='b')
		printf("BattleShip\n");
		else if(ch=='C' || ch=='c')
		printf("Cruiser\n");
		else if(ch=='D' || ch=='d')
		printf("Destroyer\n");
		else
		printf("Frigate\n");
	}
	return 0;
}
