#include<stdio.h>
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		char a ;
		scanf(" %c",&a);
		if(a=='B'||a=='b')
			printf("BattleShip\n");
		if(a=='C'||a=='c')
			printf("Cruiser\n");
		if(a=='D'||a=='d')
			printf("Destroyer\n");
		if(a=='F'||a=='f')
			printf("Frigate\n");
	}
	return 0;
}
