#include <stdio.h>
int main()
{
    int T;
  char C;
  scanf("%d",&T);
  while(T--)
  {
    scanf(" %c",&C); 
    if(C=='B' || C=='b')
    {
      printf("BattleShip\n");
    }
    else if(C=='C' || C=='c')
    {
      printf("Cruiser\n");
    }
    else if(C=='F' || C=='f')
    {
     printf("Frigate\n");
    }
    else if(C=='D' || C=='d')
    {
     printf("Destroyer\n");
    }
  }
      
	return 0;
}
