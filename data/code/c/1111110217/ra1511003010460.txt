#include <stdio.h>
int main()
{
    int n;
    scanf("%d",&n);
    int i;
    for(i=0;i<n;i++)
    {
      char c;
      scanf(" %c",&c);
      if(c=='B' || c=='b')
        printf("BattleShip");
      else if(c=='C' || c=='c')
        printf("Cruiser");
      else if(c=='D' || c=='d')
        printf("Destroyer");
      else if(c=='F' ||c=='f')
        printf("Frigate");
      printf("\n");
    }
	return 0;
}
