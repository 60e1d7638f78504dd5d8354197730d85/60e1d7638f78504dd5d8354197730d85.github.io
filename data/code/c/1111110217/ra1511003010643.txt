#include<stdio.h>
 
int main(){
	int tc,T;
	scanf("%d",&T);
	for(tc=0;tc<T;tc++){
	char ch,f;
	scanf("%c",&f);
	scanf("%c",&ch);
		if(ch=='B' || ch=='b')
			printf("BattleShip\n");
		else if(ch=='C' || ch=='c')
			printf("Cruiser\n");
		else if(ch=='D' || ch=='d')
			printf("Destroyer\n");
		else if(ch=='F' || ch=='f')
			printf("Frigate\n");
	}
	return 0;
}
