#include <stdio.h>
int main()
{
int t;
char chr;
scanf("%d",&t);
while(t--)
{
scanf(" %c",&chr);
if(chr=='c' || chr=='C')
printf("Cruiser\n");
else if(chr=='b' || chr=='B')
printf("BattleShip\n");
else if(chr=='D' || chr=='d')
printf("Destroyer\n");
else if(chr=='f' || chr=='F')
printf("Frigate\n");
}
	return 0;
}
