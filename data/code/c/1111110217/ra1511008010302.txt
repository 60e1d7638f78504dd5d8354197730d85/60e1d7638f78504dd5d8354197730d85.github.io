#include <stdio.h>
int main()
{
  int t;
	scanf("%d",&t);
	while(t--)
	{
		char ch;
		getchar();
		scanf("%c",&ch);
		if ((ch =='B')||(ch=='b')) printf("BattleShip");
		else if ((ch=='C')||(ch=='c')) printf("Cruiser");
		else if ((ch=='D')||(ch=='d')) printf("Destroyer");
		else if ((ch=='F')||(ch=='f')) printf("Frigate");
		printf("\n");
    }

	return 0;
}
