#include <stdio.h>
int main()
{
	int i;
  	char ch;
  	scanf("%d\n",&i);
  	while(i--)
    {
        scanf("%c\n", &ch);
  		if (ch=='b' || ch=='B')
    	  printf("BattleShip \n");
      	else if(ch=='c' || ch=='C')
   		  printf("Cruiser \n");
 		else if(ch=='d' || ch=='D')
          printf("Destroyer\n");
  		else if(ch=='f' || ch=='F')
   	      printf("Frigate\n");
    }
	return 0;
}
