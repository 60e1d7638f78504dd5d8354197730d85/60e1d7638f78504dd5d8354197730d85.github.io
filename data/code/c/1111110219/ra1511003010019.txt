#include<stdio.h>
int main()
{
    int t,a[10^6];
    int i,j,temp;
    scanf("%d",&t);
    for(i=0;i<t;i++)
    {
        scanf("\n%d",&a[i]);
    }
    for(i=0;i<t;i++)
    {
        for(j=i+1;j<t;j++)
        {
            if(a[i]>a[j])
            {
                temp=a[i];
                a[i]=a[j];
                a[j]=temp;
            }
        }
    }
    for(i=0;i<t;i++)
    {
        printf("\n%d",a[i]);
    }
  return 0;
}
