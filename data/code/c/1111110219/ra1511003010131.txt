#include <stdio.h>
int main()
{
    int i,j,n,a[100],swap;
   scanf("%d",&n);
   for(i=0;i<n;i++)
     scanf("%d",&a[i]);
   for(i=0;i<n;i++){
     for(j=0;j<n-i-1;j++){
     	if(a[j]>a[j+1]){
          swap=a[j];
          a[j]=a[j+1];
          a[j+1]=swap;}}}
    for(i=0;i<n;i++)
      printf("%d\n",a[i]);
	return 0;
}
