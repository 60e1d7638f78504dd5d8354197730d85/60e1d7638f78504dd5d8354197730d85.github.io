#include <stdio.h>
int main()
{

  	int n,a[100],i,j,temp;
  	scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d ",&a[i]);
  }
    for(i=0;i<n-1;i++)
    {
      for(j=0;j<n-i-1;j++)
        if(a[j+1]<a[j])
      {
        temp=a[j+1];
          a[j+1]=a[j];
          a[j]=temp;
      }
    }
  
  for(i=0;i<n;i++)
  {
    printf("%d\n",a[i]);
  }
  
	return 0;
}
