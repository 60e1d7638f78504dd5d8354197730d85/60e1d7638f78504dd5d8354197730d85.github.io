#include <stdio.h>
int main()
{
  int n,i,j,t,a[100];
  scanf("%d",&n);
  for(i=0;i<n;i++)
   scanf("%d",&a[i]);
  for(i=0;i<n;i++)
  {
    for(j=i+1;j<n;j++)
    {
      if(a[i]>a[j])
      {
        t=a[i];
        a[i]=a[j];
        a[j]=t;
      }
    }
  }

  for(j=0;j<n;j++)
  printf("\n%d",a[j]);
  return 0;
}
