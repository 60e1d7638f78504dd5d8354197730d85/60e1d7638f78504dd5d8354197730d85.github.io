#include <stdio.h>
int main()
{
	int n;
  	scanf("%d",&n);
  	int i=0,j=0;
  	int a[n];
  	int tmp;
  	for(i=0;i<n;i++) scanf("%d",&a[i]);
  	for(i=0;i<n-1;i++)
    {
      for(j=0;j<(n-i-1);j++)
      {
        if(a[j]>a[j+1])
        {
          tmp=a[j];
          a[j]=a[j+1];
          a[j+1]=tmp;
        }
      }
    }
  	for(i=0;i<n;i++) 
    {
      printf("%d",a[i]);
      if(i<n) printf("\n");
    }
      return 0;
  
}
