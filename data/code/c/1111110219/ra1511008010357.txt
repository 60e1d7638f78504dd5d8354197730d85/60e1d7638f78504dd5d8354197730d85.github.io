#include <stdio.h>
int main()
{
  int n;
  scanf("%d",&n);
  int a[n];
  int i=0,j=0,temp=0;
  for(i=0;i<n;i++)
    scanf("%d",&a[i]);
  for(i=0;i<n-1;i++)
  {
    for(j=0;j<n-1-i;j++)
    {
      if(a[j]>a[j+1])
      {
        temp=a[j];
        a[j]=a[j+1];
        a[j+1]=temp;
      }
    }
  }
  for(i=0;i<n;i++)
    printf("%d\n",a[i]);
 return 0;
}
