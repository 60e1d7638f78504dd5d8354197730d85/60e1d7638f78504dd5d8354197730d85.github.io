#include <stdio.h>
int main()
{
  int n;
  scanf("%d", &n);
  int arr[n], i, j, temp;
  
  for (i = 0; i < n; i++)
  {
    scanf("%d", &arr[i]);
  }
  
  for(i=1;i<n;i++){
      temp=arr[i];
      j=i-1;
      while((temp<arr[j])&&(j>=0)){
      arr[j+1]=arr[j];
          j=j-1;
      }
      arr[j+1]=temp;
  }
  
  for (i = 0; i < n; i++)
  {
    printf("%d\n", arr[i]);
  }

	return 0;
}
