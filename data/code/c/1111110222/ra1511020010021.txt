#include <stdio.h>
#include<string.h>
int main()
{
  int n, i;
  	scanf("%d", &n);
  	for(i=0; i<n; i++)
    {
      	int c = 0, j;
    	char b[20];
      	scanf("%s", b);
      	for(j=0; j<strlen(b); j++)
        {
        	if(b[j] == '0' && b[j+1] == '1')
            {
            	c++;
            }
        }
      	printf("%d\n", c);
    }
	return 0;
}
