#include <stdio.h>
#include <string.h>
#define MAX 100
int main()
{
    char str[MAX];
    int i,j;
    scanf("%s",str);
    for ( i = 0; str[i]!='\0';++i)
    {
        while(!((str[i]>='a' && str[i]<='z') || (str[i]>='A' && str[i]<='Z' || str[i]== '\0')))
        {
            for(j=i;str[j]!='\0';++j)
            {
                str[j]=str[j+1];
            }
           str[j] = '\0';
        }
    }
  puts(str);
  	return 0;
}
