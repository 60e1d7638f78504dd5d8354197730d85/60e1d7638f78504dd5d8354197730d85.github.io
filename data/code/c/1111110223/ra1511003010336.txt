#include <stdio.h>
int main()
{   int i,j;
    char str[50];
    scanf("%s", str);
   for(i=0; str[i]!='\0'; ++i)
    {
        while (!((str[i]>='a'&&str[i]<='z') || (str[i]>='A'&&str[i]<='Z' || str[i]=='\0')))
        {
            for(j=i;str[j]!='\0';++j)
            {
                str[j]=str[j+1];
            }
            str[j]='\0';
        }
   }
   
    printf("%s",str);        
             

	return 0;
}
