#include <stdio.h>
int main()
{
  char input[256];
  int i=0;
  
  fgets(input, 256, stdin);
  //input[strlen(input)-1]='\0';
  
  while(input[i]!='\0')
  {
    if(input[i]>='A' && input[i]<='Z')
    {
      input[i]=(input[i]-'A')+'a';
    }
    else if(input[i]>='a'&& input[i]<='z')
    {
      input[i]=(input[i]-'a')+'A';
    }
    i++;
  }
  printf("%s",input);
  	return 0;
}
