#include <stdio.h>
#include <string.h>
int main()
{
  int c = 0;
   char ch; 
  char str[10];
 
   //printf("Input a string\n");
   scanf("%s",str);
 
   while (str[c] != '\0') {
      ch = str[c];
      if (ch >= 'A' && ch <= 'Z')
         str[c] = str[c] + 32;
      else if (ch >= 'a' && ch <= 'z')
         str[c] = str[c] - 32;   
      c++;   
   }
 
   printf("%s\n", str);
	return 0;
}
