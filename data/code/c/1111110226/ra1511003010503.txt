#include <stdio.h>
#include <string.h>
int main()
{
  char str[100],temp;
  int i,j;
  scanf("%s",str);
  for(i=0;i<strlen(str);i++)
  {
    for(j=i+1;j<strlen(str);j++)
    {
      if(str[j]<str[i])
      {
        temp=str[j];
        str[j]=str[i];
        str[i]=temp;
      }
    }
  }
  printf("%s",str);
  return 0;
}
