#include <stdio.h>
#include<stdlib.h>
#include<string.h>
int main()
{
  char str[100],temp;
    int i,j;
    fgets(str,100,stdin);
    for(i=0;str[i];i++)
    {
        for(j=i+1;str[j];j++)
        {
            if(str[j]<str[i])
            {
                temp=str[j];
                str[j]=str[i];
                str[i]=temp;
            }
        }
    }
    printf("%s\n",str);
    
	return 0;
}
