#include <stdio.h>
int main()
{
	int x;	
    float init_bal,balance;
  	float charge=0.50;
	scanf("%d %f",&x,&init_bal);
    if((((int)x)%5)==0)
    {
      balance=init_bal-(float)x-charge;
      if(balance>=0)
      {
        printf("%0.1f",balance);
      }
      else
      {
        printf("%0.1f",init_bal);
      }
    }
  	else
    {
      printf("%0.1f",init_bal);
    }
      
    
  	return 0;
}
