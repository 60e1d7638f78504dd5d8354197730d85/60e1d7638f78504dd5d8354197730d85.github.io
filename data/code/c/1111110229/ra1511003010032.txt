#include <stdio.h>
#include <string.h>
 
int main(void) {
	int total, len, i;
	scanf("%d", &total);
	while (total--) {
	    int one = 0,zero = 0;
	    char buffer[100000];
	    scanf("%s", buffer);
	    len = strlen(buffer);
	    for (i = 0; i<len; i++) {
	        if (buffer[i] == '1') {
	            one++;
	        } else {
	            zero++;
	        }
	    }
	    if ((one >= len-1 && zero == 1) || (zero >= len-1 && one == 1))
	        printf("Yes\n");
        else
            printf("No\n");
	}
	return 0;
}
