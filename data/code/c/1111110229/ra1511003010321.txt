#include<stdio.h>
#include<string.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int i,c=0,d=0;
        unsigned long long l;
        char a[1000001];
        scanf("%s",a);
       l=strlen(a);
    for(i=0;i<l;i++)
    {
        if(a[i]=='0')
            c++;
            else if(a[i]=='1')
                d++;
     }
     if((c>=1 && d==1)||(d>=1 && c==1)||(c==0 && d==1)||(c==1 && d==0))
        printf("Yes\n");
     else
        printf("No\n");
    }
    return 0;
}
