#include <stdio.h>
#include <string.h>
int main() 
{
	int T;
	scanf("%d",&T);
	while(T--)
	{
	    char str[100];
	    int c1=0,c2=0,l=0,i;
	    scanf("%s",str);
	    l = strlen(str);
	    for(i=0;i<l;i++)
	    {
	        if(str[i]==48)
	        c1++;
	        else
	        c2++;
	    }
	   if (c1==1||c2==1)
	   printf("Yes\n");
	   else
	   printf("No\n");
	}
	return 0;
}
