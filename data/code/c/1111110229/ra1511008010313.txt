#include<stdio.h>
#include<string.h>
int main()
{
	int n1,n0,len,i,n;
	char arr[1000];
		scanf("%d",&n);
  	while(n--)
    {
  		scanf("%s",arr);
		n1=n0=0;
		len=strlen(arr);
		for(i=0;i<len;i++){
			if(arr[i]=='0')
				++n0;
			else
				++n1;
		}
		if(n1==len-1 || n0==len-1)
			printf("Yes\n");
		else
			printf("No\n");
    }
	return 0;
}
