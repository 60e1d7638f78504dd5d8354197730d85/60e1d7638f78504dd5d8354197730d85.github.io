#include<stdio.h>
#include<stdio.h>
#include<string.h>
int main()
{
	int t,n,f,i;
	scanf("%d",&t);
	while(t--)
	{
		int one=0,zero=0,c=0;
		char n[100001];
		scanf("%s",n);
		for(i=0;i<strlen(n);i++)
		{
			c++;
			if((n[i]-'0')==0)
				zero++;
			else 
			    one++;
		}
	
		if((c-one)==1 || (c-zero)==1)
		 printf("Yes\n");
		else
		 printf("No\n");
	}
	return 0;	
 
}
