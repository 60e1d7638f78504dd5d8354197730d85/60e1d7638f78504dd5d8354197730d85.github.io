#include <stdio.h>
int main()
{
  int i,j,n,k;
  scanf("%d",&n);
  if(n<10){
  for(i=1;i<=n;i++){
    for(j=1;j<=n;j++){
      if(j<=i){
        printf("%d",j);
      }
      else{
        printf("%d",0);
          }
    }
    for(k=n;k>=1;k--){
      if(k>i){
        printf("%d",0);
      }
      else{
        printf("%d",k);
      }
    }
    printf("\n");
  }
  }
  else{
    printf("no pattern");
  }
    return 0;
}
