#include <stdio.h>
int main()
{
  int n,i,j;
  scanf("%d", &n);
  if(n==10)
    printf("no pattern");
  else
  {
  
  for(i=1;i<=n;i++){
    for(j=1;j<=n;j++){
      if(j<=i)
        printf("%d", j);
      else 
        printf("0");
      }
    for(j=n;j>=1;j--){
      if(j<=i)
        printf("%d",j);
      else
        printf("0");
    }
    printf("\n");
  }
  }
	return 0;
}
