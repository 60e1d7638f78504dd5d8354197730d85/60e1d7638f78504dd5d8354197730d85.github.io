#include<stdio.h>
#include<string.h>
int rem(int i,char *b)
{   int remi,j,len;
    len=strlen(b);
    for(j=0;j<len;j++) b[j]-='0';
    remi=b[0]%i;
    for(j=1;j<len;j++)
    {
        remi=(remi*10+b[j])%i;
    }
    for(j=0;j<len;j++) b[j]+='0';
    return remi;
}
int hcf(int a,int b)
{
  if(b%a==0) return a;
return hcf(b%a,a);
}
