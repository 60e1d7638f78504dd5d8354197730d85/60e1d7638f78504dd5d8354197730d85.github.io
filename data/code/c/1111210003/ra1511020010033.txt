#include <stdio.h>
int gcd(int a,int b)
{
if (b==0)
return a;
else
return gcd(b,a%b);
}

for(;t>0;t--)
{

rem=0;
if(a==0)

else
{for(i=0;array[i]!='\0';i++)
{
