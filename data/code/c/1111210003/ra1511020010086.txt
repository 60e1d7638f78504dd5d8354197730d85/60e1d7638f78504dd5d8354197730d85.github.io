#include <stdio.h>
int main()
{

  int x;
  scanf("%d",&x);
  while(x--)
  {
   int n1, n2, i, gcd;

   // printf("Enter two integers: ");
    scanf("%d %d", &n1, &n2);

    for(i=1; i <= n1 && i <= n2; ++i)
    {
        // Checks if i is factor of both integers
        if(n1%i==0 && n2%i==0)
            gcd = i;
    }

    printf("%d\n", gcd);
  }
	return 0;
}
