#include<stdio.h>
int main()
{
  int t,n,i,count;
   char a[1002];
 
   scanf("%d",&t);
    while(t--)
     {
       scanf("%d",&n);
        scanf("%s",a);
         count=0;
 
     for(i=0;a[i]!=0;i++)
      if(a[i]=='1')
       {
          if(i==0&&a[i+1]=='0')
            
           a[i+1]='2';
       
           else if(i==n-1&&a[i-1]=='0')
             
            a[i-1]='2';
 
           else if(i>0&&i<n-1)
            {
                
                if(a[i-1]=='0') 
                    a[i-1]='2';
                  
                 if(a[i+1]=='0')
                a[i+1]='2';
            }
     }
          
   //printf("%s ",a);        
   for(i=0;i<n;i++)
      if(a[i]=='1'||a[i]=='2')
       count++;
 
   printf("%d\n",n-count);
}
return 0;
}
