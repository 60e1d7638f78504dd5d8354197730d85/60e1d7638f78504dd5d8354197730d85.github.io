#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#include<limits.h>
 
int main()
{
int test,n,i,j,count=0;
scanf("%d",&test);
for(i=0;i<test;i++)
{
	count=0;
scanf("%d",&n);
 
char* str;
 char* str1;

str= (char *) malloc ((n+1)* sizeof ( char));
str1= (char *) malloc ((n+1)* sizeof ( char));
 

scanf("%s",str);
strcpy(str1,str);
for(j=0;j<n;j++)
{
	
	if(j==0 && str[j]=='1')
	{
		str1[j+1]='1';
		}
		else if(j==n-1 && str[j]=='1')
		{
			str1[j-1]='1';
			}
			else if(j!=n-1&& j!=0&&str[j]=='1')
			{
				str1[j-1]='1';
				str1[j+1]='1';
				}
	}
	for(j=0;j<n;j++)
	{
		if(str1[j]=='0')
		{
			count++;
			}
		}
printf("%d\n",count);
free(str1);
free(str);
 
}
  
  return 0;
	}
