#include <stdio.h>
#include <string.h>
int main()
{
  int t,n,i,out;
	char c;
	scanf("%d",&t);
 
	while(t--)
	{
		out=0;
		scanf("%d",&n);
		scanf("%c",&c);
		char str[n+1];
		scanf("%s",str);
 
		int bls[n];
		for(i=0;i<n;i++)	bls[i]=0;
 
		for(i=0;i<n;i++)
		{
			if(i==0 && str[i]=='1')		{bls[i]=1;bls[i+1]=1;}
			else if(i==n-1 && str[i]=='1')	{bls[i]=1;bls[i-1]=1;}
			else if(i!=0 && i!=n-1 && str[i]=='1')	{bls[i]=1;bls[i-1]=1;bls[i+1]=1;}
		}
	
		for(i=0;i<n;i++)
		{
			if(bls[i]==0)	out++;
		}
 
		printf("%d\n",out);
    }
    
	return 0;
}
