#include<stdio.h>
 
int main()
{
    int T;
    int N;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d",&N);
        if((360%N)==0)
        {
                printf("y ");
        }
        else
        {
                printf("n ");
        }
        if(N<=360)
        {
            printf("y ");
        }
        else
        {
            printf("n ");
        }
        if((N*(N+1))/2<=360)
        {
            printf("y\n");
        }
        else
        {
            printf("n\n");
        }
 
    }
    return 0;
}
