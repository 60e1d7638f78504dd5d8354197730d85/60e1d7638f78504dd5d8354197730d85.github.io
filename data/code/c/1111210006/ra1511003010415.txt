#include<stdio.h>
 
int main ()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		int n;
		scanf("%d",&n);
		char a1, a2, a3;
		if(360 % n == 0)
          a1 = 'y';
      else
        a1 = 'n';
		if (n <= 360) a2 = 'y'; else a2 = 'n';
		if (n <= 26) a3 = 'y'; else a3 = 'n';
		printf("%c %c %c\n",a1,a2,a3);
	}
	return 0;
}
