#include<stdio.h>
 
int main()
{
    int t,n;
    scanf("%d",&t);
 
    while(t--)
    {
        scanf("%d",&n);
        char c1='n',c2='n',c3='n';
 
        if(360%n==0)
        c1='y';
 
        if(n<=360)
        c2='y';
 
        if(n*(n+1)/2<=360)
        c3='y';
 
        printf("%c %c %c\n",c1,c2,c3);
    }return 0;
}
