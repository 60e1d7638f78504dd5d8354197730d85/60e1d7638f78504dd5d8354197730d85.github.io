#include<stdio.h>
int main()
{
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        if(n>=1&&n<=26)
        {
            if(360%n==0)
                printf("y y y\n");
            else printf("n y y\n");
        }
        else if(n>26&&n<=360&&(360%n==0))
            printf("y y n\n");
        else if(n>26&&n<=360&&(360%n!=0))
               printf("n y n\n"); 
        else printf("n n n\n");
    }
    return 0;
}
