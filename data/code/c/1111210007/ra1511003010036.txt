#include<stdio.h>
int main()
 {
  int t,n,c,q,a=0,l,r,i;
  scanf("%d",&t);
  int arr[t];
  while(a<t)
   {
	scanf("%d %d %d",&n,&c,&q);
    while(q--)
     {
      scanf("%d %d",&l,&r);
	  if(c>=l&&c<=r)
	   c=l+r-c;
     }
    arr[a]=c;
    ++a;	
   }
  for(i=0;i<t;++i)
   printf("%d\n",arr[i]);	  
  return 0;
 }
