#include<stdio.h>
int main(){
    int t,n,c,q,i;
    scanf("%d",&t);
    while(t--){
        scanf("%d%d%d",&n,&c,&q);
        int l[q],r[q];
        for(i=1;i<=q;i++){
            scanf("%d%d",&l[i],&r[i]);
        }
        for(i=1;i<=q;i++){
            if(c>=l[i]&&c<=r[i])
                c=l[i]+r[i]-c;
        }
        printf("%d\n",c);
    }
    return 0;
}
