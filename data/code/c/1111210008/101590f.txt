#include<stdio.h>
int main()
{
    int t;
    long long int n;
    char c;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lld\n",&n);
        printf("%lld\n",n*(n-1)/2);
        if(t==0)
        break;
        do
        {
            c=getchar_unlocked();
        }while(c!='\n');
    }
    return 0;
}
