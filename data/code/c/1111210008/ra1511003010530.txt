#include<stdio.h>
#define READ_INT(_x)									\
{														\
    int _tmp = 0;										\
    register int _ch = getchar_unlocked();						\
    while (_ch < 48)             						\
	{													\
        _ch = getchar_unlocked();       						    \
	}													\
    while (48 <= _ch)									\
    {													\
		_tmp = (_tmp << 3) + (_tmp << 1) + _ch - 48;	\
		_ch = getchar_unlocked();								\
    }													\
    (_x) = _tmp;										\
}														\
 
int main ()
{
	unsigned long long int n;
	int test;
	char c;
	READ_INT (test)
	while(test--)
	{
		int i;
		READ_INT(n);
		for(i = 0; i < n; i++)
		{
			READ_INT(c);
		}
		printf("%llu\n", (n * (n - 1)) / 2);
	}
	return 0;
}
