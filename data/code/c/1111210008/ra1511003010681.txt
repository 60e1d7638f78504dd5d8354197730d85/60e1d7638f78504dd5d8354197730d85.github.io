    #include<stdio.h>
    #define get getchar_unlocked
    inline int scan()
    {
    int n=0,s=1;
    char p=get();
    if(p=='-') s=-1;
    while((p<'0'||p>'9')&&p!=EOF&&p!='-') p=get();
    if(p=='-') s=-1,p=get();
    while(p>='0'&&p<='9') { n = (n<< 3) + (n<< 1) + (p - '0'); p=get(); }
    return n*s;
    }
    int main()
    {
    int t,i,arr;
    long long int ans,n;
    t=scan();
    while(t--)
    {
    scanf("%lld\n",&n);
    for(i=0;i<n;i++) arr=scan();
    ans=(n*(n-1))/2;

    }

    }
