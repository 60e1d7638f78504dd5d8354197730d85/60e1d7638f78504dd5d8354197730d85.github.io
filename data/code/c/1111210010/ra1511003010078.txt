#include <stdio.h>
int main()
{ int t,i,n,count,min;
  scanf("%d",&t);
  while(t--)
  { scanf("%d",&n);
    int a[n];
    count=0;
    for(i=0;i<n;i++)
      scanf("%d",&a[i]);
     min=a[0];
    for(i=0;i<n;i++)
    {
      if(a[i]<min)
        min=a[i];
    }
   for(i=0;i<n;i++)
     count=count+a[i]-min;
   printf("%d\n",count);
    
  }

	return 0;
}
