#include<stdio.h>
#include<stdlib.h>
 
 
 
int main(){
  
int n,t,i,min,sum=0;
    scanf("%d",&t);
    while(t--){
        scanf("%d",&n);
        int *w=(int *)malloc(sizeof(int)*n);
        min=99999;sum=0;
        for(i=0;i<n;i++){
            scanf("%d",&w[i]);
            if(w[i]<min)min=w[i];
            }
        for(i=0;i<n;i++){
           if(w[i]!=min)sum+=w[i]-min;
            }
        printf("%d\n",sum);
        
    }
    
    
    return 0;
    
}
