#include<stdio.h>
 
int main()
{
 int arr[10005],max,min,i,n,sum,ans,t;
 
 scanf("%d",&t);
  while(t--)
  {
 scanf("%d",&n);
 scanf("%d",&arr[0]);
 sum=min=max=arr[0];
 for(i=1;i<n;i++)
   {
    scanf("%d",&arr[i]);
    sum+=arr[i];
    if(arr[i]>=max) max=arr[i];
    if(arr[i]<=min) min=arr[i];
   }
   ans=(n-1)*(max-min)-(n-2)*max+sum-min-max;
   printf("%d\n",ans);
  }
   return 0;
}
