#include <stdio.h>
int main()
{
    int t,n,i,min;
 
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        long int arr[n];
 
        for(i=0;i<n;i++)
            scanf("%ld",&arr[i]);
        long steps=0;
        min=arr[0];
        for(i=1;i<n;i++)
        {
         if(arr[i]<min)
         {
             min=arr[i];
             
         }
        }
        for(i=0;i<n;i++)
        {
          steps+=arr[i]-min;
        }
        printf("%ld\n",steps);
    }
    return 0;
}
