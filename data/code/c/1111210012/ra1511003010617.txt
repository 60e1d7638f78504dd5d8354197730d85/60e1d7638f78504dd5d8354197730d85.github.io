#include<stdio.h>
    #include<string.h>
    int main(){
    int t,i,n;
    scanf("%d\n",&t);
    char c;
    int a[26]={0};
    while(t--){
    i=0;
    n=0;
    while(i<26){a[i]=0;i++;}
    while(1){
    c=getchar_unlocked();
    if(c>='a'&&c<='z')
    {
    a[c-97]++;
    if(a[c-97]==1)n++;
    if(n>26)break;
    }
    else break;
    }
    printf("%d\n",n);
    }
     
    return 0;}
