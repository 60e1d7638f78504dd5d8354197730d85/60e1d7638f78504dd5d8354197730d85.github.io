    #include<stdio.h>
    #include<string.h>
    #define sc(t) scanf("%d\n",&t)
    #define w(t) while(t--)
     
    int main(void) 
    {
              int t,i,n;
              int a[26]={0};
              char s;
              sc(t);
              w(t)
              {
                        i=0;
                        n=0;
                        while(i<26)
                        {
                                  a[i]=0;
                                  i++;
                        }
                        while(1)
                        {
                                  s=getchar_unlocked();
                                  if(s>='a' && s<='z')
                                  {
                                            a[s-97]++;
                                            if(a[s-97]==1)n++;
                                            if(n>26)break;
                                  }
                                  else
                                  {
                                            break;
                                  }
                        }
                        printf("%d\n",n);
              }
              return 0;
    }
