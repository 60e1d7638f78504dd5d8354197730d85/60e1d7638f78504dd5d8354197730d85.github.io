#include <stdio.h>
int main()
{
long int n,c=0,a1=0,a2=0,a3=0;
    scanf("%ld",&n);
    while(c<n){
        int x;
        scanf("%d",&x);
        a1+=x;
        c++;
        if(c>=n)
            break;
        int y;
        scanf("%d",&y);
        a2+=y;
        c++;
        if(c>=n)
            break;
        int z;
        scanf("%d",&z);
        a3+=z;
        c++;
    }
    printf("%ld %ld %ld",a1,a2,a3);
   
	return 0;
}
