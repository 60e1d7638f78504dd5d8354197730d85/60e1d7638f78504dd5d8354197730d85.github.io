#include <stdio.h>
int main()
{ char bin[100];
    int i=1,count=1;
    scanf("%s",bin);
    do
    {
    	if(bin[i-1]==bin[i])
    	{
    		count++;
    		if(count==6)
    			break;
    	}
    		else count=1;
    	i++;
    }while(bin[i]!='\0');
    if(count!=6)
    	printf("Good luck!\n");
    	else printf("sorry,sorry!\n");
    return 0;
}
