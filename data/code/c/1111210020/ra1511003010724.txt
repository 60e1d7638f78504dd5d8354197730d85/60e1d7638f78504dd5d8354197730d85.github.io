long long int ans(int n,int a,int b,int c)
{long long int sum=0;
int temp,temp1;
 
if(c<=n)
    temp=c;
 
else
    temp=n;
    
    
    
for(i=0;i<=temp;i++)
 
    {   
        
        
        if(n-i<=b)
            temp1=n-i;
        else
            temp1=b;
        
        
        
        
        for(j=0;j<=temp1;j++)
 
        {
            if(i+j+a <= n)
                sum+=(a+1);
            else
            {
                if(a+1-(i+j+a-n) > 0)
                    sum+=(a+1-(i+j+a-n));
            }
 
 
 
 
        }
 
    }
 
    return sum;
}
