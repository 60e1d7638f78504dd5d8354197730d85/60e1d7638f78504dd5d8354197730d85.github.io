#include <stdio.h>
int main()
{
	int n,a,b,c;
  	int i,t,j,k,l,count=0;
  	scanf("%d",&t);
  	while(t--){
      scanf("%d%d%d%d",&n,&a,&b,&c);
      for(i=0;i<=a;i++){
        for(j=0;j<=b;j++){
          for(k=0;k<=c;k++){
            if(i+j+k<=n)
              count++;
          }
        }
      }
      printf("%d\n",count);
      count=0;
    }
	return 0;
}
