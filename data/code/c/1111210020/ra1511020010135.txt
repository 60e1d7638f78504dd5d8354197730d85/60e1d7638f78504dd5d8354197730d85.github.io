#include <stdio.h>
int main()
{
int t,n,a,b,c,i,j;
  scanf("%d",&t);
  while(t--)
  {
    scanf("%d%d%d%d",&n,&a,&b,&c);
    long int sum=0;
    for(i=0;i<=a;i++)
    {
      for(j=0;j<=b;j++)
      {
        if(i+j<=n)
        {
          sum+=min(c+1,n-i-j+1);
         }
      }
    }
   printf("%ld\n",sum);
}
  return 0;
}
 
int min(int l,int m)
{
  if(l<m)
    return l;
  else 
    return m;
}
