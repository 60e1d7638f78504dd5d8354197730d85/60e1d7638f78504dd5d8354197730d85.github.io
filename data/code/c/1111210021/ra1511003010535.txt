#include<stdio.h>
int comb(int);
int main()
{
    int cases,a,b,c,n;
    int ans;
    scanf("%d",&cases);
    while(cases--)
    {
                  scanf("%d%d%d%d",&n,&a,&b,&c);
                  ans = comb(n)-comb(n-a-1)-comb(n-b-1)-comb(n-c-1)+comb(n-a-b-2)+comb(n-a-c-2)+comb(n-b-c-2)-comb(n-a-b-c-3);
                  printf("%d\n",ans);
    }
    return 0;
}
int comb(int n)
{
    if(n<0)
           return 0;
    int ans = ((n+1)*(n+2)*(n+3));
           // combination n+r-1 C r... here r=3;
    ans /= 6;
    return ans;
}
