#include <stdio.h>
int min(int a,int b)
{
  if(a<b)
    return a;
  else
    return b;
}
int main()
{
  int t;
  scanf("%d",&t);
  while(t--)
  {
    int n,a,b,c;
    scanf("%d%d%d%d",&n,&a,&b,&c);
    long int sum=0;
    int i,j;
    for(i=0;i<=a;i++)
    {
      for(j=0;j<=b;j++)
      {
        if(i+j<=n)
        {
          sum+=min(c+1,n-i-j+1);
          
        }
      }
    }
    printf("%ld\n",sum);
  }
	return 0;
}
