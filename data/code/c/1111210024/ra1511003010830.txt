#include <stdio.h>
int main()
{
    int low, high, i, flag , t;
  	scanf ("%d" , &t) ;
  	while(t--){
    scanf("%d %d", &low, &high);
    if ( low==1 )
      low = low+1 ;
    while (low <= high){
        flag = 0;
        for(i = 2; i <= low/2; ++i){
            if(low % i == 0){
                flag = 1;
                break;
            }
        }
        if (flag == 0)
            printf("%d\n", low);
        ++low;
    }
    printf ("\n") ;
    }
    return 0;
}
