#include<stdio.h>
  
int main(){
  int t;
  scanf("%d",&t);
  int b=0;
  while(b<t)
  {
    int a,N, i, j, isPrime, n;
    scanf("%d%d",&a,&N);
     
    for(i = a; i <= N; i++)
    {
        isPrime = 0;

        for(j = 2; j <= i/2; j++){

             if(i % j == 0){
                 isPrime = 1;
                 break;
             }
        }
          
        if(isPrime==0 && N!= 1 && i!=1)
            printf("%d\n",i);
    }
   b++;
    printf("\n");
  }
   return 0;
}
