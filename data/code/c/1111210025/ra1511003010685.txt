#include <stdio.h>
#include <stdlib.h>
 
int main() {
int n, k, i, t;
char val[10];
scanf("%d%d", &n, &k);
int *a = (int*)calloc(n,sizeof(int));
 
int count = 0;
for(i=0; i<k; i++) {
scanf("%s", val);
if(val[2] == 'I') {
scanf("%d", &t);
a[t-1]++;
if(a[t-1] == 1)
count++;
else {
count--;
a[t-1] = 0;
}
}
else {
count = 0;
a = (int*)calloc(n,sizeof(int));
}
printf("%d\n",count);
}
 
return 0;
}
