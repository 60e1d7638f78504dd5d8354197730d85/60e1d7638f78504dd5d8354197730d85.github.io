#include<stdio.h>
#include<string.h>
int main()
    {
    	int n,k,i,j,add=0;
    	scanf("%d %d",&n,&k);
    	int a[1005]={0};
    	
    	for(i=0;i<k;i++)
    	{
    		char str[10];
    		int x;
    		scanf("%s",str);
    		if(!strcmp(str,"CLICK"))
    		{
    			scanf("%d",&x);
    			
    			if(a[x]==0)
    				{
    					
    					add++;
    					a[x]=1;
    					
    				}
    			else if(a[x]==1 )
    				{
    					
    					add--;
    					a[x]=0;
    				}
    			
    		}
    		else 
    		{
    			for(j=0;j<1005;j++)
    				a[j]=0;
    			add=0;
    		}	
    		printf("%d\n",add);
        }
      return 0;}
