#include<stdio.h>
int main()
{
    int n,k,count=0;
    scanf("%d%d",&n,&k);
    int a[n],i,x;
    for(i=0;i<n;i++)
        a[i]=0;
    char b[8];
    for(i=0;i<k;i++)
    {
        scanf("%s%d",b,&x);
        if(b[2]=='I')
        {
            if(a[x-1])
            {
                count--;
                a[x-1]=0;
            }
            else
            {
                count++;
                a[x-1]=1;
            }
        }
        else if(b[2]=='O')
        {
            int l;
            for(l=0;l<n;l++)
                a[l]=0;
                count=0;
        }
        printf("%d\n",count);
    }
    return 0;
}
