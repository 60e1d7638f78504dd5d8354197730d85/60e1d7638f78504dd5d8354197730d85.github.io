#include<stdio.h>
#include<string.h>
int main()
{
	int n,k,a[1001],count,i,p;
	char c[20];
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	    a[i]=0;
	scanf("%d",&k);
	while(k--)
	{
		count=0;
		scanf("%s",c);
		if(strcmp(c,"CLOSEALL")==0)
		{   printf("0\n");
		for(i=1;i<=n;i++)
	    a[i]=0;
	    }
		else
		{
			scanf("%d",&p);
			if(a[p]==0)   
		      a[p]=1;
		   else
		      a[p]=0;    
		   for(i=1;i<=n;i++)
		   {
			   if(a[i]==1)
			      count++;
			} 
			printf("%d\n",count);
		}
	}
	return(0);
}
