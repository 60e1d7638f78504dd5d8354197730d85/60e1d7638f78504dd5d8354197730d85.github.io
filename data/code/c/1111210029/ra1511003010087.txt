#include <stdio.h>
int main()
{
   int a,b,c;
    scanf("%d%d%d",&a,&b,&c);
    if((c%a==0)&&(c%b==0))
    {
       printf("strong");
    }
     else if((c%a==0)||(c%b==0))
     {
        printf("accept");
     }
     else if((c<a)||(c<b))
     {
       printf("impossible");
     }
    else
      printf("reject");
  
    
	return 0;
}
