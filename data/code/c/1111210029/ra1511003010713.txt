#include <stdio.h>
int main()
{
	int x,y,z;
  	scanf("%d%d",&x,&y);
  	scanf("%d",&z);
  	if(z>x&&z>y)
    {
	  if(z%x==0&&z%y==0)
        printf("strong");
      else if(z%x==0||z%y==0)
        printf("accept");
      else
        printf("reject");
    }
  	else
      printf("impossible");
	return 0;
}
