#include <stdio.h>
int main()
{
	int x,y,k;
    scanf("%d%d",&x,&y);
    scanf("%d",&k);
    if(k>x||k>y)
    {
    if(k%x==0&&k%y!=0||k%x!=0&&k%y==0)
      printf("accept");
    else if(k%x==0&&k%y==0)
      printf("strong");
    else
      printf("reject");
    }
    else
      printf("impossible");
    
	return 0;
}
