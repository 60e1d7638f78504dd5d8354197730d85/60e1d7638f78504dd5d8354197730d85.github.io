#include <stdio.h>
int main()
{
  int x,y;
  int key;
  scanf("%d %d",&x,&y);
  scanf("%d",&key);
  if(key%x==0 && key%y==0)
  {
    printf("strong");
  }
  else if(key%x==0 || key%y==0)
  {
    printf("accept");
  }
  else if(key<x || key<y)
  {
    printf("impossible");
  }
  else
  {
    printf("reject");
  }
  
	return 0;
}
