#include <stdio.h>
#include <math.h>
 
int main()
{
    int number, i;
    int sum = 0;
 
    scanf("%d", &number);
    sum = pow(((number * (number + 1) ) / 2),2);
    for (i = 1; i <= number; i++)
    {
        if (i != number)
            printf("%d^3 + ", i);
        else
            printf("%d^3=%d ", i, sum);
    }
    return 0;
}
