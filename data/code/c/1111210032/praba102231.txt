#include <stdio.h>
int main()
{
  int mark;
  scanf("%d",&mark);
  if (mark>100)
    printf("ERROR");
  else if((mark>=75)&& (mark<=100))  
    printf("GRADE A");
  else if((mark>=60)&& (mark<75)) 
    printf("GRADE B");
  else if((mark>=50)&& (mark<60)) 
    printf("GRADE C");
  else
    printf("FAIL");
  	return 0;
}
