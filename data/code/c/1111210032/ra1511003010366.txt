#include <stdio.h>
int main()
{
	int marks ;
  	scanf ("%d" , &marks) ;
  	if ( marks>=75 && marks<=100 )
      printf ("GRADE A") ;
  	else if ( marks>=60 && marks<=75 )
      printf ("GRADE B") ;
  	else if ( marks>=50 && marks<=60 )
      printf ("GRADE C") ;
  	else if ( marks<50 )
      printf ("FAIL") ;
  	else
      printf ("ERROR") ;
	return 0;
}
