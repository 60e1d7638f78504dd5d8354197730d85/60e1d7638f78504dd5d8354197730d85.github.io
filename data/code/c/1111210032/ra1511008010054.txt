#include <stdio.h>
int main()
{
  int m;
  scanf("%d",&m);
  if(m>=75&&m<=100)
  {
    printf("GRADE A");
  }
  else if(m>=60&&m<75)
  {
    printf("GRADE B");
  }
  else if(m>=50&&m<60)
  {
    printf("GRADE C");
  }
  else if(m<50)
  {
    printf("FAIL");
  }
  else
  {
    printf("ERROR");
  }
  return 0;
}
