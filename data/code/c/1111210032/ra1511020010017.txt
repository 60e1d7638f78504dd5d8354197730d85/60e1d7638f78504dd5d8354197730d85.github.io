#include <stdio.h>
int main()
{
	int n;
  	scanf("%d", &n);
  	if(n>=75 && n<=100)
      	printf("GRADE A");
  	else if(n>=60 && n<75)
      	printf("GRADE B");
  	else if(n>=50 && n<60)
      	printf("GRADE C");
  	else if(n>100 || n<0)
      	printf("ERROR");
  	else
      	printf("FAIL");
	return 0;
}
