#include <stdio.h>

#define MAX_SIZE 100 

int main()
{
    int arr[MAX_SIZE]; 
    int i, N;

   
    scanf("%d", &N);

    
    for(i=0; i<N; i++)
    {
        scanf("%d", &arr[i]);
    }

    
    for(i=0; i<N; i++)
    {
        
        if(arr[i] < 0)
        {
            printf("%d \t", arr[i]);
        }
    }

    return 0;
}
