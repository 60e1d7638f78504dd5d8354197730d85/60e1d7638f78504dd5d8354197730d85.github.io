#include<stdio.h>
  
int main()
{
    int inputArray[100], elementCount, counter;
    scanf("%d", &elementCount);
    for(counter = 0; counter < elementCount; counter++){
        scanf("%d", &inputArray[counter]);
    }
    for(counter = 0; counter < elementCount; counter++){
        if(inputArray[counter] < 0) 
        {
            printf("%d ", inputArray[counter]);
        }
    }
          

    return 0;
}
