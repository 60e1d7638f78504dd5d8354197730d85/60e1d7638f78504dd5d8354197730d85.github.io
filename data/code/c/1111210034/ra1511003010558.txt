#include <stdio.h>
void getword(int a)
{
  if(a==0)
    printf("Zero ");
  else if(a==1)
    printf("One ");
  else if(a==2)
    printf("Two ");
  else if(a==3)
    printf("Three ");
  else if(a==4)
    printf("Four ");
  else if(a==5)
    printf("Five ");
  else if(a==6)
    printf("Six ");
  else if(a==7)
    printf("Seven ");
  else if(a==8)
    printf("Eight ");
  else
    printf("Nine ");
}
int main()
{
	int n;
  	scanf("%d",&n);
  	while(n>0)
    {
      getword(n%10);
      n=n/10;
    }
	return 0;
}
