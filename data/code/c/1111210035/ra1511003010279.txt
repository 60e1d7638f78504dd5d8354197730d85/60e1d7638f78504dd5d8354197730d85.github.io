#include<stdio.h>
int lcm(int,int);
int main()
{
int NM,NM1,DN1,NM2,DN2,LCM;


scanf("%d%d",&NM1,&DN1);

scanf("%d%d",&NM2,&DN2);
  if(DN1==0 || DN2==0)
    printf("invalid input");
LCM = lcm(DN1,DN2);



NM1=NM1*(LCM/DN1);
NM2=NM2*(LCM/DN2);
NM=NM1+NM2;
  if(NM%2==0 && LCM%2==0)
  {
    NM=NM/2;
  LCM=LCM/2;
printf(" %d %d",NM,LCM);
  }
  else if(NM%3==0 && LCM%3==0)
  {
     NM=NM/3;
  LCM=LCM/3;
printf(" %d %d",NM,LCM);
  }
  else
    printf(" %d %d",NM,LCM);
    
return 0;
}
int lcm(int N1,int N2)
{
static int TEMP = 1;
if(TEMP % N2 == 0 && TEMP % N1 == 0)
return TEMP;
TEMP++;
lcm(N1,N2);
return TEMP;
}
