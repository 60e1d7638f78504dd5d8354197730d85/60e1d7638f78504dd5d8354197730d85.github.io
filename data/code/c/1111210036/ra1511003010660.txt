 #include <stdio.h>

  int main() {
        int numerator, denominator, wholeNo;

       
        scanf("%d", &numerator);
      
        scanf("%d", &denominator);

       
        if (numerator < denominator) {
                printf("Mixed fraction impossible!!\n");
        } else if (numerator % denominator == 0) {
              
                wholeNo = numerator / denominator;
                printf("%d\n", wholeNo);
        } else {
               
                wholeNo = numerator / denominator;
                numerator = numerator % denominator;
                printf("%d %d %d\n",
                        wholeNo, numerator, denominator);
        }
        return 0;
  }
