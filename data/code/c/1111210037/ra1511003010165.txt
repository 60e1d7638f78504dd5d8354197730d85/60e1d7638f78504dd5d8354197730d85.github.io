#include<stdio.h>
#include<assert.h>
int main()
{
    int T,N;
    scanf("%d",&T);
    assert(T>=1 && T<=100);
    while(T--)
    {
        scanf("%d",&N);
        assert(N>=1 && N<=500);
        if((N-1)%8==0)
            printf("%dLB",N+3);
        else if((N-4)%8==0)
            printf("%dLB",N-3);
        else if((N-2)%8==0)
            printf("%dMB",N+3);
        else if((N-5)%8==0)
            printf("%dMB",N-3);
        else if((N-3)%8==0)
            printf("%dUB",N+3);
        else if((N-6)%8==0)
            printf("%dUB",N-3);
        else if((N-7)%8==0)
            printf("%dSU",N+1);
        else if(N%8==0)
            printf("%dSL",N-1);
    }
    return 0;
}
