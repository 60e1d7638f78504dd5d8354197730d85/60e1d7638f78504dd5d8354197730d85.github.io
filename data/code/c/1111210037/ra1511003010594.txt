#include <stdio.h>
 
int main(void) {
    int t,n;
    int a[10];
    a[1]=4;
    a[2]=5;
    a[3]=6;
    a[4]=1;
    a[5]=2;
    a[6]=3;
    a[7]=8;
    a[8]=7;
    
    scanf("%d",&t);
    while(t--)
    {
        int s,r,p;
        scanf("%d",&n);
        s=n/8;
        r=n%8;
        if(r==0)
        {
            r=8;
            s--;
        }
        p=8*s+a[r];
        if(r==1||r==4)
        {
            printf("%dLB",p);
        }
        else if(r==2||r==5)
        {
            printf("%dMB",p);
        }
        else if(r==3||r==6)
        {
            printf("%dUB",p);
        }
        else if(r==7)
        {
            printf("%dSU",p);
        }
        else if(r==8)
        {
            printf("%dSL",p);
        }
    }
 
	return 0;
}
