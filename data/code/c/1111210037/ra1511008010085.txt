#include<stdio.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,p;
        scanf("%d",&n);
        p=n%8;
        if(p==0)
        printf("%dSL",n-1);
        else if(p==1)
        printf("%dLB",n+3);
        else if(p==2)
        printf("%dMB",n+3);
        else if(p==3)
        printf("%dUB",n+3);
        else if(p==4)
        printf("%dLB",n-3);
        else if(p==5)
        printf("%dMB",n-3);
        else if(p==6)
        printf("%dUB",n-3);
        else
        printf("%dSU",n+1);
    }
    return 0;
}
