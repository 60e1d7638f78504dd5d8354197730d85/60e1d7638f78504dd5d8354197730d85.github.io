#include<stdio.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n=0,x=0,c=0,y=0;
        scanf("%d",&n);
        n=n-1;
        x=n/8;
        n=n%8;
        if(n<=5)
        {
            c=n+3;
            c=c%6;
            y=c%3;
            switch(y)
        {case 0:
            printf("%dLB",c+x*8+1);
            break;
        case 1:
            printf("%dMB",c+x*8+1);
            break;
 
        case 2:
            printf("%dUB",c+x*8+1);
            break;
        }}
        else
        {
            if(n==6)
            {
                printf("%dSU",7+x*8+1);
            }else
            {
                printf("%dSL",6+x*8+1);
            }
        }
 
    }
}
