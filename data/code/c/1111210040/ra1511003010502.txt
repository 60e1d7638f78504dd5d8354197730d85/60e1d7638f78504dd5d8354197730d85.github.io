#include <stdio.h>
 
int main()
{
    int num, temp, remainder, reverse = 0,sum=0;
     scanf("%d", &num);
    temp = num;
    while (num > 0)
    {
        remainder = num % 10;
        reverse = reverse * 10 + remainder;
        num /= 10;
    }
    if (temp>reverse)
        printf("impossible");
    else
    {
      	sum=temp+reverse;
        printf("%d",sum);
    }
  return 0;
}
