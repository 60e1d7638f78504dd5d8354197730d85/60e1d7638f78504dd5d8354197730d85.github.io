#include <stdio.h>
int main()
{
  int i,j,n,m,max=0;
  scanf("%d%d",&n,&m);
  int a[n][m];
  for(i=0;i<n;i++)
  {
    for(j=0;j<m;j++)
    {
      scanf("%d",&a[i][j]);
      if(a[i][j]>max) max=a[i][j];
    }
  }
  for(i=0;i<n;i++)
  {
    for(j=0;j<m;j++)
    {
      printf("%d ",a[i][j]+max);
    }
    printf("\n");
  }
  return 0;
}
