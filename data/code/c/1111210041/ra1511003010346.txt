#include <stdio.h>
int main()
{
int r,c;
  scanf("%d%d",&r,&c);
  int i,j;
  int l=0;
  int a[r][c];
  for(i=0;i<r;i++){
  	for(j=0;j<c;j++){
    	scanf("%d",&a[i][j]);
      	if(a[i][j]>l){
        	l=a[i][j];
        }
    }
  }
  for(i=0;i<r;i++){
  	for(j=0;j<c;j++){
   		printf("%d ",a[i][j]+l);
    }
    printf("\n");
  }
	return 0;
}
