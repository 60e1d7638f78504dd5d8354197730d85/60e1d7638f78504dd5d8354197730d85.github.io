#include <stdio.h>
int main()
{
	int n,m;
  	scanf("%d %d",&n,&m);
  	int a[n][m],i,j,max=-1000;
  	for(i=0;i<n;i++)
    {
      for(j=0;j<m;j++)
      {
        scanf("%d",&a[i][j]);
        if(max<a[i][j])
          max=a[i][j];
      }
    }
  for(i=0;i<n;i++)
    {
      for(j=0;j<m;j++)
      {
        printf("%d ",a[i][j]+max);
      }
    printf("\n");
  }
  	
	return 0;
}
