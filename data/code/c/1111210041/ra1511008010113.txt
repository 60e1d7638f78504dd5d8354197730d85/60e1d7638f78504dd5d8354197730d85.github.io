#include <stdio.h>
int main()
{
    int r,c,i,j,max=0;
    scanf("%d %d",&r,&c);
    int a[r][c];
    for(i=0;i<r;i++)
      for(j=0;j<c;j++){
        scanf("%d",&a[i][j]);
        if(a[i][j]>max)
          max=a[i][j];
      }
    for(i=0;i<r;i++){
      for(j=0;j<c;j++)
        printf("%d ",a[i][j]+max);
      printf("\n");
    }
	return 0;
}
