#include <stdio.h>
int main()
{int arr[100],i,len,max=0,min=32767,diff,co=0;
 scanf("%d",&len);
 for(i=0;i<len;i++)
 {
   scanf("%d",&arr[i]);
   if(arr[i]>max)
     max=arr[i];
   if(arr[i]<min)
     min=arr[i];
 }
 diff=max-min;
 printf("%d",diff);
 for(i=0;i<len;i++)
 {
   if(diff==arr[i])
   {
     printf("Yes");
     co=1;
     break;
   }
 } 
 if(co==0)
 printf("No");
	return 0;
}
