#include <stdio.h>
int main()
{
int m,n;
  scanf("%d%d",&m,&n);
  int i,j,a[100][100],max=0,min=10000;
  for(i=0;i<m;i++)
  {
    for(j=0;j<n;j++)
    {
      scanf("%d",&a[i][j]);
      printf("%d ",a[i][j]);
      if(a[i][j]>max)
        max=a[i][j];
      if(a[i][j]<min)
        min=a[i][j];
    }
    printf("\n");
  }
  printf("%d %d",min,max);
	return 0;
}
