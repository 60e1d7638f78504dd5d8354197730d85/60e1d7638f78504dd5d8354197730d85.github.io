#include <stdio.h>
int main()
{ int a,b;
 scanf("%d",&a);
 scanf("%d",&b);
 if(a>0&&b>0)
   printf("point(%d,%d)lies in the First quadrant",a,b);
  else if(a<0&&b>0)
   printf("point(%d,%d)lies in the Second quadrant",a,b);
  else if(a<0&&b<0)
   printf("point(%d,%d)lies in the Third quadrant",a,b);
  else if(a>0&&b<0)
   printf("point(%d,%d)lies in the Fourth quadrant",a,b);

	return 0;
}
