#include <stdio.h>
int main()
{
    int a,b,k;
    scanf("%d%d",&a,&b);
    scanf("%d",&k);
    if(k<a||k<b)
      printf("impossible");
  
    else if(((k%a)==0 && ((k%b)==0)))
    {
       printf("strong");
    }
    else if(((k%a)==0 || ((k%b)==0)))
    {
       printf("accept");
    }  
    else
    {
      printf("reject");
    } 
	return 0;
}
