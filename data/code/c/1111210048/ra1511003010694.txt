#include <stdio.h>
int main()
{
  int x,y,key;
  scanf("%d%d",&x,&y);
  scanf("%d",&key);
  if(key>x && key>y)
  {
     if(key%x==0 && key%y==0)
      printf("strong");
     else if(key%x==0 || key%y==0)
      printf("accept");
    else
      printf("reject");
  }
  else
    printf("impossible");

	return 0;
}
