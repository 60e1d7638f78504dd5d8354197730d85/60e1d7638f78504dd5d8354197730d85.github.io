#include<stdio.h>
char arr[11][20];
int main(void)
{
   int t,n,m,i,j,same,k,count,l;
    scanf("%d",&t);
    for(k=0;k<t;k++)
     {
     scanf("%d",&m);
     scanf("%d",&n);
     count=0;
     for(i=0;i<m;i++)      
         scanf("%s",arr[i]);
     for(j=0;j<n;j++)
       {
             same =0;
             for(i=0;i<m;i++)
               {
                   if(arr[i][j]=='1')
                     same++;
               }
             count = count + same*(same-1)/2;              
       }
     printf("%d\n",count);                    
     }     
   return 0;
}
