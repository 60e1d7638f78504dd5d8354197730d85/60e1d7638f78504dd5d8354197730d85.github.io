#include<stdio.h>
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,m,i,j,count=0,sum=0;
		scanf("%d %d",&n,&m);
	   	int a[10][10];
		for(i=0;i<n;i++)
		{
			for(j=0;j<m;j++)
			{
				scanf("%1d",&a[i][j]);
			}
		}
		
		for(j=0;j<m;j++)
		{    count=0;
			for(i=0;i<n;i++)
			{
				if(a[i][j]==1)
				count++;
			}
			if(count>1)
			sum+=(count*(count-1))/2;
		}
		printf("%d\n",sum);
	}
	return 0;
}
