#include<stdio.h>
int t,i,j,n,k,m;
int main()
{
	scanf("%d",&t);
	while(t--)
	{
		int sum=0;
		scanf("%d%d",&n,&m);
		int c[n][m];
		for(i=0;i<n;i++)
		{
			for(j=0;j<m;j++)
			{
				scanf("%1d",&c[i][j]);
			}
		}
		for(j=0;j<m;j++)
		{
			for(i=0;i<(n-1);i++)
			{
				for(k=(i+1);k<n;k++)
				{
					if(c[i][j]==1 && c[k][j]==1 )
					{
						sum=sum+1;
					}
				}
			}
		}
		printf("%d\n",sum);
	}
return 0;
}
