#include <stdio.h>
int main()
{
    long t,p,i,j=11,c=0;
    long a[]={1,2,4,8,16,32,64,128,256,512,1024,2048};
    scanf("%ld",&t);
    for(i=0;i<t;i++)
    {
      scanf("%ld",&p);
      for(j=11,c=0;p>0;)
      {
        if(p-a[j]>=0)
        {
          p-=a[j];
          c++;
        }
        else
          j--;
      }
      printf("%ld\n",c);
    }
	return 0;
}
