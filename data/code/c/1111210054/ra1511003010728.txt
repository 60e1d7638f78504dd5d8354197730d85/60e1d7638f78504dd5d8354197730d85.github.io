#include<stdio.h>
 
int main()
{
    int t,k,n,arr[12]={1,2,4,8,16,32,64,128,256,512,1024,2048};
    scanf("%d",&t);
    while(t--)
    {
              int c=0;
              scanf("%d",&n);
              while(n>0)
              {
               //printf("%d ",n);         
                        for(k=11;n>0&&k>=0;)
                        {
                                          if(arr[k]<=n)
                                          {
                                                       n-=arr[k];
                                                       c++;
                                          }
                                          else 
                                          {
                                               k--;
                                          }
                        }
              }
              printf("%d\n",c);
    }
    return 0;
}
