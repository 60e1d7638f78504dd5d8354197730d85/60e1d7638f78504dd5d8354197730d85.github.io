#include <stdio.h>
#include<math.h>
int main()
{
	int t,p,i=0,c=0;
  	scanf("%d",&t);
  	while(t--){
      scanf("%d",&p);
      c=0;
      while(p>0){
        i=0;
        while(pow(2,i)<=p && i<12)
          i++;
        p=p-pow(2,i-1);
        c++;
      }
      printf("%d\n",c);
    }
	return 0;
}
