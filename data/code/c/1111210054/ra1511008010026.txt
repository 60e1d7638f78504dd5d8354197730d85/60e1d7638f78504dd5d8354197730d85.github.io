
#include<stdio.h>
#include<math.h>
int main(){
	int t,p,k;
	int c;
	scanf("%d",&t);
	while(t--){
		c=0;
		scanf("%d",&p);
		//printf("p = %d\n",p);
		while(p>0){
			k = pow(2,((int)log2((double)p)));
			//printf("k = %d\n",k);
			if(k>2048)
				k = 2048;
			while(p>=k){
				p -= k;
				c++;
				//printf("%d %d\n",p,k);
			}
		}
		printf("%d\n",c);
	}
	return 0;
}
