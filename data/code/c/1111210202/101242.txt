#include<stdio.h>

int main() {
   int n, m, i, j;
	while (scanf("%d%d",&m,&n))
	{
		int aa, bb;
		int a[101] = { 0 };
		for (i = 0; i < n; i++)
		{
			scanf("%d%d",&aa,&bb);
			a[aa]++;
			a[bb+1]--;
		}
		int sum[102];
		sum[0] = a[0];
		for (i = 1; i <= m; i++)
		{
			sum[i] = sum[i-1] + a[i];
		}
		int ok = 1;
		for (i = 1; i <= m; i++)
		if (sum[i]!=1)
		{
			ok = 0;
			printf("%d %d\n",i,sum[i]);
			break;
		}
		if (ok) 
		printf("OK");
		break;
	}
	return 0;
}
