#include <stdio.h>
#include <math.h>
int convertBinaryToDecimal(int n);

int main()
{
   int n;
   // printf("Enter a binary number: ");
    scanf("%d", &n);
    printf("%d",convertBinaryToDecimal(n));
    return 0;
}

int convertBinaryToDecimal(int n)
{
    int decimalNumber = 0, i = 0, remainder;
    while (n!=0)
    {
        remainder = n%10;
        n /= 10;
        decimalNumber += remainder*pow(2,i);
        ++i;
    }
    return decimalNumber;
}
