#include<stdio.h>
#define min(a,b) ((a<b)?a:b)
#define max(a,b) ((a>b)?a:b)
int main()
{
    int Ds,Dt,D,v,Rmax,Rmin;
    float ans;

    {
              scanf("%d%d%d",&Ds,&Dt,&D);
              Rmax=max(Ds,Dt);
              Rmin=min(Ds,Dt);
              if(D>(Rmax+Rmin)) v=(D-Rmax-Rmin);
              else
              if((Rmax-Rmin)>D) v=Rmax-D-Rmin;
              else
              v=0;
              printf("%d\n",v);
    }
    return 0;
}
