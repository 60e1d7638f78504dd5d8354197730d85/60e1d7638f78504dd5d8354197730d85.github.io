#include <stdio.h>
#include <stdlib.h>
typedef long long int LLI;
#define inchar getchar_unlocked
inline int inIntPos() {
	int n, ch;
	while ((ch = inchar()) < '0');
	n=(ch-'0');
	while((ch = inchar()) >= '0')
		n=(n<<3)+(n<<1)+(ch-'0');
	return n;
}
 
#define MAX_N 100001
int a[MAX_N],b[MAX_N];
 
 
int main(void) {
	int t,n,k,i;
	LLI res,v;
 
	t=inIntPos();
	while(t--) {
		n=inIntPos();
		k=inIntPos();
 
		for(i=0; i<n; ++i) a[i]=inIntPos();
		for(i=0; i<n; ++i) b[i]=inIntPos();
 
		res=0;
		for(i=0; i<n; ++i) {
			v=k/a[i];
			v*=b[i];
			if(v>res) res=v;
		}
 
		printf("%lld\n",res);
	}
	return 0;
}
