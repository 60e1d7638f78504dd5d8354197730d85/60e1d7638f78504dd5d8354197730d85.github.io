#include<stdio.h>
#define gc getchar_unlocked
 
inline int getn(){
    int n = 0;
    char c = gc();
    while(c < '0' || c > '9') c = gc();
    while(c >= '0' && c <= '9')
        n = (n<<3) + (n<<1) + c - '0', c = gc();
    return n;
}
 
int main()
{
    unsigned long K,T,N;
    int i=0;
    unsigned long long max;
    T = getn();
    unsigned long long arr[100005];
    while(T--)
    {
        N = getn();
        K = getn();
        max =0;
        for(i=0;i<N;i++)
            arr[i] = K / getn();
        for(i=0;i<N;i++)
        {
            arr[i]*= getn();
            if(max< arr[i]) max = arr[i];
        }
        printf("%llu\n",max);
    }
  return 0;
}
