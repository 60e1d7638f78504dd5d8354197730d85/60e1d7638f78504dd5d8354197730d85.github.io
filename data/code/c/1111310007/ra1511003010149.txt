#include <stdio.h>
int main()
{
      int t,ct,n,m,x,i;
    scanf("%d",&t);
    while(t--){
 	  scanf("%d %d",&n,&m);
      ct=0;
      for (i = 0; i < n; i++) {
      	scanf("%d",&x);
      	ct+=x;
      }
      if((ct+m)%n==0)
       	  printf("Yes\n");
       else
       	printf("No\n");
    }
 return 0;
}
