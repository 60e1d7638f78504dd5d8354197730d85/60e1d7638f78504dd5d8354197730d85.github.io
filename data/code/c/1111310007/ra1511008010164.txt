#include <stdio.h>
int main()
{

  int cases,M,N,i;
    scanf("%d",&cases);
 
    while(cases--)
    {
        scanf("%d",&N);
        scanf("%d",&M);
        int a[N];
        for(i=0;i<N;i++)
        {
            scanf("%d",&a[i]);
        }
 
        int max=a[0];
		for(i=1;i<N;i++)
		{
			if(a[i]>max)
			{
				max=a[i];
            }
 
		}
		int diff=0;
		for(i=0;i<N;i++)
		{
 
			diff=diff+(max-a[i]);
 
		}
 
        int temp=M-diff;
			if(M==diff||(temp%N==0))
		{
			printf("Yes\n");
		}
		else
			printf("No\n");
 
 
 
    }
	return 0;
}
