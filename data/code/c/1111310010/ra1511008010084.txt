#include <stdio.h>
#include <math.h>
 
int T,N,Demand;
int sum,bit,flag,i,j;
 
int main(void) 
{
    scanf("%d",&T);
    
    while(T--)
    {
        scanf("%d %d",&N,&Demand);
 
        int notes[N];
        for(i=0;i<N;i++)
            scanf("%d",&notes[i]);
        
        flag=0;
        for(i=0;i<pow(2,N);i++)
        {
            sum=0;
            for(j=0;j<N;j++)
            {
                bit = (i>>j)&1;
                
                if(bit==1)
                    sum += notes[j];
            }
            
            if(sum==Demand)
            {
                flag=1;
                break;
            }
        }
        
        if(flag==1)
            printf("Yes\n");
        else
            printf("No\n");
    }
    
	return 0;
}
