#include<stdio.h>
#include<stdlib.h>
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
    int n;
    scanf("%d",&n);
    int id[n],sum[n],idtotal=0,i,sumtotal=0;
    for( i=0;i<n;i++){
        scanf("%d%d",&id[i],&sum[i]);
    }
    for( i=0;i<n;i++){
        idtotal=idtotal+id[i];
        sumtotal=sumtotal+sum[i];
    }
    printf("%d\n",idtotal-sumtotal);
    }
  return 0;
}
