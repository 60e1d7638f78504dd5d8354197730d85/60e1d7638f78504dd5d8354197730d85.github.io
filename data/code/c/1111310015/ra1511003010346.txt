#include<stdio.h>
#include<string.h>
 
int main()
{
  long int t,i,n,j,mid,k;
  char arr[1000005];
  int flag=0,chk=0;
  
  scanf("%ld",&t);
  
  while(t--)
  {
    flag=0,chk=0;
    
    scanf("%s",arr);
    
    n=strlen(arr);
    
    for(i=0;i<n;i++)
    {
	if(arr[i]!='9')
	{
	  chk=1;
	  break;
	}
    }
    
    if(chk==1)
    {
      for(i=0,j=n-1;i<n/2,j>=n/2;i++,j--)
      {
	if(arr[i]>arr[j])//don't change middle
	  flag=1;
	
	else if(arr[i]<arr[j])//increase middle by 1
	  flag=2;
	
	arr[j]=arr[i];
      }
      
      if(flag==0||flag==2)
      {
	if(n%2==0)
	  mid=n/2-1;
	else
	  mid=n/2;
	
	k=0;
	
	while(arr[mid-k]=='9')
	{
	  arr[mid-k]='0';
	  arr[n-1-mid+k]='0';
	  k++;
	}
	
	arr[mid-k]++;
	arr[n-1-mid+k]=arr[mid-k];
      }
    
      printf("%s\n",arr);
    }
      
    else
    {
      printf("1");
      
      for(i=1;i<n;i++)
	printf("0");
      
      printf("1\n");
    }
  }
      
      return 0;
}
