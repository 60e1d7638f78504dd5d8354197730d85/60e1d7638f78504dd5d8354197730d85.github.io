#include <stdio.h>
#include<string.h>
char str[1000000];
void palin();
int main()
{
int t;
scanf("%d",&t);
while(t--)
{
scanf("%s",str);
palin();
printf("%s\n",str);
}
return 0;
}
void palin()
{
long i,len,mid;
int flag=1;
len=strlen(str);
for(i=0;i<len;i++)
{
if(str[i]!='9')
{
flag=0;
break;
}
}
if(flag==1)
{
str[0]='1';
for(i=1;i<len;i++)
str[i]='0';
str[i++]='1';
str[i]='\0';
return;
}
flag=0;
for(i=0;i<len/2;i++)
{
if(str[i]<str[len-i-1])
flag=-1;
else if(str[i]>str[len-i-1])
flag=1;
str[len-i-1]=str[i];
}
 
if(flag==0||flag==-1)
{
if(len%2==0)
mid=len/2-1;
else
mid=len/2;
int t=0;
while(str[mid-t]=='9')
{
str[mid-t]='0';
str[len-1-mid+t]='0';
t++;
}
str[mid-t]++;
str[len-1-mid+t]=str[mid-t];
}
}
