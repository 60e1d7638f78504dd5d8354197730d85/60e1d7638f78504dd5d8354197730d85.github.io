#include <stdlib.h>
#include <stdio.h>
 
typedef struct node {
	int a, d;
}node;
 
int main(){
	int n;
	node arr[101][101];
	int temp,x,y,i,j,k;
	
	while(EOF != scanf("%d",&n)){		
		for(i=0;i<n;++i){
			for(j=i;j<n;++j){
				arr[i][j].d = 999999;
			}
		}
		
		for(i=0;i<n;++i){
			scanf("%d", &arr[i][i].a);
			arr[i][i].d = 0;
		}
		
		
		
		for(i=1; i<n; ++i){
		  	for(j=0;j<n-i;++j){
				y = i+j;
				x = j;
				for(k=x; k<y; ++k){
					temp = arr[x][k].a * arr[k+1][y].a + (arr[x][k].d + arr[k+1][y].d);
					if(temp < arr[x][y].d){
						arr[x][y].a = (arr[x][k].a + arr[k+1][y].a)%100;
						arr[x][y].d = (temp);// + (arr[x][k].d + arr[k+1][y].d);
					}
				}
			}
		}
		
		printf("%d\n",arr[0][n-1].d);
	}
	
	return 0;
}
