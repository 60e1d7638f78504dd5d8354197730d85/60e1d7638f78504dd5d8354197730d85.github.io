    #include <stdio.h>
    #include <stdlib.h>
    #include <math.h>
    long long num[100005];
     
    int main()
    {
        long long N, i, k, cnt, sum = 0;
        scanf("%lld", &N);
        for(i = 0; i < N; i++)
            scanf("%lld", &num[i]);
        for(k = 30; k >= 0; k--){
            cnt = 0;
            for(i = 0; i < N; i++){
                if((1 << k) & num[i])
                    cnt++;
            }
            sum += cnt * (cnt-1) / 2 * pow(2, k);
        }
        printf("%lld\n", sum);
        return 0;
    }
