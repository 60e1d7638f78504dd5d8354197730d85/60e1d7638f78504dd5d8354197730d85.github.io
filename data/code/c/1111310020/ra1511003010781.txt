#include <stdio.h>
int main()
{
  int n;
  scanf("%d", &n);
  int a[n], i, j, sum=0;
  for(i=0; i<n; i++)
    scanf("%d", &a[i]);
  for(i=0; i<n; i++)
  {
    for(j=0; j<n; j++)
    {
      
      if(a[i]<a[j])
      {
       // printf("\n%d, %d, %d\n", a[i], a[j], sum);
        sum += a[i]&a[j];
      }
    }
  }
  printf("%d", sum);
  

	return 0;
}
