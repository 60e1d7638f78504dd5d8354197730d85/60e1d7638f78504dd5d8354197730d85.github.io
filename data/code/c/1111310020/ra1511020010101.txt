#include <stdio.h>
int main()
{
	int n,i,j,sum=0;
  scanf("%d",&n);
  int a[n];
  for(i=0;i<n;i++)
  {
   scanf("%d",&a[i]); 
  }
  for(i=0;i<n-1;i++)
  {
   for(j=i+1;j<n;j++)
   {
     if(i==j)
       continue;
     else
       sum=sum + (a[i] & a[j]);
   }
    
  }
  printf("%d",sum);
	return 0;
}
