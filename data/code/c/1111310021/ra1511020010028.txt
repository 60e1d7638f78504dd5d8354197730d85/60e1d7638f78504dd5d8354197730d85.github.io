#include <stdio.h>
#include <stdlib.h>
 
int main()
{
    int T, N, t, i, num, sum;
    scanf("%d", &T);
    while(T--){
        scanf("%d", &N);
        t = N;
        sum = 0;
         for(i = 0; i < N; i++){
            scanf("%d", &num);
            sum += num;
            if(num == 0)
                t--;
        }
        if(sum >= 100 && sum < 100 + t)
            printf("YES\n");
        else
            printf("NO\n");
    }
    return 0;
}
