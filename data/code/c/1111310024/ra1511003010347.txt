#include <stdio.h>
int main()
{
    int t,a[100],n,k,i,j,r,b[2][100];
    scanf("%u",&t);
    while(t--)
    {   r=0;
        scanf("%d %d",&n,&k);
        for(i=0;i<n;i++)
        {
            scanf("%d",&a[i]);
            b[0][i]=0;
            b[1][i]=0;
        }
        for(i=0;i<n;i++)
        {
            b[0][a[i]-1]++;
            if(a[i]-1==i) b[1][a[i]-1]=1;
        }
        for(i=0;i<n;i++)
        {
            if(b[0][i]>=k && b[1][i]!=1) r++;
        }
        printf("%d\n",r);
    }
	return 0;
}
