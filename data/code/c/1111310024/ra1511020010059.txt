#include <stdio.h>
 
int M=1000000007;
 
int main()
{
    int n,k;
 
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */
    int i,j,t;
    scanf("%d",&t);
    while(t--)
    {
        int count=0;
        scanf("%d %d",&n,&k);
        int vote[n];
        for(i=0;i<n;i++)
            vote[i]=0;
        for(i=0;i<n;i++)
        {
 
            scanf("%d",&j);
            if(i+1==j)
            {
                vote[i]=-1;
                continue;
 
            }
            if(vote[j-1]!=-1)
            {
                vote[j-1]++;
            }
 
 
        }
        for(i=0;i<n;i++)
            if(vote[i]>=k)
                count++;
        printf("%d\n",count);
 
    }
 
    return 0;
}
