#include <stdio.h>
int main()
{
  long unsigned int h[100000],i,j,max=0;
int n;
scanf("%d",&n);
for(i=0;i<n;i++){
    scanf("%lu",&h[i]);
    if(h[i]>max){
        max=h[i];
    }
}
long unsigned int s[max];
for(i=0;i<max;i++){
        s[i]=0;
}
for(i=0;i<n-1;i++){
    if(h[i]>h[i+1]){
        for(j=h[i+1];j<h[i];j++){
            s[j]++;
        }
    }
    if(h[i]<h[i+1]){
        for(j=h[i];j<h[i+1];j++){
            s[j]++;
        }
    }
}
long unsigned int k=0;
for(j=0;j<max;j++){
    if(s[j]>k)
        k=s[j];
}
printf("%lu",k);

	return 0;
}
