#include <stdio.h>

long long n,m,r,p,i,np;

static inline long long func(long long k)
{
  k %= (m*30);
  long long r,t=k*(k+1)/2;
  if(t%3 == 0)
    r = (t/3)*(2*k+1);
  else
    r = t*((2*k+1)/3);
  t = 6*t-1;
  if(r%5 == 0)
    r /= 5;
  else
    t /= 5;
  return ((r%m)*(t%m))%m;
}

int main()
{
  int t;
  scanf("%d", &t);
  while(t--)
  {
    scanf("%lld%lld", &n, &m);
    p = 1;
    for(i=0; i<20; i++)
      p = (p+n/p)/2;
    np = n/p;
    for(r=0,i=1; i<=np; i++)
      r += (((i*i)%m)*((i*i*(n/i))%m))%m;
    for(i=1; i<p; i++)
      r += func(n/i);
    r = (r-(p-1)*func(n/p))%m;
    if(r < 0)
      r += m;
    printf("%lld\n", r%m);
  }
}
