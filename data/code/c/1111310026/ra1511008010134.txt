#include <stdio.h>
 
typedef long long ll;
 
ll t,n,m,a,i,s,e,P,N,zm,lm,z[100001][2];
 
int main()
{
    for(m=1;m<100001;m++)
    {
        z[m][1]=m*m;
        if(z[m][1]>(1<<30)) { z[m][0]=z[m][1]>>30;z[m][1]&=0x3fffffff; }
        z[m][0]=( (z[m][0]*z[m][0])<<30 )+( (z[m][0]*z[m][1])<<1 ) ;
        z[m][1]*=z[m][1];
        if(z[m][1]>(1<<30)) { z[m][0]=z[m][0]+(z[m][1]>>30);z[m][1]&=0x3fffffff; }
        z[m][1]+=z[m-1][1];
        z[m][0]+=z[m-1][0];
        if(z[m][1]>(1<<30)) { z[m][0]=z[m][0]+(z[m][1]>>30);z[m][1]&=0x3fffffff; }
    }
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld%lld",&n,&m);
        lm=(1<<30)%m;
        zm=(lm*(z[m][0]%m)+z[m][1])%m;
        a=P=0;
        i=1;
        while(i<=n)
        {
            s=n/i;
            e=n/s;
            i=e+1;
            if(e<m)
            {
                if(z[e][0]) N=(lm*(z[e][0]%m)+z[e][1])%m;
                else N=z[e][1]%m;
            }
            else
            {
                N=(zm*(e/m))%m;
                e%=m;
                if(z[e][0]) N=(N+lm*(z[e][0]%m)+z[e][1])%m;
                else N=(N+z[e][1])%m;
            }
            a=( a+s*(N+m-P) )%m;
            P=N;
        }
        printf("%lld\n",a);
    }
    return 0;
}
