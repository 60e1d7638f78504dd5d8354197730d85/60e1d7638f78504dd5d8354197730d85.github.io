#include<stdio.h>
#include<string.h>
int pindex(int arr[],int low,int high)
{
	int pivot=arr[high];
	int index=low,i;
	for(i=low;i<high;i++)
	{
		if(arr[i]<=pivot)
		{
			int t=arr[i];
			arr[i]=arr[index];
			arr[index]=t;
			index++;
		}
	}
	{
		int temp=arr[index];
		arr[index]=arr[high];
		arr[high]=temp;
		return index;
	}
}
void quicksort(int arr[],int low,int high)
{
	if(low<high)
	{
		int index=pindex(arr,low,high);
		quicksort(arr,low,index-1);
		quicksort(arr,index+1,high);
	}
}
