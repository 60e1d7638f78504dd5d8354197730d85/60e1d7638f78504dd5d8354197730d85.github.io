#include<stdio.h>

int main()
{
     int t,n,x,a[1001],i;
     scanf("%d",&t);
     while(t--)
     {
         scanf("%d",&n);

         for(i=0;i<1000;i++)
             a[i]=0;

         for(i=0;i<n;i++)
         {
             scanf("%d",&x);
             a[x]++;
         }

          int max1=0,max2=0;
          for(i=0;i<1000;i++)
          {
              if(a[i]>1)
                max1=i;
          }
          for(i=0;i<1000;i++)
          {
              if((a[i]>1)&&(i<max1))
                max2=i;
          }
        if(max1&&max2){
          if(a[max1]>=4)
            printf("%d\n",max1*max1);
           else printf("%d\n",max1*max2);}

        else
        {
            if(a[max1]>=4) printf("%d\n",max1*max1);
            else printf("%d\n",-1);
        }


     }





        return 0;

}
