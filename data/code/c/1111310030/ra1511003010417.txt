#include<stdio.h>
int main()
{
	int t,i,n,j=0,k,count,flag,c[10000];
	int a[10000],b[10000];
	scanf("%d",&t);
	k=t;
	while(t>0)
	{
		flag=0;
		count=0;
		scanf("%d",&n);
		for(i=0;i<n;i++)
			scanf("%d",&a[i]);
		for(i=0;i<n;i++)
			scanf("%d",&b[i]);
		if(a[0]>=b[0])
			flag=1;
		for(i=1;i<n;i++)
		{
			if((a[i]-a[i-1])>=b[i])
				count++;
		}
		c[j]=count+flag;
		t--;
		j++;
	}
	for(i=0;i<k;i++)
	{
		if(i==k-1)
			printf("%d",c[i]);
		else
		printf("%d\n",c[i]);
	}
  return 0;
}
