 
#include <stdio.h>
#include <stdlib.h>
#include <math.h>
#include <string.h>
#include <limits.h>
enum {false, true};
 
#define ll long long int
#define gll(x) get_int(&x)
#undef getchar_unlocked
 
ll get_int(ll *i) {
	register int a, sign;
	*i = 0;
	sign = 0;
	
	while((a = getchar_unlocked()) < 33);
	if(a == '-') {
		sign = 1;
	} else {
		*i = a - '0';
	}
	while((a = getchar_unlocked()) > 33) {
		*i = (*i << 1) + (*i << 3) + a - '0';
 	}
 	*i = sign?-*i:*i;
 	return *i;
}
int main(void) {
	ll T;
	gll(T);
	while(T--) {
	 	ll N;
	 	gll(N);
	 	ll a[N];
	 	int i;
	 	for(i = 0; i < N; i++) {
	 		gll(a[i]);
	 	}
	 	ll temp, count = 0;
	 	gll(temp);
	 	if(temp <= a[0]) {
	 		count++;
	 	}
	 	for(i = 1; i < N; i++) {
	 		gll(temp);
	 		if(a[i] - a[i - 1] >= temp) {
	 			count++;
	 		}
	 	}
	 	printf("%lld\n", count);
	}
	return 0;
}
