#include <stdio.h>
 
int main()
{
	int i, t, n;
 
	scanf("%d", &t);
 
	for (i = 0; i < t; ++i) {
		scanf("%d", &n);
 
		int v[n];
 
		scanf("%d", & v[0]);
		
		int min = v[0], j;
		//int count = 0;
 
		for(j = 1; j < n; ++j) {
			scanf("%d", &v[j]);
			
			if (min > v[j]) {
				min = v[j];
			}
 
			//count += min;
		
		}
 
		
		printf("%ld\n", (long)min * (n - 1));
	}
 
	return 0;
}
