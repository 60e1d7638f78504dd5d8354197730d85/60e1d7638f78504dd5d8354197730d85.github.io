#include<stdio.h>
 
int main ()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long n;
		scanf("%ld", &n);
		long arr[n], min=100000;
		int i;
		for( i = 0 ; i < n ; i++ )
		{
			scanf("%ld",& arr[i] );
		}
		for( i = 0 ; i < n ; i++ )
		{
			if ( arr[i] < min ) min = arr[i];
		}
		long ans = min * ( n - 1 );
		printf("%ld", ans);
		printf("\n");
	}
return 0;
}
