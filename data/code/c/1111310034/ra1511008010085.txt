#include<stdio.h>
int main()
{
    int n,d;
    scanf("%d%d",&n,&d);
    int i;
    int flag=0;
    int ans=0;
    int vec[n];
    for(i=0;i<n;i++)
        scanf("%d",&vec[i]);
    for(i=0;i<n;i++)
    {
        if(vec[i]>d)
        {
            if(flag==0)
            {
                flag=1;
                continue;
            }
            else
            {
                break;
            }
        }
        else
        {
            ans++;
        }
    }
    printf("%d",ans);
    return 0;
}
