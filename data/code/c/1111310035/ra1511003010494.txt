 #include <stdio.h>
     
     
     int main()
     {
     	int t;
      long long int n,s[100000],i,count;
     	scanf("%d",&t);
     	while(t!=0)
     	{
     		count=1;
     		scanf("%lld",&n);
     		for(i=0;i<n;i++)
     		scanf("%lld",&s[i]);
     		for(i=0;i<n-1;i++)
     		{
     			if(s[i]<s[i+1])
     			 s[i+1]=s[i];
     			 else
     			 count++;
     		}
     		printf("%lld\n",count);
     		
     		t--;
     	}
     	return 0;
     }
