#include<stdio.h>
int arr [100000 + 10];
int main()
{
    int t , n ,  m , k, ans;
    scanf("%d",&t);
    while(t--){
        scanf("%d%d",&n,&arr[0]);
        ans = 1;
      int i;
        for(i = 1; i < n; i++){
            scanf("%d",&arr[i]);
            if(arr[i] <= arr[i - 1]) ans++;     
            else arr[i] = arr[i - 1];           
        }
        printf("%d\n",ans);
    }
    return 0;
}
