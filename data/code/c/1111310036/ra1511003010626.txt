#include <stdio.h>
int main()
{    int t;    
     scanf("%d",&t);
     while(t--)    
     {        
          int i,n,k,p;        
          scanf("%d",&n);
          int a[n];
          scanf("%d",&k);
          for(i=0;i<n;i++)
          {
            scanf("%d",&a[i]);
          }        
          k%=n;
          for(i=0;i<n;i++)
          {
            p = a[(i+(n-k))%n];
            printf("%d\n",p);
          } printf("\n");
      }
return 0;
}
