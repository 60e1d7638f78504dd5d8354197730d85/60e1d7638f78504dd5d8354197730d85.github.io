#include<stdio.h>
int main()
{
  int t,n,k,i;
  scanf("%d",&t);
  while(t--)
  {
    scanf("%d%d",&n,&k);
    int a[n],b[n];
    for(i=0;i<n;i++)
      scanf("%d",&a[i]);
    k=k%n;
    for(i=0;i<n;i++)
    {
      if(i+k<n)
        b[i+k]=a[i];
      else
        b[i+k-n]=a[i];
    }
    for(i=0;i<n;i++)
      printf("%d\n",b[i]);
    printf("\n");
  }
  return 0;
}
