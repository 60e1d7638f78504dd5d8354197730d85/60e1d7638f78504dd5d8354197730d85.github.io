    #include<stdio.h>
    #include<string.h>
    int fun(char *r)
    { int i;
      if(r[0]==r[1])
      return 0;
      for(i=0;i+2<strlen(r);i++)
       {if(r[i]!=r[i+2])
        return 0;
       }
      return 1;
    }
    int main()
    {int t,flag;
     scanf("%d",&t);
     char s[101];
     while(t--)
     { scanf("%s",s);

       printf("NO\n");
       else

     }
