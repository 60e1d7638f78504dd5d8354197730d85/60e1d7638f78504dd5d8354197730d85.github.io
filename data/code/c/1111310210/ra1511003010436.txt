#include<stdio.h>
#include<limits.h>
#include<ctype.h>
#include<math.h>
#include<string.h>
#include<stdlib.h>
#define gc getchar_unlocked
#define M 1000000007
#define Max(a,b) (((a)>=(b))?(a):(b))
#define Min(a,b) (((a)<=(b))?(a):(b))
 
typedef long long ll;
typedef unsigned long long ull;
 
ull getnum()
{
	char ch;
	ull num=0;
	
	while(!isdigit(ch=gc()));
	
	do
	{
		num=num*10+(ch-'0');
	
	}while(isdigit(ch=gc()));
	
	return num;
}
 
int mmin(int a, int b)
{
		if(a<=b)
			return a;
		else
			return b;
}
 
 
 
int main()
{
		int n=getnum(),tnum,bnum,mid,i,min = 100001;
		int top[100001]={0} ,bottom[100001]={0};
		
		for(i=0;i<n;++i)
		{
				
			tnum=getnum();
			bnum=getnum();
			
			if(tnum==bnum)
				top[tnum]++;
			else
			{	
					top[tnum]++;
					bottom[bnum]++;
					
			}
				
		}
		
		mid = n-n/2;
		
		for(i=0;i<100001;++i)
		{
				if(top[i] + bottom[i] <mid)
						continue;
				
				if(top[i] >= mid)
				{
					min=0;
					break;
				}
				else
					min = Min(min , mid-top[i]);
					
			
		}
		
		if(min!=100001)
			printf("%d\n",min);
		else
			printf("Impossible\n");
				
		return 0;
	
}
