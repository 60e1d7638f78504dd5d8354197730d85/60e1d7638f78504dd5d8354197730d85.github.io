#include <stdio.h>
int main()
{
int Days, Years, Weeks;

    scanf("%d",&Days); 

    // Converts days to years, weeks and days
    Years = Days/365; 
    Weeks = (Days % 365)/7;
    Days = Days- ((Years*365) + (Weeks*7));

    printf("Years:%d\n",Years);
    printf("Weeks:%d\n",Weeks);
    printf("Days:%d \n",Days);

	return 0;
}
