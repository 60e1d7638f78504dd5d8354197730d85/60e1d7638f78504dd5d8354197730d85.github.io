
#include <stdio.h>

int main()
{
    long long num;
    int count = 0;

    /* Input number from user */
    
    scanf("%lld", &num);

    /* Run loop till num is greater than 0 */
    while(num != 0)
    {
        /* Increment digit count */
        count++;

        /* Remove last digit of 'num' */
        num /= 10;
    }

    printf("Number of Digits:%d", count);

    return 0;
}
