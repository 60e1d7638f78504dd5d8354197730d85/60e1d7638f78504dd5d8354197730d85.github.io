#include <stdio.h>
int whetherPrime(int n);
int main()
{
	int n, i, flag = 0;
  	
  	scanf("%d",&n);
  	for(i = 2; i<= n/2; i++)
    {
      	if(whetherPrime(i) == 1){
          	if(whetherPrime(n-i) == 1){
         		printf("%d = %d + %d\n", n, i, n - i);
          		flag = 1;
            }
        }
    }
  	if(flag == 0){
    	printf("NOT");
    }
	return 0;
}
int whetherPrime(int n){
  	int i, isPrime = 1;
	for(i = 2; i<= n/2; i++){
      	if(n%i == 0){
          	isPrime = 0;
          	break;
        }
    }
  	return isPrime;
}
