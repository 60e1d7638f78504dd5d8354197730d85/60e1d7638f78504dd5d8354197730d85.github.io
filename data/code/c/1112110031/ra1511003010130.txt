#include <stdio.h>
#include <string.h>
#include <stdbool.h>
char s[10][10];
int main()
{
  int i,j;
    for(i=0;i<8;i++)
        scanf("%s",s[i]);//B->W
    bool flag=0;
    int x,y;
    for(i=0;i<8;i++)
        for(j=0;j<8;j++)
            if(s[i][j]=='W')
            {
                flag=1;
                x=i;
                y=j;
            }
    if(!flag)return 0*printf("8");
    int res=0;
    for(i=0;i<8;i++)
        res+=s[i][y]=='B';
    for(j=0;j<8;j++)
        res+=s[x][j]=='B';
    return 0*printf("%d",res);
}
