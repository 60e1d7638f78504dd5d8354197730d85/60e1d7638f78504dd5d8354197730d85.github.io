#include <stdio.h>
#include <math.h>
int main()
{
	int a,n, d=0, m, sum=0;
  	scanf("%d",&n);
  	m = n;
  	while(m>0){
      m=m/10;
      d++;
    }
  	m = n;
  	while(n>0){
      a = n%10;
      sum = sum + pow(a,d);
      n=n/10;
    }
  if(sum==m)
    printf("Narcissistic Number");
  else
    printf("NOT Narcissistic Number");
	return 0;
}
