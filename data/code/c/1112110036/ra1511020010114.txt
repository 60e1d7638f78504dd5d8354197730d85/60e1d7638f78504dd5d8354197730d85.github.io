#include<stdio.h>
#include<math.h>
int main(){
int a,n,d=0,m,sum=0;
//printf("\"Program To Check Whether It Is Narcissistic Or Not\" \n");
//printf("Enter the number \n");
scanf("%d",&n);
m=n;
//for counting the number of digits in number
while(m>0){
        m=m/10;
        d++;
}
m=n;
while(n>0){
        a=n%10;
        sum=sum+pow(a,d);
        n=n/10;
}
if(sum==m)
          printf("Narcissistic Number");
else
      printf("NOT Narcissistic Number");
return 0;
}
