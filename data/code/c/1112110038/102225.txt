#include <stdio.h>
int cp(int n);
int main()
{
int n,i,f=0;
  scanf("%d",&n);
  
  for(i=2;i<=n/2;++i)
  {
    if(cp(i)==1)
    {
      if(cp(n-i)==1)
      {
        printf("%d = %d + %d\n",n,i,n-i);
        f=1;
      }
    }
  }
  if(f==0)
  {
    printf("NOT");
  }
	return 0;
}
int cp(int n)
{
  int i,sp=1;
  for(i=2;i<=n/2;++i)
  {
    if(n%i==0)
    {
      sp=0;
      break;
    }
  }
  return sp;
}
