#include <stdio.h>
int checkprime(int n)
{int i,isprime=1;
  for(i=2;i<=n/2;i++){
  if(n%i==0)
  {isprime=0;
   break;
  }
  }
 return isprime;
}

int main()
{int i,n,flag=0;
 scanf("%d",&n);
 for(i=2;i<=n/2;i++)
 {
   if (checkprime(i)==1)
   {
     if (checkprime(n-i)==1)
     {
       printf("%d = %d + %d\n",n,i,n-i);
       flag=1;
     }
   }
 }
	if (flag==0)
    {
    printf("NOT");
    }
	return 0;
}
