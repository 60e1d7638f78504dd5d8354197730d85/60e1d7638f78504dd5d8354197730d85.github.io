#include <stdio.h>
int prime(int n)
{
  int i,isprime=1;
  for(i=2;i<=n/2;++i)
  {
    if(n%i==0)
    {
      isprime=0;
      break;
    }
  }
  return isprime;
}
int main()
{
   int n,i ,flag=0;
   scanf("%d",&n);
  for(i=2;i<=n/2;++i)
  {
    if(prime(i)==1)
    {
      if(prime(n-i)==1)
      {
        printf("%d = %d + %d\n",n,i,n-i);
        flag=1;
      }
    }
  }
    if(flag==0)
      printf("NOT");
  
	return 0;
}
