#include<stdio.h>
int main()
{
  long long int t,i,c,d,l,min,max;
  scanf("%lld",&t);
  for(i=0;i<t;i++)
  {
    scanf("%lld",&c);
    scanf("%lld",&d);
    scanf("%lld",&l);
    if(c>2*d)
      min=4*d+4*(c-2*d);
    else
      min=4*d;
    max=4*(d+c);
    if(l>=min && l<=max && l%4==0)
      printf("yes\n");
    else
      printf("no\n");
  }

	return 0;
}
