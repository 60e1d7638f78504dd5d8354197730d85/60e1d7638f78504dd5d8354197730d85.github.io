#include<stdio.h>
int main()
{
   int n;
   scanf("%d",&n);
   if(n<=25){
       if(n==21)
        printf("4\n");
       else if(n<=10||n>20)
        printf("0\n");
       else if(n>10 && n<20)
        printf("4\n");
       else if(n==20)
        printf("15\n");
       }
  else
    printf("Invalid Input");
return 0;
}
