
#include<stdio.h>
 
int main() {
   int i, j, rows;
   int count = 1;
   
   //printf("Enter the number of rows\n");
   scanf("%d", &rows);
    
   for (i = 1; i <= rows; i++) {
      for (j = 1; j <= i; j++) {
         printf("%d ", count);
         count++;
      }
      printf("\n");
   }
   return(0);
}
