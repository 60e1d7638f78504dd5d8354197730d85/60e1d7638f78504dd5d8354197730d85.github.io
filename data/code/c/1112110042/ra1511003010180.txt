#include <stdio.h>
#include <string.h>
int main()
{ int t,i,k;
    char s[100],j[100];
    scanf("%d",&t);
 	while(t--){
        int count=0;
        int ans[130];
        for(i=0;i<130;i++)
            ans[i]=0;
    	scanf("%s",j);
        scanf("%s",s);
        for(i=0;i<strlen(j);i++){
                 ans[(int)j[i]]+=1;
        }  
        for(i=0;i<strlen(s);i++){
         		if(ans[(int)s[i]])
                  count++;
        }
        printf("%d\n",count);        
    }
	return 0;
}
