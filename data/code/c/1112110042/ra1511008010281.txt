#include <stdio.h>
#include <string.h>
int main()
{
   int t,i,j1,s1,k,count;
   char s[1001],j[1001];
   scanf("%d",&t);
   while(t--)
   {         count=0;
             scanf("%s",j);
             scanf("%s",s);
             j1=strlen(j);
             s1=strlen(s);
             for(i=0;i<s1;i++)
             for(k=0;k<j1;k++)
                             
                             if(s[i]==j[k])
                             {count++;
                            
                              break;}
                              
                              
                            printf("%d\n",count);
                            
   }
	return 0;
}
