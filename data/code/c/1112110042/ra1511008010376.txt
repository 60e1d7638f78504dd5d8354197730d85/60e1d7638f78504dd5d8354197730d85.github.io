#include <stdio.h>
#include<string.h>
int main()
{
  int t;
  scanf("%d",&t);
  while(t--)
  {
    char j[100],s[100];
    int i,p,e=0;
    scanf("%s",j);
    scanf("%s",s);
    for(i=0;i<strlen(s);i++)
    {
      for(p=0;p<strlen(j);p++)
      {
        if(s[i]==j[p])
        {
          e++;
          break;
        }
      }
    }
    printf("%d\n",e);
  }
	return 0;
}
