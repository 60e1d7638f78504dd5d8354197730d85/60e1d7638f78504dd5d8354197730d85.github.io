#include<stdio.h>
#include<string.h>
#include<stdlib.h>
 
int main()
{
 int t,i,k,count;
 char j[105],s[105];
 scanf("%d",&t);
 while(t--)
 {
  scanf("%s",j);
  scanf("%s",s);
  count=0;
  for(i=0;i<strlen(s);i++)
  {
   for(k=0;k<strlen(j);k++)
   {
    if(s[i]==j[k])
    {
     count++;
     break;
    }
   }
  }
  printf("%d\n",count);
 }
 return 0;
}
