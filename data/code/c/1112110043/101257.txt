#include <stdio.h>
int main()
{
int t,n1,n2,N[1000],F[1000];
	int k,j,i;
    scanf("%d",&t);
	while(t--){
		k=0;
		j=0;
		scanf("%d",&n1);
		for(i=0;i<n1;i++){
		scanf("%d",&N[i]);}
		scanf("%d",&n2);
		for(i=0;i<n2;i++){
		    scanf("%d",&F[i]);
	    }
		for(i=0;i<n1;i++){
			if(N[i]==F[j]){
				k++;
				j++;
			}if(k==n2){
			    printf("Yes\n");
			    break;
		    }
     	}
     	if(k!=n2)
     	printf("No\n");
	}
	return 0;
}
