#include <stdio.h>
int main()
{
int t;
  long int n,r,i,j,d;
  scanf("%d",&t);
  while(t--)
  {d=0,j=0;
   scanf("%ld",&n);
   long int a[n];
   for(i=0;i<n;i++)
   {
     scanf("%ld",&a[i]);
   }
   scanf("%ld",&r);
   long int b[r];
   for(i=0;i<r;i++)
   {
     scanf("%ld",&b[i]);
   }
   for(i=0;i<n;i++)
   {
     if(a[i]==b[j])
     {
       d++;
       j++;
     }
     if(d==r)
       break;
   }
   if(d==r)
     printf("Yes\n");
   else
     printf("No\n");
  }
  
  return 0;
}
