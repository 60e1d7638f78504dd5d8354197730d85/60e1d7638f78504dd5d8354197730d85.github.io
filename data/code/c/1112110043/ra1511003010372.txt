#include <stdio.h>
int main()
{
int t, n, f, arr1[1000]={}, arr2[1000]={}, arr3[1000]={}, i, j, check;
	scanf("%d", &t);
	while(t--){
		scanf("%d", &n);
		for (i = 0; i < n; i++)
			scanf("%d", &arr1[i]);
		scanf("%d", &f);
		for (i = 0; i < f; i++)
			scanf("%d", &arr2[i]);
		
		for (i = 0; i < f; i++)
			for(j = 0; j < n; j++)
				if (arr2[i] == arr1[j]){
					arr3[i] = 1;
					break;
				}
		check = 1;
		for (i = 0; i < f; i++)
			if (arr3[i] == 0){
				check = 0;
				break;
			}
		if (check == 1)
			printf("Yes\n");
		else
			printf("No\n");  	 
		
		for (i = 0; i < n; i++)
			arr1[i] = 0;
		for (i = 0; i < f; i++){
			arr2[i] = 0;
			arr3[i] = 0; 	 
		}
    }
	return 0;
}
