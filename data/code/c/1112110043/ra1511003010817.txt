  #include<stdio.h>
     
    int main()
    {
    	int n;
    	scanf("%d",&n);
    	while(n>0)
    	{
    		n--;
    		int x,y,i,j;
    		scanf("%d",&x);
    		int a[x];
    		for(i=0;i<x;i++)
    			scanf("%d",&a[i]);
    		scanf("%d",&y);
    		int b[y];
    		for(i=0;i<y;i++)
    			scanf("%d",&b[i]);
    		int flag=0;
    		for(i=0,j=0;i<x && j<y;i++)
    		{
    			if(a[i]==b[j])
    			{
    				flag++;
    				j++;
    				if(flag==y)
    				break;
    			}
    		}
    		if(flag==y)
    			printf("Yes\n");
    		else
    			printf("No\n");
    	}
    	return 0;
    }
