#include <stdio.h>
int main()
{
	int a,k,c,d,e,i;
	scanf("%d",&a);
	while(a--)
	{
		scanf("%d",&k);
		int A[k];
		for(c=0;c<k;c++)
		{
			scanf("%d",&A[c]);
		}
		scanf("%d",&d);
		int B[d];
		for(e=0;e<d;e++)
		{
			scanf("%d",&B[e]);
		}
		i=0;
		for(c=0;c<k;c++)
		{
	        if(A[c]==B[i])
	        {
	        	i++;
	        }
	        
			if(i==d)
			{
				printf("Yes\n");
				break;
			}
	    }
		 if(i<d)
		{
			printf("No\n");
		}
	}
	return 0;
}
