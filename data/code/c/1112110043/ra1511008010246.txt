#include <stdio.h>
int main()
{

  int t,n,m,i;
   scanf("%d",&t);
   while(t--)
   {   
       scanf("%d",&n);
       int a[n];
       for(i=0;i<n;i++)
         scanf("%d",&a[i]);
         scanf("%d",&m);
         int b[m];
       for(i=0;i<m;i++)
         scanf("%d",&b[i]);
         int j=0;
            
         for(i=0;i<n;i++)
          {
              if(a[i]==b[j])
                {j++;
                    if(j==m)
                   break;               
          }
        }  
          if(j==m)
           printf("Yes\n");
           else
            printf("No\n");
   }
	return 0;
}
