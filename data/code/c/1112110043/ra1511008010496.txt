#include<stdio.h>
int main()
{
int t,n,k,a1[n],a2[k],i,j,flag;
scanf("%d",&t);
while(t--)
{
scanf("%d",&n);
for(i=0;i<n;i++)
{
scanf("%d",&a1[i]);
}
scanf("%d",&k);
for(j=0;j<k;j++)
{
scanf("%d",&a2[j]);
}
j=0,flag=0;
for(i=0;i<n;i++)
{
if(a1[i]==a2[j])
{
if(j==k-1)
{
flag=1;
break;
}
j++;
}
}
if(flag==1)
{
printf("Yes\n");
}
else
{
printf("No\n");
}
}
return 0;
}
