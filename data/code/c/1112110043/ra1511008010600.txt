#include <stdio.h>
#include <stdlib.h>
int main()
{
  int t;
    	int n,f;
    	int i,j;
     
    	scanf("%d",&t);
    	while(t--)
    	{
    		scanf("%d",&n);
    		int arr[n];
     
    		for(i=0;i<n;i++)
    			scanf("%d",&arr[i]);
     
    		scanf("%d",&f);
    		int seq[f];
     
    		for(i=0;i<f;i++)
    			scanf("%d",&seq[i]);
     
    		for(i=0,j=0; i<n&&j<f; i++)
    		{
    			if(arr[i] == seq[j]) 
    				j++;
    		}
     
    		if(j==f)
    			printf("Yes\n");
    		else 
    			printf("No\n");
  
}
	return 0;
}
