#include<stdio.h>
int main()
{
    int t,i,h,j;
    scanf("%d",&t);
    while(t--){
   		int n;
    	scanf("%d",&n);
    	int a[n];
    	for(i=0;i<n;i++){
    		scanf("%d",&a[i]);
    	}
   		int f;
    	scanf("%d",&f);
    	int b[f];
    	for(j=0;j<f;j++){
    		scanf("%d",&b[j]);
    	}
    	int g=0;int s=0;
    	for(h=0;h<n;h++){
    		if(a[h]==b[g]){
    			s++;g++;
    	}
    }
    if(s==f){
      printf("Yes\n");
      continue;
    }
    else{
      printf("No\n");
      continue;
    }
 }
 return 0;
}
