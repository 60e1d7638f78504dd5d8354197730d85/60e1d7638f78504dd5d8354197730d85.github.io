#include<stdio.h>
#include<string.h>
int main()
{
int i,j,k,T,a,b,count;
char M[25010],W[25010];
scanf("%d",&T);
for(i=0;i<T;i++)
{
scanf("%s %s",M,W);
a=strlen(M);
b=strlen(W);
count=0;
k=-1;
if(a>b)
{
for(j=0;j<b;j++)
for(k=k+1;k<a;k++)
if(M[k]==W[j])
{
count++;
break;
}
if(count==b)
printf("YES\n");
else
printf("NO\n");
}
else if(a<b)
{
for(j=0;j<a;j++)
for(k=k+1;k<b;k++)
if(W[k]==M[j])
{
count++;
break;
}
if(count==a)
printf("YES\n");
else
printf("NO\n");
}
else if(a==b)
{
for(j=0;j<b;j++)
if(M[j]!=W[j])
{
j=-1;
break;
}
if(j==-1)
printf("NO\n");
else
printf("YES\n");
}
}
return 0;
}
