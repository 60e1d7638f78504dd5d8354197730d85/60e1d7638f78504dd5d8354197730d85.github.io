#include<stdio.h>
#include<string.h>
int main()
{
	char m[25001],b[25001];int t,i,l1,l2;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s%s",m,b);
		l1=strlen(m);l2=strlen(b);
		if(l1>l2)
		{
			int z=0,j=0;
			for(i=0;i<l1;i++)
			{
				if(m[i]==b[z])
				{
					z++;
				//	printf("%c ",m[z]);
					if(z==l2)
					{
						j=1;break;
					}
				}
			}
			if(j==0)
				printf("NO\n");
			else
				printf("YES\n");
		}
		else if(l1==l2)
		{
			if(strcmp(m,b)==0)
				printf("YES\n");
			else
				printf("NO\n");
		}
		else
		{
			int z=0,j=0;
			for(i=0;i<l2;i++)
			{
				if(b[i]==m[z])
				{
					z++;
				//	printf("%c ",m[z]);
					if(z==l1)
					{
						j=1;break;
					}
				}
			}
			if(j==0)
				printf("NO\n");
			else
				printf("YES\n");
		}
	}
	return 0;
}
