#include<stdio.h>
 
int main() {
    int t;
  int i, j;
    scanf("%d", &t);
    for( i = 0; i < t; i++) {
        int n;
        scanf("%d", &n);
        int seq_n[n];
        for( j = 0; j < n; j++) {
            scanf("%d", &seq_n[j]);
        }
        int f;
        scanf("%d", &f);
        int seq_f[f];
        for( j = 0; j < f; j++) {
            scanf("%d", &seq_f[j]);
        }
        int k = 0;
        for( j = 0; j < n; j++) {
            if(seq_n[j] == seq_f[k]) k++;
        }
        if(k == f) printf("Yes\n");
        else printf("No\n");
    }
    return 0;
}
