#include<stdio.h>
int main()
{
   int T,N,A[1000],F,B[1000],i,k,t1;
  scanf("%d",&T);
  while(T--)
  {
    scanf("%d",&N);
    for(i=0;i<N;i++)
    {
      scanf("%d",&A[i]);
    }
    scanf("%d",&F);
    for(i=0;i<F;i++)
    {
      scanf("%d",&B[i]);
    }
    k=0;t1=0;
    for(i=0;i<N;i++)
    {
         if(A[i]==B[k])
         {
          t1++;
          k++;
         }
         
       if(t1==F)
       {
         break;
       }
    }
    if(t1==F)
      printf("Yes\n");  
    else
      printf("No\n");
  }
   return 0;
}
