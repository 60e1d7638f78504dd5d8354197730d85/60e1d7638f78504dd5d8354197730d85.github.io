#include <stdio.h>
#include<string.h>
int main()
{
int i;
  int t,n1,n2;
  char A[10];
  char B[10];
  scanf("%d",&t);
  
  while(t!=0)
  {
    scanf("%d",&n1);
    for(i=0;i<n1;i++)
    {
      scanf("%s",&A[i]);
    }
    scanf("%d",&n2);
    for(i=0;i<n2;i++)
    {
      scanf("%s",&B[i]);
    }
  if(strstr(A,B))
    printf("Yes");
  else
    printf("No");
    t=t-1;
    printf("\n");
  }
	return 0;
}
