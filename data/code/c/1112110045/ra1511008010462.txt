#include <stdio.h>
int main()
{
int t,n,l[1000],f,k[1000],i,j,c;
 
    scanf("%d",&t);
 
    while(t--)
    {
        scanf("%d",&n);
 
        for(i=0;i<n;i++)
            scanf("%d",&l[i]);
 
        scanf("%d",&f);
 
        for(i=0;i<f;i++)
            scanf("%d",&k[i]);
 
        c=0;
        j=0;
        for(i=0;i<n;i++)
        {
 
            if(l[i]==k[j])
                {
                    c++;
                    j++;
                }
 
            if(c==f)
            {
                break;
            }
        }
        if(c==f)
            printf("Yes\n");
        else
            printf("No");
 
    }
	return 0;
}
