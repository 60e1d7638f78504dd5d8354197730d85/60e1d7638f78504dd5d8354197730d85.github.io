#include <stdio.h>
 
int main() {
int t;
scanf("%d",&t);
while(t--) {
int n, m, i, j;
scanf("%d",&n);
int a[n];
for(i=0; i<n; i++)
scanf("%d",a+i);
 
scanf("%d",&m);
int b[m];
for(i=0; i<m; i++)
scanf("%d",b+i);
 
i=0, j=0;
while(i < n) {
if(a[i] == b[j])
j++;
i++;
if(j == m)
break;
}
 
if(j == m)
printf("Yes\n");
else
printf("No\n");
}
return 0;
}
