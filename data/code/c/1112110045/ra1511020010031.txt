#include<stdio.h>
int main()
{int T,N,F,j,i,k;
scanf("%d",&T);
while(T--)
{
k=0;
scanf("%d",&N);
int a[N];
for(j=0;j<N;j++)
scanf("%d",&a[j]);
scanf("%d",&F);
int b[F];
for(j=0;j<F;j++)
scanf("%d",&b[j]);
for(j=0;j<F;j++)
{for(i=0;i<N;i++)
{if(a[i]==b[j])
{k++;
break;
}
else
k=k;
}
}
if(k==F)
printf("\nYes");
else
printf("\nNo");
}
return 0;
}
