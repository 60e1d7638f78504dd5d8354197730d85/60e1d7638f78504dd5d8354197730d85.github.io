#include<stdio.h>
 
int main()
{
    int t;
    scanf("%d", &t);
    while(t--) {
        int n, f;
        scanf("%d", &n);
        int sq[n];
        int i, j, k, count;
        for(i = 0; i < n; i++) {
            scanf("%d", &sq[i]);
        }
        scanf("%d", &f);
        int p[f];
        for(i = 0; i < f; i++) {
            scanf("%d", &p[i]);
        }
        j = 0;
        count = 0;
        for(i = 0; i < n; i++) {
            if(sq[i] == p[j]) {
                k = i + 1;
                count += 1;
                j += 1;
                for(; j < f; j++) {
                    if(sq[k] == p[j])
                        count += 1;
                    else
                        break;
                }
                k += 1;
            }
            if(count == f)
                break;
        }
        if(count == f)
            printf("Yes\n");
        else
            printf("No\n");
    }
    return 0;
}
