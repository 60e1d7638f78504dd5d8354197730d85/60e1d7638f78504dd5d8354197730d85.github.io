#include <stdio.h>
 
int main(void) {
	
	int t,n,m,i,j,count;
	
	scanf("%d",&t);
	
	while(t--){
		scanf("%d",&n);
		
		int a[n];
		
		for(i=0;i<n;i++){
		scanf("%d",&a[i]);
		}
		
		scanf("%d",&m);
		
		int b[m];
		
		for(i=0;i<m;i++){
		scanf("%d",&b[i]);
		}
		count=0;
		i=0;
		j=0;
		
		while(i<n && j<m){
			if(a[i]==b[j]){
				j++;
				count++;
			}
			if(count==m)
			break;
			
			i++;
		}
		
		if(count==m)
		printf("Yes\n");
		else
		printf("No\n");
	}
	
	return 0;
}
