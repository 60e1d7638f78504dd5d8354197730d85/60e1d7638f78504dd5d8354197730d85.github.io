#include <stdio.h>
#include <math.h>
int main()
{
	int test,x[22],y[22],n,i,j;
	long long int sum;
	double area;
	scanf("%d",&test);
	while(test--)
	{
		sum=0;
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%d",&x[i]);
		}
		
		for(i=0;i<n;i++)
		{
			scanf("%d",&y[i]);
		}
		
		for(i=0;i<n;i++)
		{
			if(i==(n-1))
			{
				j=0;
			}
			
			else
			j=i+1;
			
			sum=sum+((x[i]*y[j])-(y[i]*x[j]));
		}
		
		area=sum/2.0;
		if(area<0){
		area=-area;
        }
      	if(area ==0){
		printf("-%0.1f\n",area);
        }
      	else
		printf("%0.1lf\n",area);
    }
  	
	return 0;
}
