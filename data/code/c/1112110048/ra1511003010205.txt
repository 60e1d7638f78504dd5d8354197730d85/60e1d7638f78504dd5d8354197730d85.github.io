#include <stdio.h>
int main()
{
int a,n,k;
    scanf("%d%d%d",&a,&n,&k);
    int i;
    int b[k],c=0,e=a;
    for(i=0;i<k;i++)
        b[i]=0;
    for(;(a>0);a--)
    {
        if(a==e)
        {
            if(n==0)
            {
             c=1;
             break;
            }
            else
            {
                b[0]=1;
            }
        }
        else
        {
            b[0]++;
            if(b[0]>n)
            {
                b[1]++;
                b[0]=0;
               int j=1;
               for(;j<(k);j++)
               {
                   if(j!=(k-1))
                      {
                   if(b[j]>n)
                   {
                      b[j+1]++;
                      b[j]=0;
                   }
                   else
                    break;
                  }
                  else
                  {
                      if((b[j])>n)
                      b[j]=0;
                  }
               }
            }
        }
    }
    if(c==1)
    {
        for(i=0;i<k;i++)
            printf("0 ");
            printf("\n");
    }
    if(c==0)
    {
    for(i=0;i<k;i++)
    {
        printf("%d ",b[i]);
    }
    printf("\n");
    }
	return 0;
}
