#include<stdio.h>
#include<math.h>
int main()
{
    long long int t,n,j,i,l,co;
    scanf("%lld",&t);
    for(i=0;i<t;i++)
       {
           scanf("%lld",&n);
           l=sqrt(n);
           co=0;
           for(j=1;j<=l;j++)
             {
                 if(n%j==0)
                 co+=1;
             }
           if(co==1)
           printf("yes\n");
           else
           printf("no\n");
       }
    return 0;
}
