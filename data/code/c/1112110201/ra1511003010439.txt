#include <stdio.h>
int main()
{
   int n, c, t, i;
   scanf("%d",&t);
 
   for(i=0; i<t; i++)
   {
       scanf("%d",&n);
       for ( c = 2 ; c <= n - 1 ; c++ )
          {
            if ( n%c == 0 )
               {
                  printf("no\n");
	              break;
               }
          }
            if ( c == n )
            printf("yes\n");
 
   }
   return 0;
}
