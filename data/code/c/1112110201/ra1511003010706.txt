#include <stdio.h>
int main()
{
  int i=0,tc,f=0,n;
  scanf("%d",&tc);
  while(tc!=0)
  {
    scanf("%d",&n);
    for(i=2;i<(n/2);i++)
    {
      if(n%i==0)
      {
        f=1;
        break;
      }
    }
    if(f==0)
      printf("yes\n");
    else printf("no\n");
    tc--;
    f=0;
  }
   
	return 0;
}
