#include <stdio.h>
 
int main()
{
    int t;
    scanf(" %d", &t);
 
    while(t--)
    {
        int n, i, x=0;
        scanf(" %d", &n);
        for(i=2; i<n;i++)
        {
            if(n%i==0)
            {
                x=1;
            }
 
        }
        if(x==1)
        {
            printf("no\n");
        }
        else
        {
            printf("yes\n");
        }
    }
    return 0;
}
