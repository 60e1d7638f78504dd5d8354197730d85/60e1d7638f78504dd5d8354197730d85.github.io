#include <stdio.h>
int prime(int n)
{
  int i,j,k;
  if(n==2)
    return 1;
  for(i=2;i<n/2;i++)
    if(n%i==0)
      return 0;
  return 1;
}
int main()
{
  int t,i,j,k,z;
  scanf("%d",&t);
  for(i=0;i<t;i++)
  {
    scanf("%d",&k);
    int z=prime(k);
    if(z==1)
      printf("yes\n");
    else
      printf("no\n");
  }
	return 0;
}
