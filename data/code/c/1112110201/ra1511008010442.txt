#include <stdio.h>
int main()
{

	int T, N,i,temp;
	scanf("%d", &T);
	while (T--)
	{
		scanf("%d", &N);
		temp = 0;
		for (i = 2; i < N; i++)
		{
			if ((N%i) == 0)
			{
				temp++;
				break;
			}
 
		}
		if (temp != 0)
		{
			printf("no\n");
		}
		else
		{
			printf("yes\n");
		}
 
	}
	

	return 0;
}
