#include <stdio.h>
int main()
{
    int ar[100],i,n,j,counter;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
     scanf("%d",&ar[i]);
    }
     for(i=0;i<n;i++)
     {
       counter=0;
       for(j=2;j<ar[i];j++)
       {
         if(ar[i]%j==0)
          { counter=1;
            printf("no\n");
            break;
          }
       }
        if(counter==0)
       {
          printf("yes\n");
           }
     }

	return 0;
}
