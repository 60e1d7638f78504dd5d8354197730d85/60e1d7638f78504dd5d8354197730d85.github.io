#include<stdio.h>
#include<string.h>
 main()
{
 int t,i,temp,l,flag;
 scanf("%d",&t);
 while(t--)
 {flag=0;
  char s[50];
  int a[125]={0};
  scanf("%s",s);
  for(i=0;s[i]!='\0';i++)
  {
   temp=s[i];
   a[temp]++;
  }
  l=strlen(s);
  if(l%2==0)
  {
   for(i=97;i<=122;i++)
   {
    if(a[i]==l/2)
    {flag=1;
     printf("YES\n");
     break;
    }
   }
  }
   if(flag==0)
   printf("NO\n");
 }
 }
