#include <stdio.h>
 
int main()
{
    int num1,n, num2,i, gcd, lcm, remainder, numerator, denominator;
scanf("%d",&n);
  for(i=0;i<n;i++){
    scanf("%d %d", &num1, &num2);
    if (num1 > num2)
    {
        numerator = num1;
        denominator = num2;
    }
    else
    {
        numerator = num2;
        denominator = num1;
    }
    remainder = numerator % denominator;
    while (remainder != 0)
    {
        numerator   = denominator;   
        denominator = remainder;
        remainder   = numerator % denominator;
    }
    gcd = denominator;
    lcm = num1 * num2 / gcd;
    printf("%d %d\n",gcd,lcm);}
  	return 0;
}
