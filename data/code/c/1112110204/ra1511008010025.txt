#include<stdio.h>
long int gcd(long int a,long int b);
long int gcd(long int a,long int b)
{
    if(b==0)
        return a;
    return(gcd(b,a%b));
}
int main()
{
    int t;
    long int a,b,g,l;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%ld%ld",&a,&b);
        g=gcd(a,b);
        l=(a*b)/g;
        printf("%ld %ld\n",g,l);
    }
return 0;
}
