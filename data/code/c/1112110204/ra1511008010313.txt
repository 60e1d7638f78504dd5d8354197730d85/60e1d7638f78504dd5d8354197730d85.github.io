#include<stdio.h>
int main()
{
    int t;
    long a,b,x,y;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%ld %ld",&x,&y);
        a=x;
        b=y;
        while(a!=b)
        {
            if(a>b)
            {
                a-=b;
            }
            else
            {
                b-=a;
            }
        }
        printf("%ld %ld\n",a,(x*y)/a);
    }
}
