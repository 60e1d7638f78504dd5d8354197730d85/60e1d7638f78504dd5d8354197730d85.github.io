#include <stdio.h>
long int gcd(long int a,long int b)
{
  if(a==b)
    return a;
  else if(a>b)
    return gcd(a-b,b);
  else
    return gcd(a,b-a);
}
long int lcm(long int a,long int b)
{
  return (a*b)/gcd(a,b);
}
int main()
{
  int t;
  long int a,b;
  scanf("%d",&t);
  while(t--)
  {
    scanf("%ld %ld",&a,&b);
    printf("%ld %ld\n",gcd(a,b),lcm(a,b));
  }
}
