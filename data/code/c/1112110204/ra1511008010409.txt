#include <stdio.h>
int main()
{
	int t;
  scanf("%d",&t);
  while(t--)
  {
  	int a,b,gcd;
  	scanf("%d %d",&a,&b);
  	int i,lcm;
  	for(i=1;i<=a && i<=b;++i)
 	 {
    	if((a%i==0)&&(b%i==0))
    	{
      		gcd=i;
   	 	}
   
    }
    lcm=(a*b)/gcd;
  	printf("%d %d\n",gcd,lcm);
  }
  	
	return 0;
}
