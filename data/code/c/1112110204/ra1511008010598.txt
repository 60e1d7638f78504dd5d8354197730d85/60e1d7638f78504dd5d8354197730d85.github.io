#include<stdio.h>
int main()
{
 int t,i;
 long long int a,b,n1,n2,temp,gcd,lcm;
 scanf("%d",&t);
 for(i=0;i<t;i++)
 {
  scanf("%lld%lld",&n1,&n2);
 
  a=n1;
  b=n2;
 
  while(b!=0)
  {
   temp = a%b;
   a=b;
   b=temp;
  }
  gcd = a;
  lcm = (n1*n2)/gcd;
  printf("%lld %lld\n",gcd,lcm);
 }
return 0;
}
