#include <stdio.h>
int main()
{
   int row, c, n, temp;
 
   scanf("%d",&n);
 
   temp = n;
 
   for ( row = 1 ; row <= n ; row++ )
   {
      for ( c = 1 ; c < temp ; c++ )
         printf(" "); // space
 
      temp--;
 
      for ( c = 1 ; c <= 1*row ; c++ )
         printf("*");
 
      printf("\n");
   }
 
   return 0;
}
