#include<stdio.h>
#include<stdlib.h>
 
int main() {
	int test;
int baseSalary, hra, da;
	scanf("%d", &test);
	while(test--) {
		scanf("%d", &baseSalary);
		if(baseSalary<1500) {
			hra = (baseSalary*10)/100;
			da = (baseSalary*90)/100;
		}
		else {
			hra = 500;
			da = (baseSalary*98)/100;
		}
		printf("%.2d\n", baseSalary+hra+da);
	}
	return 0;
}
