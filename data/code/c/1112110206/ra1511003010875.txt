#include <stdio.h>
 
int main() {
	int i,t,a;
	float hra,da,s;
	scanf("%d",&t);
	i=0;
	while(i<t)
	{
	    scanf("%d",&a);
	    if(a<1500)
	    {
	        hra=(float)a*10.0/100;
	        da=(float)a*90.0/100;
	        s=(float)a+hra+da;
	    }
	    else 
	    {
	        hra=500;
	        da=(float)a*98.0/100;
	        s=(float)a+hra+da;
	    }
	    printf("%1.0f\n", s);
	    i++;
	}   
	return 0;
}
