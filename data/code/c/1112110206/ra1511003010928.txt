#include <stdio.h>
int main()
{
  int a[1000],n;
  int i;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  for(i=0;i<n;i++)
  {
    if(a[i]<1500)
    {
      a[i]=2*a[i];
    }
    else
    {
      a[i]=500+((a[i]/100)*98)+a[i];
    }
  }
    for(i=0;i<n;i++)
    {
      printf("\n%d",a[i]);
     }
	return 0;
}
