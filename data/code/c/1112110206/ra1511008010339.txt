#include <stdio.h>
     
    int main(void) {
        int t, s;
            
        scanf ("%d\n", &t);
        while (t--) {
            
            scanf ("%d", &s);
            
            if (s < 1500)
                printf("%g\n", s + s * 0.1 + s * 0.9);
            else
                printf("%g\n", s + s * 0.98 + 500);
        }
        
        return 0;
    }
