#include <stdio.h>
#include<stdlib.h>
int main()
{
       int t;
       float sal,dra,hra,gross;
       scanf("%d",&t);
       while(t--)
       {    scanf("%f",&sal);
            hra=0;
            dra=0;
        if(sal<1500)
            {
                hra=0.1*sal;
                dra=0.9*sal;
            }
        else
           {
            hra=500;
            dra=0.98*sal;
            }
            gross=sal+hra+dra;
            printf("%g\n",gross);
       }
       return 0;
}
