#include <stdio.h>
int main()
{
  int i, d, n, t, ans, sum, ans1, x;
    scanf("%d", &t);
    while(t--){
        ans  = 0, sum = 0, ans1 = 0, x = 0;
        scanf("%d %d", &d, &n);
        x = d-1;
        for(i = 1; i <= n; i++){
            sum += i;
        }
        ans1 = sum;
        if(x == 0){
            printf("%d\n", sum);
        }
        else{
            while(x--){
                ans = 0;
                for(i = 1; i <= ans1; i++){
                    ans += i;
                }
                ans1 = ans;
            }
            printf("%d\n", ans);
        }
    }
	return 0;
}
