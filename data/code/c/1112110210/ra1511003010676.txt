#include <stdio.h>
int sum(int s)
{
	int i,su=0;
  	for(i=1;i<=s;i++)
      su+=i;
	return su;
}
int main()
{
  int i,test,d[16],n[16],j;
  scanf("%d",&test);
  for(i=0;i<test;i++)
    scanf("%d %d",&d[i],&n[i]);
  for(i=0;i<test;i++)
  {
    for(j=0;j<d[i];j++)
    {
      n[i]=sum(n[i]);
    }
    printf("\n%d",n[i]);
  }
    return 0;
}
