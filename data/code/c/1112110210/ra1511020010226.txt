#include <stdio.h>
int main()
{
int x=0,y,z;
  scanf("%d",&x);
  while( x>0)
  {
    scanf("%d%d",&y,&z);          
    while (y>0)
    {
      z=sum(z);   
      y--;
  }
    printf("%d\n",z);
    x--;
  } 
	return 0;
}
int sum (int a)
{
  int x=0,sum=0;
  while(x<=a)
  {
    sum = sum+x;
    x++;
  }
return sum ;
}
