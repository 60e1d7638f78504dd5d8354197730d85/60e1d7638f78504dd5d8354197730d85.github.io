#include <stdio.h>
int main()
{
int tc,h,t,i=0,grade[5],j=0,temp;
  float c;
  scanf("%d",&tc);
  temp=tc;
  while(tc!=0)
  {
    scanf("%d %f %d",&h,&c,&t);
    if(h>50&&c<0.7&&t>5600) grade[i]=10;
    else if(h>50&&c<0.7) grade[i]=9;
    else if(c<0.7&&t>5600) grade[i]=8;
    else if(h>50&&t>5600) grade[i]=7;
    else if(h>50||c<0.7||t>5600) grade[i]=6;
    else grade[i]=5;
    i++;
    tc--;
  }
  for(j=0;j<temp;j++)
    printf("%d\n",grade[j]);
	return 0;
}
