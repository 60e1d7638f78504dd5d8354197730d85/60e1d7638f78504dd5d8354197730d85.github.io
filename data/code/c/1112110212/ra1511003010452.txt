#include <stdio.h>
 
int main(void) {
 
    int T, h,s,grade;
    float c;
    scanf("%d",&T);
 
    while (T > 0){
 
        scanf("%d",&h);
        scanf("%f",&c);
        scanf("%d",&s);
 
        if(h > 50 && c < 0.7 && s > 5600){
            grade = 10;
 
        }
 
 
        if(h > 50 && c < 0.7 && s <= 5600){
            grade = 9;
 
        }
 
 
        if(h <= 50 && c < 0.7 && s > 5600){
            grade = 8;
 
        }
 
        if(h > 50 && c >= 0.7 && s > 5600){
            grade = 7;
 
        }
 
 
        if(h > 50 && c >= 0.7 && s <= 5600){
            grade = 6;
 
        }
 
        if(h <= 50 && c < 0.7 && s <= 5600){
            grade = 6;
 
        }
 
        if(h <= 50 && c >= 0.7 && s > 5600){
            grade = 6;
 
        }
 
        if(h <= 50 && c >= 0.7 && s <= 5600){
            grade = 5;
        }
 
        printf("%d\n",grade);
        T--;
    }
 
    return 0;
}
