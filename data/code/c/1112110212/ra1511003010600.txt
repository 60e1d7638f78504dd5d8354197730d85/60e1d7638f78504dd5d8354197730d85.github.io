#include <stdio.h>
 
int main(void) {
	int t,H,T;
	float C;
	scanf(" %d",&t);
	while(t--)
	{
	    scanf(" %d %f %d",&H,&C,&T);
	    if(H>50&&C<.7&&T>5600)          printf("%d\n",10);
	    else if(H>50&&C<.7)          printf("%d\n",9);
	    else if(C<.7&&T>5600)          printf("%d\n",8);
	    else if(H>50&&T>5600)          printf("%d\n",7);
	    else if(H>50||C<.7||T>5600)          printf("%d\n",6);
	    else                            printf("%d\n",5);
	}
	return 0;
}
