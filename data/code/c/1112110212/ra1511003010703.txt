#include <stdio.h>
int main()
{
  float h, c, t;
  int T;
  int grade;
  scanf("%d", &T);
  while(T--) {
    scanf("%f%f%f", &h, &c, &t);
    if (h > 50 && c < 0.7 && t > 5600)
      grade = 10;
    else if (h > 50 && c < 0.7)
      grade = 9;
    else if (c < 0.7 && t > 5600)
      grade = 8;
    else if (h > 50 && t > 5600)
      grade = 7;
    else if (h > 50 || c < 0.7 || t > 5600)
      grade = 6;
    else if (h <= 50 && c >= 0.7 && t <= 5600)
      grade = 5;
    printf("%d\n", grade);
  }
  return 0;
}
