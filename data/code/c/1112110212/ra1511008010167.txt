#include <stdio.h>
#include <math.h>
 
int main()
{
	int T;
	float hard,car,str;
	int c1,c2,c3;
 
	scanf("%d", &T);
 
	while(T--)
	{
		c1 = c2 = c3 = 0;
 
		scanf("%f %f %f", &hard, &car, &str);
 
		if(hard > 50)
		{
			c1 = 1;
		}
 
		if(car < 0.7)
		{
			c2 = 1;
		}
 
		if(str > 5600)
		{
			c3 = 1;
		}
 
		if((c1 == 1)&&(c2 == 1)&&(c3 == 1))
		{
			printf("10\n");
		}
		else if((c1==1)&&(c2==1))
		{
			printf("9\n");
		}
		else if((c2==1)&&(c3==1))
		{
			printf("8\n");
		}
		else if((c1==1)&&(c3==1))
		{
			printf("7\n");
		}
		else if((c1==1)||(c2==1)||(c3==1))
		{
			printf("6\n");
		}
		else
		{
			printf("5\n");
		}
	}
	
	return 0;
}
