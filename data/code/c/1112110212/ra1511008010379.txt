#include <stdio.h>
int main()
{
  int t;
  int h,j,g;
  float c;
  scanf("%d",&t);
  while(t--)
  {
    scanf("%d",&h);
    scanf("%f",&c);
    scanf("%d",&j);
    if(h>50 && c<0.7 && j>5600)
      g=10;
    else if(h>50 && c<0.7)
      g=9;
    else if(c<0.7 && j>5600)
      g=8;
    else if(h>50 && j>5600)
      g=7;
    else if(h>50 || c<0.7 || j>5600)  
      g=6;
    else
      g=5;
    printf("%d\n",g);
  }
	return 0;
}
