#include <stdio.h>
int main()
{
  int t=0,grade=0,i=0;
  float h,c,ts;
  scanf("%d",&t);
  for(i=0;i<t;i++)
  {
    scanf("%f%f%f",&h,&c,&ts);
    if(h>=50&&c<=0.7&&ts>=5600){
      grade=10;
    }
    else if(h>=50&&c<=0.7){
      grade=9;
    }
    else if(c<=0.7&&ts>=5600){
      grade=8;
    }
    else if(h>=50&&ts>=5600){
      grade=7;
    }
    else if(h>=50 ||c<=0.7 ||ts>=5600){
      grade=6;
    }
    else{
      grade=5;
    }
    printf("%d\n",grade);
      
  }

	return 0;
}
