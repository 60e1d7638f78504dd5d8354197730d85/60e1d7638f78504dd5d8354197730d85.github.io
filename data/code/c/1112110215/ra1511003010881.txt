#include <stdio.h>
int main()
{  int  fib1 = 0, fib2 = 1, fib3, limit, count = 0;
 
   // printf("Enter the limit to generate the Fibonacci Series \n");
    scanf("%d", &limit);
   // printf("Fibonacci Series is ...\n");
    printf("%d", fib1);
    printf("%d", fib2);
    count = 2;
    while (count < limit)
    {
        fib3 = fib1 + fib2;
        count++;
        printf("%d", fib3);
        fib1 = fib2;
        fib2 = fib3;
    }

	return 0;
}
