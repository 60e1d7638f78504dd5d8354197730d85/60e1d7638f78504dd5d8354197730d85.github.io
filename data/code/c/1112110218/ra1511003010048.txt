#include <stdio.h>
int main()
{
   int num, reverse_num=0, remainder,temp;
  
   scanf("%d", &num);


   temp=num;
   while(temp!=0)
   {
      remainder=temp%10;
      reverse_num=reverse_num*10+remainder;
      temp/=10;
   } 


   if(reverse_num==num) 
      printf("%d is a palindrome ",num);
   else
      printf("%d is not a palindrome ",num);
  return 0;
}
