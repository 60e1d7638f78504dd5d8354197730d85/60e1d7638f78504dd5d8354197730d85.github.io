#include <stdio.h>
int main()
{
	int n, rev=0, temp;
  	scanf("%d", &n);
  	
  	temp = n;
  
  	while(temp != 0)
    {
      rev = rev*10;
      rev = rev+temp%10;
      temp = temp/10;
    }
  	
  	if(n == rev)
      printf("%d is a palindrome", n);
  	else
      printf("%d is not a palindrome", n);
	return 0;
}
