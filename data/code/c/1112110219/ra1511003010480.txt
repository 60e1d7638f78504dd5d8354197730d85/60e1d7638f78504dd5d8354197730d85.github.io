#include <stdio.h>
#include <math.h>
int main()
{
  int num,sum=0,temp;
  scanf("%d",&num);

  temp=num;
  while(temp!=0)
  {
    sum+=pow(temp%10,3);
    temp/=10;
  }
  
  if(sum==num)
  {
    printf("%d is an Armstrong number.",num);
  }
  else
  {
    printf("%d is not an Armstrong number.",num);
  }
	return 0;
}
