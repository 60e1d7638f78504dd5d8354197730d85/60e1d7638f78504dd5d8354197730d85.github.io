#include <stdio.h>
int main()
{
    int n, i, flag = 0;

    
    scanf("%d",&n);

    for(i=2; i<=n/2; ++i)
    {
        // condition for nonprime number
        if(n%i==0)
        {
            flag=1;
            break;
        }
    }

    if (flag==0)
        printf("Give to Scientist Newton");
    else
        printf("Dont Give to Scientist Newton");
    
    return 0;
}
