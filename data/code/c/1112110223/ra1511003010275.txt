#include <stdio.h>
int main()
{
int n;
		int i;
 
		int counter=0;
		int counter1=0;
		scanf("%d",&n);
		int a[n];
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
			if(a[i]%2==0)
			{
				counter=counter+1;
			}
			else
			{
				counter1=counter1+1;
			}
		}
		if(counter>counter1)
		{
			printf("READY FOR BATTLE");
		}
		else
		{
			printf("NOT READY");
        }
	return 0;
}
