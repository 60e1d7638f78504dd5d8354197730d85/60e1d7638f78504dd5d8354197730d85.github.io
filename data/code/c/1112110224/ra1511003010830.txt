#include <stdio.h>
typedef long long ll;
int main()
{
    ll t, n, m, z, l, r, b;
    ll p,h,c1,c2,c3,c,lb,rb,mid;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld %lld %lld %lld %lld %lld",&n,&m,&z,&l,&r,&b);
        p = n*m;
        if((z+l+r)>= p)
            printf("%lld\n",p);
        else
        {
            h = m%2 ? m/2 +2 :m/2;
	c1 = p-l-r-z;
	c2 = n*h;
	c3 = (p+n-l-r)/2;
	lb=0;
	rb=b+1;
	while(rb-lb>1)
	{
	 mid = (lb+rb)>>1;
	 if(mid<=c1 && mid<=c2 && mid<=c3)
	  lb=mid;
	 else
  	  rb=mid;
  	}
    c = l+r+z+lb;
           printf("%lld\n",c);
        }
    }
  return 0;
}
