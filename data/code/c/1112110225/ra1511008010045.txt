#include <stdio.h>
int main()
{
   int t;
    signed int s1,s2,gcd,i,f;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d",&s1,&s2);
          for(i=1;i<=abs(s1)&&i<=abs(s2);i++)
        {
            if(s1%i==0 && s2%i==0)
            gcd=i;
        }
        f=abs(s1-s2);
        printf("%d\n",f/gcd);
    }
	return 0;
}
