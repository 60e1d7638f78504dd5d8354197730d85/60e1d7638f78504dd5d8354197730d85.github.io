#include <stdio.h>
int main()
{ int n,temp,reverse;	
  scanf("%d",&n);
 
   temp = n;
 
   while( temp != 0 )
   {
      reverse = reverse * 10;
      reverse = reverse + temp%10;
      temp = temp/10;
   }
 
   if ( n == reverse )
      printf("Mirror Image");
   else
      printf("Not a Mirror Image");

	return 0;
}
