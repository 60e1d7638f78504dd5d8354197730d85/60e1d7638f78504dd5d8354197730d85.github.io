#include <stdio.h>
int inp[1003];
int main()
{ int test,n,i,j,k,time=0;
  scanf("%d",&test);
  while(test--)
  {
    time=0;
    scanf("%d %d",&n,&k);
    inp[0]=0;
    for(i=1;i<=n;i++)
    {
      scanf("%d",&inp[i]);
    }
    j=1;
    for(i=n;i>0;i--)
    {
      time+=(((j-1)/k)+1)*(inp[i]-inp[i-1]-1);
      j++;
    }
    printf("%d\n",time);
  }
 
	return 0;
}
