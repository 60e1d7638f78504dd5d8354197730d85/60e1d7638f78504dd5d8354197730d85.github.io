#include <stdio.h>
int main()
{
    int t; 
    scanf("%d",&t);
    while(t--){
        int n,k,i,a[1000]={[0 ... 999]=0},j,m;
        scanf("%d %d",&n,&k);
        int sum=0;
     
      	for(i=1;i<=n;i++)
            scanf("%d",&a[i]);
        
      	j=1;
        for(i=n;i>0;i--){
                 sum += (((j-1)/k)+1)*(a[i]-a[i-1]-1);
                 j++;
        }
        printf("%d\n",sum);
    }
    return 0;
}
