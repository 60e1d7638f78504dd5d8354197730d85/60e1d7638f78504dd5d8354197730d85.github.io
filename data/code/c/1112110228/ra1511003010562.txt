#include <stdio.h>
int main()
{
  int T,N,K,X[1000],i,j,ans,total,moves;
  scanf("%d",&T);
  while(T--)
  {
    ans=0;
    scanf("%d %d",&N,&K);
    for(i=0;i<1000;i++)
      X[i]=0;
    for(i=0;i<N;i++)
    {
      scanf("%d",&j);
      X[j]=1;
    }
    total=0;
    while(1)
    {
      while(j&&X[j])
      {
        total++;
        j--;
      }
      if(!j)
        break;
      moves=0;
      while(j && !X[j])
      {
        moves++;
        j--;
      }
      if(total%K)
        ans+=(moves*((total/K)+1));
      else
        ans+=(moves*(total/K));
    }
    printf("%d\n",ans);
  }
	return 0;
}
