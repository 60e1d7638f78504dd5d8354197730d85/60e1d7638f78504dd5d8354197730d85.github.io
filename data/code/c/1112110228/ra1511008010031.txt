#include<stdio.h>
int main()
{
    int t,hash[1001],n,k,num,pivot,i,j,val;
    scanf("%d",&t);
//    arr[0]=0;
    hash[0]=0;
    while(t--)
    {
        int ans=0;
        int group=0;
        scanf("%d%d",&n,&k);
        int tot=0;
        for(i=1;i<=1000;i++)
        {
            hash[i]=0;
        }
        for(i=1;i<=n;i++)
        {
            scanf("%d",&num);
            if(i==n)
                pivot=num;
            hash[num]=1;
        }
        for(i=pivot;i>=1;i--)
        {
            if(hash[i]==1)
                group++;
            else
                ans+=(group+k-1)/k;
        }
        printf("%d\n",ans);
    }
 
    return 0;
}
