#include <stdio.h>
 
int main()
{
	int i, k, n, count, t, l, index, flag;
	int arr[1001];
	
	scanf("%d", &t);
	
	while(t--)
	{
		for(i=0; i<1001; i++)
			arr[i] = 0;
		count = 0;
		scanf("%d%d", &n, &k);
		
		for(i=0; i<n; i++)
		{
			scanf("%d", &l);
			arr[l-1] = 1;
		}
		
		index = l-1;
		if(index == n-1)
			printf("0\n");
		else
		{
			while(index > n-1)
			{
				flag = 0;
				for(i=1; i<=index; i++)
				{
					if(arr[index-i+1]==1 && arr[index-i]==0)
						flag=1;
					if(flag==1)
						break;
				}
				if(flag==1)
				{
					arr[index-i] = 1;
					if(i<=k)
						count++;
					else
					{
						if(i%k == 0)
							count+=(i/k);
						else
							count+=((i/k)+1);
					}
					index--;
				}
			}
			printf("%d\n", count);
		}
	}
	return 0;
}
