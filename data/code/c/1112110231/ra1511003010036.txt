#include <stdio.h>
int main()
{
   int  num2, flag_var, i, j;
   scanf("%d",&num2);
  for(i=2; i<=num2; ++i)
   {
      flag_var=0;
      for(j=2; j<=i/2; ++j)
      {
         if(i%j==0)
         {
            flag_var=1;
            break;
         }
      }
      if(flag_var==0)
         printf("%d ",i);
  }
  return 0;
}
