#include <stdio.h>
int main()
{int i,n,j,flag;
 scanf("%d",&n);
  for (i = 3; i < n; i = i + 2)
    {
        flag = 0;
        for (j = 2; j <= i / 2; j++)
        {
            if ((i % j) == 0)
            {
                flag = 1;
                break;
            }
        }
        if (flag == 0)
        {
            printf("%d ", i);
            
        }
  }
	return 0;
}
