#include<stdio.h>
//#include<limit.h>
 
 
int main()
{
    int i,j,k,n,m,a,min,max;
    scanf("%d %d",&n,&m);
    for(i=0;i<n;i++)
    {
        scanf("%d",&a);
        if(i==0){min=a;max=a;}
        if(a<min){min=a;}
        if(a>max){max=a;}
    }
 
    for(i=0;i<m;i++)
    {
        scanf("%d",&a);
        if(a>=min&&a<=max){printf("Yes\n");}
        else printf("No\n");
    }
 
    return 0;
}
