
#include<stdio.h>
int main(){
    int t;
    scanf("%d",&t);
    while (t--) {
        int n,m;
        char waste;
        scanf("%d%d%c",&n,&m,&waste);
        int i,j;
        char a[n][m];
        for (i=0;i<n; i++) {
            for (j=0;j<m; j++) {
                scanf("%c",&a[i][j]);
            }
            char temp;
            scanf("%c",&temp);
        }
        int sum[m];
        for (i=0;i<m ;i++) {
            int temp=0;
            for (j=0;j<n;j++) {
                if(a[j][i]=='1')
                    temp++;
            }
            sum[i]=temp;
        }
        int ans=0;
        for (i=0; i<m; i++) {
            if(sum[i]!=0)
            ans+=(sum[i]*(sum[i]-1))/2;
        }
        printf("%d\n",ans);
    }
    return 0;
}
