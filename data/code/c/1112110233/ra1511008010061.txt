#include<stdio.h>
int main()
{
	int t,n,m,i,j,a[10],sum;
	char s[10];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d\n",&n,&m);
		sum=0;
		for(i=0;i<10;i++)
			a[i]=0;
		for(i=0;i<n;i++)
		{
			scanf("%s",s);
			for(j=0;j<m;j++)
				if(s[j]=='1')
				  a[j]++;
		}
		for(j=0;j<m;j++)
			sum=sum+(a[j]*(a[j]-1)/2);
		printf("%d\n",sum);
	}
	return 0;
}
