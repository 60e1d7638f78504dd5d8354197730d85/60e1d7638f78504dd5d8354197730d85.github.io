#include <stdio.h>
#include <stdlib.h>
 
int main()
{
	int t,n,m,i,j,k,l,a,p;
	int sum[100];
	char string[10];
	int g[10];
	scanf("%d",&t);
	for(i=0;i<=t-1;i++)
	{
		for(p=0;p<=9;p++)
            g[p]=0;
        sum[i]=0;
		scanf("%d%d",&n,&m);
		for(j=0;j<=n-1;j++)
		{
			scanf("%s",string);
			for(k=0;k<=m-1;k++)
			{
				if(string[k]=='1')
					g[k]=g[k]+1;
				if(string[k]=='0')
					continue;
			}
		}
		for(l=0;l<=m-1;l++)
			sum[i]=sum[i]+g[l]*(g[l]-1)/2;
	}
	for(a=0;a<=t-1;a++)
		printf("\n%d",sum[a]);
  return 0;
}
