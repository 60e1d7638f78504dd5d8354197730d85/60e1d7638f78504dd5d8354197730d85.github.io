#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int main()
{
    int t,n,m,i,j,count,temp,col=0;
    char *a[10],word[10];
    scanf("%d",&t);
    while(t--){
        col = 0;
        scanf("%d%d",&n,&m);
        for(i=0;i<n;i++){
            scanf("%s",word);
            a[i] = (char*)malloc((strlen(word)+1)*sizeof(char));
            strcpy(a[i],word);
        }
        for(i=0;i<m;i++){
            count = 0;
            for(j=0;j<n;j++){
                if(a[j][i]=='1')
                    count++;
            }
        if(count<=1)
            col+=0;
        else
            col+=(count*(count-1)/2);
        }
        printf("%d\n",col);
    }
    return 0;
}
