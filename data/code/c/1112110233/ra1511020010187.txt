
#include<stdio.h>
int main(){
int col[50],t,M[15][15],ans,sum,i,j,n,m;
char str[20];
scanf("%d",&t);
col[0]=0;
for(i=1;i<=15;i++)
    col[i]=col[i-1]+i-1;
while(t--){
 
    scanf("%d%d",&n,&m);
    for(i=1;i<=n;i++){
        scanf("%s",str);
        for(j=0;j<m;j++){
            M[i][j+1]=str[j]-48;
        }
 
    }
   for(j=1,ans=0;j<=m;j++){
 
    for(i=1,sum=0;i<=n;i++){
        if(M[i][j]==1){
            sum++;
        }
 
    }
    ans+=col[sum];
 
   }
 
   printf("%d\n",ans);
 
 
}
 
 
 
return 0;
}
