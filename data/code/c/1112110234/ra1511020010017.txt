#include<stdio.h>
#include<stdlib.h>
int cmp(const void *a,const void *b)
{
return (*(int*)a-*(int*)b);
}
 
int main()
{
	int t,n,j,i,b;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		scanf("%d",&b);
		int a[n],sum=0;
		for(i=0;i<n;i++)
		scanf("%d",&a[i]);
 
		qsort(a,n,sizeof(int),cmp);
		for(i=b;i<n-b;i++)
		{
			sum=sum+a[i];
		}
		printf("%f\n",sum/(n-2.0*b));		
	}
	return 0;
}
