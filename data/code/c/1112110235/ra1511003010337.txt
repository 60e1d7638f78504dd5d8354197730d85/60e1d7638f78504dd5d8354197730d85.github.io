#include <stdio.h>
int main()
{
  float f;
  scanf("%f",&f);
  if(f<2.0)
    printf("Micro");
  
  if(f>=2.0 && f<=3.0)
    printf("Very minor");
  
  if(f>=3.0 && f<=4.0)
    printf("Minor");
	
  if(f>=4.0 && f<=5.0)
    printf("Light");

  if(f>=5.0 && f<=6.0)
    printf("Moderate");

  if(f>=6.0 && f<=7.0)
    printf("Strong");
  
  if(f>=7.0 && f<=8.0)
    printf("Major");
 
  if(f>=8.0 && f<=10.0)
    printf("Great");
  
  if(f>10.0)
    printf("Meteoric");
  
	return 0;
}
