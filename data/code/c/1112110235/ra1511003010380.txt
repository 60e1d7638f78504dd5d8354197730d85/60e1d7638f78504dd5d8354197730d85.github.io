#include <stdio.h>
int main()
{  float n;
 scanf("%f",&n);
 if(n>=10)
   printf("Meteoric");
 else if(n>=8 && n<10)
   printf("Great");
 else if(n>=7 && n<8)
   printf("Major");
 else if(n>=6 && n<7)
   printf("Strong");
 else if(n>=5 && n<6)
   printf("Moderate");
 else if(n>=4 && n<5)
   printf("Light");
 else if(n>=3 && n<4)
   printf("Minor");
 else if(n>=2 && n<3)
   printf("Very minor");
 else if(n<2)
   printf("Micro");

	return 0;
}
