#include <stdio.h>
int main()
{
float a;
  scanf("%f",&a);
  if(a<2.0) printf("Micro");
  else if(a>=2.0 && a<3.0) printf("Very minor");
  else if(a>=3.0 && a<4.0) printf("Minor");
  else if(a>=4.0 && a<5.0) printf("Light");
  else if(a>=5.0 && a<6.0) printf("Moderate");
  else if(a>=6.0 && a<7.0) printf("Strong");
  else if(a>=7.0 && a<8.0) printf("Major");
  else if(a>=8.0 && a<10.0) printf("Great");
  else if(a>=10.0) printf("Meteoric");
  else ;
  
	return 0;
}
