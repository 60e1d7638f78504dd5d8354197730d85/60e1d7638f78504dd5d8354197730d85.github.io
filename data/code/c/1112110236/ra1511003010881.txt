#include <stdio.h>
int main()
{  long long int t;
  scanf("%lld",&t);
  while(t--)
  {
  long long int n,k,sum=0;
  scanf("%lld%lld",&n,&k);
  long long int dp[n][n],i,j;		
  for(i=0;i<n;i++)
   {   scanf("%lld",&dp[0][i]);
   	   if(dp[0][i]>=k)
   	    sum++;
		 }
		 	  
  //for( i=1;i<n;i++)
  i=1;	  
	  { for(j=i;j<n;j++)
	     {
	     	dp[i][j]=(dp[i-1][j-1]|dp[0][j]);
	     	if(dp[i][j]>=k)
	     	 sum++;
	     	if(j==n-1)
	     	  {i=i+1;
	     	   j=i-1;
			   }
		 }
	  }
	  printf("%lld\n",sum);  
  }	

	return 0;
}
