#include<stdio.h>
#include<math.h>
#include<stdlib.h>
#include<string.h>
long long int a[100005];
int main()
{
	long long int i,j,k,l,t,n,k1,s,s1,p,p1,p2;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&n,&k);
		for(i=1;i<=n;i++)
		scanf("%lld",&a[i]);
		s=0;
		p=1;
		s1=0;
		for(i=1;i<=n;i++)
		{
			s=s|a[i];
			if(s>=k)
			{
			    k1=0;
			    p1=j;
				for(j=i;j>=p;j--)
				{
					k1=k1|a[j];
					if(k1>=k)
					{
						p1=j;
						break;
					}
					else
					s=k1;
				}	
				s1=s1+(p1-p+1)*(n-i+1);
				p=p1+1;
			}
		}
		printf("%lld\n",s1);
	}
	return 0;
}
