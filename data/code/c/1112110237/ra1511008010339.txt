#include <stdio.h>
unsigned long long int gcd(unsigned long long int a,unsigned long long int b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}
int main()
{
	int i,j,t,n;
	scanf("%d",&t);
	while(t--)
	{
		unsigned long long int min=99999999,lcm;
		scanf("%d",&n);
		unsigned long long int arr[n];
		for(i=0;i<n;i++)
		scanf("%llu",&arr[i]);
		for(i=0;i<n;i++)
		{
			for(j=i+1;j<n;j++)
			{
				if(arr[i]>min||arr[j]>min)
				continue;
				lcm=(arr[i]*arr[j])/gcd(arr[i],arr[j]);
				if(lcm<min)
				min=lcm;
			}
		}
		printf("%llu\n",min);
	}
	return 0;
}
