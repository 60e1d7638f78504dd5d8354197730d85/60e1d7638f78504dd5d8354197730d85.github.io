 #include<stdio.h>
int hcf(int a,int b)
 { int c;
   if(a<b) {c=a;a=b;b=c;}
   while(b>0)
    { c=a%b;
      a=b;
      b=c;
    }
   return(a);
 }
 
unsigned long long int lcm(int a,int b,int c)
  { unsigned long long int d=a*(b/c);
     return(d);
  }
