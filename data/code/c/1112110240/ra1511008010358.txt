#include<stdio.h>
#include<string.h>
int main()
{
  	int t,n0=0,n1=0,len,i;
  	char arr[100005];
  	scanf("%s", arr);
  	len=strlen(arr);
  	for(i=0;i<len;i++)
    {
      	if(arr[i]=='0')
          	++n0;
      	else
          	++n1;
    }
  	if(n1==len-1||n0==len-1)
      	printf("\nYes");
  	else
      	printf("\nNo");
  	return 0;
}
