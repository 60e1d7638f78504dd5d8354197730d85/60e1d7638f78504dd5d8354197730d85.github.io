#include <stdio.h>
#include<stdlib.h>
#include<math.h>
#include<string.h>
int main()
{char knows[27];
	scanf("%s",knows);
	int i,j,k;
	int n,flag;
	scanf("%d",&n);
	for (i=0;i<n;i++)
	{
		char word[13];
		scanf("%s",word);
		for (j=0;j<strlen(word);j++)
		{
			flag = 0;
			for (k=0;k<strlen(knows);k++)
			{
				if (knows[k]==word[j])
					flag = 1;
			}
			if (flag==0)
				break;
		}
		if (flag==0)
			printf("No\n");
		else
			printf("Yes\n");
    }

	return 0;
}
