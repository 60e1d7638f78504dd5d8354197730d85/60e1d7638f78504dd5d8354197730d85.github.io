#include <stdio.h>
#include <stdlib.h>
 

 
int main(){
    unsigned long long int i,j;
    int *primes;
  int limit;
    scanf("%d",&limit);
 
    primes = malloc(sizeof(int) * limit);
 
    for (i = 2;i < limit; i++)
        primes[i] = 1;
 
    for (i = 2;i < limit; i++)
        if (primes[i])
            for (j = i;i * j < limit; j++)
                primes[i * j] = 0;
 
   // printf("\nPrime numbers in range 1 to 100 are: \n");
    for (i = 3;i < limit; i++)
        if (primes[i])
            printf("%lld\n", i);
 
return 0;
}
