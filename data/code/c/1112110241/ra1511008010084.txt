#include <stdio.h>
#include<string.h>
int n,i,j,k,c,x[1000][1000],l,y[1000];
char a[30],b[15];
main()
{
scanf("%s",a);
scanf("%d",&n);
for(i=0;i<n;i++)
{
l=0;
scanf("%s",b);
y[i]=strlen(b);
for(j=0;b[j]!='\0';j++)
{c=0;
for(k=0;a[k]!='\0';k++)
{
if(b[j]==a[k])
c++;
}
x[i][j]=c;
}
}
for(i=0;i<n;i++)
{c=0;
for(j=0;j<y[i];j++)
{
if(x[i][j]==0)
c++;
}
if(c==0)
printf("Yes\n");
else
printf("No\n");
}
  return 0;
}
