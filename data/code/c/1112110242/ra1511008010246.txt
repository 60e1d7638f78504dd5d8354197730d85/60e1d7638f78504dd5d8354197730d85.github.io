#include <stdio.h>
#include <string.h>
int main()
{
    int t,n,j,b,c;
    char a[100000];
    scanf("%d",&t);
    while(t--)
    {
        b=0;
        c=0;
        scanf("%s",a);
        n=strlen(a);
        for(j=0;j<n;j++)
        {
            if(a[j]=='0')
                b++;
            else
                c++;
        }
        
        if((b==1) || (c==1))
            printf("Yes\n");
        else 
            printf("No\n");
        
    }

	return 0;
}
