#include <stdio.h>
#include <stdlib.h>
 
int main()
{
    int t;
    long int coin,n,k,i;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%ld%ld",&n,&k);
        coin=0;
        for (i=1;i<=k;i++)
        {
            if ((n%i)>coin)
                coin=n%i;
        }
        printf("%ld\n",coin);
    }
    return 0;
}
