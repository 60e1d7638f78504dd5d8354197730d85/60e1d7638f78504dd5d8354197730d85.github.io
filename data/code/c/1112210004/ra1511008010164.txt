#include <stdio.h>
#include <math.h>
int main()
{

   int t,n,i,s;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        i = 1;
        s = 0;
        while(n>=fib(i))
        {
            s++;
            i++;
        }
        s--;
        printf("%d\n",abs(s));
    }
	return 0;
}
int fib(int n)
{
  int F[2][2] = {{1,1},{1,0}};
  if (n == 0)
    return 0;
  power(F, n-1);
  return F[0][0];
}

int power(int F[2][2], int n)
{
  if( n == 0 || n == 1)
      return;
  int M[2][2] = {{1,1},{1,0}};
 
  power(F, n/2);
  multiply(F, F);
 
  if (n%2 != 0)
     multiply(F, M);
  return 0;
}
 
int multiply(int F[2][2], int M[2][2])
{
  int x =  F[0][0]*M[0][0] + F[0][1]*M[1][0];
  int y =  F[0][0]*M[0][1] + F[0][1]*M[1][1];
  int z =  F[1][0]*M[0][0] + F[1][1]*M[1][0];
  int w =  F[1][0]*M[0][1] + F[1][1]*M[1][1];
 
  F[0][0] = x;
  F[0][1] = y;
  F[1][0] = z;
  F[1][1] = w;
  return 0;
}
