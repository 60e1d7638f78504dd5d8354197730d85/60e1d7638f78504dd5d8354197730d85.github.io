#include <stdio.h>
#include<string.h>
int main()
{
  char a[1001][1001]={0};
  int ans[1001],i,len,n;
  char str[1001];
  scanf("%s",str); 
  ans[0]=1;
  n=strlen(str);
  for(i=0;i<n;i++)
  {
    for(len=1;len<=i+1;len++)
      if(str[i]==str[i+1-len]&&(len<=2 || a[i-1][len-2]))
      {
        a[i][len]=1;
        ans[i+1]=(ans[i+1]+ans[i+1-len])%1000000007;
      }
  }
  printf("%d\n",ans[n]); 
  return 0;
}
