#include<stdio.h>
#include<string.h>
#define MAX 1000000007
char str[1005];
char a[1005][1005]={0};
check(int i,int j)
{	
	if(str[i]!=str[j])
	return(0);
	i++;j--;
	while(j>i)
	{
		if(a[i][j])
		return(1);
		else return(0);
	
	}
	return(1);
}
int main()
{
char c;
int b[1005]={0},i=0,j,k,l;
b[0]=1;
scanf("%s",str);
l=strlen(str);
 
	for(i=0;i<l;i++)
	a[i][i]=1;
	
	for(j=0;j<l;j++)
	{
		k=j+1;
		for(i=0;i<=j;i++)
		{
			if(a[i][j]=check(i,j))
			{
				b[k]=(b[k]+b[i])%MAX;	
				
			}
		}
		
	}
	printf("%d\n",b[l]%MAX);
	return(0);
}
