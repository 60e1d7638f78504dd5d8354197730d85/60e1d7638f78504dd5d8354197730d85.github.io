#include<stdio.h>
#include<string.h>

char s[1001];
char A[1000][1001]={{0}};
unsigned int ans[1001]={0};

int main()
{
	int i,len=1,n;
	scanf("%s",s);
	n=strlen(s);
	ans[0]=1;
	for(i=0;i<n;i++)
	{
		for(len=1;len<=i+1;len++)
		{
			if(s[i]==s[i+1-len] && (len<=2 ||A[i-1][len-2]))
			{
				A[i][len]=1;
				ans[i+1]=(ans[i+1]+ans[i+1-len])%1000000007;
			}
 
		}
	}
	printf("%d",ans[n]);
	return 0;
}
