#include<stdio.h>
#include<math.h>
int main() {
    unsigned long long int l, d, s, c, t;
    //FILE *file = fopen("/home/rohit/Desktop/jio/1.txt", "r");
    fscanf(stdin, "%llu", &t);
    while (t--) {
        fscanf(stdin, "%llu %llu %llu %llu", &l, &d, &s, &c);
        if (log(c + 1.0) * (d - 1) >= log(l) - log(s))
            puts("ALIVE AND KICK");
        else
            puts("DEAD AND ROT");
    }
    return 0;
}
