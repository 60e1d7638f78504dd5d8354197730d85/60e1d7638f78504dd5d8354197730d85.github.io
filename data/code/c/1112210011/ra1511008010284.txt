#include <stdio.h>
int main()
{
    int t,count=0,count1=0;
  	scanf("%d",&t);
  	
  	int i,j,k,s;
  	for(i=0;i<t;i++)
    {   
       
   		scanf("%d",&s);
      	int a[s];
      	
      	for(j=0;j<s;j++)
        {
        	scanf("%d",&a[j]);
        }
       
        count1=0;
      	for(j=0;j<s;j++)
        {
          count=0;
          	for(k=0;k<j;k++)
          	{
             	if(a[k]>a[j])
             	{
               	count++;
             	}
            }
          if(count==j)
            count1++;
         }
      printf("%d\n",count1);
      
      
    }
	return 0;
}
