#include<stdio.h>
    int main()
    {
    long double s,t,d;
    int n;
    scanf("%d",&n);
    while(n--)
    {
     scanf("%Lf %Lf %Lf",&s,&t,&d);
     if(d>=s+t)
     printf("%Lf\n",d-(s+t));
     else if(s>=(d+t))
     printf("%Lf\n",s-(d+t));
     else if(t>=(d+s))
     printf("%Lf\n",t-(d+s));
     else
     printf("0.0\n");
    }
    return 0;
    }
