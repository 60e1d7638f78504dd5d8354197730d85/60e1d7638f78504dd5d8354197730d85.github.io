#include<stdio.h>
int main()
{
    float s,t,d,min;
     int T;

    scanf("%d",&T);
    while(T--)
    {
        scanf("%f%f%f",&s,&t,&d);
        if((s+t)<=d)
            printf("%f\n",d-s-t);
        else if(s>=(d+t))
            printf("%f\n",s-d-t);
        else if(t>=(d+s))
            printf("%f\n",t-d-s);
        else
            printf("0.0\n");
    }

    return 0;   
}
