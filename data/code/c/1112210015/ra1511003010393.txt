#include<stdio.h>
int main()
{
	float t,ds,dt,d,i,l,sm,s; float ans;
	scanf("%f",&t);
	while(t--)
	{
		scanf("%f%f%f",&ds,&dt,&d);
		if(ds>dt)
		s=ds-dt;
		else
		s=dt-ds;
		if(d>ds+dt)
		printf("%f\n",d-ds-dt);
		else if(d<=ds+dt&&d>=s)
		printf("0.0\n");
		else
		{
			if(ds<dt)
			{
			sm=ds;l=dt;}
			else{
			
			sm=dt;l=ds;}
			if(l>d+sm)
			ans=l-d-sm;
			else
			ans=d+sm-l;
			printf("%f\n",ans);
		}
	
	}
	
	return 0;
}
