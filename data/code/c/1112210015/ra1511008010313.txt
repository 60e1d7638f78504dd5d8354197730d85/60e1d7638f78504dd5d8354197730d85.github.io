 #include<stdio.h>
     
    int main()
	{
        
        int t,ds,dt,d,i;
        scanf("%d",&t);
        while(t--)
		{
            scanf("%d %d %d",&ds,&dt,&d);
            i = ds+dt;
            if(ds>(d+dt))
            {    
              printf("%d.000000\n",ds-dt-d);
            }
             else if(dt>(d+ds))
            {   
               printf("%d.000000\n",dt-ds-d);
            }
             else if( i > d)
             {
               printf("0.0\n");
             }
              else
              {  
                printf("%d.000000\n",d-i);
              }
        }
        return 0;
    }
