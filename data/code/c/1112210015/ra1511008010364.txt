#include<stdio.h>
     
    int main()
	{
        
        float t,ds,dt,d,i;
        scanf("%f",&t);
        while(t--)
		{
            scanf("%f %f %f",&ds,&dt,&d);
            i = ds+dt;
            if(ds>(d+dt))
                printf("%.6f\n",ds-dt-d);
            else if(dt>(d+ds))
                printf("%.6f\n",dt-ds-d);
            else if( i > d)
                printf("0.0\n");
            else
                printf("%f\n",d-i);
            
        }
        return 0;
    }
