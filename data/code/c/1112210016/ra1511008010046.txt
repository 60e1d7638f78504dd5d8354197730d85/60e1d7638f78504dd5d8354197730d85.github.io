#include <stdio.h>
inline void fastread_lint(long  int  *a)
{
    char c=0; *a=0;
    while(c<33){c=getchar_unlocked();}
    while(c>33){*a=(*a<<3)+(*a<<1)+c-'0'; c=getchar_unlocked();}
}
inline void fastread_llint(long long int  *a)
{
    char c=0; *a=0;
    while(c<33){c=getchar_unlocked();}
    while(c>33){*a=(*a<<3)+(*a<<1)+c-'0'; c=getchar_unlocked();}
}
inline void fastread_lluint(unsigned long long int  *a)
{
    char c=0; *a=0;
    while(c<33){c=getchar_unlocked();}
    while(c>33){*a=(*a<<3)+(*a<<1)+c-'0'; c=getchar_unlocked();}
}
int main(void) {
	long long int n;
	fastread_llint(&n);
	long int m;
	fastread_lint(&m);
	while(m--)
	{
		unsigned long long int q;
		fastread_lluint(&q);
		if(q<=n+1)
		printf("0\n");
		else if(q<=2*n+1)
		printf("%llu\n",q-n-1);
		else
        {
          if(3*n-q+1>100000)printf("0\n");
          else
		printf("%llu\n",3*n-q+1);}
		
	}
	
	return 0;
}
