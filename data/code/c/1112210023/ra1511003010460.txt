#include <stdio.h>
#include<string.h>
int a[1000001];
int vis[1000001]={0};
int temp[1000001];
int main(){
    int t,n,res,count,x,y;
    long long int i,j;
    vis[0]=1;
    vis[1]=1;
    for(i=2;i<=1000000;i++){
        if(vis[i]==0){
            vis[i]=i;
            for(j=i*i;j<=1000000;j+=i)
                if(vis[j]==0)
                    vis[j]=i;
        }
    }
    scanf("%d",&t);
    while(t--){
        scanf("%d",&n);
        for(i=0;i<n;i++)
            scanf("%d",&a[i]);
        memset(temp,-1,1000001*sizeof(int));
        count=0;
        res=0;
        for(i=0;i<n;i++){
            x=a[i];
            if(x==1)
                count++;
            else{
                while(x>1){
                    y=vis[x];
                    if(temp[y]==i){}
                    else if(temp[y]==-1){
                        temp[y]=i;
                    }
                    else{
                        if(temp[y]<i-count)
                            temp[y]=i;
                        else{
                            if(count>res)
                                res=count;
                            count=i-temp[y]-1;
                            temp[y]=i;
                        }
                    }
                    x=x/y;
                }
                count++;
            }
        }
        if(count>res)
            res=count;
        printf("%d\n",(res>1)?res:-1);
    }
    return 0;
}
