#include <stdio.h>
int rec(int k,int l)
{
    int c;
    while(k>0)
    {
        c=(l*(l+1))/2;
        l=c;
        k--;
    }
    return l;
}
 
int main(void) {
    int t; scanf("%d",&t);
    while(t--)
    {
        int y,z; scanf("%d %d",&y,&z);
        printf("%d\n",rec(y,z));
    }
	// your code goes here
	return 0;
}
