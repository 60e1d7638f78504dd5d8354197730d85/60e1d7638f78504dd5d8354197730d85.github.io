#include <stdio.h>
#include <stdlib.h>
#include <math.h>
 
int main()
{   long long c,d,e;
	scanf("%lld\n",&c);
	while(c--)
	{
	    scanf("%lld %lld\n",&d,&e);
	    long long count=0;
	    while(d!= e)
	    {
		count+=1;
		if (d>e)
		{
			d /=2;
		}
		else if( e>d)
		{
			e /=2;
		}
	  }
	   printf("%lld\n",count);
	}
	return 0;
}
