#include<stdio.h>
int main()
{
    int t,n,l,i,j;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        int a[n];
        l=n;
        for(i=0;i<n;i++)
        {
            scanf("%d",&a[i]);
        }
        for(i=0;i<n;i++)
        {
            for(j=i+1;j<n;j++)
            {
                if(a[j]==a[i]&&a[j]!=0)
                {
                    a[j]=0;
                    l=l-1;
                }
            }
        }
        printf("%d\n",l);
    }
}
