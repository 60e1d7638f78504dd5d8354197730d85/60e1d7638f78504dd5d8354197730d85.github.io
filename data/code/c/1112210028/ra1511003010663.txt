#include <stdio.h>
#include<stdlib.h>
#include<string.h>
 
int main()
{
	int t,i;
	scanf("%d",&t);
	while(t--)
	{
	int n,count=0;
	scanf("%d",&n);
	int a[n];
	int b[101]={0};
	for(i=1;i<=n;i++){  
	scanf("%d",&a[i]);
	if(b[a[i]]==0)
	{
	b[a[i]]=1;
	count++;
}
}
 
	printf("%d\n",count);
	}
return 0;
}
