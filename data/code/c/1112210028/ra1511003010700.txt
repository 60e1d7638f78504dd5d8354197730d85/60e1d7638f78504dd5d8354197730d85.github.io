#include <stdio.h>
 
int main(void) {
int t,i,n;
scanf("%d",&t);
while(t--)
{
    scanf("%d", &n);
    int a[n],k=0,j=0;
    for(i=0;i<n;i++)
    scanf("%d",&a[i]);
    
    if(n<=50)
    {
        k=n;
    for(i=0;i<n-1;i++)
    {
        
        for(j=i+1;j<n;j++)
        {
            
            if(a[i]==a[j])
           {
            --k;
             break;
        }   }
    }
    }
    printf("%d\n",k);
}
 
return 0;
}
