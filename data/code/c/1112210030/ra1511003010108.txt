#include<stdio.h>
int main()
{
	int n,i,p1,p2,d1=0,d2=0,l1=0,l2=0;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d %d",&p1,&p2);
		d1=d1+p1;
		d2=d2+p2;
		if(d1>d2)
		{
			if(l1<(d1-d2))
			{
				l1=d1-d2;
			}
		}
		else
		{
			if(l2<(d2-d1))
			{
				l2=d2-d1;
			}
		}
		
	}
	if(l1>l2)
	printf("1 %d",l1);
	else
	printf("2 %d",l2);
	return 0;
}
