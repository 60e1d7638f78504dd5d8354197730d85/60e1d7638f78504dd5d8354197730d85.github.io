 
   #include<stdio.h>
    int main()
    {
    	int c=0,n,s,temp1=0,ca=0,temp2=0,i,t,value=1;
    	scanf("%d",&n);
    	
    	for(i=0;i<n;i++)
    	{    	  
    	    scanf("%d%d",&s,&t);
    	    temp1=temp1+s;
    	    temp2=temp2+t;
    	    if(temp1>temp2)
    	    {
    	        if((temp1-temp2)>c)
    	        c=temp1-temp2;
    	        value=1;
    	    }
    	    else 
    	    {
    	        if((temp2-temp1)>ca)
    	        ca=temp2-temp1;
    	        if(c<ca)
    	        value=2;
    	    }
    	    
    	}
    	    if(value==1)
    	    printf("%d %d",value,c);
    	    else
    	    printf("%d %d ",value,ca);
    	    return 0;
    }
