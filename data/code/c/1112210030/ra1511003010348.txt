# include <stdio.h>
 
int main ()
{
    
    
    int t,i,cum1=0,cum2=0,lead1=0,lead2=0,one,two;
    scanf("%d",&t);
    for(i=1;i<=t;i++)
    {
       scanf("%d %d",&one,&two);
       cum1+=one;
       cum2+=two;
       if(cum1>cum2)
       {
          if(lead1<(cum1-cum2))
          lead1=(cum1-cum2);
       
       }
       else
       {
       if(lead2<(cum2-cum1))
          lead2=(cum2-cum1); 
       }
    }
    if(lead1>lead2)
    printf("1 %d",lead1);
    else
    printf("2 %d",lead2);
    
    return 0;
}
