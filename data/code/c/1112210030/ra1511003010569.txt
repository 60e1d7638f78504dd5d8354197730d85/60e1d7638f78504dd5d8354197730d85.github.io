#include<stdio.h>
#include<stdlib.h>
int main()
{
        int i,h,c=0,d=0,a,b,e=0,f=0;
        scanf("%d\n",&h);
        for(i=1;i<=h;i++)
        {
           scanf("%d",&a);
           scanf("%d",&b);
           c=c+a;
           d=d+b;
           if (c-d>0)
           {
 
               if(c-d>e)
               {
                   e=c-d;
               }
           }
           else
           {
 
               if(d-c>f)
               {
                   f=d-c;
               }
           }
        }
 
 
if(e>f)
{
    printf("1");
    printf(" %d\n",e);
}
else
{
    printf("2");
    printf(" %d\n",f);
}
  return 0;
}
