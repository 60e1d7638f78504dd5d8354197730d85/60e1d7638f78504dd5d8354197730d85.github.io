#include<stdio.h>
int main()
{
	int a[100], i, n, even, odd;
	scanf("%d", &n);
	for (i = 0; i<n; i++)
		scanf("%d", &a[i]);
	even = 0;
	odd = 0;
	for (i = 0; i < n; i++)
	{
		if (a[i] % 2 == 0)
			even++;
		else if (a[i] % 2 != 0)
			odd++;
	}
	if (even>odd)
	{
		printf("READY FOR BATTLE");
 
	}
	else
	{
		printf("NOT READY");
	}
 
return 0;
}
