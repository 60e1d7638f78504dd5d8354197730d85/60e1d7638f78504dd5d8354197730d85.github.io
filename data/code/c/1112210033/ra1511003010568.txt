#include <stdio.h>
int main()
{
  int L,N;
  scanf("%d %d",&L,&N);
  while(N--)
  {
    int W, H;
    scanf("%d %d",&W,&H);
    if(W>=L && H>=L)
    {
      if(W==H)
        printf("ACCEPTED\n");
      else
        printf("CROP IT\n");
    }
    else
      printf("UPLOAD ANOTHER\n");
  }
	return 0;
}
