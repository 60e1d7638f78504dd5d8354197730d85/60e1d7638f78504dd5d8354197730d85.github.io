#include <stdio.h>
int main()
{
	long int t;
	scanf("%ld",&t);
	while(t--)
	{
	    int n,m,j,i,flag=1;
	    scanf("%d",&n);
	    scanf("%d",&m);
	    char arr[n][m+1];
	    for(i=0;i<n;i++)
	    {
	         scanf("%s",arr[i]);
	    }
	    for(i=0;i<n;i++)
	    {
	         for(j=0;j<m;j++)
	         {
	              if(arr[i][j]=='W')
	              {
	                    if(i>0)
	                    {
	                         if(arr[i-1][j]=='B')
	                         {
	                              flag=0;
	                         }
	                    }
	                    if(j==0||j==m-1)
	                    {
	                         flag=0;     
	                    }
	                    else
	                    {
	                         if(arr[i][j-1]=='A')
	                         flag=0;
	                         else if(arr[i][j+1]=='A')
	                         flag=0;
	                    }
	              }
	              else if(arr[i][j]=='A')
	              {
	                    if(i>0)
	                    {
	                         if(arr[i-1][j]=='B'||arr[i-1][j]=='W')
	                         flag=0;
	                    }
	              }
	         }
	    }
	    if(flag==1)
	    printf("yes\n");
	    else
	    printf("no\n");
    }
	return 0;
}
