#include <stdio.h>
int main()
{
  int t;
	long long int i,j,x,y,l,d,s,c;
	scanf("%d",&t);
	for(i=0;i<t;i++){
	    
	    scanf("%lld",&l);
	    scanf("%lld",&d);
	     scanf("%lld",&s);
	     scanf("%lld",&c);
	     y=0;
	     for(j=0;j<d;j++){
	         if(j==0)
	         x=s;
	         else
	         x=x+x*c;
	         if(x>=l){
	              y=1;
	         break;
	        
	         }
	         
	     }
	    if(y==0)
	    printf("DEAD AND ROTTING\n");
	    else
	    printf("ALIVE AND KICKING\n");
    }

	return 0;
}
