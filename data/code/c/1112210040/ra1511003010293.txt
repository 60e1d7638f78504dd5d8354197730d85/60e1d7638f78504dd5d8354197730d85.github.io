#include <stdio.h>
 
inline long long read_int()
{
	long long ret = 0;
	int c = getchar_unlocked();
	
	while(c<'0' || c>'9')
		c = getchar_unlocked();
	while(c>='0' && c<='9')
	{
		ret = (ret<<3) + (ret<<1) + c - '0';
		c = getchar_unlocked();
	}
        return ret;
}
 
inline void write_int (int n)
{
	int n2=n;
	int i=20;
	char output_buffer[20];
	do
	{
		output_buffer[--i]=(n2%10)+'0';
		n2/=10;
	}while(n2);
	do
	{
		putchar_unlocked(output_buffer[i]);
	}while(++i<20);
}
 
int main()
{
	long long N,M,q,count=0;
	N = read_int();
	M = read_int();
// 	scanf("%d%d",&N,&M);
	while(M--)
	{
		q = read_int();
// 		scanf("%d",&q);
		if(q < (N+1))
			count = 0;
		else if(q > 2*N)
			count = 3*N-q+1;
		else
			count = q-N-1;
		write_int(count);
		putchar_unlocked('\n');
// 		printf("%d\n",count);
	}
	return 0;
}
