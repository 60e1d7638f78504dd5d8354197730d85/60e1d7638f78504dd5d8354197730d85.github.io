#include <stdio.h>
 
int main()
{
    long long int n,m,q;
    scanf("%lld %lld",&n,&m);
    while(m--)
    {
        scanf("%lld",&q);
        if(q-n-1>0)
        {
            if(q<=2*n+1)
            {
                printf("%lld\n",q-n-1);
            }
            else
            {
                printf("%lld\n",3*n-q+1);
            }
        }
        else
            printf("%d\n",0);
    }
    return 0;
}
