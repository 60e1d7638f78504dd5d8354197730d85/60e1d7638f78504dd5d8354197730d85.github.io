#include<stdio.h>
#define gc getchar_unlocked
#define pc putchar_unlocked
inline long long int inp()
{
    long long int n = 0, c = gc(), f = 1;
    while(c != '-' && (c < '0' || c > '9')) c = gc();
        if(c == '-')
        {
            f = -1;
            c = gc();
        }
    while(c >= '0' && c <= '9')
        n = (n<<3) + (n<<1) + c - '0', c = gc();
    return n * f;
}
 
int main()
{
long long int n,m,i;
n = inp(); m=inp();
for(i=0;i<m;i++)
{
    long long int q,ans=0;
    q = inp();
    if(q<(2*n+1) && (q>=n+2))
    {
        ans = (q-n-1);
        printf("%lld\n",ans);
    }
    else if(q>=(2*n+1))
      {
         ans = (3*n +1 -q);
         printf("%lld\n",ans);
      }
        else
            printf("0\n");
}
  return 0;
}
