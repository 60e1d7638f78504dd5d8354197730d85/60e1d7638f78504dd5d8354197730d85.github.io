#include <stdio.h>
long long int mulmod(long long int a,long long int b,long long int c)
{
    long long int x = 0,y=a%c;
    while(b > 0)
	{
        if(b%2 == 1)
		{
            x = (x+y)%c;
        }
        y = (y*2)%c;
        b /= 2;
    }
    return x%c;
}
long long int power_modulo(int a,long long int b,long long int c)
{
    long long int x=1,y=a; 
    while(b > 0)
	{
        if(b%2 == 1)
		{
            x=(x*y)%c;
        }
        y = (y*y)%c; 
        b /= 2;
    }
    return x%c;
}
