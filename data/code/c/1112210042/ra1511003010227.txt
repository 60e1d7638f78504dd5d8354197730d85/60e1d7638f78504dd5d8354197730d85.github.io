#include <stdio.h>
#include <string.h>
#include <stdlib.h>
 
#define MAX1 100000007
 
int epow(int a, int b)
{
	if (b==0) return 1;
	if (b%2==0)
	{
		long long int t = epow(a,b/2);
		return (t*t)%MAX1;
		
	}
	return (epow(a,b-1)*a)%MAX1;
}
 
int pow1(int a, int b)
{
	int i;
	int rem=1;
	for (i=1; i<=b; i++)
	{
		rem=(rem*a)%MAX1;
	}
	return rem;
}
 
 
 
int arr_to_int(int a[], int low, int high)
{
	int i=low;
	int num=0;
	for (i=low; i<=high; i++)
	{
		num=(num*10)+a[i];
	}
	return num;
}
