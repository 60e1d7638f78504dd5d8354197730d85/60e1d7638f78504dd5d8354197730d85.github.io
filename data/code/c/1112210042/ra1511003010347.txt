#include <stdlib.h>
#define NUM 200000014
#define NUM1 100000007
long long int find_pow(int n,long long int r)
{
    long long int t1,t2,t3;
    if(r == 0)
    {
        return 1;
    }
    else if(r == 1)
    {
        return n;
    }
    else if(r%2 == 0)
    {
        t1 = find_pow(n,r/2);
        t2 = t1%NUM;
        t3 = (t2*t2)%NUM;
        return t3;
    }
    else
    {
        t1 = find_pow(n,r/2);
        t2 = t1%NUM;
        t3 = (t2*t2*n)%NUM;
        return t3;
    }
}
 
void find_soln(long long int n)
{
    long long int p_2,p_3,p_4,p_21;

    p_4 = find_pow(4,n);
    t1 = (p_3+1)/2 - p_2;
    t1 = t1%NUM;
    if(t1 < 0)
    {
     t1 += NUM;
    }
    sol1 = t1;
    t1 = p_4/2 + p_2/2 - p_3 - sol1;
    t1 = t1%NUM;
    if(t1 < 0)
    {
     t1 += NUM;
    }
    sol2 = t1;
    sol1 = sol1%NUM1;

    printf("%lld %lld\n",sol1,sol2);

}
int main()
