#include <stdio.h>
#include<math.h>
void isPrime(int p)
{
	int i,f=1;
	for(i=2;i<=sqrt(p);i++)
	{
		if(p%i==0)
		{
			f=0;break;
		}
	}
	if(f && p>1)
	 printf("%d\n",p);
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int m,n,i;
		scanf("%d %d",&m,&n);
		for(i=m;i<=n;i++)
		 isPrime(i);
      	printf("\n\n");
	}
	
	
	return 0;
}
