#include<stdio.h>
 
int main()
{
  int T,k,n,r;
  scanf("%d",&T);
  while(T--)
  {
    scanf("%d %d", &n, &k);
    int ans = 1;
    n = n-1;
    k = k-1;
    if(k > n/2)
    {
      r = n-k;
    }
    else
    {
      r = k;
    }
    int i;
    for (i = 0; i < r; i++) {
      ans *= (n-i);
      ans /= (i+1);
    }
    printf("%d\n",ans);
  }
  return 0;
}
