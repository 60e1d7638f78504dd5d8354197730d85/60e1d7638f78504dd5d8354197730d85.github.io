#include <stdio.h>
int main(void) 
{
	int i,t,n,k;
	long long int a[1003],sum=0;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&n,&k);
		for(i=0;i<n;i++)
		{
			scanf("%lld",&a[i]);
			sum+=a[i];
		}
		if(sum%2==0)
		{
			if(k==1)
			printf("odd\n");
			else
			printf("even\n");
		}
		else
		{
			printf("even\n");	
		}
	}
	return 0;
}
