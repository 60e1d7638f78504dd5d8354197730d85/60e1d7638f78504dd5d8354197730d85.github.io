#include<stdio.h>
int main(){
	int T,N,K,i,j,sum;
	int a[1000];
	scanf("%d",&T);
	for(i=0;i<T;i++){
		sum=0;
		scanf("%d %d",&N,&K);
		for(j=0;j<N;j++){
			scanf("%d",&a[j]);
			sum=sum+a[j];
		}
		//printf("%d",sum);
		if(sum%2!=0)	//initial sum is odd
			printf("even\n");
		if(sum%2==0&&K==1)	//initial sum is even
			printf("odd\n");
		if(sum%2==0&&K!=1)
		printf("even\n");
			
	}
	//printf("%d",0%2);
  return 0;
}
