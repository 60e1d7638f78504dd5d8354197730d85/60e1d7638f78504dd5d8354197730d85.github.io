#include<stdio.h>
int main(){
	int T,N,K,i,j,sum;
	int a[1000];
	scanf("%d",&T);
	for(i=0;i<T;i++){
		sum=0;
		scanf("%d %d",&N,&K);
		for(j=0;j<N;j++){
			scanf("%d",&a[j]);
			sum=sum+a[j];
		}
		
		if(sum%2!=0)	
			printf("even");
		if(sum%2==0&&K==1)	
			printf("odd");
		if(sum%2==0&&K!=1)
		printf("even");
			
	}
	return 0;
}
