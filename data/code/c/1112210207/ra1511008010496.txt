#include <stdio.h>
#include<math.h>
int main()
{
	int t,n,i;
	long int k,a[1001],sum,sum1,temp;
	scanf("%d",&t);
	while(t--)
	{
		sum=0;
		sum1=0;
		scanf("%d %ld",&n,&k);
		for(i=0;i<n;i++)
		scanf("%ld",&a[i]);
		for(i=0;i<n;i++)
		sum=sum+(a[i]);
		sum+=1;
		temp=pow(2,k-1);
		sum1=temp*sum;
		//printf("%ld",sum1);
		if(sum1%2==0)
		printf("even\n");
		else
		printf("odd\n");
	}
	return(0);
}
