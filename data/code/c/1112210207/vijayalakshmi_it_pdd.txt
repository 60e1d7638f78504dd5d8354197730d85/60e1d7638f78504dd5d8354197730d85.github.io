    #include <stdio.h>
     
    int main(void) {
    	int t,n;
    	unsigned long long int k,i,j,j1,sum,num;
    	scanf("%d",&t);
    	while(t--){
    		sum=j=j1=0;
    		scanf("%d %Ld",&n,&k);
    		for(i=0;i<n;i++){
    			scanf("%Ld",&num);
    			sum+=num;
    		}
    		if(sum%2==0&&k==1)  printf("odd\n");
    		else if(sum%2==0 && k>1) printf("even\n");
    		else if(sum%2!=0) printf("even\n");
    	}
    	return 0;
    }
