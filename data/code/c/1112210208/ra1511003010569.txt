void sort(int n,int l,int h)
{
    if(l!=h){
    //int temp[n];
    int mid=l+(h-l)/2;
    sort(n,l,mid);
    sort(n,mid+1,h);
    merge(n,l,mid,mid+1,h);
    copy(l,h);
    }
}
void merge(int n,int l1,int h1,int l2,int h2)
{
    int i,j,k;
    for(i=l1,j=l2,k=l1;((i<=h1)&&(j<=h2));k++)
    {
        if(d[i]<d[j])

        j++;}
        else
        {
            temp[k]=d[i];
            i++;
        }
    }
    for(;i<=h1;i++)
    {
        temp[k]=d[i];
        k++;
    }
    for(;j<=h2;j++)
    {
        temp[k]=d[j];
        k++;
    }
}
void copy(int l,int h)
{
    int i;
    for(i=l;i<=h;i++)

        d[i]=temp[i];
    }
}
