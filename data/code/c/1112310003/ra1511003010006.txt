# include <stdio.h>
int check(int n,int m,int a[])
{
	if(m==0)
	return 1;
	else if(n==0 &&m!=0)
	return 0;
	else if(a[n-1]>m)
	return check(n-1,m,a);
	else 
	return (check(n-1,m,a)||check(n-1,m-a[n-1],a));
}
int main()
{
	int t,n,i,m;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&n,&m);
		int a[n];
		for( i=0;i<n;i++)
		scanf("%d",&a[i]);
		if(check(n,m,a)==0)
		printf("No\n");
		else
		printf("Yes\n");
	}
	return 0;
}
