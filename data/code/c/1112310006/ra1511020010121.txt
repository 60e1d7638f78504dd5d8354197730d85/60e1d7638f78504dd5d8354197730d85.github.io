#include<stdio.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long long int n,m,j;
        scanf("%lld%lld",&n,&m);
        long double l=2;
        while(l<=m)
        {
            l*=2;
        }
        long double s=1.0;
        j=m-l/2;
        s=s*(2*j+1);
        s=s/l *1.0;
        s=s*1.0*n;
        printf("%Lf\n",s);
    }
    return 0;
}
