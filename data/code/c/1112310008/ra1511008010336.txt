#include <stdio.h>
 
#define gc getchar_unlocked
inline int getn(){
	int n = 0, c = gc(), f = 1;
	while(c != '-' && (c < '0' || c > '9')) c = gc();
	if(c == '-') f = -1, c = gc();
	while(c >= '0' && c <= '9')
		n = (n<<3) + (n<<1) + c - '0', c = gc();
	return n * f;
}
 
int main(){
	char s[101][101];
	int T = getn(),N, i,j, x[500],y[500], a,b,c,d,n,r;
	while(T--){
		for(i = 0; i <= 100; i++)
			for(j = 0; j <= 100; j++)
				s[i][j] = 0;
		N = getn(), r = 0;
      if(N==0)
      {
        printf("3");
        return 0;
      }
		for(i = 0; i < N; i++){
			x[i] = getn()+50, y[i] = getn()+50;
			s[x[i]][y[i]] = 1;
		}
		for(i = 0; i < N-1; i++){
			a = x[i], b = y[i];
			for(j = i+1; j < N; j++){
				c = x[j], d = y[j];
				if(b == d){
					n = b + (a>c ? a-c : c-a);
					if(n < 101 && s[a][n] && s[c][n])
						r++;
				}else if(a<c && b>d && a+b-d<101 && b+c-a<101 && s[a+b-d][b+c-a]
						&& c+b-d<101 && d+c-a<101 && s[c+b-d][d+c-a])
					r++;
				else if(a>c && b<d && d+a-c<101 && c+d-b<101 && s[c+d-b][d+a-c]
						&& a+d-b<101 && b+a-c<101 && s[a+d-b][b+a-c])
					r++;
			}
		}
		printf("%d\n",r);
	}
	return 0;
}
