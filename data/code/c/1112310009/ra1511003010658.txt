#include <stdio.h>
#include <math.h>
 
int prime(int n) {
if(n == 1)
return 0;
int a = sqrt(n);
int i;
 
for(i=2; i<=a; i++) {
 if(n%i == 0)
  return 0;
}
 
return 1;
}
 
int main() {
int t;
scanf("%d",&t);
 
while(t--) {
 int m, n, i;
 scanf("%d%d",&m,&n);
 
 if(m%2 == 0 && m != 2)
  m++;
 
 if(m == 1 && n > 2)
  printf("2\n");
 
 if(m == 2) {
  printf("2");
  m++;
 }
 
 for(i=m; i<=n; i+=2) {
  if(prime(i))
   printf("%d\n",i);
 }
 
// printf("\n");
}
 
return 0;
}
