#include<stdio.h>
void isprime(int p)
{ int count=0;
 
     int i;
    for(i=2;i*i<=p;i++)
         {
             if(p%i==0){count++;break;}
         }
    if(count==0 && p>1)printf("%d\n",p);
 
    return ;
}
 
int main()
{
 
int t;
scanf("%d",&t);
while(t--)
 {
    int  n,m;
   scanf("%d%d",&m,&n);
   int i;
   for(i=m;i<=n;i++)
   isprime(i);
 
 }
}
