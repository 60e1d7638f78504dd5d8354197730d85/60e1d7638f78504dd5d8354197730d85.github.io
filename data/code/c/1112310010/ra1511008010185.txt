#include <stdio.h>
 
#define ll long long
#define gc getchar_unlocked
 
int getn(){
  int n = 0, c = gc(), f = 1;
  while(c != '-' && (c < '0' || c > '9')) c = gc();
  if(c == '-') f = -1, c = gc();
  while(c >= '0' && c <= '9') n = (n<<3) + (n<<1) + c - '0', c = gc();
  return n * f;
}
 
ll nck(int n, int k){
  int d;
  ll r = 1;
  if(k > n/2) k = n-k;
  for(d = 1; d <= k; ++d, --n) r *= n, r /= d;
  return r;
}
 
int main(){
  int T,N,K;
 
  T = getn();
  while(T--){
    N = getn(), K = getn();
 
    printf("%lld\n", nck(N-1, K-1));
  }
  return 0;
}
