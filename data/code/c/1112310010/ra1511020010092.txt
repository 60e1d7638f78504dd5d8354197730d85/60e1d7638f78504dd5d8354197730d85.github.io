#include<stdio.h>
long long int co(long long int a,long long int b)
{ 
  long long int n=1ll,i;
  if(b>a/2)b=a-b;
  for(i=0;i<b;i++)
  {
    n*=a-i;
    n/=i+1;
  } 
  return n;
}
 
int main()
{
  long long int t,a,b;
  scanf("%lld\n",&t);
  while(t--)
  {
    scanf("%lld %lld\n",&a,&b);
    
    if(a==b)printf("1\n");
    else
  {
      
      printf("%lld\n",co(a-1,b-1));
    }
  }
  return 0;
}
