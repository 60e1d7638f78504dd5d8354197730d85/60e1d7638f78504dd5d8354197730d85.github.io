#include<stdio.h>
#include<stdlib.h>
#include<math.h>
#include<string.h>
#define REP(i,a,b) for(i=a;i<b;i++)
#define rep(i,n) REP(i,0,n)
 
int n;
int id[100], sum[100];
int res[1001];
int state[100];
 
int main(){
  int i,sumsum;
  int T;
 
  scanf("%d",&T);
  while(T--){
    scanf("%d",&n);
    rep(i,n) scanf("%d%d",id+i,sum+i);
    rep(i,1001) res[i] = 0;
 
    rep(i,n) state[i] = 0;
    rep(i,n) if(sum[i]==0) state[i] = 1;
 
    sumsum = 0;
    rep(i,n) sumsum -= sum[i];
    rep(i,n) sumsum += id[i];
 
    printf("%d\n",sumsum);
  }
 
  return 0;
}
