#include <stdio.h>
#include <stdlib.h>
#include <string.h>
 
int inset(int *a,int *mem,int n,int i){
	int *b;
	int k,mask=1,cnt=0;
	mask<<=i;
	b = (int*)malloc(sizeof(int)*n);
	for(k=0;k<n;k++){
		if((mask&a[k]) && mem[k]){
			cnt++;
			b[k]=1;
		}
		else{
			b[k]=0;
		}
	}
	if(cnt>1){
		memcpy(mem,b,n*sizeof(int));
	}
	free(b);
	return cnt;
 
}
 
int main(){
	int *a;
	int *mem;
	int i,j;
	int n;
	scanf("%d",&n);
	a = (int*)malloc(sizeof(int)*n);
	for(i=0;i<n;i++)
		scanf("%d",a+i);
	mem = (int*)malloc(sizeof(int)*n);
	for(i=0;i<n;i++){
		mem[i]=1;
	}
	for(i=sizeof(int)*8-1;i>=0;i--){
		if(inset(a,mem,n,i)==2)
			break;
	}
	for(i=0;!mem[i];i++);
	for(j=i+1;!mem[j];j++);
	printf("%d\n",a[i]&a[j]);
	free(a);
	free(mem);
	return 0;
}
