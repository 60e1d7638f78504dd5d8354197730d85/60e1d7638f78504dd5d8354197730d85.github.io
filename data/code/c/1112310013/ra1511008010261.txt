#include <stdio.h>

int fastread()
{
int i=0;
char c=getchar_unlocked();
while(c<48) c=getchar_unlocked();
while(c>47)
{
i=i*10+c-48;
c=getchar_unlocked();
}
return i;
}

int main(void) {
		int n=fastread();
		long int a[n],i,j;
		for( i=0;i<n;i++){
			a[i]=fastread();
		}
		int max=0;
		for(i=0;i<n-1;i++){
			if(a[i]>=max)
			for( j=i+1;j<n;j++){
		int c=a[i]&a[j];
		if(c>max )
		
			max=a[i]&a[j];
		}
		}
	printf("%d",max);
	return 0;
}
