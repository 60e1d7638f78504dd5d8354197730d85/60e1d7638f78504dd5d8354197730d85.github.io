#include <stdio.h>
 
unsigned long long int power(int x)
{
	if (x == 0) {
		return 1;
	}
	else if (x == 1) {
		return 2;
	} else {
		unsigned long long int n = power(x / 2);
		n = (n * n) % 1000000007;
		if (x & 1) {
			n <<= 1;
			n %= 1000000007;
		}
 
		return n;
	}
}
 
int main()
{
	int t, n;
	register int i;
	unsigned long long int x, s;
 
	scanf("%d", &t);
 
	while (t--) {
		scanf("%d", &n);
 
		x = power((n + 1) / 2);
		s = x;
		s--;
		s <<= 1;
		if (s > 1000000007) {
			s -= 1000000007;
		}
 
		if (n % 2 == 0) {
			s += x;
			if (s > 1000000007) {
				s -= 1000000007;
			}
		}
 
		printf("%lld\n", s);
	}
 
	return 0;
}
