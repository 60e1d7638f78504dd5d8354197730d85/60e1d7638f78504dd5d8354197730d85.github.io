#include<stdio.h>
 
 
int main()
{
	int n,m,t,min[31]={0,2,5,8,12,16,20,24,29,34,39,44,49,54,59,64,70,76,82,88,94,100,106,112,118,124,130,136,142,148,154};
	int max[31]={0,2,5,9,14,20,27,35,44,54,65,77,90,104,119,135,152,170,189,209,230,252,275,299,324,350,377,405,434,464,495};
 
	scanf("%d",&t);
 
	while(t--)
	{
		scanf("%d %d",&n,&m);
		if(min[n]>m)
			printf("-1\n");
		else if(m>max[n])
			printf("%d\n",m-max[n]);
		else
			printf("0\n");
	}
 
	return 0;
}
