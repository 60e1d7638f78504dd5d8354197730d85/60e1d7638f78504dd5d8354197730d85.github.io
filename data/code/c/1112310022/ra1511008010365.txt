#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>   
#include<stdbool.h> 
 
int aa(unsigned int a[][100],long long int n,long long int x,long long int y)
{int i;
 long long int p;
 for(i=0;i<=n;i++)
 {p=a[x][i]&a[y][i];
  
     if(p)
      return 1;
 }    
 return 0;
}    
int main() 
{long long int i,j,k,n,f,o;
 char ch[2001][2001];
 static unsigned int a[2001][100];
 long long int ans=0;
 scanf("%lld",&n);
 for(i=0;i<n;i++)
 {scanf("%s",ch[i]);     
 }  
 
 for(i=0;i<n;i++)
 {f=0;
  k=0;
  for(j=0;j<n;j++)
  {
   if(ch[j][i]=='1')
     a[i][f]=a[i][f]+pow(2,k);
      k++;
   if(k==32)
   {f++;
    k=0;
   }
  }
  o=f;
 }
 for(i=0;i<n;i++)
 {for(j=i+1;j<n;j++)
   {if(ch[i][j]=='1')
       continue;
    if(aa(a,o,i,j))
    {ans=ans+2;
    }    
   }   
 } 
 printf("%lld\n",ans);   
    return 0;
}
