#include <stdio.h>
#define MAX_SIE 1000
int main()
{
    int k,t,n,i,count;
    int a[MAX_SIE];
    scanf("%d",&t);
    for(k=0;k<t;k++)
    {
        scanf("%d",&n);
        for(i=0;i<n;i++)
        {
            scanf("%d",&a[i]);
        }
        count=0;
        for(i=0;i<n;i++)
        {
            if(a[i]%2==0)
                count=count+1;
        }
        printf("%d\n",count);
    }
  return 0;
}
