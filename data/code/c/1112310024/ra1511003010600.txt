#include<stdio.h>
int main()
{


    int t,n,s;
    scanf("%d",&t);
    while(t--)
    {   s=0;

        scanf("%d",&n);
        int a[100000],x;
        for(x=0;x<n;x++)
        {

            scanf("%d",&a[x]);
            if(a[x]%2==0)
                s++;
        }
        printf("%d\n",s);
    }

return 0;
}
