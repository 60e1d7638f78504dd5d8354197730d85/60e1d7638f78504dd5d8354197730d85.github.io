#include <stdio.h>
 
int getlb (int n) {
        if(n==2) return 5;
        if(n==1) return 2;
        if(n==0) return 0;
        return n+1+getlb(n/2)+getlb((n-1)/2);
}
 
 
int main() {
 
        int t,n,m;
 
        scanf("%d", &t);
        while (t--) {
                scanf("%d %d", &n, &m);
 
                int ub = ((n+1)*(n+2))/2-1;
                int lb = getlb(n);
 
 
                if (m>=ub) {
                        printf("%d\n", m-ub);
                }
 
                else if (m>=lb) {
                        printf("0\n");
                }
 
                else {
                        printf("-1\n");
                }
        }
 
        return 0;
}
