int cache[11]={0,9,81,648,4536,27216,136080,544320,1632960,3265920,3265920};
 
int match(char *t,int len)
{
  int i,j,k,p,ans=0;
  int map[10]={0};
  char str[11]={"1000000000"};
  
  for(i=0;i<len;i++)
  {    
     if(i&&(map[t[i-1]-'0']>1))
     break;
     
     if(str[i]<t[i])
     {
       p=t[i]-str[i];
       
       for(j=t[i]-'0'-1;j>=0;j--)
        if(map[j])
         p--;
       
       map[t[i]-'0']++;         
       for(j=i+1,k=9-i;j<len;j++,k--)
       p*=k;   
       ans+=p;            
     }
          
     else if(str[i]==t[i])
     map[t[i]-'0']++;
     
     else
     break;                
  }   
  
  if((i==len)&&(map[t[i-1]-'0']==1))
  ans++;
  //printf("yo : %d\n",ans);
  return ans;
}
 
int isUnique(char *a,int len)
{
 int i,flag=0,map[10]={0};
 for(i=0;i<len;i++)
 {
  if(map[a[i]-'0'])
  {
    flag=1;
    break;                
  }
  map[a[i]-'0']=1;
 }  
 return flag?0:1;  
}
