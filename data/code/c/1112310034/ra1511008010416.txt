inline void fastRead(char *a, int *d) {
	register char c = 0;
	while(c < 33)
		c = getchar_unlocked();
	int n = 0;
	while(c > 33) {
		if(n <= 10) {
		    *a = c;
		    ++a;
		    n++;
		}
    	c = getchar_unlocked();
	}
	*d = n;
}
