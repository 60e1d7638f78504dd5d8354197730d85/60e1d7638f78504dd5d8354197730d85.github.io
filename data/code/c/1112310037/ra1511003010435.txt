#include<stdio.h>
#include<math.h>
 
int main()
{
    long int p;
    int t,i,count;
    scanf("%d",&t);
    while(t--)
    {
              scanf("%ld",&p);
              i=0;
              count=0;
              while(p!=0)
              {
                         if((p)&(1<<i))
                         {
                                        if(i<=11)
                                        count++;
                                        else
                                        {
                                            count+=pow(2,i-11);
                                        }
                                        p=(p)-(1<<i);
                         }
                         i++;
              }
              printf("%d\n",count);
    }
    return(0);
}
