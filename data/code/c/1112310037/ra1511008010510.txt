#include<stdio.h>
#include<math.h>
 
   int find_ones(long n)
   {
      int count=0;
         while(n)
         {
         n=n&(n-1);
           count++;
         }
      return count;
   } 
   
int main()
{
       int t;
      long n;
        scanf("%d",&t);
          while(t--)
          {
          scanf("%ld",&n);
            if(n<=2048)
            printf("%d\n",find_ones(n));
            else
            {
            printf("%ld\n",((n/2048)+find_ones(n%2048)));
            }
            
          }
          
       return 0;
}
