#include <stdio.h>
#include <math.h>
int main()
{
	int t, i;
	scanf("%d", &t);
	for(i=0;i<t;i++)
	{
		int ls, b, temp;
		float rsa, rsb;
		scanf("%d %d", &b, &ls);
		temp = pow(ls, 2) - pow(b,2);
		rsa=sqrt(temp);
		temp = pow(ls, 2) + pow(b,2);
		rsb=sqrt(temp);
		printf("%f %f\n", rsa, rsb);
	}
	return 0;
}
