#include <stdio.h>
#include <math.h>
int main()
{
	int t;
  	float a,b,min,max;
  	scanf("%d",&t);
  	while(t--)
    {
      scanf("%f",&a);
      scanf("%f",&b);
      if(a>b){
      	min = sqrt(pow(a,2)-pow(b,2));
      }
      else
      {
        min = sqrt(pow(b,2)-pow(a,2));
      }
      max = sqrt(pow(a,2)+pow(b,2));
      printf("%f %f\n",min,max);
    }
	return 0;
}
