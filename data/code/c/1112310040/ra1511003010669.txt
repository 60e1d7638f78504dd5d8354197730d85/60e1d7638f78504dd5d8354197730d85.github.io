#include <stdio.h>
#include <stdlib.h>
#include<math.h>
 
 
int main()
{
    int t,r,c1,c2,s1,s2,h1,h2;
    double h,s,c;
    scanf("%d",&t);
    while(t--)
    {
             scanf("%d",&r);
             scanf("%d%d",&c1,&c2);
             scanf("%d%d",&h1,&h2);
             scanf("%d%d",&s1,&s2);
             h=pow((pow(h1-c1,2)+pow(h2-c2,2)),0.5);
            // printf("%lf\n",h);
             s=pow( (pow(s1-c1,2)+pow(s2-c2,2)),0.5);
             //printf("%lf\n",s);
             c=pow( (pow(h1-s1,2)+pow(h2-s2,2)),0.5);
             //printf("%lf\n",c);
             if(h<=r && s<=r && c<=r)
             printf("yes\n");
             else if(h>r && s>r && c>r)
             printf("no\n");
             else if(h>r && s<=r && c<=r)
             printf("yes\n");
             else if(c>r && s<=r && h<=r)
             printf("yes\n");
             else if(s>r && c<=r && h<=r)
             printf("yes\n");
             else printf("no\n");
             }
             
             
             
             
             
  
  //system("PAUSE");	
  return 0;
}
