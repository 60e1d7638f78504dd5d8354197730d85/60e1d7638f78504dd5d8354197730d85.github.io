#include <stdio.h>
#include<math.h>
 
 
int pos[3][2];
 
float dis(int i,int j)
{   float temp= ((pos[j][0]-pos[i][0])*(pos[j][0]-pos[i][0])+(pos[j][1]-pos[i][1])*(pos[j][1]-pos[i][1]));
 
if(temp>0)
return sqrt(temp);
else
return 0;
}
 
int main()
{
  int i,j,k,r,count,result_count,ntc;
float first_nbr,sec_nbr,other_two_nbr;
scanf("%d",&ntc);
 
while(ntc--){
 
count=0;
result_count=0;
 
scanf("%d",&r);
 
 
for(i=0;i<3;i++){ 
for(j=0;j<2;j++)
scanf("%d",&pos[i][j]);
}
 
 
 
for(k=0;k<3;k++)
{ first_nbr=dis(k,(k+1)%3);
  sec_nbr=dis(k,(k+2)%3);
other_two_nbr=dis((k+1)%3,(k+2)%3);
if(first_nbr<=r&&sec_nbr<=r)
{ count=1;
}
else if(first_nbr>r&&sec_nbr>r)
   count=0;
 
  else if(first_nbr>r)
{  if(sec_nbr<=r&&other_two_nbr<=r)
   count=1;
else
count=0;
}
 
else if(sec_nbr>r)
{  if(first_nbr<=r&&other_two_nbr<=r)
   count=1;
else
count=0;
}
    
if(count==1)
result_count++;
}
 
//printf("here");
if(result_count==3)
printf("\nyes");
else
printf("\nno");
}

	return 0;
}
