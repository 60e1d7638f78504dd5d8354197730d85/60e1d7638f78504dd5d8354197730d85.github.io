#include <stdio.h>
#include <math.h>
#define sc(t) scanf("%d",&t)
#define w(t) while(t--)
 
int main(void)
{
	int t;
	sc(t);
	w(t)
	{
		unsigned long long int a,b;
		scanf("%llu%llu",&a,&b);
		if(a>b)
		{
			printf(">\n");
		}
		else if (a<b)
		{
			printf("<\n");
		}
		else
		{
			printf("=\n");
		}
	}
	return 0;
}
