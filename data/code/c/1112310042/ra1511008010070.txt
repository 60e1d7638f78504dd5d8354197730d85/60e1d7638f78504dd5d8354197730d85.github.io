#include <stdio.h>
char compare(int x , int y)
{
    char cmp;
    if(x == y) cmp = '=';
    else cmp = (x > y) ? '>' : '<' ;
    return cmp;
}
 
int main ()
{
   
   char (*p)(int,int);
   p = compare;
   int T,x,y;
   scanf("%d",&T);
   while(T--)
   {
       scanf("%d%d",&x,&y);
       printf("%c\n",p(x,y));
   }
   
   return 0 ;
}
