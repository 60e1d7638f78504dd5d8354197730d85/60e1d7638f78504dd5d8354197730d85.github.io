#include<stdio.h>
int main()
{
	long long int a,b;
	int t;
	scanf("%d",&t);
	while(t--)
        {
            scanf("%lld%lld",&a,&b);
            if(a>b)
                printf(">");
            else if(a<b)
                printf("<");
            else
                printf("=");
            printf("\n");
		}
 
 
	return 0;
}
