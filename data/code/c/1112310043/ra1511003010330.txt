#include<stdio.h>
#include<string.h>
#include<stdlib.h>
int main()
{
   int t,i,c;
  char s1[100],s2[100];
   scanf("%d",&t);
   while(t--)
   {
   	  
   	  scanf("%s",s1);
   	  scanf("%s",s2);
   	  int l=strlen(s1);c=0;
   	  for(i=0;i<l;i++)
   	  {
   	  	  if(s1[i]==s2[i]||s1[i]=='?'||s2[i]=='?')
   	  	      c++;
   	  }
   	  if(c==l)
   	    printf("Yes\n");
   	  else printf("No\n");  
   } 
     
}
