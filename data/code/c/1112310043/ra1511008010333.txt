#include <stdio.h>
 
int main(void)
{
	int t,i,ctr;
	char a[11],b[11];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",a);
		scanf("%s",b);
		ctr=0;
		for(i=0;a[i]!='\0';i++)
		{
			if(a[i]=='?'||b[i]=='?');
			else if(a[i]!=b[i])
 
			{
				ctr=1;
				break;
			}
 
		}
		if(ctr==1)
		printf("No\n");
		else
		printf("Yes\n");
	}
 
	return 0;
}
