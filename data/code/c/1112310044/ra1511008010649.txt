#include<stdio.h>
 
int main(void) 
{
    int t,g;
    float c1, c2, c3;
    
    scanf(" %d", &t);
    
    while(t--)
    {
        scanf(" %f %f %f", &c1, &c2, &c3);
        if(c1>50&&c2<0.7&&c3>5600){g=10;}
        else if(c1>50&&c2<0.7&&!(c3>5600)){g=9;}
        else if(!(c1>50)&&c2<0.7&&c3>5600){g=8;}
        else if(c1>50&&!(c2<0.7)&&c3>5600){g=7;}
        else if(((c1>50)^(c2<0.7))^(c3>5600)){g=6;}
        else {g=5;}
        
        printf("%d\n",g);
    }
    
    return 0;
}
