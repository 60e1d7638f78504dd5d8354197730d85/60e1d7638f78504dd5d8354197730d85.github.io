#include <stdio.h>
#include <stdlib.h>
void lazyjem(long long int n,long long int b,long long int m,long long int sum);
 
int main()
{
	setvbuf(stdout,NULL,_IONBF,0);
	long long int n,b,m;
	int t,k;
	scanf("%d",&t);
	for(k=0;k<t;k++)
	{
		long long int sum=0;
		scanf(" %lld",&n);
		scanf(" %lld",&b);
		scanf(" %lld",&m);
		lazyjem(n,b,m,sum);
 
	}
	return 0;
}
	void lazyjem(long long int n,long long int b,long long int m,long long int sum)
	{
		long long int x1,x2;
	if(n==0)
	{
		printf("%lld\n",sum-b);
		return;
	}
	sum=sum+b;
		if(n%2==0)
			x1=n/2;
		else
			x1=(n+1)/2;
		x2=n-x1;
		sum=sum+m*x1;
		lazyjem(x2,b,2*m,sum);
 
	}
