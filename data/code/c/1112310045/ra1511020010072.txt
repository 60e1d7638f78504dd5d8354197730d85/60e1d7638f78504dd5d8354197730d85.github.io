#include<stdio.h>
 
int main( void )
{
	int t;
	scanf("%d", &t);
	while(t--)
	{
		long n, b ;
		long long m, time = 0;
		scanf("%ld %ld %lld", &n, &b, &m);
		while(n)
		{
			long doing, rem;
			if( n%2 ) 
			{
				doing = ( n+1 )/2;
				rem = n - doing;
			}
			else
			{
				doing = n/2;
				rem = n - doing;
			}
			time += doing * m;
			time += b;
			m *= 2;
			n = rem;
		}
		time -= b;
		printf("%lld\n", time);
	}
 
return 0;
}
