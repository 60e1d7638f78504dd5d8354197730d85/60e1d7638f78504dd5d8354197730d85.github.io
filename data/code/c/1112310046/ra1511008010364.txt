#include <stdio.h>
 
int main(void) {
	int N,A[105],i,j,temp,k,p=0,n,max=0,B[105],l,f=0;
	scanf("%d",&N);
	for(i=0;i<N;i++){
	    scanf("%d",&A[i]);
	    B[i] = 0;
	}
	if(N <=1){
	    max =-1;
	    goto R;
	}
	for(i=0;i<N-1;i++)
	    for(j=0;j<N-1-i;j++)
	        if(A[j] > A[j+1]){
	            temp = A[j];
	            A[j] = A[j+1];
	            A[j+1] = temp;
	        }
 
	 
	 n = N;
	 for(i=0;i<N-3-f;i++){
	     for(j=i+1;j<N-2-f;j++){
	         k = N-2;
	         while(k != j){
	            if(B[k] == -1){
	                k--;
	            }
	            else if((A[i]+A[N-1-f]) == (A[j]+A[k])){
	                p++;
	                B[k] = -1;
	              // printf("%d %d %d %d\n",A[i],A[N-1-f],A[j],A[k]);
	                k--;
	                break;
	            }
	            else if(A[i]+A[N-1-f] > A[j]+A[k]){
	                break;
	            }
	            else if(A[i]+A[N-1-f] < A[j]+A[k]){
	                k--;
	            }
	         }
	    }
	     //printf("%d ",i);
 
        if(f == N-i-4){
            f =0;
        }
        else{
            f++;
            i--;
        }
        for(l=0;l<n;l++){
            B[l] = 0;
        }
	    if(p>max){
	         max = p;
	     }
	     p = 0;
 
	 }
 
    R: printf("%d\n",(max+1)*2);
	return 0;
}
