#include <stdio.h>
#include <math.h>
int main()
{
	int T;
	long long int A,B,count,temp,i;
	scanf("%d",&T);
	while(T--)
	{
		scanf("%lld%lld",&A,&B);
		count = 0;
		for(i=1;i<61;i++)
		{
			temp = pow(2,i);
			if(temp>=A && temp <=B)
				count++;
			temp *= 3;
			while(temp<=B)
			{
				if(temp>=A)
					count++;
				temp *= 3;	
			}	
		}
		if(A==1)
			count++;
		printf("%lld\n",count);
	}
	return 0;
}
