#include<stdio.h>
int main(){
	int t;
	unsigned long long l,r ,ans ,val ,curr;
	scanf("%d" ,&t);
	while(t--){
		scanf("%llu%llu" ,&l ,&r);
		ans =0;
		val =2;
		while(val<=r){
			curr =val;
			while(curr<=r){
				if(curr>=l&&curr<=r)	ans++;
				curr *=3;
				}
				val *= 2;
		}
		if(1>=l&&1<=r)	ans++;
		printf("%llu\n" ,ans);
	}
	return 0;
}
