#include<stdio.h>
int main() {
    int t,r,c,i,j;
    scanf("%d",&t);
    while(t--) {
        scanf("%d %d",&r,&c);
        char a[r+1][c+1],s[60];
        for(i=0;i<r;i++) {
            scanf("%s",s);
            for(j=0;j<c;j++)
            a[i][j]=s[j];
        }
        int res=0;
        for(i=2;i<r-2;i++) {
            for(j=2;j<c-2;j++) {
                if((a[i][j]=='^') && (a[i-1][j]=='^') && (a[i-2][j]=='^') && (a[i+1][j]=='^') &&
                   (a[i+2][j]=='^') && (a[i][j-1]=='^') && (a[i][j-2]=='^') && (a[i][j+1]=='^') && (a[i][j+2]=='^'))
                    ++res;
            }
 
        }
        printf("%d\n",res);
    }
    return 0;
}
