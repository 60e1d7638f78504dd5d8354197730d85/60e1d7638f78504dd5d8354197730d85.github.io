#include<stdio.h>
#include<stdlib.h>
#include<math.h>
int main()
{
int index[30],t,n,a,out=1,r,i,j,k,l,root,count[30],prime[30],m;
scanf("%d",&t);
while(t--)
{
scanf("%d",&n);
l=0;
r=0;
out=1;
for(j=0;j<30;j++)
{
count[j]=0;
prime[j]=0;
index[j]=-1;
}
for(j=0;j<n;j++)
{
scanf("%d",&a);
root=sqrt(a);
for(k=2;k<=(root+1);k++)
{
if(a%k==0)
{
for(m=0;m<r;m++)
{
if(prime[m]==k)
{
l=index[m];
break;
}
}
if(m==r)
{
prime[r]=k;
index[r]=r;l=r;
r++;
}
while(a%k==0)
{
count[l]=count[l]+1;
a=a/k;
}
}
}
if(a>1)
{
for(m=0;m<r;m++)
{
if(prime[m]==a)
{
l=index[m];
break;
}
}
if(m==r)
{
prime[r]=a;
index[r]=r;l=r;
r++;
}
count[l]++;
}
}
for(j=0;j<r;j++)
{
count[j]=count[j]+1;
out=out*count[j];
}
printf("%d\n",out);
}
return 0;
}
