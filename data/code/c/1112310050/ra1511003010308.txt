#include<stdio.h>
#include<string.h>
char a[1000000],b[1000000];
int main()
{
  long long int len,i,acount=1,j=0,bcount=0,big=0,k;
  char c;
  scanf("%lld",&len);
  scanf("%s%s",a,b);
 
  c=a[0];
  for(i=0;i<len;i++)
  {
    if(b[i]==c)
       {
           k=1;
           i++;
           bcount=1;
           while(a[k]==b[i]&&i<len)
            {
                bcount++;
                k++;
                i++;
            }
 
       }
       if(bcount>big)
       {
           big=bcount;
           j=i-bcount;
           bcount=0;
       }
 
  }
  printf("%lld",j);
  return 0;
}
