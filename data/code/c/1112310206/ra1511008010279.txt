#include<stdio.h>
int main()
{
    int n, i, j, sum = 0;
    scanf("%d", &n);
    int mat[n-1][3];
    for(i=0;i<n-1;i++)
        for(j=0;j<3;j++)
            scanf("%d", &mat[i][j]);
    sum = (n-2)*mat[n-2][2];
    for(i=0;i<n-2;i++)
    {
        for(j=i+1;j<n-1;j++)
        {
            if(mat[i][2]>mat[j][2])
                sum += mat[j][2];
            else
                sum += mat[i][2];
            sum += (n-2)*mat[i][2];
        }
    }
    printf("%f", (float)sum / ((float)((n-1)*(n-2)*3)/2));
    return 0;
}
