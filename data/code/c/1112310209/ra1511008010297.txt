#include <stdio.h>
 
int main(void) {
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int e,i,j,p;
        long long int m,k,n,y=0,l,swap;
		scanf("%lld%lld%d%lld",&n,&k,&e,&m);
		long long int a[n];
		for(i=0;i<n;i++)
		{
		    y=0;
		    if(i==(n-1))
		        e=e-1;
		    for(j=0;j<e;j++)
		    {
		        scanf("%lld",&l);
		        y=y+l;
		    }
		    a[i]=y;
		    //printf("$ %lld\n",a[i]);
	    }
	
		p=n-1;
		for(i=0;i<p;i++)
		{
		    for(j=0;j<p-i-1;j++)
		    {
			    if(a[j]<a[j+1])
			    {
				    swap=a[j];
				    a[j]=a[j+1];
				    a[j+1]=swap;
		    	}
		    }
	
	    }	
	a[k-1]=a[k-1]-a[n-1];
    if((a[k-1]>=0) && (a[k-1]+1<=m)){
        a[k-1]=a[k-1]+1;
        printf("%lld\n",(a[k-1]));
    }
    else if(a[k-1]<0)
        printf("0\n");
    else
        printf("Impossible\n");
	
		
	}
	return 0;
}
