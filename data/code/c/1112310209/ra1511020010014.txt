#include <stdio.h>
void qsort(long long int * d, int l, int r) {
  int ll, rr;
  long long int p,a;
  if (r > l) {
    ll = l;
    rr = r;
    p = d[(rr + ll) >> 1];
    while (ll <= rr) {
      while (ll < r && d[ll] < p)
	++ll;
      while (rr > l && d[rr] > p)
	--rr;
      if (ll <= rr) {
	a = d[ll];
	d[ll] = d[rr];
	d[rr] = a;
	++ll;
	--rr;
      }
    }
    if (l < rr)
      qsort(d, l, rr);
    if (ll < r)
      qsort(d, ll, r);
  }
}

    for(i=0;i<n-1;i++)
        a[i]=0;
     for(i=0;i<n-1;i++)
