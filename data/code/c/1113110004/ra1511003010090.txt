#include<stdio.h>
#define max(a, b) (((a) >= (b)) ? (a) : (b))
int main()
{
	int t, n, no, i; scanf("%d", &t);
	while(t-- && scanf("%d", &n))
    {
		int h[1000001] = {0}, m1 = 1, m2 = 0;
		while(n-- && scanf("%d", &no)) 
          h[no]++, m1 = max(m1, no);
		for(i = 1; i <= m1; i++) 
          m2 = max(m2, h[i]);
		for(i = m1; i >= 1; i--) 
          if(h[i] == m2) 
            printf("%d ", i);
		printf("\n");
}
}
