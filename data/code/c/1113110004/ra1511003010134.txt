#include <stdio.h>

int mode(int a[],int n) {
   int maxValue = 0, maxCount = 0, i, j;

   for (i = 0; i < n; ++i) {
      int count = 0;
      
      for (j = 0; j < n; ++j) {
         if (a[j] == a[i])
         ++count;
      }
      
      if (count > maxCount) {
         maxCount = count;
         maxValue = a[i];
      }
   }

   return maxValue;
}

int main() {
    int n;
    int t;
    scanf("%d",&t);
  while(t--){
    scanf("%d",&n);
  	int i;
    int a[n];
  	for(i=0;i<n;i++){
    	scanf("%d",&a[i]);  
    }
	
   printf("%d \n", mode(a,n));
  }

   return 0;
}
