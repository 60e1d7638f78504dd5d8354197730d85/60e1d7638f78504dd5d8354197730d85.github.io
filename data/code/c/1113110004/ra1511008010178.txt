#include <stdio.h>

int mode(const int a[], int n) {
    int maxValue = 0, maxCount = 0, i, j;
    for (i = 0; i < n; i++) {
        int count = 0;

        for (j = 0; j < n; j++) {
            if (a[i] == a[j]) {
                count++;
            }
            if (count > maxCount) {
                maxCount = count;
                maxValue = a[i];
            }
        }
    }
    return maxValue;
}

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int n, i;
        scanf("%d", &n);
        int a[n];
        for (i = 0; i < n; i++) {
            scanf("%d", &a[i]);
        }
        printf("%d \n", mode(a, n));
    }
    return 0;
}
