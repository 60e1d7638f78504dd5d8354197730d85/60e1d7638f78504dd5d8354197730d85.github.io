#include <stdio.h>
int main()
{
	int t;
  	scanf("%d",&t);
  	while(t--)
    {
      int a,b;
      scanf("%d %d",&a,&b);
      int i;
      long int s=0;
      for(i=a;i<=b;i++){
        int rev=0,rem,temp=i;
        while(temp!=0)
        {
          rem=temp%10;
          rev=rev*10+rem;
          temp/=10;
        }
        if(rev==i)
          s+=i;
      }
      printf("sum is %ld\n",s);
    }
	return 0;
}
