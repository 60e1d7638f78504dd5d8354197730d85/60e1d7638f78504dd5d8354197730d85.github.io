#include<stdio.h>
int main()
{
	int n,i,a,b,num,num1,palin,sum,rem;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
	scanf("%d%d",&a,&b);
	sum=0;
	for(num1=a;num1<=b;num1++)
	 {
	 	num=num1;
	 	palin=0;
	 	rem=0;
		while(num>0)
		{
			rem=num%10;
			palin=palin*10+rem;
			num/=10;
		}	
		if(palin==num1)
		{
			sum+=palin;
		}
	 }
	 printf("sum is %d\n",sum);
	}
		
	
	return 0;
}
