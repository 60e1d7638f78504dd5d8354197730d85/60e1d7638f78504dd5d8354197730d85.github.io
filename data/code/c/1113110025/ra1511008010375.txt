#include <stdio.h>
 
int main()
{
    int t,i,j,l,r,rem,sum;
    scanf("%d",&t);
    for(i=0;i<t;i++)
     { sum=0;
       scanf("%d%d",&l,&r);
       for(j=l;j<=r ;j++)
        {
          int no=j,rev=0;
          while(no)
          {
           rem=no%10;
           rev=rev*10+rem;
           no=no/10;
          }
          if(rev==j)
            sum=sum+j;
        }
       printf("sum is %d\n",sum);
     }
	return 0;
}
