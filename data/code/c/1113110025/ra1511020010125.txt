#include <stdio.h>
int main()
{
int t,i;
  scanf("%d",&t);
  while(t--)
  {
    long long int c=0;
    int l,r,reverse=0,temp;
    scanf("%d %d",&l,&r);
    for(i=l;i<=r;i++)
    {
      temp=i;
      reverse=0;
      while(temp !=0)
      {
        reverse = reverse * 10;
        reverse = reverse + temp%10;
        temp=temp/10;
      }
      if(i==reverse)
        c+=i;
    }
    printf("sum is %lld\n",c);
  }
	return 0;
}
