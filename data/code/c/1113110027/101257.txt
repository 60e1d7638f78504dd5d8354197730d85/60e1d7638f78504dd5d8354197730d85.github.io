#include <stdio.h>
int main(){
 
	int t;
 
	scanf("%d",&t);
 
	while(t--){
 
		float h,c,t;
 
		scanf("%f %f %f",&h,&c,&t);
 
		int x = 0,y = 0,z = 0;
 
		if(h > 50.00)x = 1;
		if(c < 0.70)y = 1;
		if(t > 5600.00)z = 1;
 
		if((x == 0)&&(y == 0)&&(z == 0))printf(" Grade 5\n");
		else{
 
			if((x == 1) && (y == 1) && (z == 1))printf("Grade 10\n");
			else{
 
				if((x == 1) && (y == 1))printf("Grade 9\n");
				if((x == 1) && (z == 1))printf("Grade 7\n");
				if((y == 1) && (z == 1))printf("Grade 8\n");
 
				if((((y == 1) && (z == 1)) || ((y == 1) && (x == 1)) || ((x == 1) && (z == 1))) == 0)printf("Grade 6\n");
			}
		}
	}
 
	return 0;
}
