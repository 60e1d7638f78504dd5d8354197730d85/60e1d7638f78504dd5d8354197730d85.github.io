#include <stdio.h>
int main()
{
  int h,s,t;
  float c;
  scanf("%d",&t);
  while(t--)
  {
    scanf("%d",&h);
    scanf("%f",&c);
    scanf("%d",&s);
    if((h>50)&&(c<0.7)&&(s>5600))
    {
      printf("Grade 10\n");
    }
    else if((h>50)&&(c<0.7))
    {
      printf("Grade 9\n");
    }
    else if((c<0.7)&&(s>5600))
    {
      printf("Grade 8\n");
    }
    else if((h>50)&&(s>5600))
    {
      printf("Grade 7\n");
    }
    else if((h>50)||(c<0.7)||(s>5600))
    {
      printf("Grade 6\n");
    }
    else if((h<50)&&(c>0.7)&&(s<5600))
    {
      printf("Grade 5\n");
    }
  }
	return 0;
}
