#include <stdio.h>
int main()
{
	int i,ts,hard,grade,n;
  	float carb;
  scanf("%d",&n);
  for(i=1;i<=n;i++)
  {
  	scanf("%d %f %d",&hard,&carb,&ts);
    if((hard>=50 && carb<0.7) && ts>=5600)
      grade = 10;
  	else if(hard>=50 && carb<0.7)
      grade =9;
  else if (carb<0.7 && ts>=5600)
    grade =8;
  else if (hard>=50 && ts>=5600)
    grade = 7;
  else if ((hard>50 || ts>5600) || carb<0.7)
    grade = 6;
  else 
    grade = 5;
  printf("Grade %d\n",grade);
  }
	return 0;
}
