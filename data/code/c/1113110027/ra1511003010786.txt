#include <stdio.h>
int main()
{ 
  int n,i,a,c,grade;
  float b;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d%f%d",&a,&b,&c);
    if(a>50 && b<0.7 && c>5600)
    {
      grade=10;
    }
    else if(a>50 && b<0.7)
    {
      grade=9;
    }
    else if(b<0.7 && c>5600)
    {
      grade=8;
    }
    else if(a>50 && c>5600)
    {
      grade=7; }
    else if(a>50 || b<0.7 || c>5600)
    {
      grade=6;
    }
    else
    {
      grade=5;
    }
    printf("Grade %d\n",grade);
  }
    

	return 0;
}
