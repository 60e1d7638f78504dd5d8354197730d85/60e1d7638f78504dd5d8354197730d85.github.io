#include <stdio.h>
int main()
{
  int t,g,i;
  double h,c,s;
  scanf("%d",&t);
  for( i=0;i<t;i++)
  { 
    scanf("%lf",&h);
    scanf("%lf",&c);
    scanf("%lf",&s);
    if(h>50 && c<=0.7 && s>5600)
    {
      g=10;
    }
    else if(h>50 && c<0.7)
    {
      g=9;
    }
    else if(c<0.7 && s>5600)
    {
      g=8;
    }
    else if(h>50 && s>5600)
    {
      g=7;
    }
    else if(h>50 || c<0.7 || s>5600)
    {
      g=6;
    }
    else
    {
      g=5;
    }
    printf("Grade %d\n",g);
    
  }
  

	return 0;
}
