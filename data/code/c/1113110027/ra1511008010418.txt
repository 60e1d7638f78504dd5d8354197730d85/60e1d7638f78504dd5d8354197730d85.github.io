#include <stdio.h>
int main()
{int n;
 int i;
 int h[100];
 float c[100];
 int t[100];
 
 scanf("%d",&n);
 for( i=1;i<=n;i++){
   scanf("%d",&h[i]);
   scanf("%f",&c[i]);
   scanf("%d",&t[i]);
   if(h[i]>50 && c[i]<0.7 && t[i]>5600){
     printf("Grade 10\n");
   }
   else if(h[i]>50 && c[i]<0.7){
     printf("Grade 9\n");
   }
   else if(c[i]<0.7 && t[i]>5600){
     printf("Grade 8\n");
   }
   else if(h[i]>50 && t[i]>5600){
     printf("Grade 7\n");
   }
   else if(h[i]>50||c[i]<0.7||t[i]>5600){
     printf("Grade 6\n");
   }
   else if(h[i]<50||c[i]>0.7||t[i]<5600){
     printf("Grade 5\n");
   }
 }

	return 0;
}
