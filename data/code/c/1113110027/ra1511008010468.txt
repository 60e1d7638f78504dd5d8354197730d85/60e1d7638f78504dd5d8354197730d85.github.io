#include <stdio.h>
int main()
{
long t=0,grade=0,i=0;
  long hs,ts;
  float cc;
  scanf("%lu",&t);
  for(i=0;i<t;i++)
  {
    scanf("%lu %f %lu",&hs, &cc,&ts);
    if(hs>=50&&cc<=0.7&&ts>=5600)
    {
    grade=10;
  }
  else if(hs>=50&&cc<=0.7)
  {
    grade=9;
  }
  else if(cc<=0.7&&ts>=5600)
  {
    grade=8;
  }
  else if (hs>=50&&ts>=5600)
  {
    grade=7;
  }
  else if (hs>=50||cc<=0.7||ts>=5600)
  {
    grade=6;
  }
  else
  {
    grade=5;
  }
  printf("Grade %lu\n",grade);
}
    
	return (0);
}
