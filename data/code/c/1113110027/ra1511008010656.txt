#include <stdio.h>

int grade (int h, float c, int t)
{
  if (h>50 && c<0.7 && t>5600)
  {
    return 10;
  }
  else if (h>50 && c<0.7)
  {
    return 9;
  }
  else if (c<0.7 && t>5600)
  {
    return 8;
  }
  else if (h>50 && t>5600)
  {
    return 7;
  }
  else if (h>50 || c<0.7 || t>5600)
  {
    return 6;
  }
  else 
  {
    return 5;
  }
}
