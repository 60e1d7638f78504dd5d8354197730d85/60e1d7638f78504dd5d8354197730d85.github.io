#include <stdio.h>
int main()
{
  int n;
  int h,t;
  float c;
  int g;
  scanf("%d",&n);
  int i;
  for(i=0;i<n;i++)
  {
    scanf("%d",&h);
    scanf("%f",&c);
    scanf("%d",&t);
    if(h>50 && c<0.7 && t>5600)
      g=10;
    else if(h>50 && c<0.7)
      g=9;
    else if(c<0.7 && t>5600)
      g=8;
    else if(h>50 && t>5600)
      g=7;
    else if(h>50 || c<0.7 || t>5600)
      g=6;
    else
      g=5;
    printf("Grade %d\n",g);
      
  }

	return 0;
}
