#include <stdio.h>
int main()
{

  long long int t,i,n,k,j,max,m;
    long long int a[100007];
    scanf("%lld",&t);
    for(i=0;i<t;i++)
    {
        scanf("%lld%lld",&n,&k);
        m=0;
        if(k==1)
        {
            printf("0\n");
        }
        else if(k>n)
        {
            printf("%lld\n",n);
        }
        else
        {
        for(j=2;j<=k;j++)
        {
            a[m]=n%j;
            m++;
        }
        max=-10;
        for(j=0;j<m;j++)
        {
            if(max<a[j])
            max=a[j];
        }
        printf("%lld\n",max);
        }
    }
	return 0;
}
