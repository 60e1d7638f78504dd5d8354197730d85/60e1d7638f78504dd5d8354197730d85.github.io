#include <stdio.h>
int main()
{
  int h,t,c,grade,i; double car;
  scanf("%d\n",&c);
  for(i=0;i<c;i++)
  {
    scanf("%d %lf %d\n",&h,&car,&t);
    if(h>50 && car<=0.7 && t>5600)
      grade=10;
    if(h>50 && car<=0.7 && t<=5600)
      grade=9;
    if(car<=0.7 && t>5600 && h<=50)
      grade=8;
    if(h>50 && t>5600 && car>0.7)
      grade=7;
    if((h>50 && car>0.7 && t<=5600)||(h<=50 && car<=0.7 && t<=5600)||(h<=50 && car>0.7 && t>5600))
      grade=6;
    if(h<=50 && car>0.7 && t<=5600)
      grade=5;
    printf("Grade %d\n",grade);
  }

	return 0;
}
