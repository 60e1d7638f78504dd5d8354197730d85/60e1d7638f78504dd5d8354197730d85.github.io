#include <stdio.h>
int main()
{
  int t,hard,tensile;
  float carbon;
  scanf("%d",&t);
  while(t--)
  {
    scanf("%d %f %d",&hard,&carbon,&tensile);
    if (hard>50 && carbon<=0.7 && tensile>5600)
      printf("Grade 10\n");
    else if(hard>50 && carbon<0.7)
      printf("Grade 9\n");
    else if(carbon<0.7 && tensile>5600)
      printf("Grade 8\n");
    else if(hard>50 && tensile>5600)
      printf("Grade 7\n");
    else if(hard>50 || carbon<0.7 || tensile>5600)
      printf("Grade 6\n");
    else
      printf("Grade 5\n");
  }
	return 0;
}
