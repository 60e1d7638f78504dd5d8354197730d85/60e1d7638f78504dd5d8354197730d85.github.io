#include <stdio.h>
int main()
{
    int h,ts,t,i;
   float cc;
    scanf("%d",&t);
    while(t--)
    {
      scanf("%d %f %d",&h,&cc,&ts);
      if(h>50&&cc<0.7&&ts>5600)
      {
        printf("Grade 10\n");
      }
      else if(h>50&&cc<0.7)
      {
        printf("Grade 9\n");
      }
       else if(cc<0.7&&ts>5600)
       {
         printf("Grade 8\n");
       }
       else if(h>50&&ts>5600)
       {
         printf("Grade 7\n");
       }
       else if(h>50||cc<0.7||ts>5600)
       {
         printf("Grade 6\n");
       }
       else 
       {
         printf("Grade 5\n");
       }
    }
        
	return 0;
}
