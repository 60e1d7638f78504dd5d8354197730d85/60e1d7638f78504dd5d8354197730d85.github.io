#include <stdio.h>
int main()
{
  int t;
  scanf("%d",&t);
  int i;
  for(i=1;i<=t;i++)
  {
    int a;
    float b;
    int c;
    scanf("%d %f %d",&a,&b,&c);
    if((a>=50)&&(b<=0.7)&&(c>=5600))
      printf("Grade 10\n");
    else if((a>=50)&&(b<=0.7))
      printf("Grade 9\n");
    else if((b<=0.7)&&(c>=5600))
      printf("Grade 8\n");
    else if((a>=50)&&(c>=5600))
      printf("Grade 7\n");
    else if((a>=50)||(b<=0.7)||(c>=5600)){
      printf("Grade 6\n");}
    else{
      printf("Grade 5\n");}
  }
	return 0;
}
