#include <stdio.h>
int main()
{	int i,h,t,n;
	float c;
	scanf("%d",&n);
for(i=0;i<n;i++)
{
 	scanf("%d",&h);
 	scanf("%f",&c);
 	scanf("%d",&t);
 	
 	if((h>50)&&(c<0.7)&&(t>5600))
{
	printf("Grade 10\n");
}

else if((h>50)&&(c<0.7))
{
	printf("Grade 9\n");
}
else if((c<0.7)&&(t>5600))
{
	printf("Grade 8\n");
}
else if((h>50)&&(t>5600))
{
	printf("Grade 7\n");
}
else if((h>50)||(c<0.7)||(t>5600))
{
	printf("Grade 6\n");
}
else 
{
	printf("Grade 5\n");
}
}
	return 0;
}
