#include <stdio.h>
int main()
{
	int n,h[10],t[10],i,a,b,d,g;
  float c[10];
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    a=b=d=0;
    scanf("%d %f %d",&h[i],&c[i],&t[i]);
    if(h[i]>50)
      a=1;
    if(c[i]<=0.7)
      b=1;
    if(t[i]>5600)
      d=1;
    if(a==1 && b==1 && d==1)
      g=10;
    else if(a==1 && b==1)
      g=9;
    else if(b==1 && d==1)
      g=8;
    else if(a==1 && d==1)
      g=7;
    else if(a==1 || b==1 || d==1)
      g=6;
    else
      g=5;
    printf("Grade %d\n",g);
  }
	return 0;
}
