#include <stdio.h>
int main()
{
    long h,t,n;
    double c;
    scanf("%ld",&n);
    while(n--)
    {
        scanf("%ld%lf%ld",&h,&c,&t);
        if(h>=50 & c<=0.7 & t>=5600)
            printf("Grade 10\n");
        else if (h>=50 & c<=0.7)
            printf("Grade 9\n");
        else if(c<=0.7 &t>=5600)
            printf("Grade 8\n");
        else if(t>=5600 & h>=50)
            printf("Grade 7\n");
        else if(t>=5600 || h>=50 || c<=0.7)
            printf ("Grade 6\n");
        else
            printf("Grade 5\n");
    }
	return 0;
}
