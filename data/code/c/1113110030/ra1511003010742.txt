#include<stdio.h>
int main()
{
    float h,c,t;
    int n;
    scanf("%d",&n);
    while(n--)
    {
        scanf("%f %f %f",&h,&c ,&t);
        if(h>50 && c<0.7&& t>5600)
            printf("Grade 10\n");
        else if(h>50 && c<0.7)
             printf("Grade 9\n");
        else if(c<0.7 && t>5600 )
           printf("Grade 8\n");
         else if(h>50 && t>5600)
            printf("Grade 7\n");
        else if (h>50 || c<0.7 || t>5600)
            printf("Grade 6\n");
         else
            printf("Grade 5\n");
 
    }
    return 0;
}
