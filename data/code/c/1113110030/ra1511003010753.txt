#include <stdio.h>
int main()
{int p,q,r,i,t;
	float a,b,c;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%f%f%f",&a,&b,&c);
		p=a>=50.00f;
		q=b<=0.7f;
		r=c>=5600.00f;
		if(p&&q&&r)
		{
			printf("Grade 10\n");
		}
		else if(p&&q)
		{
			printf("Grade 9\n");
		}
		else if(q&&r)
		{
			printf("Grade 8\n");
		}
		else if(p&&r)
		{
			printf("Grade 7\n");
		}
		else if(p||q||r)
		{
			printf("Grade 6\n");
		}
		else 
		{
			printf("Grade 5\n");
		}
    }

	return 0;
}
