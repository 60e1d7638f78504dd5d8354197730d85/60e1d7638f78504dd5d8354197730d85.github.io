#include <stdio.h>
int main()
{
    int t,h,ten;
    float c;
    scanf("%d",&t);
    while(t--)
    {
      scanf("%d %f %d",&h,&c,&ten);
      if(h>=50 && c<=0.7 && ten>=5600)
        printf("Grade 10\n");
      else if(h>=50 && c<=0.7)
        printf("Grade 9\n");
      else if(c<=0.7 && ten>=5600)
        printf("Grade 8\n");
      else if(h>=50 && ten>=5600)
        printf("Grade 7\n");
      else if(h>=50 || c<=0.7 || ten>=5600)
        printf("Grade 6\n");
      else
        printf("Grade 5\n");
    }
	return 0;
}
