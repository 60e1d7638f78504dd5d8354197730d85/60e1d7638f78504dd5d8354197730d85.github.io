#include <stdio.h>
int main()
{
long int n=0,i=0,grade=0;
  long int h,t;
  float c;
 scanf("%li",&n);
  for(i=0;i<n;i++)
  {
    scanf("%li%f%li",&h,&c,&t);
  {
  if(h>50&&c<0.7&&t>5600)
  {
    grade=10;
  }
    else if(h>50&&c<0.7)
    {
    grade=9;
    }
  else if(c<0.7&&t>5600)
  {
    grade=8;
  }
    else if(h>50&&t>5600)
    {
      grade=7;
    }
    else if(h>50||c<0.7||t>5600)
    {
      grade=6;
    }
    else
    {
      grade=5;
  }
  }
   printf("Grade %li\n",grade);
  }
	return 0;
}
