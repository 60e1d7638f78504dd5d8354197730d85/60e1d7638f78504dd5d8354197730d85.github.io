#include<stdio.h>
 
int main()
{
	int T;
	float h,cc,ts;
	scanf("%d",&T);
	while(T--)
	{
		scanf("%f%f%f",&h,&cc,&ts);
		if(h>50 && cc<0.7 && ts>5600)
		printf("Grade %d\n",10);
		else if(h>50 && cc<0.7 && ts<=5600)
		printf("Grade %d\n",9);
		else if(h<=50 && cc<0.7 && ts>5600)
		printf("Grade %d\n",8);
		else if(h>50 && cc>=0.7 && ts>5600)
		printf("Grade %d\n",7);
		else if((h<=50 && cc<0.7 && ts<=5600)||(h>50 && cc>=0.7 && ts<=5600)||(h<=50 && cc>=0.7 && ts>5600))
		printf("Grade %d\n",6);
		else if(h<=50 && cc>=0.7 && ts<=5600)
		printf("Grade %d\n",5);
	}
	return 0;
}
