#include <stdio.h>
int main()
{
  float n , a , b , c , i , x=0 , y=0 , z=0;
  scanf("%f",&n);
  for(i=0 ; i<n ; i++)
  {
     scanf("%f",&a);
    	if(a>50)
          x=1;
    
     scanf("%f",&b);
    if(b<.7)
      y=1;
     scanf("%f",&c);
    if(c>5600)
      z=1;
    if(x==1 && y==1 && z==1)
      printf("Grade 10");
    else if(x==1 && y==1)
      printf("Grade 9");
    else if(y==1 && z==1)
      printf("Grade 8");
    else if(x==1 || y==1 || z==1)
      printf("Grade 6");
    else if(x==0 && y==0 && z==0)
      printf("Grade 5");
    z=0 , x=0 , y=0;
    printf("\n");
  }

	return 0;
}
