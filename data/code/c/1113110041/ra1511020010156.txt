#include <stdio.h>
int main()
{
      int firstNumber, secondNumber, temporaryVariable;

    
      scanf("%d", &firstNumber);

      
      scanf("%d",&secondNumber);

      temporaryVariable = firstNumber;

      firstNumber = secondNumber;

      secondNumber = temporaryVariable;

      printf("%d\n", firstNumber);
      printf("%d", secondNumber);

      return 0;
}
