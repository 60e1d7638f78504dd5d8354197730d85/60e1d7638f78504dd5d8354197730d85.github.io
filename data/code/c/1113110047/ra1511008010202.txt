#include <stdio.h>
 
struct complex
{
   int realpart, imaginary;
};
 
main()
{
    struct complex a, b, c;
    scanf("%d", &a.realpart);
    scanf("%d", &a.imaginary);
    scanf("%d", &b.realpart);
    scanf("%d", &b.imaginary);
    c.realpart = a.realpart + b.realpart;
    c.imaginary = a.imaginary + b.imaginary;
    if (c.imaginary >= 0)
        printf("%d+%di\n", c.realpart, c.imaginary);
    else
        printf("%d%di\n", c.realpart, c.imaginary);
    return 0;
}
