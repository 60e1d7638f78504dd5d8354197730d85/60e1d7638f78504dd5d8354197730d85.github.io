#include <stdio.h>
#include <string.h>
int main()
{
   int t,num,len,i;
	scanf("%d",&t);
	while(t--){
		char str[105];
		scanf("%s",str);
		len=strlen(str);
		num=0;
		if(str[0]==str[1]) printf("NO\n");
		else{
			for(i=0;i<len;i++){
				if(i%2){
					if(str[i]!=str[1]){
						num=1;
						break;
					}  
				}
				else{
					if(str[i]!=str[0]){
						num=1;
						break;
					}
				}
			}
			if(num==1) printf("NO\n");
			else printf("YES\n");
		}
    }
	return 0;
}
