#include <stdio.h>
#include <string.h>
int main()
{
    int i,j,k,n,t,flag,x;
    char a[105];
    scanf("%d",&t);
    for(i=1;i<=t;i++)
    {
        flag=0;
        scanf("%s",a);
        n=strlen(a);
        if(n==2)
        {
            if(a[0]!=a[1])
              printf("YES\n");
            else
              printf("NO\n");
            continue;  
        }
        if(n%2!=0)
          x=3;
        else
          x=2;
        if(a[0]==a[1])
        {
            printf("NO\n");
            continue;
        }
        for(j=2;j<=n-x;j=j+2)
        {
            if(a[0]!=a[j]||a[1]!=a[j+1])
            {
                flag=1;
                break;
            }
        }
        if(flag)
          printf("NO\n");
        else
        {
           if(n%2==0)
             printf("YES\n");
           else
           {
               if(a[0]==a[n-1])
                 printf("YES\n");
               else
                 printf("NO\n");
           }
        }
    }
    return 0;
}
