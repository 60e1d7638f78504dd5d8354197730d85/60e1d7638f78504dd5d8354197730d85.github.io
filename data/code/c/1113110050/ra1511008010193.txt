#include <stdio.h>
#include<string.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        char s[1000];
        scanf("%s",s);
        int l,i,f=0;
        l=strlen(s);
        int ar[1000];
        for(i=0;i<26;i++)
        ar[i]=0;
        
        for(i=0;i<l;i++)
        {
            ar[(s[i]-'A')]++;
        }
        for(i=0;i<26;i++)
        {
            if(ar[i]>0)
            f++;
        }
        if(f==2)
        printf("YES\n");
        else
        printf("NO\n");
    }
    return(0);
}
