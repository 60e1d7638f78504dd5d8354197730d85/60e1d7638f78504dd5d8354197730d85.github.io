#include <stdio.h>
int main()
{
  int num,diff1,diff2,i,j;
  scanf("%d",&num);
  for(i=num;;i++)
  {
    for(j=2;j<i;j++)
    {
      if(i%j==0)
        break;
    }
    if(j==i)
    {
      diff1=i-num;
      break;
    }
  }
  for(i=num;i>0;i--)
  {
    for(j=2;j<i;j++)
    {
      if(i%j==0)
        break;
    }
    if(j==i)
    {
      diff2=num-i;
      break;
    }
  }
  if(diff1>diff2)
    printf("%d",diff2);
  else
    printf("%d",diff1);
	return 0;
}
