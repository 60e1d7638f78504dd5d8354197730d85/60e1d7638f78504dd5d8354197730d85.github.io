#include <stdio.h>
#include<string.h>
int main()
{
    char s[100000];int i,t,l,z,o;
    scanf("%d",&t);
    while(t)
 {
    	scanf("%s",s);
    	l=strlen(s);
    	z=0;
    	o=0;
        for(i=0;i<l;i++)
        {
            if(s[i]=='0')
                z++;
            else
                o++;
		}
        if(l==1)
            printf("Yes\n");
        else
        {
            if(z==1||o==1)
                printf("Yes\n");
            else
                printf("No\n");
        }
        t--;
    }
    return 0;
}
