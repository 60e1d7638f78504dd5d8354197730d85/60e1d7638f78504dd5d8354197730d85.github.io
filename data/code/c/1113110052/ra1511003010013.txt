#include <stdio.h>
#include <string.h>
int main()
{
  int t,n1,n0,len,i;
	char arr[100005];
	scanf("%d",&t);
	while(t--){
		scanf("%s",arr);
		n1=n0=0;
		len=strlen(arr);
		for(i=0;i<len;i++){
			if(arr[i]=='0')
				++n0;
			else
				++n1;
		}
		if(n1==len-1 || n0==len-1)
			printf("Yes\n");
		else
			printf("No\n");
    }

	return 0;
}
