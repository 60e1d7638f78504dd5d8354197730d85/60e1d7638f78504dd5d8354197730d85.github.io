#include<stdio.h>
#include<string.h>
int main()
{
    int t;
    long int len,i,coun;
    char str[100000];
    scanf("%d",&t);
    while(t--)
    {
        scanf(" %s",str);
        len = strlen(str);
        i = coun = 0;
        while(str[i] != '\0')
        {
            if(str[i] == '1')
                coun++;
 
            i++;
        }
 
        if(coun == len - 1 || coun == 1)
            printf("Yes\n");
 
        else
            printf("No\n");
    }
 
    return 0;
}
