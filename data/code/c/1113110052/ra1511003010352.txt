#include <stdio.h>
#include <string.h>
int main(void)
{
int t,i,temp,len;
	char s[100001];
	scanf("%d",&t);
	while(t--) {
		temp = 0;
		scanf("%s",s);
		len = strlen(s);
		
		for(i=0;i<len;i++)
			if(s[i]=='0') temp++;
		
		if(temp == 1 || temp == len-1) printf("Yes\n");
		else printf("No\n");
    }
	return 0;
}
