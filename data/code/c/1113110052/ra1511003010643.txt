 #include<stdio.h>
#include<string.h>
 
int main(){
  long int  t,len,zero,one,i;
   char  str[100001];
   scanf("%ld",&t);
   while(t--){
     scanf("\n%s",str);
     len =0,zero=0,one=0;
     
    for(i=0; str[i]!='\0';i++)
       {  len++;
            if(str[i]=='1')
               one++;
            else
               zero++;   
       }
       if((one >= len-1 && zero == 1) || (zero >= len-1 && one == 1)|| (len==1) )
         printf("Yes\n");
        else
           printf("No\n");
   }
   return 0;
}
