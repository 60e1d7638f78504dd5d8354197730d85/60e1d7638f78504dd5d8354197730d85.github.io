#include <stdio.h>
#include <string.h>
 
int main(void) {
	int t,len;
	char str[100001];
	scanf("%d",&t);
	while(t--)
	{
	    scanf("%s",str);
	    len = strlen(str);
	    if(len==1)
	    {
	        printf("Yes\n");
	    }
	    else
	    {
	        int count=0,i;
	        for(i=0;i<len;i++)
	        {
	            if(str[i]=='1')
	                count++;
	        }
	        if(len-count==1 || count==1)
	        {
	            printf("Yes\n");
	        }
	        else
	            printf("No\n");
	    }
	}
	return 0;
}
