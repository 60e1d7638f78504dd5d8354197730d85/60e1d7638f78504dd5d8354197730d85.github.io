#include <stdio.h>
int main()
{
int i,D,n,temp,length,count;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  { length=0;
    count=0;
    scanf("%d",&D);
    while(D!=0)
    {
      temp=D%10;
      if(temp==1)
        count++;
      D/=10;
      length++;
    }
    if(count==1 || count+1==length)
      printf("Yes\n");
      else
          printf("No\n");
  }
	return 0;
}
