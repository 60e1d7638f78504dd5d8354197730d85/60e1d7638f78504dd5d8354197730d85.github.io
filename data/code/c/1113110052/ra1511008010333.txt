#include <stdio.h>
int main()
{
char a[100001];
	int t;
	scanf("%d",&t);
	while (t>0)
	{
		int z = 0, o = 0, i;
		scanf("%s", a);
		i = 0;
		while (a[i] != '\0')
		{
			if (a[i] == '0')
				++z;
			if (a[i] == '1')
				++o;
		    i++;
		}
		if ((z == 1) || (o == 1))
			printf("Yes\n");
		else
			printf("No\n");
	
	    t--;
    }
	return 0;
}
