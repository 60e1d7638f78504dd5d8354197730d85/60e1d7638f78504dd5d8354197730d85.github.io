#include<stdio.h>
int main()
{
    int t,i,j;
    char a[100000];
    int count1,count2;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%s",a);
        i=0;
        count1=0;count2=0;
        while(a[i]!='\0')
        {
            if(a[i]=='1')
                count1++;
            if(a[i]=='0')
                count2++;
            i++;
        }
        if(count1==1||count2==1)
            printf("Yes\n");
        else
            printf("No\n");
    }
    return 0;
}
