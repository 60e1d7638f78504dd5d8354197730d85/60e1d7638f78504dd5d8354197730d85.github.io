#include <stdio.h>
 
int main(void) 
{ long int j,o,c,t;char a[100001];
   scanf("%ld",&t);
   while(t--)
   {
       scanf("%s",a);
   
     c=0; o=0;
      for(j=0;a[j]!='\0';j++)
      { if(a[j]=='1')
         c++;
         else
         o++;
      }
    
       if(c==1||o==1)
      printf("Yes\n");
      else
      printf("No\n");
   }
	return 0;
}
