#include <stdio.h>
int main()
{
  int t,num,temp,a=0,b=0;
  scanf("%d",&t);
  while(t--)
  {
    a=0;
    b=0;
    scanf("%d",&num);
   while(num!=0)
   {
     temp=num%10;
     if(temp==1)
       a++;
     else
       b++;
     num=num/10;
   }
    if(a==1||b==1)
      printf("Yes\n");
    else
      printf("No\n");
  }

	return 0;
}
