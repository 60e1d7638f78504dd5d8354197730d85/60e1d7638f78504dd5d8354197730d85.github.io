#include<stdio.h>
#include<string.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        char a[123456];
        scanf("%s",a);
        int i,count=0;
        for(i=0;a[i]!=0;i++)
        {
            if(a[i]=='0')
                count++;
        }
        int l=strlen(a);
        if(count==1||count==l-1)
            printf("Yes\n");
        else
            printf("No\n");
    }
    return 0;
}
