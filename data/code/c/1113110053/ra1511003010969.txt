#include <stdio.h>
int main()
{
    int t;
    long int n,c,i;
    scanf("%d",&t);
    while(t--)
    {
       scanf("%ld",&n);
       c=0;
       for(i=1;i<=n;i++)
       {
           if(n>=i)
           {
               n=n-i;c++;
           }
           else
            break;
       }
        printf("%ld\n",c);
    }

	return 0;
}
