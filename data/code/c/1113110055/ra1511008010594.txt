#include <stdio.h>
inline int scan()
{
int t=0;
char c;
c=getchar_unlocked();
while(c<'0' || c>'9')
c=getchar_unlocked();
while(c>='0' && c<='9')
{
t=(t<<3)+(t<<1)+c-'0';
c=getchar_unlocked();
}
return(t);
}
inline void print(int a)
{
char num[20];
int i=0;
do
{
num[i++]=a%10+48;
a=a/10;
}while(a!=0);
i=i-1;
while(i>=0)
putchar_unlocked(num[i--]);
putchar_unlocked('\n');
}
int main(void) {
	int t,n,m1,m2,a;
	t=scan();
	while(t--)
	{m1=m2=10000000;
		n=scan();
		while(n--)
		{a=scan();
			if(m1>a)
			{m2=m1;m1=a;}
			else if(m2>a&&m1!=a)
			m2=a;
		}print(m1+m2);
	}
	return 0;
}
