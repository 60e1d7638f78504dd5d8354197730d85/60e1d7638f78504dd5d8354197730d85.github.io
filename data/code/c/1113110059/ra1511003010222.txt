#include <stdio.h>
int main()
{
	int n, a[1000], i, j;
  	scanf("%d",&n);
  	for(i = 0; i < n; i++)
      scanf("%d",&a[i]);
  	for(i = 0; i < n; i++)
      for(j = i + 1; j < n; j++)
        if(a[j] < a[i])
        {
          int t = a[i];
          a[i] = a[j];
          a[j] = t;
        }
  	for(i = 0; i < n; i++)
      printf("%d\n",a[i]);
	return 0;
}
