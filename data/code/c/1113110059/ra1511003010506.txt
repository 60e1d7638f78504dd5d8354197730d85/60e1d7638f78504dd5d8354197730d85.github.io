#include <stdio.h>
int main()
{
	int n,temp,j;
  	scanf("%d",&n);
  	int arr[n];
  	int i=0;
  	while(i<n){
      scanf("%d",&arr[i++]);
    }
  
  	for(i=0; i<n; i++)
    {
        for(j=i+1; j<n; j++)
        {
            if(arr[i] > arr[j])
            {
                temp     = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }
  	for(i=0; i<n; i++)
  	{
      printf("%d\n", arr[i]);
    }
  
	return 0;
}
