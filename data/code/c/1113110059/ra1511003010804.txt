#include <stdio.h>
int main()
{
int array[100];
  int i,j,n,temp;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&array[i]);
  }
  for(i=0;i<n;i++)
  {
    for(j=0;j<(n-i-1);j++)
    {
      if(array[j]>array[j+1])
      {
        temp=array[j];
        array[j]=array[j+1];
        array[j+1]=temp;
      }
    }
  }
       for(i=0;i<n;i++)
       {
         printf("%d\n",array[i]);
       }
        
	return 0;
}
