#include <stdio.h>
int main()
{   int t,i,j,n,a[10];
     scanf("%d",&n);
     for(i=0;i<n;i++)
       scanf("%d",&a[i]);
     
     for(i=0;i<n-1;i++)
       for(j=0;j<n-i-1;j++)
         if(a[j]>a[j+1])
         {  t=a[j];
            a[j]=a[j+1];
            a[j+1]=t;
         }
     
     for(i=0;i<n;i++)
       printf("%d\n",a[i]);
           

	return 0;
}
