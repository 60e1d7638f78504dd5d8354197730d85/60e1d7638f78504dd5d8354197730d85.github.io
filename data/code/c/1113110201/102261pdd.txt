#include <stdio.h>
typedef long long ll;
ll a[120];

int main()
{
  int i;
  ll a[120];
    ll n;
    a[1] = 2;
    a[2] = 3;
    scanf("%lld", &n);
    for (i = 3; i < 100; i++)
    {
        a[i] = a[i - 1] + a[i - 2];
    }
    for ( i = 1; i < 100; i++)
    {
        if (a[i] > n)
        {
            printf("%d\n", i - 1);
            break;
        }
    }
    return 0;
}
