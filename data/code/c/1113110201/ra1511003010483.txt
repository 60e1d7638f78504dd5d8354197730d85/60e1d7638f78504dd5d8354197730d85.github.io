#include <stdio.h>
int main()
{
  int n;
  scanf("%d",&n);
  if(n>=7 && n<=10)
  {
    switch(n)
    {
      case 7 : printf("Heptagon");
        break;
      case 8: printf("Octagon");break;
      case 9: printf("Nanogon");break;
      case 10:printf("Decagon");break;
    }
  }
  else
  {
    printf("Input should be from 7 to 10");
  }
	return 0;
}
