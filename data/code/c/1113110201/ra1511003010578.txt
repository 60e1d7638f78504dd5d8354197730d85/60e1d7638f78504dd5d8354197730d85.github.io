#include <stdio.h>
int main()
{
	int s;
  	scanf("%d",&s);
  	switch(s){
      case 7: printf("Heptagon"); 
        break;
      case 8: printf("Octagon"); 
        break;
      case 9: printf("Nanogon"); 
        break;
      case 10: printf("Decagon"); 
        break;
      default: printf("Input should be from 7 to 10");
        break;
    }
	return 0;
}
