#include <stdio.h>
int main()
{
	int a;
  	scanf("%d",&a);
  	if(a<=10 && a>=7)
    {
      switch(a)
      {
        case 7:
          printf("Heptagon");
          break;
        case 8:
          printf("Octagon");
          break;
        case 9:
          printf("Nanogon");
          break;
        case 10:
          printf("Decagon");
          break;
      }
    }
  else
    printf("Input should be from 7 to 10");
	return 0;
}
