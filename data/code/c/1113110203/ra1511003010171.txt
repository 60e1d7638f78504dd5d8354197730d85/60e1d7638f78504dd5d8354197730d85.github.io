#include <stdio.h>

#include<math.h>
int main()
{
    char sequence[100];
    int length, no_of_zeroes = 0, no_of_ones = 0,i;
    scanf("%d %s", &length, sequence);

    for(i = 0;i< length;i++)
    {
        no_of_zeroes += (sequence[i] == '0');
        no_of_ones += (sequence[i] == '1');
    }

    int no_of_survivors = length - 2*min(no_of_ones, no_of_zeroes);

    printf("%d\n", no_of_survivors);
    return 0;
}

int min(int a, int b)
{
  if(a>b)
    return b;
  else
    return a;
  
}
