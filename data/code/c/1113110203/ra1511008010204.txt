#include<stdio.h>
#include<string.h>

int main()
{
  int n,p,q,i;
char a[200000];
    scanf("%d",&n);
    scanf("%s",a);
    q=0,p=0;
    for( i=0; i<n; i++)
    {
        if(a[i]=='1')
            p++;
        if(a[i]=='0')
            q++;
    }
    if(p>=q)
        printf("%d",p-q);
    else
        printf("%d",q-p);
    return 0;
}
