#include<stdio.h>
int main()
{
    char sequence[100];
    int length, no_of_zeroes = 0, no_of_ones = 0,i;
    scanf("%d %s", &length, sequence);

    for(i = 0; i < length; i++)
    {
        no_of_zeroes += (sequence[i] == '0');
        no_of_ones += (sequence[i] == '1');
    }

    int no_of_survivors = length - 2*((no_of_ones< no_of_zeroes)?no_of_ones:no_of_zeroes);

    printf("%d\n", no_of_survivors);
    return 0;
}
