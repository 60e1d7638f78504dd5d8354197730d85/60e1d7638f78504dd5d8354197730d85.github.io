#include <stdio.h>
int main()
{
	int a, b, t;
  	scanf("%d", &t);
    while(t>0)
    {
      	scanf("%d%d", &a, &b);
      	if(a<b)
        {
          printf("<\n");
          t--;
        }
        else if(a>b)
        {
          printf(">\n");
          t--;
        }
        else
        {
          printf("=\n");
          t--;
        }
      
    }
	return 0;
}
