#include <stdio.h>
int main()
{
int N,n,v,i,sum=0;
  
  scanf("%d",&N);
  while(N!=0)
  {
    scanf("%d",&n);
    scanf("%d",&v);
     for(i=0;i<n;i++)
     {
      v=fun(v);
     }
    printf("%d\n",v);
    N=N-1;
  }
   
	return 0;
}
int fun(int a)
{
  int i,sum=0; 
  for(i=1;i<=a;i++)
    sum=sum+i;
  return sum;
}
