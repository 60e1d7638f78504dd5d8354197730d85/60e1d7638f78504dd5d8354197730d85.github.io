#include<stdio.h>
long long int x,k,fib[45],ans;
void chefonacci(long long int i,long long int sum,long long int c,long long int z)
{
    if((sum+fib[i])==x && (c+1)==k)
        ans++;
    else if((sum+fib[i])<x && (c+1)<k)
    {
        if(sum+fib[i]*(k-c) >= x)
        {   chefonacci(i,sum+fib[i],c+1,z+1);
            if(i>0)
                chefonacci(i-1,sum+fib[i],c+1,0);
        }
    }
    if(i>0 && z==0)
    {
        if(sum+fib[i-1]*(k-c) >= x)
        chefonacci(i-1,sum,c,0);
    }
}
int main()
{
    long long int q,i;
    scanf("%lld",&q);
    fib[0]=1;fib[1]=2;
    for(i=2;i<44;i++)
        fib[i]=fib[i-1]+fib[i-2];
    while(q--)
    {
        scanf("%lld%lld",&x,&k);
        ans=0;
        for(i=42;i>0;i--)
        {
            if(fib[i]<=x)
            {
                break;
            }
        }
        chefonacci(i,0,0,0);
        printf("%lld\n",ans);
    }
    return 0;
}
