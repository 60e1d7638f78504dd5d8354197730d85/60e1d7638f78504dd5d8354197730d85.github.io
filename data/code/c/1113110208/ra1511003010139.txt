#include <stdio.h>
#include <stdlib.h>
 
 
int num_ways(int X,int m,int K,int *chefo_memo)
{
	
	
	if(X<0)
	{
		return 0;
	}
	if(m<0)
	{
		return 0;
	}
	
	if(K>0)
	{
		int answer;
		answer = (num_ways(X-chefo_memo[m],m,K-1,chefo_memo)%1000000007 + num_ways(X,m-1,K,chefo_memo)%1000000007)%1000000007;
		return (answer)%1000000007; 
	}
	else if(K==0)
	{
		if(X>0)
		{
			return 0;
		}
		else if(X<0)
		{
			return 0;
		}
		else
		{
			return 1;
		}
	}
	else
	{
		return 0;
	}
	
}
 
int chefonacci(int num,int *chefo_memo)
{
	if(chefo_memo[num]==0)
	{
		if(num == 0)
		{
			chefo_memo[num] = 1;
			return 1;
		}
		else if(num == 1)
		{
			chefo_memo[num] = 2;
			return 2;
		}
		else if(num>1)
		{
			chefo_memo[num] = chefonacci(num-1,chefo_memo) + chefonacci(num-2,chefo_memo);
			return chefo_memo[num];
		}
		
	}
	else
	{
		return chefo_memo[num];
	}
}
 
int main()
{
	
	int Q;
	scanf("%d",&Q);

		printf("%d\n",answer);

}
