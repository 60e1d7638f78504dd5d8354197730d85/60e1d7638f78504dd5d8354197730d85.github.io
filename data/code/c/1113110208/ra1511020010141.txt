#include <stdio.h>
#include <stdlib.h> 
int possiblesets;
int x,k;
long long *arr;
int num;
void find(int ind,int sum,int count);
int main(){
	int t,i;
	arr=(long long *)malloc(sizeof(long long)*(45));
	scanf("%d",&t);
	for(i=0;i<45;i++){
			if(i==0){arr[i]=1;}
			else if(i==1){arr[i]=2;}
			else{arr[i]=arr[i-1]+arr[i-2];}
 
		}
 
	while(t--){
		scanf("%d %d",&x,&k);
		possiblesets=0;
		num=0;
		find(44,x,k);
		printf("%d\n", possiblesets%1000000007);
	}
	return 0;
}
 
void find(int ind,int sum,int count){
	int i;
	if(sum==0 && count==0){
		possiblesets=(possiblesets+1)%100000007;
		return;}
	else if(sum>0 && count>0){
	for(i=ind;i>=0;i--){
		if(arr[i]*count<sum)
			break;
		else if(arr[i]<=sum){find(i,sum-arr[i],count-1);}
		}
	}
}
