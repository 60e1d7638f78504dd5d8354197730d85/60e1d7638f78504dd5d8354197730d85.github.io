#include<stdio.h>
int main()
{
int n,i,sumtime,mintime=0,maxtime=0;
    scanf("%d%d",&n,&sumtime);
    int a[n],b[n];
    for(i=0;i<n;i++)
    {
    	scanf("%d%d",&a[i],&b[i]);
    	mintime+=a[i];
    	maxtime+=b[i];
    }
	if(sumtime<mintime || sumtime>maxtime)
	{
		printf("NO\n");
		return 0;
	}
	printf("YES\n");
	for(i=0;i<n;i++)
	{
		if(b[i]<(a[i]+sumtime-mintime))
        {
          printf("%d ",b[i]);
          sumtime-=b[i];
        }
      else
      {
        printf("%d ",(a[i]+sumtime-mintime));
         sumtime-=a[i]+sumtime-mintime;
      }
		  
      
		mintime-=a[i];
    }
return 0;
}
