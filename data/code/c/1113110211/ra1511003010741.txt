#include <stdio.h>
#include<string.h>
int main(void)
{
  int total;
  scanf("%d",&total);
  while(total--)
  {
    char s[5001];
    int a=0,b=0,len,i;
    scanf("%s",s);
    len=strlen(s);
    for(i=0;i<len;i++)
    {
      if(s[i]=='A')
        a++;
      if(s[i]=='B')
        b++;
    }
    if(a==2)
      printf("A\n");
    else if(b==2)
      printf("B\n");
    else
      printf("-1\n");
  }
    

	return 0;
}
