#include <stdio.h>
#include<string.h>
int main()
{
    int t,c,d,e,i;
    scanf("%d",&t);
    while(t--)
    {
        d=0;
        e=0;
        char a[10000];
        scanf("%s",a);
        c=strlen(a);
        for(i=0;i<c;i++)
        {
            if(a[i]=='A')
                d++;
        }
        e=c-d;
        if(d==2)
        {
            printf("A\n");
        }
        else if(e==2)
            printf("B\n");
        else printf("-1\n");
        
 
    }
	return 0;
}
