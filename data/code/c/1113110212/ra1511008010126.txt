#include <stdio.h>
int main()
{
  int colour[41]={0}, cost[41]={0}, dp1[41][41]={0}, dp2[41][41]={0};
  
  int t, n, m, i, j, a, b;
  double num, den;
  scanf("%d", &t);
  while(t--)
  {
    for(i=0; i<41; i++)
    {
      cost[i]=0;
    colour[i]=0;
    for(j=0; j<41; j++)
    {
      dp1[i][j]=0;
      dp2[i][j]=0;
    }
  }
  scanf("%d", &n);
  scanf("%d", &m);
  for(i=0; i<n; i++)
  {
    scanf("%d", &a);
    scanf("%d", &b);
    colour[a]++;
    cost[a]+=b;
  }
  dp1[0][0]=1;
  for(i=1; i<41; i++)
  {
    dp1[i][0]=1;
    for(j=1; j<i+1; j++)
    {
      if(colour[i]>0)
      {
        dp1[i][j] = dp1[i-1][j] + dp1[i-1][j-1]*((1LL<<colour[i])-1);
      }
      else
        dp1[i][j] = dp1[i-1][j];
    }
  }
  
  for(i=1; i<41; i++)
  {
    for(j=1; j<i+1; j++)
    {
      if(colour[i]>0)
      {
        dp2[i][j] = dp2[i-1][j] + dp2[i-1][j-1]* ((1LL<<colour[i])-1) + dp1[i-1][j-1]*cost[i]* (1LL<<(colour[i]-1));
      }
      else
          dp2[i][j] = dp2[i-1][j];
          }
       }
                               num = 0; 
                               den = 0;
                                      for(i=m; i<=n; i++)
                                      {
                                        num+=dp2[40][i];
                                        den+=dp1[40][i];
                                      }
                                      printf("%.9lf\n", (double)(num/den));
                                      }
                                      
                                      
    

	return 0;
}
