#include <stdio.h>
int main()
{
   int n,a[20],i,count=0,sum=0;
   scanf("%d",&n);
   for(i=0;i<n;i++)
     scanf("%d",&a[i]);
   for(i=0;i<n;i++)
   {
     if(a[i]<0)
     {
       sum=sum+a[i];
     }
     else
     {
       count=count+1;
     }
   }
  printf("Positive Count=%d\n",count);
  printf("Negative Sum=%d",sum);
  return 0;
}
