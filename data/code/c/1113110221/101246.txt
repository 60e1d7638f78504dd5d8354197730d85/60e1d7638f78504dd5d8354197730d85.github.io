#include <stdio.h>
int main()
{
   int m,n, AND_opr,OR_opr,XOR_opr,NOT_opr ;
   scanf("%d",&m);
   scanf("%d",&n);
   AND_opr = (m&n);
   OR_opr = (m|n);
   NOT_opr = (~m);
   XOR_opr = (m^n);
   printf("%d\n",AND_opr );
   printf("%d\n",OR_opr );
   printf("%d\n",NOT_opr );
   printf("%d\n",XOR_opr );
   printf("%d\n", m << 1);
   printf("%d\n", m >> 1);
	return 0;
}
