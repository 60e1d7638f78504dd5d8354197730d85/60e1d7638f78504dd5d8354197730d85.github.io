#include <stdio.h>
int main()
{
int a[15],i,n,count=0,sum1=0;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  scanf("%d",&a[i]); 
  for(i=0;i<n;i++)
  {
    if(a[i]>0)
    {
      count++;
    }
   if(a[i]<0)
   {
     sum1+=a[i];
   }
  }
  printf("Positive Count=%d\n",count);
  printf("Negative Sum=%d",sum1);
    
return 0;
}
