#include <stdio.h>
int main()
{
  int n,i,count,sum,a[100];
  sum=0;
  count=0;
  scanf("%d",&n);
  for (i=0;i<n;i++)
   scanf("%d",&a[i]);
  for (i=0;i<n;i++){
    if(a[i]<0)
      sum+=a[i];
    else
      count++;
  }
  printf("Positive Count=%d\n",count);
  printf("Negative Sum=%d",sum);
  
	return 0;
}
