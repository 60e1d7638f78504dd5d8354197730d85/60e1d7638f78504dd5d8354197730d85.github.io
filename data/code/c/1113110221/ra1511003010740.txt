#include <stdio.h>
int main()
{int n,arr[1000],count=0,sum=0,i;
 scanf("%d",&n);
 for(i=0;i<n;i++)
 {scanf("%d",&arr[i]);
 }
 for(i=0;i<n;i++)
 {
   if(arr[i]>0)
     count=count+1;
   else if(arr[i]<0)
   {sum=sum+arr[i];
   }
   else continue;
 }
 printf("Positive Count=%d\nNegative Sum=%d",count,sum);

	return 0;
}
