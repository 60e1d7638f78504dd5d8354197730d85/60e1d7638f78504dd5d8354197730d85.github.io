#include <stdio.h>
int main()
{
	int a[100],n,i,c=0,sum=0;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
      scanf("%d",&a[i]);
    }
    for(i=0;i<n;i++)
    {
        if(a[i]>0)
        c++;
        if(a[i]<0)
        sum+=a[i];
    }
    printf("Positive Count=%d\n",c);
    printf("Negative Sum=%d",sum);
	return 0;
}
