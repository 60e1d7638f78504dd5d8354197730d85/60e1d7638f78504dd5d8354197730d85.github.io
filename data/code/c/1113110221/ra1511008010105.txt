#include <stdio.h>
int main()
{
int a[20],n,i,count=0,sum=0;
  scanf("%d",&n);
  for (i=0;i<n;i++)
  { 
    scanf("%d",&a[i]);
  }
  for (i=0;i<n;i++)
  {
    if (a[i]>0)
    {
      count++;
    }
    else 
    {
      sum=sum+a[i];
    }
  }
    printf("Positive Count=%d\n",count);
    printf("Negative Sum=%d",sum);
	return 0;
}
