#include <stdio.h>
int main()
{
  	int n,i,count=0,sum=0;	
  	scanf("%d\n",&n);
  	int a[n];
  	for(i=0;i<n;i++){
      scanf("%d ",&a[i]);
    }
  	for(i=0;i<n;i++){
      if(a[i]>0)
        count++;
      else if(a[i]<0)
        sum=sum+a[i];
    }
  	printf("Positive Count=%d\n",count);
  	printf("Negative Sum=%d",sum);
	return 0;
}
