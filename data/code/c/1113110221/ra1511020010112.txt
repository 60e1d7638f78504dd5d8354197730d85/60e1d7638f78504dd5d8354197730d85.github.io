# include <stdio.h>
int main() 
{ 
 int a[20], i,n,nsum = 0,count=0 ; 

 scanf("%d", &n) ; 

 for(i = 0 ; i < n ; i++) 
  scanf("%d", &a[i]) ; 
  for(i=0;i<n;i++)
  {
     if(a[i]>=0)
        {
            count++;
     }
    
  }
  printf("Positive Count=%d\n",count);
 for(i = 0 ; i < n ; i++) 
 { 
  if(a[i]<0) 
   nsum = nsum + a[i] ; 
 } 

 printf("Negative Sum=%d", nsum) ;
return 0;
}
