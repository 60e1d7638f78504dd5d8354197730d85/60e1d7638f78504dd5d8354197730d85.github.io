#include <stdio.h>
void convert(int,int);

int main()
{
  int num;
  scanf("%d%d%d",&num,&num,&num);
  printf("%d",num);
  printf(" ");
  convert(num,8);
  printf(" ");
  convert(num,16);
	return 0;
}
void convert(int num,int base)
{
  int rem=num%base;
  if(num==0)
    return;
  convert(num/base,base);
  if(rem<10)
    printf("%d",rem);
  else
    printf("%c",rem-10+'a');
}
