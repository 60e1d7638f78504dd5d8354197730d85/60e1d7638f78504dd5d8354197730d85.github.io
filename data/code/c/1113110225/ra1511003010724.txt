
#include<stdio.h>
int main(){
    
    int i,v,c,ch,d,s,o;
    o=v=c=ch=d=s=0;
    char line[20];
  scanf("%[^\n]",line);
    for(i=0;line[i]!='\0';i++)
    {
        if(line[i]=='a' || line[i]=='e' || line[i]=='i' || line[i]=='o' || line[i]=='u' || line[i]=='A' || line[i]=='E' || line[i]=='I' || line[i]=='O' || line[i]=='U')
            v+=1;
        else if((line[i]>='a'&& line[i]<='z') || (line[i]>='A'&& line[i]<='Z'))
            c+=1;
        else if(line[i]>='0'&&c<='9')
            d+=1;
        else if (line[i]==' ')
            s+=1;
    }
    printf("%d %d %d %d",v,c,d,s);
    return 0;
}
