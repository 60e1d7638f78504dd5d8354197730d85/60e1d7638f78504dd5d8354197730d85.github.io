#include <stdio.h>
int main()
{
  int i,v,c,ch,d,s,o;
  o=v=c=ch=d=s=0;
  char line[25];
    scanf(" %[^\n]s",line);
  for(i=0;line[i]!='\0';++i)
  {
    if(line[i]=='a' || line[i]=='e' || line[i]=='o' || line[i]=='i' || line[i]=='u' || line[i]=='A' || line[i]=='E' || line[i]=='I' || line[i]=='O' || line[i]=='U')
      ++v;
    else if((line[i]>='a' && line[i]<='z') || (line[i]>='A' && line[i]<='Z'))
      ++c;
    else if(line[i]>='0' && line[i]<='9')
      ++d;
    else if(line[i]==' ')
      ++s;
  }
    printf("Vowels:%d\nConsonants:%d\nDigits:%d\nWhite spaces:%d",v,c,d,s);

	return 0;
}
