#include <stdio.h>
int main()
{
 int n,i,j,a;
    long long int pro=1,net;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        pro=1;
        scanf("%d",&a);
        for(j=1;j<=a;j++)
        {
            pro=pro*2;
            pro=pro%100000;
        }
        net=((pro-1)%100000);
        printf("%lld\n",net);
    }
	return 0;
}
