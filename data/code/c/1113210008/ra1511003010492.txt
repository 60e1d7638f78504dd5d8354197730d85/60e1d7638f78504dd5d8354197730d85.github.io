    int first(int n,int k)
    {
    long double nlogn=n*log10l(n);

    long long e_b_sq(int bs,int po,int mod)
    {
    long long y=bs%mod,x=1;
    int i;
    while(po>0)
    {
    if(po&1)
    {
    long long temp=0;
    for(i=0;i<x;i++)
    temp+=y;
    x=temp%mod;
    }
    long long temp=0;
    for(i=0;i<y;i++)
    temp+=y;
    y=temp%mod;
    po>>=1;
    }
    return x;
    }
