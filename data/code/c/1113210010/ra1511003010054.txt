# include <stdio.h>
# include <stdlib.h> 
int main()
{
	int t,i,j;
	scanf("%d",&t);	
	for (i = 0; i < t; i++)
	{
		int N,M,s;
		scanf("%d%d%d",&N,&M,&s); 
		int A[1005];
		unsigned long long int t1[1005];
		unsigned long long int t2[1005]; 
		for(j=0;j<M;j++)
		{
			scanf("%d",&A[j]);
		} 
		for(j=1;j<=N;j++)
		{
			t1[j] = 0;
			t2[j] = 0;
		}
		int k;
		int count = 0;
		int flag =0; 
		while(count != M)
		{
			int i = A[count];
			if(count == 0)
			{
				if(s-i>=1)
					{t1[s-i]++;
					}
				
				if(s+i <= N)
					{
						t1[s+i]++;
					}
				flag =1; 
			}
			else
			{
				if(flag == 1) 
				{ 
					for(j=1;j<=N;j++)
					{
						t2[j] = 0;
					}
 
					for(k=1;k<=N;k++)
					{
						if(t1[k] > 0)
						{
							if(k-i>=1)
							{
								t2[k-i] = t2[k-i] + t1[k];
							}
							if(k+i <= N)
							{
								t2[k+i] = t2[k+i] + t1[k];
							}
						} 
					}
					flag = 2;
				}
				else if(flag == 2) 
				{					
					for(j=1;j<=N;j++)
					{
						t1[j] = 0;
					}					
					for(k=1;k<=N;k++)
					{
						if(t2[k] > 0)
						{
							if(k-i>=1)
							{
								t1[k-i] = t1[k-i] + t2[k];
							}
							if(k+i <= N)
							{
								t1[k+i] = t1[k+i] + t2[k];
							}
						}
 
					}					
					flag = 1;
				}
			}			
			count++;
		} 
		int z; 
		if(flag == 1)
		{
			for(z=1;z<=N;z++)
			{
				printf("%llu ",(t1[z]%1000000007));
			}
		}
		else if(flag == 2)
		{
			for(z=1;z<=N;z++)
			{
				printf("%llu ",(t2[z]%1000000007));
			}
		}
		printf("\n");		
	}
	return 0;
}
