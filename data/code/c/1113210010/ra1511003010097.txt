#include<stdio.h>
#include<math.h>
#include<stdlib.h>
#define c 1000000007
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long  m,s,j,n,i;
        scanf("%ld%ld%ld",&n,&m,&s);
        long a[m+1];
        long ans[m+1][n+1];
        for(i=1;i<=m;i++)
            scanf("%ld",&a[i]);
        for(i=1;i<=n;i++)
        {
            if(i!=s)
                ans[0][i]=0;
            else
                ans[0][i]=1;
        }
        for(i=1;i<=m;i++)
        {
            for(j=1;j<=n;j++)
            {
                if(j-a[i]>=1 && j+a[i]<=n)
                {
                    ans[i][j]=(ans[i-1][j-a[i]]+ans[i-1][j+a[i]])%c;
                }
                else if(j-a[i]>=1 && j+a[i]>n)
                {
                    ans[i][j]=ans[i-1][j-a[i]];
                }
                else if(j-a[i]<1 && j+a[i]<=n)
                {
                    ans[i][j]=ans[i-1][j+a[i]];
                }
                else
                {
                    ans[i][j]=0;
                }
            }
        }
        for(i=1;i<=n;i++)
        {
            printf("%ld ",ans[m][i]);
        }
        printf("\n");
    }
    return 0;
}
