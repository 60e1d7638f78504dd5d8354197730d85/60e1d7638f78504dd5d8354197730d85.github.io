# include <stdio.h>
# include <stdlib.h>
 
 
int main()
{
	int t,i,j;
	scanf("%d",&t);
	
	for (i = 0; i < t; i++)
	{
		int N,M,s;
		scanf("%d%d%d",&N,&M,&s);
 
		int A[1005];
		unsigned long long int t1[1005];
		unsigned long long int t2[1005];
 
		for(j=0;j<M;j++)
		{
			scanf("%d",&A[j]);
		}
 
		for(j=1;j<=N;j++)
		{
			t1[j] = 0;
			t2[j] = 0;
		}
 
		int k;
		int count = 0;
		int flag =0;
 
		while(count != M)
		{
			int i = A[count];
			if(count == 0)
			{
				if(s-i>=1)
					t1[s-i] = (t1[s-i]+1)%1000000007;
								
				if(s+i <= N)
		     		t1[s+i] = (t1[s+i]+1)%1000000007 ;
					
				flag =1;
 
			}
			else
			{
				if(flag == 1) // we will travese in 1st array
				{
					
 
					for(j=1;j<=N;j++)
					{
						t2[j] = 0;
					}
 
					for(k=1;k<=N;k++)
					{
						if(t1[k] > 0)
						{
							if(k-i>=1)
							{
								t2[k-i] = (t2[k-i] + t1[k])%1000000007;
							}
							if(k+i <= N)
							{
								t2[k+i] = (t2[k+i] + t1[k])%1000000007;
							}
						}
 
					}
 
 
					flag = 2;
				}
				else if(flag == 2) // we will travere in 2nd array
				{
					
					for(j=1;j<=N;j++)
					{
						t1[j] = 0;
					}
 
					
					for(k=1;k<=N;k++)
					{
						if(t2[k] > 0)
						{
							if(k-i>=1)
							{
								t1[k-i] = (t1[k-i] + t2[k])%1000000007;
							}
							if(k+i <= N)
							{
								t1[k+i] = (t1[k+i] + t2[k])%1000000007;
							}
						}
 
					}
					
 
					
					flag = 1;
				}
			}
 
			
			count++;
		}
 
		int z;
 
		if(flag == 1)
		{
			for(z=1;z<=N;z++)
			{
				printf("%llu ",(t1[z]%1000000007));
			}
		}
		else if(flag == 2)
		{
			for(z=1;z<=N;z++)
			{
				printf("%llu ",(t2[z]%1000000007));
			}
		}
		printf("\n");
		
 
		
	}
 
	return 0;
}
