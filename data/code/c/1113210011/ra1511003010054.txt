#include<stdio.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,i,d,j,min;
        scanf("%d",&n);
        int A[n];
        for(i=0;i<n;i++)
            scanf("%d",&A[i]);
        min=1000000000;
        for(i=0;i<n;i++)
            for(j=i+1;j<n;j++)
            {
                d=A[i]>A[j]?A[i]-A[j]:A[j]-A[i];
                if(d<min)
                    min=d;
            }
        printf("%d\n",min);
    }
  return 0;
}
