#include <stdio.h>
void partition(int *arr,int l,int m,int h)
{
  int temp[10000],i=l,j=m+1,k=l;
  while(i<=m&&j<=h)
  {
    if(arr[i]<arr[j])
    {
      temp[k]=arr[i];
      i++;
    }
    else
    {
      temp[k]=arr[j];
      j++;
    }
    k++;
  }
  
  while(i<=m)
  {
    temp[k++]=arr[i++];
  }
  while(j<=h)
  {
    temp[k++]=arr[j++];
  }
  for(i=l;i<k;i++)
  {
    arr[i]=temp[i];
  }
}
void mergesort(int *arr,int low,int high)
{
  int mid;
  mid=(low+high)/2;
  if(low<high)
  {
    mergesort(arr,low,mid);
    mergesort(arr,mid+1,high);
    partition(arr,low,mid,high);
  }
}
