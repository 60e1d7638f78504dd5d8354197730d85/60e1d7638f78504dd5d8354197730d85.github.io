#include <stdio.h>
int main()
{
    	int a[20],n,i,j,k,d,times,u;
	scanf("%d",&times);
  if(times==0)
  {
    printf("0");
    
  }
  else
  {
    for(u=0;u<times;u++)
    {
	scanf("%d",&n);
	
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	for(i=0;i<n-1;i++)
	{
		for(j=0;j<n-i-1;j++)
		{
			if (a[j]>a[j+1])
			{
				k=a[j];
				a[j]=a[j+1];
				a[j+1]=k;
			}
		}
    }
  d=a[1]-a[0];
  printf("%d",d);
  }
  }
	return 0;
}
