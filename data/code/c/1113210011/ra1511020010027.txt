#include<stdio.h>
#include<limits.h>
#define ll long long
 
void merge(ll int a[],ll int l,ll int mid,ll int r)
{
	ll int i,j,k,n1,n2;
	n1 = mid-l+1;
	n2 = r-mid;
	ll int L[n1],R[n2];
 
	for(i=0;i<n1;i++)
		L[i] = a[l+i];
	for(i=0;i<n2;i++)
		R[i] = a[mid+i+1];
	
	i=0;
	j=0;
	k=l;
	while(i<n1 && j<n2)
	{
		if(L[i]<=R[j])
		{
			a[k]=L[i];
			i++;
			k++;
		}
		else
		{
			a[k]=R[j];
			j++;
			k++;
		}
	}
	while(i<n1)
	{
		a[k]=L[i];
		i++;
		k++;
	}
	while(j<n2)
	{
		a[k]=R[j];
		j++;
		k++;
	}
}	
 
void mergesort(ll int a[],ll int l,ll int r)
{
	if(l<r)
	{
		ll int mid = (l+r)/2;
		mergesort(a,l,mid);
		mergesort(a,mid+1,r);
		merge(a,l,mid,r);
	}
}
 
int main()
{
	ll int t,n,i,j,min;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		ll int a[n];
		for(i=0;i<n;i++)
			scanf("%lld",&a[i]);
		mergesort(a,0,n-1);
		min = INT_MAX;
		for(i=0;i<n-1;i++)
		{
			if((a[i+1]-a[i])<min)
				min = a[i+1]-a[i];
		}
		printf("%lld\n",min);
	}
return 0;
}
