#include <stdio.h>
#include<limits.h>
 
int main(void) {
    int t, p,i,j;
    scanf("%d",&t);
    while(t--){
 
    	p=INT_MAX;
    	int n;
    	scanf("%d", &n);
    	int a[n];
    	for( i=0; i<n; i++){
    		scanf("%d", &a[i]);
    	}
 
 
    	for (i = 0; i < n; ++i){
    		for (j = i+1; j < n; ++j){
    			if(p > abs(a[i]-a[j])) p=abs(a[i]-a[j]);
    		}
    	}
 
    	printf("%d\n",p);
    }
	return 0;
}
