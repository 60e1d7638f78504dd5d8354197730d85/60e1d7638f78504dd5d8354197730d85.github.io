#include<stdio.h>
int main()
{
	int t,n;
	scanf("%d",&t);
	while(t--)
	{
      long int i;
	    long int min;
	    scanf("%d",&n);
	    long int arr[n];
	    for( i=0;i<n;i++)
	        scanf("%ld",&arr[i]);
	    for( i=1;i<n;i++) 
	    {
	        int j=i-1;
	        long int key =arr[j+1];
	        while(j>=0 && key<arr[j])
	        {
	            arr[j+1]=arr[j];
	            j--;
	        }
	        arr[j+1]=key;
	    }
	    for( i=n-1;i>0;i--)
	    {
	        if(i==n-1 || (arr[i]-arr[i-1])<min)
	        min=(arr[i]-arr[i-1]);
	    }
	    printf("%ld\n",min);
	}
	return 0;
}
