#include<stdio.h>
#include<math.h>
int main()
{
  float t,x1,x2,x3,y1,y2,y3;
  int cnt=0;
  scanf("%f",&t);
  while(t--)
  {
    scanf("%f %f %f %f %f %f",&x1,&y1,&x2,&y2,&x3,&y3);
   
   if((y1-y2)*(y1-y3)==-1*(x1-x2)*(x1-x3) || (y2-y3)*(y2-y1)==-1*(x2-x3)*(x2-x1) || (y3-y1)*(y3-y2)==-1*(x3-x1)*(x3-x2))cnt++;
  }
  printf("%d\n",cnt);
  return 0;
  
}
