#include <stdio.h>
#define inchar getchar_unlocked
inline int inIntPos() {
	int n = 0, ch;
	while ((ch = inchar()) < '0');
	while (ch >= '0') {
		n = (n << 3) + (n << 1) + (ch - '0');
		ch = inchar();
	}
	return n;
}
int main() {
	int n,x1,y1,x2,y2,x3,y3,res=0;
	int x12,x13,x23,y12,y13,y23;
	int s1,s2,s3;
 
	n = inIntPos();
	do {
		x1 = inIntPos();
		y1 = inIntPos();
		x2 = inIntPos();
		y2 = inIntPos();
		x3 = inIntPos();
		y3 = inIntPos();
 
		x12 = (x1-x2);
		x13 = (x1-x3);
		x23 = (x2-x3);
 
		y12 = (y1-y2);
		y13 = (y1-y3);
		y23 = (y2-y3);
 
		s1 = x12*x12+y12*y12;
		s2 = x13*x13+y13*y13;
		s3 = x23*x23+y23*y23;
 
		if(s1+s2==s3 || s1+s3==s2 || s2+s3==s1) ++res;
	} while(--n);
 
	printf("%d\n",res);
	return 0;
}
