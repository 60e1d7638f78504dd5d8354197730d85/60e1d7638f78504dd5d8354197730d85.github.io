#include<stdio.h>
#include<math.h>
 
int main(void)
{
int t,i,j,p,q,n;
scanf("%d",&t);
for(i=1;i<=t;i++)
  {
    scanf("%d",&n);
    p=(int)(sqrt(n));
    for(j=p;j>=1;j--)
      {
         if(n%j==0)
            { q=(n/j)-j;
              printf("%d\n",q);
              break;
            }        
      
      }
    
  }
return 0;
}
