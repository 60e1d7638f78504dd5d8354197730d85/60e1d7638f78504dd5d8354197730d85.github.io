#include<stdio.h>
 
int main(){
    int n,i;
	long long int s[100]={0},k[100]={0};
	scanf("%d",&n);
	for(i=0;i<n;i++){
	scanf("%lld %lld",&s[i],&k[i]);
	}
	for(i=0;i<n;i++){
	if(s[i]>0 && k[i]>0)
	printf("%lld %lld\n",s[i]/k[i],s[i]%k[i]);
	else if(s[i]==0 || k[i]==0)
	printf("%d %lld\n",0,s[i]);
	}
	return 0;
}
