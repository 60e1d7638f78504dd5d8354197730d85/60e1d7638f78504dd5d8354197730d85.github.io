#include<stdio.h>
 
 
 
int main()
{
	int i, t, x, y, sum=0;
	scanf("%d", &t);
	
	for( i = 0; i < t; i++ )
	{
		scanf("%d %d", &x, &y);
		sum = x+y;
		int j,k=1;
		for(j=2; j <= (sum+k)/2; j++)
		{
			if( (sum+k)%j == 0 )
			{
				k++;
				j = 2;
				continue;
			}
			if( j > ((sum+k)/2) )
				break;
		}
		
		printf("%d\n", k);
	}
	
	return 0;
}
