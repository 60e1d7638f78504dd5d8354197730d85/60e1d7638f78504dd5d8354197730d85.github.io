#include<stdio.h>
int main()
{
    int T;
    scanf("%d",&T);
    while(T--)
    {
        int x,y,z,c,i;
        scanf("%d%d",&x,&y);
        z=x+y;c=z;z++;
        i=2;
        while(i<z)
            if(z%i==0){
                z++;
                i=2;}
                else
                    i++;
        printf("%d\n",z-c);
    }
    return 0;
}
