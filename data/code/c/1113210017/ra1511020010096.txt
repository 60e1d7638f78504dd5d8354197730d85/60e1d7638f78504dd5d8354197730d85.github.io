#include <stdio.h>
int main()
{
   int n,a,b,p,i,j,c=0;
  scanf("%d",&n);

  while(n--)
  {
    j=0;
    scanf("%d%d",&a,&b);
    a=a+b;
    p=a;
    a++;
    while(j==0)
    { c=0;
      for(i=1;i<a;i++)
      {
        if(a%i==0)
          c++;
      }
      if(c==1)
      {
        printf("%d\n",a-p);
        break;
      }
      else
      a++;
      
    }
    
  }
	return 0;
}
