long long int power_module(long long int times)
{
	long long int prod=1,base=26;
	while(times>0)
	{
		if((times & 1)==1) prod=(prod*base) % 10000009;
		times=times>>1;
		base=(base*base) % 10000009;
	}
	return prod;
}
