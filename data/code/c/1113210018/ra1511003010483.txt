#include<stdio.h>
#include<string.h>
 
 
int main()
{ int t;
  scanf("%d",&t);
 
  while(t--)
  { char arr[1000001];

    long n=strlen(arr);
       
    long i;

    if(n%2==0)odd=0;
 
    for(i=0;i<n/2;i++)
    { if((arr[i]==arr[n-i-1])||(arr[i]=='?')||(arr[n-i-1]=='?'))
	{  if(arr[i]=='?')arr[i]=arr[n-i-1];
           if(arr[n-i-1]=='?')arr[n-i-1]=arr[i];
        }
	else {palin=1; break;}
    }
    
    if(palin==0)
	{ long count=0;
          
 	  if((odd==1)&&(arr[n/2]=='?'))count++;
          
	  for(i=0;i<n/2;i++)
          { if(arr[i]=='?')count++;}
            long ans=1;
	   while(count--)
      	  { ans=ans*26;
            ans=ans%10000009;}
         
          printf("%ld\n",ans);
 
        }
    else printf("0\n");
 }
 return 0;
}
