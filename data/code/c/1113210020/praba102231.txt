#include<stdio.h>
    #define MOD 1000000007
    #define MAX 1003
    int ans[MAX][MAX];
    void in()
    {
    int i,j;
    ans[0][0]=1;
    for(i=1;i<MAX;i++)
    {
    ans[i][0]=ans[i-1][i-1];
    for(j=1;j<=i;j++)
    {
    ans[i][j]=(ans[i][j-1]+ans[i-1][j-1])%MOD;
    }
    }
    }
    int main()
    {
    int T,N;
    in();
    scanf("%d",&T);
    while(T--)
    {
    scanf("%d",&N);
    printf("%d\n",ans[N][0]);
    }
    return 0;
    }
