    #include <stdio.h>
    typedef long long int LLI;
    #define inchar getchar_unlocked
    inline int inIntPos() {
    	int ch,n;
    	while((ch = inchar())<'0');
    	n =(ch-'0');
    	while((ch = inchar())>='0')
    		n=(n<<3)+(n<<1)+(ch-'0');
    	return n;
    }
    inline LLI inLLI() {
    	int ch;
    	LLI n;
    	while((ch = inchar())<'0');
    	n =(ch-'0');
    	while((ch = inchar())>='0')
    		n=(n<<3)+(n<<1)+(ch-'0');
    	return n;
    }
    int main() {
    	int t,i,n;
    	LLI res,k,r,a,m;
     
    	t=inIntPos();
    	while(t--) {
    		n=inIntPos();
    		k=inLLI();
     
    		res=0;
    		r=0;
    		for(i=0; i<n; ++i) {
    			a=inIntPos();
     
    			if(r==0) { ++res; r=k; }
     
    			if(r>=a) {
    				r-=a;
    			}
    			else { 
    				a-=r;
    				if(k<a) {
    					m=a/k;
    					res+=m;
    					a-=(m*k);
     
    					if(a>0) {
    						r=k-a;
    						++res;
    					}
    					else r=0;
    				}
    				else {
    					r=k-a;
    					++res;
    				}
    			}
     
    			if(r>=1) --r;
    		}
    		printf("%lld\n",res);
    	}
    	return 0;
    }
