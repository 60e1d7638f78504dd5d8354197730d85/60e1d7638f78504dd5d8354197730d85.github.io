unsigned long long int power(unsigned long long int x, unsigned long long int bin){
	unsigned long long int b;
	if(bin==1){
		
		return x;
	}
	else if(bin%2==0){
		b=power(x,bin/2);
		return b*b;
		}
	else{
		b=power(x,bin/2);
		return b*b*x;
	}
 
}
