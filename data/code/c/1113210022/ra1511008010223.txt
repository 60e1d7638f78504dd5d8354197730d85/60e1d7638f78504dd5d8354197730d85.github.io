#include<stdio.h>
#include<stdbool.h>
#include<math.h>
#define MX 1000001
#define llint long long int
#define dprint if(dbg==1)printf
int dbg=0;
int A[MX];
bool P[MX]={false};
void gen_prime(){
    int i,j;
    P[0]=true;
    P[1]=true;
    for(i=2;i*i<MX;i++){
        if(!P[i]){
            for(j=i*2;j<MX;j+=i)P[j]=true;
        }
    }
}
int main(){
    int ans,t,i,j,lim;
    llint l,r,temp;
    gen_prime();
    scanf("%d",&t);
    while(t--){
        scanf("%lld %lld",&l,&r);
        for(i=0;i<MX;i++)A[i]=0;
        lim=(int)sqrt(r);
        //count no of divisors
        for(i=1;i<=lim;i++){
            temp=(llint)i*(llint)i;
            if(temp>=l)j=(int)(temp-l);
            else j=(i-(int)(l%i))%i;
            /*
            if(i>=l)j=i-l;
            else j=(i-(int)(l%i))%i;
            */
            dprint("Divisor : %d %d\n",i,j);
 
 
            for(;j<=(int)(r-l);j=j+i){
                dprint("\tDivisor : %d %lld\n",i,j+l);
                if((llint)(l+j)==(llint)i*i)A[j]++;
                else A[j]+=2;
 
            }
            dprint("Ai:%d\n",A[i]);
        }
        ans=0;
        for(i=0;(llint)i<=r-l;i++){
            dprint("%lld %d\n",i+l,A[i]);
            if(!P[A[i]]){
                ans++;
 
            }
 
        }
        printf("%d\n",ans);
    }
    return 0;
}
