#include <stdio.h>
int main()
{
  int count = 0,i,j;
	scanf("%d", &count);
	int subCount;
	for(i=0;i<count;i++){
		scanf("%d", &subCount);
		int temp[subCount];
		for(j=0;j<subCount;j++){
			scanf("%d", &temp[j]);
			if(subCount > 1){
				temp[j] = 0;	
			}else{
				temp[j] = temp[j];
			}
		}
		if(subCount > 1){
			printf("%d\n",0);
		}else if(subCount ==1){
			printf("%d\n",temp[0]);
		}else{
			printf("%d\n",0);
		}
    }
	return 0;
}
