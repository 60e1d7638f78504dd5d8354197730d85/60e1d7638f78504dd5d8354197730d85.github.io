#include <stdio.h>
int poss(int a[],int key,int pos);
 
int main()
{
	int k;
	scanf("%d",&k);
	while(k>0)
	{
		int n,arr[20],i,fg,key;
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%d",&arr[i]);
		}
		scanf("%d",&key);
		fg=poss(arr,key,n-1);
		if(fg)
		{
			printf("YES\n");
		}
		else
		{
			printf("NO\n");
		}
		k--;
	}
	return 0;
}
 
int poss(int a[],int key,int pos)
{
	if(key==0)
	{
		return 1;
	}
	if(pos==0)
	{
		if(a[0]==key)
		{
			return 1;
		}
		else
		{
			return 0;
		}
	}
	else
	{
		int m=poss(a,key,pos-1);
		if(m==1)
		return 1;
		int n=poss(a,key-a[pos],pos-1);
		if(n==1)
		return 1;
	}
	return 0;
}
