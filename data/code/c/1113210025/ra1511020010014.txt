#include <stdio.h>
#include<math.h>
int main()
{
 int t;
    
    scanf("%d",&t);
    while(t--) {
        int limit,n,i,j,k,sum;
        
        scanf("%d",&n);
        int a[n];
        limit = (int) pow(2,n);
        
        for(i=0; i<n; scanf("%d",&a[i++]));
        scanf("%d",&sum);
        
        for(i=0; i<limit; i++) {
            int tmpSum=0;
            
            for(k=0,j=1; j<32768; j*=2,k++) {
                if (i&j)
                    tmpSum+=a[k];
                
                if (tmpSum==sum) {
                    printf("YES");
                    goto done;
                }
            }
        }
        printf("NO");
        done:
        printf("\n");
    }
	return 0;
}
