#include <stdio.h>
int main()
{
  int t,n,z;
    long ai;
    
    scanf("%d",&t);
    
    while (t--) {
        scanf("%d%d",&z,&n);
        
        while(n--) {
            scanf("%ld",&ai);
            z&=ai;
        }
        
        if (z==0) {
            printf("Yes\n");
        } else {
            printf("No\n");
        }
    }
	return 0;
}
