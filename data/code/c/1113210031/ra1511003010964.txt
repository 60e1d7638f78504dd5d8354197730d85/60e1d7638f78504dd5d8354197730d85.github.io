#include <stdio.h>
int main()
{
	int i,n;
  	scanf("%d",&n);
  	int a[n];
  	for(i=0;i<n;i++)
    {
      scanf("%d",&a[i]);
    }
  int q;
  	for(i=n-1;i>=0;i--)
    {
      while(a[i]!=0)
      {
        q=a[i]%10;
        printf("%d",q);
        a[i]=a[i]/10;
      }
      printf("\n");
    }
	return 0;
}
