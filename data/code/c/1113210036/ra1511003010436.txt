#include<stdio.h>
 
void binary(unsigned int);   // Prototype Declaration
 
int main()
{
unsigned int num;

 scanf("%u",&num);
binary(num);   // Function Call
return 0;
}
//========================================================
void binary(unsigned int num)
{
unsigned int mask=32768;   //mask = [1000 0000 0000 0000]
 
while(mask > 0)
   {
   if((num & mask) == 0 )
         printf("0");
   else
         printf("1");
  mask = mask >> 1 ;  // Right Shift
   }
}
