void binary(unsigned int);
 
int main()
{
unsigned int num;
scanf("%u",&num);
binary(num);

}
void binary(unsigned int num)
{
unsigned int mask=32768;

while(mask > 0)
   {
   if((num & mask) == 0 )
         printf("0");
   else
         printf("1");
  mask = mask >> 1 ; 
   }
}
