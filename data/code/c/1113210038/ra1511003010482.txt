#include<stdio.h>
#define bool int
 
/* Function to check if x is power of 2*/
bool isPowerOfTwo(int n)
{
  if (n== 0)
    return 0;
  while (n != 1)
  {
    if (n%2 != 0)
      return 0;
    n = n/2;
  }
  return 1;
}
int main()
{	
  int x;
  scanf("%d",&x);
  isPowerOfTwo(x)? printf("power series\n"): printf("not\n");
  return 0;
}
