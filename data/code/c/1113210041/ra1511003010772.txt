#include <stdio.h>
#include<math.h>
int isPrime(int a)
{
  int flag = 0,i,h;
  if(a==1)
    return 0;
  h = sqrt(a);
  for(i=2;i<=h;i++)
  {
    if(a%i == 0)
    {
      flag = 1;
      break;
    }
  }
  if(flag == 1)
    return 0;
  else
    return 1;
}
int main()
{
  int g,a,b,i,j;
  scanf("%d%d",&a,&b);
  for(i=a;i<=b;i++)
  {
    g = isPrime(i);
    if(g==1)
      printf("%d\n",i);
  }
  return 0;
}
