#include <stdio.h>
int countEvenOdd(int n)
{
    int even_count = 0;
    int odd_count = 0;
    while (n > 0) {
        int rem = n % 10;
        if (rem % 2 == 0)
            even_count++;
        else
            odd_count++;
        n = n / 10;
    }
    if (even_count % 2 == 0 && odd_count % 2 != 0)
        return 1;
    else
        return 0;
}
int main()
{
    int n;
    scanf("%d",&n);
    int t = countEvenOdd(n);
    if (t == 1)
        printf("YES");
    else
        printf("NO");
    return 0;
}
