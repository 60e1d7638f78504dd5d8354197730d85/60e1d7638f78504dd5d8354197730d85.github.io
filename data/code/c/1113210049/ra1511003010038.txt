#include <stdio.h>
int main()
{
  int n,a[100],i,c=0;
  scanf("%d",&n);
  for(i=0;i<n;i++)
    scanf("%d",&a[i]);
  if(n==1)
    printf("YES");
  if(n==2)
  {
    for(i=0;i<n;i++)
    {
      if(a[i]==a[i+1])
        c=1;
    }
    if(c==1)
      printf("YES");
    else
      printf("NO");
  }
  return 0;
}
