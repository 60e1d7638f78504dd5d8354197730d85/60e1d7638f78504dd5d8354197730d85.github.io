#include <stdio.h>
int main()
{
  int i,j,temp=0,n;
  scanf("%d", &n);
  int arr[n];
  for(i=0; i<n; i++)
  {
    scanf("%d", &arr[i]);
  }
  for (i=0; i<n; i++)
    {
        for (j=i+1; j<n; j++)
        {
            if (arr[i]>arr[j])
            {
                temp =  arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
  	}
  for(i=0; i<n; i++)
  {
    printf("%d\n", arr[i]);
  }
return 0;
}
