#include <stdio.h>
int main()
{
  int a[100],n,i;
   scanf("%d",&n);
   for(i=0;i<n;i++)
   {
     scanf("%d",&a[i]);
   }
   for(i=0;i<n;i++)
   {int reverse=0,remainder=0;
     while (a[i] > 0)
    {
        remainder = a[i] % 10;
        reverse = reverse * 10 + remainder;
        a[i] /= 10;
       
     }
     a[i]=reverse;
   }
   for(i=0;i<n;i++)
   {
     printf("%d\n",a[i]);
   }
	return 0;
}
