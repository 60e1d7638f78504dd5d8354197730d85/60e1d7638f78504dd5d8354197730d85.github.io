 #include<stdio.h>
     
 int main()
    {
	    int t=0,num=0,temp=0;
     
    	scanf("%d",&t);
     
    while(t--)
    {
   		 scanf("%d",&num);
     
    	while(num)
   	 {
    	temp += num % 10;
    	temp = temp * 10;
    	num /= 10;
    	}
    		printf("%d\n",temp/10);
    		temp=0;
    }
    return 0;
 }
