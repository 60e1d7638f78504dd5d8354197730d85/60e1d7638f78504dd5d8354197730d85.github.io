#include<stdio.h>
int long long lookup[1000000];
int long long euler[1000000];
int main()
{
	int i,j;
	lookup[1]=0;
	for(i=2;i<=1000000;i++)
	{
		euler[i]=i;	
	}
	for(i=2;i<=1000000;i++)
	{
		if(euler[i]==i) // i is prime
		{
			for(j=i;j<=1000000;j=j+i)
			{
				euler[j]=((euler[j])/i)*(i-1);
			}
		}
			lookup[i]=lookup[i-1]+euler[i];
	}
		
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
      if(n==0)
        printf("0");
      else
		printf("%lld\n",lookup[n]);
	}
	return 0;
}
