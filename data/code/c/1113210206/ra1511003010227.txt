#include<stdio.h>
int a[1000001],i,j;
long long int b[1000001];
inline void initialise()
{
a[1]=1;
for(i=2;i<=1000000;i++)
a[i]=i;
}
inline void prepare()
{
for(i=2;i<=1000000;i++)
if(a[i]==i)
{
a[i]=i-1;
for(j=2;j*i<=1000000;j++)
if(a[i*j]==i*j&&a[j]!=j)
{
if(j%i==0)
a[i*j]=i*a[j];
else
a[i*j]=a[i]*a[j];
}
}
}
inline void sum()
{
a[1]=0;
for(i=2;i<=1000000;i++)
b[i]=b[i-1]+a[i];
}
int main()
{
initialise();
prepare();
sum();
scanf("%d",&i);
while(i--)
{
scanf("%d",&j);
printf("%lld\n",b[j]);
}
return 0;
}
