#include<stdio.h>
int fact(int z)
{
	int f=1,i;
	if(z==0 || z==1)
	{
		return 1;
	}
	else
	{
		for(i=1;i<=z;i++)
	{
            f=f*i;
	}
    }
    return(f);
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,m,i,j,k,sum=0,num,c=0;
		scanf("%d %d",&n,&m);
		int a[n][m];
		for(i=0;i<n;i++)
		{
			scanf("%d",&num);
			for(j=m-1;j>=0;j--)
			{
				a[i][j]=num%10;
				num/=10;
			}
		}
		for(i=0;i<m;i++)
	{
		for(j=0;j<n;j++)
		{
			if(a[j][i]==1)
			{
				c++;
			}
		}
		if(c!=1)
		{
		k=fact(c)/(2*fact(c-2));
		sum+=k;
	    }
		c=0;
	}
	printf("%d\n",sum);
	}
	return 0;
}
