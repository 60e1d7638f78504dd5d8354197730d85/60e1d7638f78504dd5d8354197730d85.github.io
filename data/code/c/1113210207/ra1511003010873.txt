#include<stdio.h>
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,m,i,j,count,col=0;
		scanf("%d %d",&n,&m);
		char mat[n][m];
		for(i=0;i<n;i++)
		{
			for(j=0;j<m;j++)
			{
			    scanf(" %c",&mat[i][j]);
			    //printf("%c",mat[i][j]);
			}
			//printf("\n");
			//fflush(stdin);
		}
		for(j=0;j<m;j++)
		{
			count=0;
			for(i=0;i<n;i++)
			{
				if(mat[i][j]=='1')
				{
				    count++;
				    //printf("%d %d\n",i,count);
				}
				
			}
			//printf("%d\n",count);
			col=col+((count*(count-1))/2);
		}
		printf("%d\n",col);
	}
  return 0;
}
