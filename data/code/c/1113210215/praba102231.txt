#include "stdio.h"
int main(){
 
 
	int tc;
	long long i,ans,a,n,k;
	scanf("%d" , &tc);
 
	while(tc--){
 
		scanf("%lld %lld" , &n , &k);
 
		 ans = 0;
		for( i=1;i<=n;i++){
 
			scanf("%lld",&a);
 
			if(a <= 0){
 
				ans = ans + 1;
			}
		}
 
		if(ans < k){
 
            printf("YES\n");
		}
		else{
 
            printf("NO\n");
		}
	}
}
