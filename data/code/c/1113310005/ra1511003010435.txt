#include<stdio.h>
 
main()
{
      int t,n,a,b,i;
      printf(" ");
      scanf("%d",&t);
      for(i=1;i<=t;i++)
      {
                       scanf("%d",&n);
                       a=n/8;
                       b=n%8;
                       switch(b)
                       {case 1:
                             printf("%dLB\n",4+8*a);
                             break;
                       case 2:
                            printf("%dMB\n",5+8*a);
                            break;
                       case 3:
                            printf("%dUB\n",6+8*a);
                            break;
                       case 4:
                            printf("%dLB\n",1+8*a);
                            break;
                       case 5:
                            printf("%dMB\n",2+8*a);
                            break;
                       case 6:
                            printf("%dUB\n",3+8*a);
                            break;
                       case 7:
                            printf("%dSU\n",8+8*a);
                            break;
                       case 0:
                            printf("%dSL\n",7+8*(a-1));
                            break;
                       
                       }
      }
      
      return 0;
}
