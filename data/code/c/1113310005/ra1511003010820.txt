#include<stdio.h>
int main()
{
    int i,t,n;
    scanf("%d",&t);
    for(i=1;i<=t;++i)
    {
        scanf("%d",&n);
        if(n%8<5 && n%8!=0)
        {
            if(n%8==1)
                printf("%dLB\n",n+3);
            if(n%8==2)
                printf("%dMB\n",n+3);
            if(n%8==3)
                printf("%dUB\n",n+3);
            if(n%8==4)
                printf("%dLB\n",n-3);
        }
        else
        {
           if(n%8==5)
                printf("%dMB\n",n-3);
            if(n%8==6)
                printf("%dUB\n",n-3);
            if(n%8==7)
                printf("%dSU\n",n+1);
            if(n%8==0)
                printf("%dSL\n",n-1);
        }
    }
    return 0;
}
