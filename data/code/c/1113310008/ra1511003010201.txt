#include <stdio.h>
int main()
{
int n,temp,j;
  scanf("%d",&n);
  int i,a[12];
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  temp=a[0];
  for(i=0;i<n;i++)
  {
    for(j=0; j<n-i-1; j++)
    if(a[j]>a[j+1])
    {
      temp=a[j];
      a[j]=a[j+1];
      a[j+1]=temp;
    }
  }
  for(i=0;i<n;i++)
  {
    printf("%d ",a[i]);
  }
	return 0;
  
}
