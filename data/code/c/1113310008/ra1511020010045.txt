#include <stdio.h>
int main()
{
int b,a[50],temp,i,j;
  scanf("%d",&b);
  for(i=0;i<b;i++)
  {
    scanf("%d",&a[i]);
  }
  for(i=0;i<b-1;i++)
  {
    for(j=0;j<b-i-1;j++)
    {
      if(a[j]>a[j+1])
         {
           temp=a[j];
           a[j]=a[j+1];
           a[j+1]=temp;
         }
    }
  }
  for(i=0;i<b;i++)
    printf("%d ",a[i]);

	return 0;
}
