#include <stdio.h>
 
int main()
{
    long long int t,a[1000000],c,n;
    scanf("%lld",&t);
    while(t--)
    {
    	long long int ans=0,i,r=0;
    	scanf("%lld%lld%lld%lld",&a[0],&a[1],&c,&n);
    	if(n==1)
    	ans=a[0];
    	else if(n==2)
    	ans=(a[0]*10+a[1]);
    	else
    	{
    			ans=(a[0]*10+a[1]);
    		for(i=2;i<=n-1;i++)
    		{
    			a[i]=(a[i-1]*c+a[i-2])%10;
    			r=ans%41;
    			ans=r*10+a[i];
    			//ans%=41;
    		}
    	}
    	//printf("%lld ",ans);
    	if(ans%41==0)
    	printf("YES\n");
    	else printf("NO\n");
    }
    return 0;
}
