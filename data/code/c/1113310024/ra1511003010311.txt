#include<stdio.h>
int gcd(int A, int B)
{
if(B==0)
return A;
else
return gcd(B, A % B);
}
int main()
{
int n,a[50],g,i;
long long int f=1,ans,c;
c=1000000007;
scanf("%d",&n);
for(i=0;i<n;i++)
scanf("%d",&a[i]);
g=a[0];
for(i=1;i<n;i++)
{
g=gcd(g,a[i]);
}
for(i=0;i<n;i++)
{
f=((f%c)*(a[i]%c))%c;    
}
 
 
ans=1;
while(g!=0)
{
if(g%2==1)
{
ans=(ans*f)%c;
}
f=(f*f)%c;
g=g/2;
}
printf("%lld",ans);
return 0;

}
