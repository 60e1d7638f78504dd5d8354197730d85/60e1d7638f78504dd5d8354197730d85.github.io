#include<stdio.h>
#include<stdlib.h>
int cal_gcd(int A,int B)
{
    int tmp;
    tmp=A;
    A=B;
    B=tmp%B;
    if(B==0)
        return A;
    else
        return cal_gcd(A,B);
}
int main()
{
    int input,gcd,i,*arr;
    unsigned long long mul=1,ans=1;
    scanf("%d",&input);
    arr=(int *)malloc(sizeof(int)*input);
    for(i=0;i<input;i++)
        scanf("%d",&arr[i]);
    gcd=arr[0];
    for(i=0;i<input-1;i++)
    {
        gcd=cal_gcd(gcd,arr[i+1]);
        if(gcd==1)
            break;
    }
    for(i=0;i<input;i++)
    {
        mul=mul*arr[i];
        mul%=1000000007;
    }
    for(i=0;i<gcd;i++)
    {
        ans=ans*mul;
        ans%=1000000007;
    }
    printf("%llu",ans);
 return 0;
}
