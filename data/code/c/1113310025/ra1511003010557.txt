#include<stdio.h>
 
int primes(int k) {
  int p[k],i,j,r;
  p[0]=0;
  p[1]=0;
  p[2]=1;
  for(i=3;i<=k;i++)
    p[i]=-1;
  //for(i=0;i<=k;i++)
  //printf("%d",p[i]);
  for(i=2;i<=k;i++) {
    if(p[i]==-1)
      p[i]=1;
    if(p[i]==1) {
      j=i+i;
      while(j <= k) {
  	p[j]=0;
  	j=j+i;
      }
    }
  }
  int f=0;
  //for(i=0;i<=k;i++)
    //printf("%d ",p[i]);
  for(i=0;i<=k;i++) {
    r=0;
    if(p[i]==1) {
      r=k-i;
      //printf("\n%d ",r);
      if(p[r]==1) {
	f=1;
	return 1;
      }
    }
  }
  if(f==0)
  return 0;
  return 0;
}
 
int main() {
  int t=0;
  scanf("%d",&t);
  while(t--) {
    int k,res=0;
    scanf("%d",&k);
    res=primes(k);
    if(res==1)
      printf("Deepa\n");
    else
    printf("Arjit\n");
  }
  return 0;
}
