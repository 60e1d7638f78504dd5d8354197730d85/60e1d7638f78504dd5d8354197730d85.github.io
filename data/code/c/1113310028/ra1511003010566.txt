#include<stdio.h>
int prime(int n)
{
	int i,res;
	res=n;
	for(i=2;i*i<=n;i++)
	{
		if(n%i==0)
		{
			res=res-res/i;
			while(n%i==0)
			{
				n=n/i;
			}	
		}
	}
	if(n>1)
	{
		res=res-res/n;
	}
	return res;
}
int main()
{
	int m,p,t;
	scanf("%d",&t);
	while(t--)
	{
	scanf("%d",&m);
	p=prime(m);
	printf("%d\n",p);
	}
	return 0;
}
