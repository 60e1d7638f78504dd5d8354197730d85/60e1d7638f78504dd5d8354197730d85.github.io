#include <stdio.h>
int main()
 {
    int data[100],i,n,steps,temp;
    
    scanf("%d",&n);
    for(i=0;i<n;++i)
      {
       
       scanf("%d",&data[i]);
    }
    for(steps=0;steps<n;++steps)
    for(i=steps+1;i<n;++i)
     {
         if(data[steps]>data[i])  

          {
             temp=data[steps];
             data[steps]=data[i]; 
             data[i]=temp;
         }
    }
    
    for(i=0;i<n;++i)
        printf("%d\n",data[i]);
    return 0;
}
