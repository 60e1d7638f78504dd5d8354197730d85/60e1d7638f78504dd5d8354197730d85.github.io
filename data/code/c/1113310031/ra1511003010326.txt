#include <stdio.h>
void swap(int *x,int *y){
	int temp=*x;
 	*x=*y;
  	*y=temp;
}
void bubs(int arr[],int n){
	 int i, j;
   for (i = 0; i < n-1; i++)      
		for (j = 0; j < n-i-1; j++) 
           if (arr[j] > arr[j+1])
              swap(&arr[j], &arr[j+1]);
    
}
int main()
{
	int n,i;
  	scanf("%d",&n);
  	int a[n];
  	for(i=0;i<n;i++){scanf("%d",&a[i]);}
  	bubs(a,n);
  	for(i=0;i<n;i++){
    	printf("%d\n",a[i]);
    }
  	
	return 0;
}
