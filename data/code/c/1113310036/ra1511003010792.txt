#include<stdio.h>
int inp()
{
	int n=0;
	char ch=getchar();
	while(ch>='0' && ch<='9')
	{
		n=(n<<3)+(n<<1)+ch-'0';
		ch=getchar();
	}
	return n;
}
int max(int a,int b)
{
	if(a>b)
	return a;
	return b;
}
