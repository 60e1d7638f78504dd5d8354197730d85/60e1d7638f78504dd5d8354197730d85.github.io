
#include<stdio.h>
int main()
{
	int t,p[10000],l=0,flag,i,j;
	scanf("%d",&t);
	for(i=2;i<=10000;i++)
	{
		flag=0;
		for(j=2;j<=i/j;j++)
		{
			if(i%j==0)
			{
				flag=1;
				break;
			}
		
		}
		if(flag==0)
		{
			p[l]=i;
			l++;
		}
	}
	while(t--)
	{
		int n,s1=1000000000,x,m,temp,sum,r,s2=0,ans;
		scanf("%d",&n);
		int a[n];
		scanf("%d",&a[0]);
		for(i=1;i<n;i++)
		{
			scanf("%d",&x);
			if(x<a[i-1])
			{
				a[i]=a[i-1];
				s2=s2+a[i]-x;
			}
			else
			a[i]=x;
			
		}
		for(m=0;m<l;m++)
		{
			temp=p[m];
			sum=0;
			for(j=0;j<n;j++)
			{
				r=a[j]%temp;
				if(r!=0)
				sum=sum+temp-r;
			}
			if(sum<=s1)
			{
				s1=sum;
			}
			ans=s1+s2;
		}
		printf("%d\n",ans);
	}
	return 0;
}
