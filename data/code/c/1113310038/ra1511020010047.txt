#include<stdio.h>
int main()
{
    int t,i,k,key,n,j,q,temp;
    scanf("%d", &t);
    while(t>0)
    {
        scanf("%d",&n);
        int a[n];
        for(i=0;i<n;i++)
        {
            scanf("%d",&a[i]);
        }
        scanf("%d", &k);
        q=a[k-1];
        for(i=0;i<n;i++)
        {
              for(j=0;j<n-1;j++)
              {
              if(a[j]>a[j+1])
                {
                    temp=a[j];
                    a[j]=a[j+1];
                    a[j+1]=temp;
                }
              }
        }
        for(i=0;i<n;i++)
        {
            if(a[i]==q)
            {
                printf("%d\n",i+1);
                break;
            }
        }
        t--;
    }
        return 0;
}
