int ls(int,int);
int a[102],b[102];
int main()
{int t,n,c=0,i,x;

while(t--)
{scanf("%d",&n);
for(i=0;i<n;i++)
{scanf("%d",&a[i]);}
b[0]=1;
for(i=1;i<n;i++)
{x=ls(a[i],i);b[x]=i+1;if(x<(i-x))c=c+x;else c+=i-x;}

int ls(int n,int l)
{int i,r=0,j;
    if(n==0){for(j=l-1;j>=0;j--){b[j+1]=b[j];}return 0;}
    for(i=0;i<l;i++)
    {if(b[i]==n){r=i+1;break;}}
  if(i!=l)
  {for(j=l-1;j>i;j--)
    {b[j+1]=b[j];}}
    else return l;
    return r;
}
