#include <stdio.h>
 
int main(void) {
    // your code goes here
 
    int t, n, i, j, a, s, k, l;
    int p[124];
        
    scanf ("%d", &t);
    
    while (t--) {
        
        scanf ("%d", &n);
        s = 0;
        
        for (i = 1; i <= n; ++i) {
            scanf ("%d", &a);
            
            if (a == 0) {
                for (j = i; j >= 1; --j) {
                    p [j+1] = p[j];
                }
                
                p[1] = i;
                
            } else {
                for (j = 1; j < i; ++j) {
                    if (p[j] == a) {
                        s += (j < (i - j - 1))?j:(i - j - 1);
                        
                        k = p[++j];
                        p[j] = i;
                        
                        while (j < i) {
                            l = p[++j];
                            p[j] = k;
                            k = l;
                        }
                        
                        break;
                    }
                }
            }
        }
        
        printf ("%d\n", s);
    }
   
    return 0;
}
