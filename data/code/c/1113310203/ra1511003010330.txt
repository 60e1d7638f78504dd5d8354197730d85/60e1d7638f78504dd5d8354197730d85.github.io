#include <stdio.h>  
 
#include <string.h>  

 


int vl,vr,ml,mr;  
int judge(int g, int b)
{  
    if(b<g-1)    return 0;  
    else if(b>2*g+2) return 0;  
    else return 1;  
}  
  
int main()  
{  
    scanf("%d%d%d%d",&vl,&vr,&ml,&mr);  
    if(judge(vl,mr) || judge(vr,ml))   
      printf("YES");  
    else  printf("NO"); 
    return 0;  
}
