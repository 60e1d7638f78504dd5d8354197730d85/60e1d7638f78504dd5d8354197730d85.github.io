#include<stdio.h>
#include<string.h>
int main()
{
    char s[101];
    int t , n ,x,i,p,j;
    scanf("%d",&t);
    while(t--)
    {
        int count=0;
        p=0;
        fflush(stdin);
        scanf("%s",s);
        scanf("%d",&n);
        for(i=0;i<strlen(s);i++)
        {
            if(s[i]=='T')
                p=p+2;
            else
                p=p+1;
        }
        //printf("%d\n",p);
 
 
        for(x=1;x<=12*n;x++)
        {
            i=0;
            for(j=1;(x+j*p)<=n*12;j++)
            {
                count++;
                i=1;
            }
            if(i==0)
                break;
 
        }
        printf("%d\n",count);
 
    }
    return 0;
}
