#include<stdio.h>
#include<string.h>
int main()
{
    long long t;
    scanf("%lld",&t);
    while(t--)
    {
        char s[101];
        long long n,i,j,sum1,sum2=0,ans=0;
        scanf("%s",s);
        scanf("%lld",&n);
        sum1=12*n;
        long long arr[sum1+1];
        long long len=strlen(s);
        for(i=0;i<len;i++)
            if(s[i]=='T')
                sum2+=2;
            else sum2+=1;
        for(i=1;i<=sum1;i++)
            arr[i]=0;
        for(i=1;i<=sum1;i++)
            {
                arr[i]=(sum1-i)/sum2;
                if(arr[i]==0)
                    break;
            }
        for(i=1;i<=sum1;i++)
        {
            //printf("%lld ",arr[i]);
            if(arr[i]!=0)
                ans+=arr[i];
            else break;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
