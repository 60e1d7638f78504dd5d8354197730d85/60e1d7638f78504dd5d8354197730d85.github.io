#include <stdio.h>
 
int main(void) {
	long long int t,N;
	scanf("%lld",&t);
	while(t--){
		char str[101];
		scanf("%s",str);
		scanf("%lld",&N);
		N=12*N;
		int i=0;
		long long int n=0;
		while(str[i]!='\0'){
			if(str[i]=='S')
				n++;
			else
				n=n+2;
				
				i++;
		}
		i=1;
		long long int ans=0;
		while(N-i*n>=0){
			ans=ans+N-(i*n);
			i++;
		}
		printf("%lld\n",ans);
	}
	return 0;
}
