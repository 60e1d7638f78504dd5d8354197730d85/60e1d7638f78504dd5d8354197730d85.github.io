#include<stdio.h>
 
 long long  binarysearch(long long a[], long long  low, long long  high, long long target) {
    while (low <= high) {
        long long  middle = low + (high - low)/2;
        if (target < a[middle])
            high = middle - 1;
        else if (target > a[middle])
            low = middle + 1;
        else
            return middle;
    }
    return -1;
}
