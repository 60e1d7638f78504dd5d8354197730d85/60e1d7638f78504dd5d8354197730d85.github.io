long long binary (long long x[],long long y[],long long low,long long high,long long xo,long long yo,long long n)
{
long long co=0,mid,i;
while(1)
{
if(low<high)
{mid=(low+high)/2;
if(x[mid]==xo)
{
break;
}
else if(x[mid]>xo)
{
high=mid-1;
}
else if(x[mid]<xo)
{
low=mid+1;
}
}
else
break;
 
}
for(i=mid;i<=n;i++)
{
if(x[i]==xo)
{
if(y[i]==yo)
co++;
}
else break;
}
for(i=mid-1;i>=1;i--)
{
if(x[i]==xo)
{
if(y[i]==yo)
co++;
}
else break;
}
return co;
}
