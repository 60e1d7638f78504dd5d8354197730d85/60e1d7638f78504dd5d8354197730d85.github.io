 #include<stdio.h>
    #define MOD 1000000007
    int main()
    { 
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int dp[105][105]={0};
        int i,j,k,n,m;
        scanf("%d%d",&n,&m);
        int x[105];
        int y[105];
        for(i=1;i<=m;i++)
            scanf("%d%d",&x[i],&y[i]);
        dp[0][0]=1;
        for(i=1;i<=m;i++)
            for(j=0;j<=n;j++)
                for(k=x[i];k<=y[i];k++)
                  {if(j>=k)
                    {
                    dp[i][j]+=dp[i-1][j-k];
                    if(dp[i][j]>= MOD)
                        dp[i][j]-= MOD;
                    }
                    else
                        break;
                  }
     
        printf("%d\n",dp[m][n]);
    }
        return 0;
    }
