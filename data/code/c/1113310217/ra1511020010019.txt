
 
#include<stdio.h>

 

void decToBinary(int n)
{

    int binaryNum[1000];
 	int j;
 
    int i = 0;
    while (n > 0) {
 
    
        binaryNum[i] = n % 2;
        n = n / 2;
        i++;
    }

    for (j = i - 1; j >= 0; j--)
        printf("%d", binaryNum[j]);
}
 

int main()
{
    int n;
  	scanf("%d",&n);
    decToBinary(n);
    return 0;
}
