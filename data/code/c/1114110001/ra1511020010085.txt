int main()
{
	int n;
  	scanf("%d",&n);

    {

    }
  	else if(n%100==0)
    {
      printf("%d is not a leap year",n);
    }
  	else if(n%4==0)
    {
      printf("%d is a leap year",n);
    }
  	else
    {
      printf("%d is a leap year",n);
    }
	return 0;
}
