#include <stdio.h>
int main()
{
int n,i,sum=0;
  scanf("%d",&n);
  int a[10];
  for(i=0;i<n;i++)
  { scanf("%d",&a[i]);}
  if(n==1)
  {printf("%d",a[0]);}
  if(n>1)
  {
    for(i=1;i<n;i++)
    {
      if(a[i]==0)
      {
        int j=i;
        while(a[j]==0)
        {j=j-1;}
        a[j]=0;
      }
    }
    for(i=0;i<n;i++)
      sum=sum+a[i];
    printf("%d",sum);
  }
  
	return 0;
}
