#include <stdio.h>
int main()
{
	int a[10];
  	int i,n;
  int sum=0;
  	scanf("%d",&n);
  	for(i=0;i<n;i++)
    {
    	scanf("%d",&a[i]);
    }
  for(i=1;i<n;i++)
    {
    	if(a[i]==0)
        {
          a[i-1]=0; 
        }
   }
  for(i=0;i<n;i++)
    {
 		sum=sum+a[i];
    }
  printf("%d",sum);
  
	return 0;
}
