#include <stdio.h>
int main()
{
  int n,sum=0,i;
  int a[10];
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  for(i=0;i<n;i++)
  {
    if(i!=0)
    {
      sum+=a[i];
      if(a[i]==0)
        sum-=a[i-1];
    }
    else
    {
      sum+=a[i];
    }
  }
  printf("%d",sum);

	return 0;
}
