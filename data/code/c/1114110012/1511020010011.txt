#include <stdio.h>

int main()
{
    int year;

    //printf("Enter a year: ");
    scanf("%d",&year);

    if(year%4 == 0)
    {
        if( year%100 == 0)
        {
            // year is divisible by 400, hence the year is a leap year
            if ( year%400 == 0)
                printf("Leap Year");
            else
                printf("Not a Leap year");
        }
        else
             printf("Leap Year");
    }
    else
         printf("Not a Leap year");
    
    return 0;
}
