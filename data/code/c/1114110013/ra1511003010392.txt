#include <stdio.h>
int main()
{
	float a,b;
  scanf("%f%f",&a,&b);
  if(a>b)
  {
    if(a-b == 0.5)
    {
      printf("Approximate number");
    }
    else
      printf("Not an Approximate number");
  }
  else
  {
    if(b-a <= 0.5)
    {
      printf("Approximate number");
    }
    else
      printf("Not an Approximate number");
  }
	return 0;
}
