#include <stdio.h>
int main()
{
  float num1,num2;
  scanf("%f",&num1);
  scanf("%f",&num2);
  if(num1>=num2)
  {
    if(num1-num2<=0.5)
      printf("Approximate number");
    else
      printf("Not an Approximate number");
  }
  else
  {
    if(num2-num1<=0.5)
      printf("Approximate number");
    else
      printf("Not an Approximate number");
  }
  return 0;
}
