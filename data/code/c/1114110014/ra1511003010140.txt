#include<stdio.h>
int main()
{
int r,a,b,c;
  scanf("%d%d%d%d",&r,&a,&b,&c);
  if(r>a&&a<b&&a<c)
  {
    printf("First person wins!!");
  }
  else if(r>b&&b<a&&b<c)
  {
    printf("Second person wins!!");
  }
  else if(r>c&&c<a&&c<b)
  
    printf("Third person wins!!");
  
  else
  {
    printf("No person wins!!");
  }
	return 0;
}
