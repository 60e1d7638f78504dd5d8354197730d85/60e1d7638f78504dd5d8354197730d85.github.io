#include <stdio.h>
int main()
{
int n,a,b,c;
  scanf("%d %d %d %d",&n,&a,&b,&c);
  if((a<b)&&(a<c)&&(a<n))
     {
       printf("First person wins!!");
     }
     else if((b<a)&&(b<c)&&(b<n))
     {
  printf("Second person wins!!");
     }
  else if((c<a)&&(c<b)&&(c<n))
  {
    printf("Third person wins!!");
  }
  else
  {
    printf("No person wins!!");
  }
	return 0;
}
