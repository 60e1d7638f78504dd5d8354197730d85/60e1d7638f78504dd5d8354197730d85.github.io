#include <stdio.h>
int main()
{
  int t,a,b,c;
 scanf("%d",&t);
  scanf("%d %d %d",&a,&b,&c);
 
  
  if(a<t && a<b && a<c)
    printf("First person wins!!");
  else if(b<t && b<a && b<c)
    printf("Second person wins!!");
  else if(c<t && c<a && c<b)
    printf("Third person wins!!");
  
  else
    printf("No person wins!!");
  
  
	return 0;
}
