#include <stdio.h>
int main()
{int n,a,b,c;
scanf("%d",&n);
 scanf("%d",&a);
 scanf("%d",&b);
 scanf("%d",&c);
if(a<n&&a<b&&a<c){printf("First person wins!!");} 
 else if(b<n&&b<a&&b<c){printf("Second person wins!!");}
 else if(c<n&&c<a&&c<b){printf("Third person wins!!");}
 else printf("No person wins!!");
return 0;
}
