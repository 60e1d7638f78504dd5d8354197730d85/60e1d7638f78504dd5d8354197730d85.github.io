 #include<stdio.h>
int main(){
  int n,sum=0,count=0,i=1,b;
  //printf("Enter the value of n\n");
  scanf("%d",&n);
  while(i<=n)
  {
    //printf("Enter the number\n");
    scanf("%d",&b);
    i++;
    if(b>=0)
    {
      count++;
      sum=sum+b;
    }
  }
  printf("Positive Numbers=%d\n",count);
  printf("Sum=%d",sum);
  return 0;
}
