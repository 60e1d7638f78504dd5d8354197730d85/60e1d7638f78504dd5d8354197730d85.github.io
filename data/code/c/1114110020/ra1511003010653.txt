#include <stdio.h>
int main()
{
  int n,sum=0,count=0, i, a[25];
  scanf("%d", &n);
for(i=0; i<n; i++)
{
  scanf("%d", &a[i]);
}
  for(i=0; i<n; i++)
  {
    if(a[i]>0)
    {
     count++;
      sum= sum+ a[i];
    }
  }
  printf("Positive Numbers=%d\n", count);
  printf("Sum=%d",sum);
  
  	return 0;
}
