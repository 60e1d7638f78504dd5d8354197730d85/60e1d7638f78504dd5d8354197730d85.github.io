#include <stdio.h>
int main()
{
	int n,i,j,count=0,count1=0;
  scanf("%d",&n);
  int a[n];
  for(i=0;i<n;i++)
    scanf("%d",&a[i]);
  for(j=0;j<n;j++)
  {
    if(a[j]>0)
      count++;
      else
        count1++;
  }
  printf("positive numbers count=%d\nnegative number count=%d",count,count1);
	return 0;
}
