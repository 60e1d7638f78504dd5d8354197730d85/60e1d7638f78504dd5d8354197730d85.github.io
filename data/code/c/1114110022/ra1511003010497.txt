#include <stdio.h>
int main()
{
int n,e=0,i,o=0;
  scanf("%d",&n);
  if(n%2==0)
  {
    for(i=1;i<=n;i++)
    {
      if(i%2==0)
        e=e+i;
      else
        o=o+i;
    }
    printf("Sum of all odd numbers=%d\n",o);
    printf("Sum of all even numbers=%d",e);
  }
    else
      printf("Wrong Input");
	return 0;
}
