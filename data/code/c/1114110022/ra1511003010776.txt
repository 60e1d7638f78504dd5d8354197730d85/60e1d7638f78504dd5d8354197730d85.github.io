#include <stdio.h>
int main()
{
  int n,i,oddsum=0,evensum=0;
  scanf("%d",&n);
  if(n%2==1)
  {printf("Wrong Input");}
  else
  {for(i=1;i<=n;i++)
  {
    if(i % 2 == 0)
    {evensum=evensum+i;}
    else
    {oddsum=oddsum+i;}
  }
   printf("Sum of all odd numbers=%d\n",oddsum);
  printf("Sum of all even numbers=%d\n",evensum);
  }
   return 0;
}
