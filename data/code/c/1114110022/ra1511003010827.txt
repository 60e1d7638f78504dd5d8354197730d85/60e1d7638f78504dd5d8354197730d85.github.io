#include <stdio.h>
 
int main()
{
    int i, num, odd_sum = 0, even_sum = 0;
 
    //printf("Enter the value of num\n");
    scanf("%d", &num);
    for (i = 1; i <= num; i++)
    {
        if (i % 2 == 0)
            even_sum = even_sum + i;
        else
            odd_sum = odd_sum + i;
    }
  if(num%2!=0){
    printf("Wrong Input");
}
else
{
    printf("Sum of all odd numbers=%d\n", odd_sum);
    printf("Sum of all even numbers=%d\n", even_sum);
}
return 0;
}
