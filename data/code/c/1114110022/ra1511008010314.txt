#include <stdio.h>
int main()
{
  int i,n,odd=0,even=0;
  scanf("%d",&n);
  if(n%2!=0)
  {
    printf("Wrong Input");
    
  }
  else
  {
  for(i=1;i<=n;i++)
  {
    
    if(i%2==0)
    {
      even+=i;
    }
    else
      odd+=i;
  }
    printf("Sum of all odd numbers=%d\n",odd);
    printf("Sum of all even numbers=%d",even);
  }
 

	return 0;
}
