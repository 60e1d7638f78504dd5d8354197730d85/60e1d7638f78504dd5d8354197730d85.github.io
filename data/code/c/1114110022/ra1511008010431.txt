#include <stdio.h>
int main()
{
  
    int i, num, odd_sum = 0, even_sum = 0;
    scanf("%d", &num);
  if(num%2!=0)
  {
    printf("Wrong Input");
  }
  else
  {
    for (i = 1; i <= num; i++)
    {
        if (i % 2 == 0)
            even_sum = even_sum + i;
        else
            odd_sum = odd_sum + i;
    }
 
    printf("Sum of all odd numbers=%d", odd_sum);
    printf("\nSum of all even numbers=%d", even_sum);
  }
    return 0;
}
