#include <stdio.h>
int main()
{
  int  n, i, odd=0, even=0;
	scanf("%d",&n);
  
  	if (n%2==0){
  	for(i=1; i<=n; i++)
      if(i%2==0)
        even=even+i;
        else
          odd=odd+i;
      printf("Sum of all odd numbers=%d\n",odd);
  printf("Sum of all even numbers=%d",even);
    }
  
  	else 
      printf("Wrong Input");
 
  
	return 0;
}
