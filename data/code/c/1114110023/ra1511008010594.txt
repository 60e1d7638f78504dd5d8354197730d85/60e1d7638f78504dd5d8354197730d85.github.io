#include<stdio.h>
int fact(int n)
{
    int i=1,res=1;
    for(i=1;i<=n;i++)
        res*=i;
    return res;
}
int main()
{
    int n,k,d1,d2,d3;
    scanf("%d%d",&n,&k);
    if(k>n)
        printf("0\n");
    else
    {
        d1=fact(n);
        d2=fact(k);
        d3=fact(n-k);
        d1=d1/(d2*d3);
        printf("%d\n",d1);
    }
    return 0;
}
