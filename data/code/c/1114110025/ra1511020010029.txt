#include <stdio.h>
int main ()
{
    printf("1\n");
    printf("3 1 2 2\n");
    printf("3 3 4 4\n");
    int n,f;
    scanf("%d",&n);
    printf("2\n");
    if(n==0)
        f=5;
    else if(n==(-1))
        f=3;
    else if(n==(-2))
        f=4;
    else if(n==1)
        f=1;
    else if(n==2)
        f=2;
    else if(n==3)
        return 0;
    printf("%d\n",f);
return 0;
}
