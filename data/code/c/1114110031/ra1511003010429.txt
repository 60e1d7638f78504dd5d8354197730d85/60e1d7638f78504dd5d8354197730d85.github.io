#include <stdio.h>

int main()
{
    int rows, i, j;

    /* Input rows and columns from user */
    
    scanf("%d", &rows);
    
    

    for(i=1; i<=rows; i++)
    {
        for(j=1; j<=rows; j++)
        {
            // Print the current column number
            printf("%d", j);
        }

        printf("\n");
    }

    return 0;
}
