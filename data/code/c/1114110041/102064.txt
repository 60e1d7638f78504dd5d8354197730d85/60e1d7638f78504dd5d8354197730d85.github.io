#include <stdio.h>
int main()
{
int num, temp, remainder, reverse = 0;
 
       scanf("%d", &num);
    /*  original number is stored at temp */
    temp = num;
    while (num > 0)
    {
        remainder = num % 10;
        reverse = reverse * 10 + remainder;
        num /= 10;
    }
    
    printf("%d", reverse);
	return 0;
}
