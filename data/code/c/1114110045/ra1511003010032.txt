#include <stdio.h>
 
int main()
{
   int array[10],position,c,n,value;
   scanf("%d",&n);
 
   for (c=0;c<n;c++)
   scanf("%d",&array[c]);
 
   scanf("%d",&value);
   scanf("%d",&position);
 
   for (c=n-1;c>=position-1;c--)
   {
     array[c+1] = array[c];
   }
   array[position-1]=value;
   
    for (c = 0; c <= n; c++)
    {
      printf("%d ", array[c]);
    }
    return 0;
}
