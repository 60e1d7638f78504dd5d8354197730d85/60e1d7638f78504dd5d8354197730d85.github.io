#include <stdio.h>
int main()
{
  int arr[10],num,i,location,element;
  scanf("%d",&num);
  for(i=0;i<num;i++)
    scanf("%d",&arr[i]);
  scanf("%d",&element);
  scanf("%d",&location);
for (i = num; i >= location; i--) {
      arr[i] = arr[i - 1];
   }
   num++;
   arr[location - 1] = element;
  
  for(i=0;i<num;i++)
    printf("%d ",arr[i]);
	return 0;
}
