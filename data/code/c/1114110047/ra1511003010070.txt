#include <stdio.h>
int main()
{
    long bin1,bin2;

    while(bin1!=0 || bin2!=0)
    {
      sum[i++]=(bin1%10 + bin2%10 + rem)%2;
      rem=(bin1%10 + bin2%10 +rem)/2;
      bin1=bin1/10;

    }
   if(rem!=0)
     sum[i++]=rem;
  --i;

      printf("%d",sum[i--]);
	return 0;
}
