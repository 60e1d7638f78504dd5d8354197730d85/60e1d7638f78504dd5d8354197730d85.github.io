#include <stdio.h>
#include<string.h>
int main()
{
  int num, temp, remainder, reverse = 0;
    scanf("%d", &num);
    temp = num;
    while (num > 0)
    {
        remainder = num % 10;
        reverse = reverse * 10 + remainder;
        num /= 10;
    }
  while(reverse>0)
  {
    printf("%d ",reverse%10);
    reverse/=10;
  }
}
