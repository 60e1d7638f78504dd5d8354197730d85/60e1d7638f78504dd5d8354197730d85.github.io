#include<stdio.h>
int main()
{
    int t;
    int n;
    int f;
    int i,j,k;
    scanf("%d",&t);
    for(i=0;i<t;i++){
     scanf("%d",&n);
     int a[n];
     for(j=0;j<n;j++){
        scanf("%d",&a[j]);
     }
     scanf("%d",&f);
     int b[f];
     for(j=0;j<f;j++){
        scanf("%d",&b[j]);
     }
     int k=0;
    for(j=0;j<n;j++){
        if(a[j]==b[k]){
          k++;
        }
        if(k==f){
        break;
     }
        }
       if(k==f){
        printf("Yes\n");
       }
       else{
        printf("No\n");
       }
    }
    return 0;
    }
