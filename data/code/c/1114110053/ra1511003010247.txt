#include <stdio.h>
int main()
{
  int t;
  scanf("%d",&t);
  while(t--)
  {
    int n,r,i,l;
    scanf("%d",&n);
    int ar[n];
    for(i=0;i<n;i++)
    {
      scanf("%d",&ar[i]);
    }
    scanf("%d",&r);
    int arr[r];
    for(i=0;i<r;i++)
    {
      scanf("%d",&arr[i]);
    }
    l=0;
    for(i=0;i<n;i++)
    {
      if(ar[i]==arr[l])
        l++;
      if(l==r)
        break;
    }
    if(l==r)
      printf("Yes\n");
    else
      printf("No\n");
  }
	return 0;
}
