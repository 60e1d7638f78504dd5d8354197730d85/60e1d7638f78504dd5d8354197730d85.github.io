#include <stdio.h>
 
int main(void) {
    int T,length1,length2,i,j;
    scanf("%d",&T);
    while(T--)
    {
       scanf("%d",&length1); 
       int seq1[length1];
       for(i=0;i<length1;i++)
       scanf("%d",&seq1[i]);
       
       scanf("%d",&length2); 
       int seq2[length2];
       for(i=0;i<length2;i++)
       scanf("%d",&seq2[i]);
       j=0;
       for(i=0;i<length1;i++)
       {
          if(seq1[i]==seq2[j])j++;
          if(j==length2)
          break;
       }
         printf("%s\n",j==length2?"Yes":"No");
    }
	return 0;
}
