#include<stdio.h>
 
int main ( void )
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int m;
		scanf("%d", &m);
		int txt[m], i;
		for( i = 0 ; i < m ; i++ )
		{
			scanf("%d", txt + i );
		}
		int n;
		scanf("%d", &n);
		int seq[n];
		for( i = 0 ; i < n ; i++ )
		{
			scanf("%d", seq + i );
		}
		int j = 0, k;
 
		for( k = 0 ; k < m ; k++ )
		{
			if( txt[k] == seq[j])
			{
				j++;
			}
			if( j == n ) 
			{
				printf("Yes");
				break;
			}
		}
		if( j < n ) printf("No");
		printf("\n");
	}
return 0;
}
