#include <stdio.h>
 
int main(void) {
	int t;
	scanf("%d",&t);
	while(t--){
	    int n,i;
	    scanf("%d",&n);
	    int a[n],k=0;
	    for(i=0;i<n;i++)
	    scanf("%d",a+i);
    	int m;
    	scanf("%d",&m);
    	int b[m];
    	for(i=0;i<m;i++)
	    scanf("%d",b+i);
	    
	    for(i=0;i<n;i++)    {
	        if(a[i]==b[k]){
	            k++;
	        }
	        if(k==m){
	            break;
	        }
	}
	if(k==m){
	    printf("Yes\n");
	}
	else{
	    printf("No\n");
	}
	}
	return 0;
}
