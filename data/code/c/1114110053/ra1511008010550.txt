#include <stdio.h>
int main()
{
 int a,x,m,y,op[10000],n,od[10000],i,j,flag,flag1,count;
	scanf("%d",&a);		
	for(x=0;x<a;x++)
	{
		scanf("%d",&m);
		for(y=0;y<m;y++)
		{
			scanf("%d",&op[y]);
		}
		scanf("%d",&n);
		for(y=0;y<n;y++)
		{
			scanf("%d",&od[y]);
		}
		j=0;
		for(i=0;i<m;i++)
		{
			if(op[i]!=od[j])
			{
				continue;
			}
			j++;
			if(j==n)
			break;
		}
				
								
		if(j==n)
		printf("Yes\n");
		else
		printf("No\n");
    }
	return 0;
}
