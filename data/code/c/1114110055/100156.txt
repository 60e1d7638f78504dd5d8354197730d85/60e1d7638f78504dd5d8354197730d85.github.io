#include<stdio.h>
#include <string.h>
int main()
{
    char s[105];
    int t,l,i,ac;
    scanf("%d",&t);
    while(t--)
    {
        ac=0;
        scanf("%s",s);
        l=strlen(s);
        if(s[0]!=s[1])
            ac=1;
        for(i=2;i<l;i++)
        {
            if(s[i]!=s[i-2])
            {
                ac=0;
                break;
            }
        }
 
        if(ac==1)
            printf("YES\n");
        else
            printf("NO\n");
 
    }
    return 0;
}
