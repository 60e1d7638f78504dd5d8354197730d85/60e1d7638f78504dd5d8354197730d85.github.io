#include <stdio.h>
#include <string.h>
#include <malloc.h>
int main()
{
int t,i;
   char str[100];
   scanf("%d",&t);
   while (t--)
   {
      scanf("%s",str);
      int len = strlen(str);
      if(len==2)
      {
      	if(str[0]!=str[1])
      		printf("YES\n");
      	else
      		printf("NO\n");
      }
      else
      {
      	if(str[0]!=str[1])	
      	{
      		int flag=0;
      		for(i=0;i<len-1;i++)
      		{
      			if(i%2==0)
      			{
      				if(str[0]!=str[i] || str[1]!=str[i+1])
      				{
      					flag=1;
      					break;
      				}
      			}
      			else
      			{
      				if(str[0]!=str[i+1]|| str[1]!=str[i])
      				{
      					flag=1;
      					break;
      				}
      			}
      		}
      		if(flag==0)
      			printf("YES\n");
      		else
      			printf("NO\n");
      	}
      	else
      		printf("NO\n");
      }
   }
	return 0;
}
