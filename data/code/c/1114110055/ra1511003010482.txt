#include <stdio.h>
#include<string.h>
int main()
{	
  int i,t,p,q,n,flag;
  char a[101];
  scanf("%d",&t);
  
  while(t--)
  {
    flag=0;
    scanf("%s",a);
    n=strlen(a);
    p=a[0];
    q=a[1];
    if(p==q)
    {
      flag=1;
    }
    else
      for(i=2;i<n;i++)
      {
        if(i%2==0)
        {
        	if(a[i]!=p)
            {
            	flag=1;
            	break;
            }
        }
        else
        {
          if(a[i]!=q)
          {
            flag=1;
            break;
          }
        }
      }
    	if(flag==0)
        {
          printf("YES\n");
        }
    else
    {
      printf("NO\n");
    }
  }

	return 0;
}
