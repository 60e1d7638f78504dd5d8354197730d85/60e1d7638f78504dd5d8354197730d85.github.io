#include <stdio.h>
int main(){
        long long int c, d, l, t, max, min;
        scanf("%lld", &t);
        while(t--){
                scanf("%lld%lld%lld", &c, &d, &l);
                max = c*4 + d*4;
                if(c <= 2*d){
                        min = d*4;
                }
                else if(c > 2*d){
                        min = d*4 + (c-2*d)*4;
                }
                if(l >= min && l <= max && l % 4 == 0){
                	printf("yes\n");
                }
                else
                        printf("no\n");
        }
        return 0;
}
