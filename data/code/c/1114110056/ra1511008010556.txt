#include <stdio.h>
int main()
{
  int cats,dogs,legs,t;
scanf("%d",&t);
while(t>0)
{
scanf("%d",&cats);
scanf("%d",&dogs);
scanf("%d",&legs);
if(cats<=2*dogs)
{
if((legs<=4*(cats+dogs))&&(legs>=4*dogs)&&((legs%4)==0))
printf("yes\n");
else
printf("no\n");
}
if(cats>2*dogs)
{
if((legs<=4*(cats+dogs))&&(legs>=4*(dogs+(cats-2*dogs)))&&((legs%4)==0))
printf("yes\n");
else
printf("no\n");
}
t=t-1;
}
	return 0;
}
