#include <stdio.h>
#include <math.h>

int main()
{
    int num, swappedNum;
    int firstDigit, lastDigit, digits;
    scanf("%d", &num);
    lastDigit  = num % 10; 
    digits     = (int) log10(num); 
    firstDigit = (int) (num / pow(10, digits));
    swappedNum  = lastDigit;
    swappedNum *= (int) round(pow(10, digits));
    swappedNum += num % ((int)round(pow(10, digits)));
    swappedNum -= lastDigit;
    swappedNum += firstDigit;
    printf("%d", swappedNum);
    return 0;
}
