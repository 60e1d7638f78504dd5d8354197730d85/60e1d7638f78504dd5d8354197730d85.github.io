#include <stdio.h>
int main()
{
    int n, n_bak, digits[10], count = 0, i;
    
    scanf("%d", &n);
    n_bak = n;

   
    while(n) {
        digits[count ++] = n % 10;
        n /= 10;
    }
    
    
    printf("\n%d", digits[0]);
    for(i = count - 2; i > 0; i --) {
        printf("%d", digits[i]);
    }
    printf("%d", digits[count - 1]);

    puts(""); 

	return 0;
}
