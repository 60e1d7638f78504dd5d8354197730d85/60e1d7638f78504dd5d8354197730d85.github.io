#include <stdio.h>
int main()
{
     int num, decimal_num, remainder, base = 1, binary = 0;
 
  
    scanf("%d", &num);
    decimal_num = num;
    while (num > 0)
    {
        remainder = num % 2;
       
        binary = binary + remainder * base;
        num = num / 2;
        base = base * 10;
    }
 
    printf("%d\n", binary);
   
	return 0;
}
