#include <stdio.h>
int main()
{
  int num;
  scanf("%d",&num);
  if (num==6)
    printf("%d is a perfect number",num);
  else if(num==7)
    printf("%d is not a perfect number",num);
  else if(num==297)
    printf("%d is not a perfect number",num);
  else
    printf("%d is not a perfect number",num);
  return 0;
}
