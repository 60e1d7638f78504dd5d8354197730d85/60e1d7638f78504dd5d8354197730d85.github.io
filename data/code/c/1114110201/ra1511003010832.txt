#include <stdio.h>
int main()
{
int number, rem, sum = 0, i;
 
    scanf("%d\n", &number);
    for (i = 1; i <= (number - 1); i++)
    {
        rem = number % i;
	if (rem == 0)
        {
            sum = sum + i;
        }
    }
    if (sum == number)
        printf("%d is a perfect number",number);
    else
        printf("%d is not a perfect number",number);
	return 0;
}
