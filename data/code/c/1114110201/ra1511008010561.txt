#include <stdio.h>
int main()
{
     int num,i,sum=0;
    scanf("%d",&num);
    for(i=1;i<num;i++)
    {
      if(num%i==0)
     {
        sum=sum+i;
         
     }
    }
      if(num==sum)
      {
        printf("%d is a perfect number",num);
      }
      else
      {
         printf("%d is not a perfect number",num);
      }
  
	return 0;
}
