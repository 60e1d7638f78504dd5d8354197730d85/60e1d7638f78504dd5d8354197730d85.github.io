#include <stdio.h>
int main()
{
int t,n,i,j,ta,k;
	scanf("%d",&t);
	while(t--)
	{ ta=0;
	  scanf("%d",&n);
	  int ar[n];
	  for(i=0;i<n;i++)
	   scanf("%d",&ar[i]);
	  for(i=0;i<n;i++)
	  {  k=1;
	     for(j=i+1;j<n;j++)
	      {  if(ar[i]==ar[j])
	          k=0;
		    }
		   if(k==1) 
		    ta++;
		   }
		printf("%d\n",ta);    	
    }
	return 0;
}
