#include<stdio.h>
int main()
{
    int t, n, cnt, i, p; 
    scanf("%d",&t); 
    while(t--)
    {
        scanf("%d",&n); 
        int a[101]={0};
        cnt=0;
        for(i=0;i<n;i++)
        {
            scanf("%d",&p); 
            if(a[p]==0)
            {
                 a[p]=1;
                 cnt++;
            }
        }
        printf("%d\n",cnt);
    }
    return 0;
}
