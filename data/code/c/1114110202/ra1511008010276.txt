#include <stdio.h>
int main()
{
    int T,i,n,j,k,m;
   scanf("%d",&T);
   for(i=0;i<T;i++)
   {
     scanf("%d",&n);
     int arr[n],count=0;
     for(j=0;j<n;j++)
     {
       scanf("%d",&arr[j]);
     }
     for(j=0;j<n-1;j++){
       for(k=j+1;k<=n-1;k++){
         if(arr[j]==arr[k])
           count++;
       }
     }
       printf("%d\n",n-count);
   }
	return 0;
}
