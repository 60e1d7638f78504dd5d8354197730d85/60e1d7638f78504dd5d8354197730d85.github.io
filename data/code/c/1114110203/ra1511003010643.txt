#include<stdio.h>
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int x,y,i,n,ans=1;
        scanf("%d",&n);
        for(i=0;i<n;i++){
            scanf("%d%d",&x,&y);
            
        }
        for(i=2;i<=n;i++)
            ans ^= i;
        printf("%d\n",ans);
    }
    return 0;
}
