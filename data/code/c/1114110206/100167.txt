#include <stdio.h>
int main()
{
  int t,horses[5000],n,i,j,k,diff,temp,gaps[8]={701,301,132,57,23,10,4,1};
    scanf("%d",&t);
               while(t--)
               {
                 scanf("%d",&n);
                 for(i=0;i<n;i++)
                   scanf("%d",&horses[i]);
                 for(k=0;k<8;k++)
                 for(i=gaps[k];i<n;++i){
                   temp=horses[i];
                   for(j=i;j>=gaps[k]&&horses[j-gaps[k]]>temp;j-=gaps[k])
                       horses[j]=horses[j-gaps[k]];
                       horses[j]=temp;
                       }
                       diff=horses[1]-horses[0];
                       for(i=1;i<n;i++)
                       if(horses[i]-horses[i-1]<diff)
                       diff=horses[i]-horses[i-1];
                       printf("%d\n",diff);
                       }
                       
	return 0;
}
