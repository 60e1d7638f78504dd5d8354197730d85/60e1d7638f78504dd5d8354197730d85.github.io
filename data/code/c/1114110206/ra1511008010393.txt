#include <stdio.h>
int main()
{
  int z;
  scanf("%d",&z);
        while(z--)
        {
          int n,temp,diff;
          scanf("%d",&n);
          int a[n],i,j;
          for(i=0;i<n;i++)
          {
            scanf("%d",&a[i]);
          }
          for(i=0;i<n;i++)
          {
            for(j=0;j<n-1;j++)
            {
              if(a[j]>a[j+1])
              {
                temp=a[j];
                a[j]=a[j+1];
                a[j+1]=temp;
              }
            }
          }
          int min=abs(a[1]-a[0]);
          for(i=1;i<n;i++)
          {
            diff=abs(a[i]-a[i-1]);
            if(min>diff)
              min=diff;
          }
          printf("%d\n",min);
        }
	return 0;
}
