#include <stdio.h>
 
int main()
{
	int t, n;
	int temp, j, i, index;
 
	scanf("%d", &t);
 
	while(t--)
	{
		scanf("%d", &n);
		int s[n];
 
		for(i=0;i<n;i++)
			scanf("%d", &s[i]);
		
		int min;
 
		for(i=0; i<n; i++)
		{
			for(j=i+1; j<n; j++)
			{
				if(s[i]>s[j])
				{
					temp = s[j];
					s[j] = s[i];
					s[i] = temp;
				}
			}
		}
 
		min = s[1]-s[0];
		for(i=0;i<n-1;i++)
		{
			if((min>s[i+1]-s[i]))
				min = s[i+1] - s[i];
		}
		printf("%d\n", min);
	}
	return 0;
}
