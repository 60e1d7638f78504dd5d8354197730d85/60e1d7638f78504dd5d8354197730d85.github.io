#include <stdio.h>
int main()
{
   int i, j, a[10][10], sum, rows, columns;
   scanf("%d", &rows); 
   scanf("%d", &columns);   
   for (i = 0; i < rows; i++)
      for (j = 0; j < columns; j++) {
         scanf("%d", &a[i][j]);
      }
   sum = 0;
  if(rows==columns)
  {
   for (i = 0; i < rows; i++)
      for (j = 0; j < columns; j++) {
        
         if (i < j) {
            sum = sum + a[i][j];
         }
      }
   printf("%d", sum);
  }
  else
  {
    printf("entered row and columns is not square matrix");
  }
    return 0;
}
