#include <stdio.h>
int main()
{
int i, j, a[10][10], sum, rows, columns;
 
 
   scanf("%d", &rows);
 
   scanf("%d", &columns);
  if(rows!=columns){printf("entered row and columns is not square matrix");}
else{
   for (i = 0; i < rows; i++)
      for (j = 0; j < columns; j++) {
         scanf("%d", &a[i][j]);
      }
   sum = 0;
   for (i = 0; i < rows; i++)
      for (j = 0; j < columns; j++) {
         if (i < j) {
            sum = sum + a[i][j];
         }
      }
 
   printf("%d", sum);}
	return 0;
}
