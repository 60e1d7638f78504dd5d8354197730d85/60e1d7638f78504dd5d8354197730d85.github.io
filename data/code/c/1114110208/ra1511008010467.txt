#include <stdio.h>
int main()
{
	int r,c,a[40][40],i,j,sum=0;
  	scanf("%d",&r);
  	scanf("%d",&c);
  	if(r==c)
    {
     	for(i=0;i<r;i++)
        {
          	for(j=0;j<c;j++)
            {
            	scanf("%d",&a[i][j]);
            }
        }
      	for(i=0;i<r;i++)
        {
          for(j=0;j<c;j++)
          {
           	 if(i<j)
             {
               	sum+=a[i][j];
             }
          }
    	}
    printf("%d",sum);
    }
  	else
    {
     	printf("entered row and columns is not square matrix");
    }
	return 0;
}
