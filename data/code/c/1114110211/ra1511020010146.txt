#include <stdio.h>
int main()
{
	int n,i,j,c;
  	scanf("%d",&n);
  	int a[n];
  	for(i=0; i<n; i++)
      scanf ("%d",&a[i]);
  for(i=0; i<n; i++)
  {
    c=1;
    for(j=i; j<n; j++)
    {
      if(a[i]==a[j] && a[i]!=0 && i!=j)
      {
        c++;
        a[j]=0;
      }
    }
    if(c>1)
      printf("%d ",a[i]);
  }
    
	return 0;
}
