#include <stdio.h>
int main()
{
  int array[100], maximum, size, c, location = 0;
 
  
  scanf("%d", &size);
 

 
  for (c = 0; c < size; c++)
    scanf("%d", &array[c]);
 
  maximum = array[0];
 
  for (c = 1; c < size; c++)
  {
    if (array[c] > maximum)
    {
       maximum  = array[c];
       location = c;
    }
  }
 
  printf("Peak Element=%d\n",maximum);
  printf("Index Value=%d\n",location);
  
  return 0;
}
