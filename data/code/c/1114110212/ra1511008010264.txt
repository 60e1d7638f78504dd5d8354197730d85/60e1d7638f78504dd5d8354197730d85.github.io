#include<stdio.h>
int main()
{
  int i,n;
  scanf("%d",&n);
  int a[n];
  
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
    
  }
  int count=0;
  int max=a[0];
  for(i=0;i<n;i++)
  {
   if(a[i]>max)
   {
     max=a[i];
     count=i;
   }
    
  }
  printf("Peak Element=%d\n",max);
   printf("Index Value=%d",count);
  
  return 0;
}
