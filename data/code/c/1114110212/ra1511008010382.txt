#include <stdio.h>
int main()
{
  int n,i,index,max;
  scanf("%d",&n);
  
  int a[n];
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  
  max=a[0];
  for(i=0;i<n;i++)
  {
    if(a[i]>max)
    {
      max=a[i];
      index=i;
    }
  }
  printf("Peak Element=%d\n",max);
  printf("Index Value=%d\n",index);
    

	return 0;
}
