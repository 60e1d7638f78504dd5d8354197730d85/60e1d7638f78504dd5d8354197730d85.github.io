#include <stdio.h>
int main()
{
int n,i,max,index;
  scanf("%d",&n);
  int a[n];
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  for(i=0;i<n;i++)
  {
    if(a[i]>max)
    {
      max=a[i];
      index=i;
    }
  }
  printf("Peak Element=%d\n",max);
  printf("Index Value=%d\n",index);
	return 0;
}
