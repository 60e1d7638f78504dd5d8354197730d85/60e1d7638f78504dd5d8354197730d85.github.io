#include<stdio.h>
void merge(int c[], int low, int mid, int high)
{
    int i=low,j=mid+1,k=low,b[100000];
    while(i<=mid&&j<=high)
    {
        if(c[i]<=c[j])
        {
            b[k]=c[i];
            i++;
        }
        else
        {
            b[k]=c[j];
            j++;
        }
        k++;
    }
    for(;i<=mid;i++)
    {
        b[k]=c[i];
        k++;
    }
    for(;j<=high;j++)
    {
        b[k]=c[j];
        k++;
    }
    for(i=low;i<=high;i++)
        c[i]=b[i];
}
void merge_sort(int c[],int low, int high)
{
    int mid;
    if(low<high)
    {
        mid=(low+high)/2;
        merge_sort(c,low,mid);
        merge_sort(c,mid+1,high);
        merge(c,low,mid,high);
    }
}
