#include<stdio.h>
 
typedef long long unsigned llu;
 
const int m = 2*(1000000000+7);
 
llu fastf(llu n) {
	// exp. matrix
	llu a=1, b=1, c=0;
 
	// act matrix
	llu pa=1, pb=0, pc=1;
	while(n>0) {
		if(n%2==1) {
			llu npa = pa*a + pb*b;
			llu npb = pa*b + pb*c;
			llu npc = pb*b + pc*c;
			pa = npa % m;
			pb = npb % m;
			pc = npc % m;
		}
 
		llu na = a*a + b*b, nb = a*b + b*c, nc = b*b + c*c;
		a = na % m;
		b = nb % m;
		c = nc % m;
 
		n /= 2;
	}
	return pb;
}
 
 
int main() {
	int t;
	scanf("%d", &t);
 
	while(t-->0) {
		llu h;
		scanf("%llu", &h);
		
		llu r = fastf(h);
		r = r/2 + r%2;
		r = r % (m/2);
	
		printf("%llu\n", r);
	}
}
