#include <stdio.h>
int main()
{
  int counter, N, i, isPrime, primeFactorSum = 0;
   scanf("%d", &N);   
 
       for(counter = 2; counter <= N; counter++) {
     isPrime = 1;
        for(i = 2; i <=(counter/2); ++i) {
            if(counter%i==0) {
                isPrime = 0;
                break;
            }
        }
    
        if(isPrime==1)
            primeFactorSum += counter;
    }
    
    printf("%d",primeFactorSum);
    

	return 0;
}
