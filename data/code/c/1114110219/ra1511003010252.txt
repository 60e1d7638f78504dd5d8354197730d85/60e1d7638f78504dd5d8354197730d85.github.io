#include<stdio.h>
#include<math.h>
int main()
{
    int T,N,i,f,h;
    scanf("%d",&T);
    while(T--)
    {
        f=0,h=0;
        scanf("%d",&N);
        if(N%2==0&&N>2)
        {
            printf("no\n");
            h++;
        }
 
        else
        {
             for(i=2;i<=sqrt(N);i++)
        {
            if(N%i==0)
            {
                f++;
            }
        }
        }
        if(f>0&&h==0)
        {
            printf("no\n");
        }
        if(f==0&&h==0)
        {
            printf("yes\n");
        }
    }
    return 0;
}
