#include<stdio.h>
int main()
{
    int t,i,s;
    float gs;
    scanf("%d",&t);
    for(i=1;i<=t;i++)
    {
        scanf("%d",&s);
        if(s<1500)
        {
            gs=2*s;
        }
        else
        {
            gs=s+500+(0.98*s);
        }
        printf("%.2f\n",gs);
    }
    return 0;
}
