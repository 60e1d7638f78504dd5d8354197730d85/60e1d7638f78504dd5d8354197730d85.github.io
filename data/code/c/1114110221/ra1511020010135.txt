#include<stdio.h>
 
 
int main()
{
    int t;
    scanf("%d",&t);
  while(t--){
        float gross_salary;
        int salary;
        scanf("%d",&salary);
        if(salary<1500)
        {
            gross_salary = salary+0.1*salary+0.9*salary;
        }
        else
        {
            gross_salary= salary+500+0.98*salary;
        }
        
        printf("%.2f\n",gross_salary);
  }
    return 0;
}
