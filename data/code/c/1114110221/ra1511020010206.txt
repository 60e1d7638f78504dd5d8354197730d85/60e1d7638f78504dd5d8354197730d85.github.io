#include <stdio.h>
int main()
{
	int t,bs,i;
  	float hra,da,gs;
  	scanf("%d",&t);
  	for(i=1;i<=t;i++)
    {
    	scanf("%d",&bs);
      	if(bs<1500)
        {
        	hra=0.1*bs;
          	da=0.9*bs;
        }
      	else
        {
        	hra=500.0;
          	da=0.98*bs;
        }
      	gs=hra+da+bs;
      	printf("%.2f\n",gs);
    }
	return 0;
}
