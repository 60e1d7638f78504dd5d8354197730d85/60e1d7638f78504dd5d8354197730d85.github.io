#include <stdio.h>

int sum(int a,int b)
{
  int s=0,i;
  if(a==1)
  {
    for(i=1;i<=b;i++)
    s+=i;
    return s;
  }
  else
  {
  while(a>0)
  {
    b=sum(1,b);
    a--;
  }
  
  
  return b;
  }
}


    
int main()
{

  int a,b,n,i;
  
  scanf("%d",&n);
  
  for(i=0;i<n;i++)
  {
  scanf("%d %d",&a,&b);
  int ans=sum(a,b);
  
  printf("%d\n",ans);
