#include <stdio.h>
int sum(int n){
	int i,sum=0;
  	for(i=1;i<=n;i++){
    	sum+=i;
    }
  	return sum;
}
int main()
{
	int t,n,d,i,ans;
  	scanf("%d",&t);
  	while(t){
    	scanf("%d %d",&d,&n);
      	for(i=0;i<d;i++){
        	ans=sum(n);
          	n=ans;
        }
      	printf("%d\n",ans);
    	t--;
    }
	return 0;
}
