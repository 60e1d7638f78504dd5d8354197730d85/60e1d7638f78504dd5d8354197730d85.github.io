int sum(int D, int N)
{
  if(D<=0)
    return N;
  int i;
  int sum1=0;
  for(i=0;i<=N;i++)
  {
    sum1=sum1+i;
  }
  if(D>0){
    return sum(D-1,sum1);
  }
}
