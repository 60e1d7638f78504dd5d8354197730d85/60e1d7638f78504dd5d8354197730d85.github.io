#include<stdio.h>

#include<stdlib.h>

 

int calcSum(int n) {

    int sum = 0,i;

    for(i=1; i<=n; i++)

    {

        sum=sum+i;
    }

    return sum;

}

 

int main() {

    int test,i,d,n,result;

    scanf("%d",&test);

    while(test--) {

        scanf("%d %d",&d,&n);

        result = n;
        for(i=0; i<d; i++) {

            result = calcSum(result);

        }

        printf("%d\n", result);

    }
    return 0;

}
