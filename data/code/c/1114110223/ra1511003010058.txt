#include <stdio.h>
int main()
{
  int T,i,j;
    scanf("%d",&T);
    for(j=0;j<T;j++)
    {
        int n,sum=0;
        scanf("%d",&n);
        int a[100];
        for(i=0;n!=0;i++)
        {
            a[i]=n%10;
            n=n/10;
        }
        sum=a[0]+a[i-1];
        printf("%d\n",sum);
    }
	return 0;
}
