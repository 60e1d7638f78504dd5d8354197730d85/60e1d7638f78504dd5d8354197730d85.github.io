#include <stdio.h>
int sum_end_start(int ) ;
int main()
{
 int t,no,sum ;
 scanf("%d",&t) ;
 while(t--)
 {
  scanf("%d",&no) ;
  sum=sum_end_start(no) ;
  printf("%d\n",sum);
 }
}

int sum_end_start(int no)
{
 int i,r,sum=0 ;
 i=no ;
 while(no!=0)
 {
  r=no%10 ;
  if((no==i)||(no/10==0))
   sum=sum+r ;
  no=no/10 ;
 }
 return sum ;
}
