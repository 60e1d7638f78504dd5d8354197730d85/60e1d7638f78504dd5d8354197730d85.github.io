#include <stdio.h>
int main()
{
 int num, sum=0, firstDigit, lastDigit,n,i;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
  scanf("%d", &num);
  lastDigit = num % 10;
    

    /* Copy num to first digit */
    firstDigit = num;

    /* Find the first digit by dividing num by 10 until first digit is left */
    while(num >= 10)
    {
        num = num / 10;
    }
    firstDigit = num;


    /* Find sum of first and last digit*/
    sum = firstDigit + lastDigit; 

    printf("%d\n", sum);
  }
	return 0;
}
