#include <stdio.h>
int main()
{
  int n,h,t,i,g;
  float c;
  
  scanf("%d",&n);
  
  for(i=0;i<n;i++)
  {
     scanf("%d %f %d",&h,&c,&t);
     
    if(h>50 && c<0.7 && t>5600)
       g=10;
    else if(h>50 && c<0.7) 
       g=9;
    else if(c<0.7 && t>5600)
       g=8;
    else if(h>50 && t>5600)
       g=7;
    else if(h>50 || c<0.7 || t>5600)
       g=6;
    else 
       g=5;
      
    printf("%d\n",g);  
    
  }

	return 0;
}
