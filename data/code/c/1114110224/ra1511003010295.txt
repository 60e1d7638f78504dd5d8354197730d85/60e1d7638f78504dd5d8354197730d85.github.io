#include <stdio.h>
int main()
{
  int t;
  float hard,carbon,tensile;
  
  scanf("%d",&t);
  while(t--)
  {
    scanf("%f %f %f",&hard,&carbon,&tensile);
    if(hard>50 && carbon<0.7 && tensile>5600)
    {
       printf("10\n");
    }
    else if(hard>50 && carbon<0.7 && tensile<=5600)
    {
       printf("9\n");
    }
    else if(hard<=50 && carbon<0.7 && tensile>5600)  
    {
       printf("8\n");
    }
    else if(hard>50 && carbon>=0.7 && tensile>5600)
    {
      printf("7\n");
    }
    else if(hard<=50 && carbon>=0.7 && tensile<=5600)
    {
       printf("5\n");
    }
    else
    printf("6\n");
  }            
  
	return 0;
}
