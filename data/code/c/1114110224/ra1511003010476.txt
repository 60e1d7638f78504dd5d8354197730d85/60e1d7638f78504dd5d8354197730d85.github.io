int main()
{
  	int T,h,t,i;
  	float c;
  	scanf("%d\n",&T);
  	while(T--)
    {
      scanf("%d %f %d\n",&h,&c,&t);
      if(h>50 && c<0.7 && t>5600)
        printf("10\n");
      else if(h>50 && c<0.7)
      	printf("9\n");
      else if(c<0.7 && t>5600)
        printf("8\n");

        printf("7\n");
      else if(h>50 || c<0.7 || t>5600)
        printf("6\n");

    }

}
