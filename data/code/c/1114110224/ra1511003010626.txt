#include <stdio.h>
int main()
{
	int t;
  	scanf("%d",&t);
  	while(t--)
    {
      int h,ts,grade;
      float c;
      scanf("%d %f %d",&h,&c,&ts);
      if(h>50 && c<0.7 && ts>5600)
      {
        grade=10;
      }
      else if(h>50 && c<0.7)
      {
        grade=9;
      }
      else if(c<0.7 && ts>5600)
      {
        grade=8;
      }
      else if(h>50 && ts>5600)
      {
      	grade=7;
      }
      else if(h>50 || c<0.7 || ts>5600)
      {
        grade=6;
      }
      else
      {
        grade=5;
      }
      printf("%d\n",grade);
      
    }
	return 0;
}
