#include <stdio.h>
 
int main()
{
    int t,hrd,ts,gd,tst=1; float cc;
    scanf("%d",&t);
    for(tst=1;tst<=t;tst++)
    {
        scanf("%d%f%d",&hrd,&cc,&ts);
        
        if(hrd>50 && cc<0.7 && ts>5600)
        	gd=10;
        else if(hrd>50 && cc<0.7)
        	gd=9;
        else if(cc<0.7 && ts>5600)
        	gd=8;
        else if(hrd>50 && ts>5600)
        	gd=7;
        else if(hrd>50 || cc<0.7 || ts>5600)
        	gd=6;
        else 
        	gd=5;
        printf("%d\n",gd);
    }
    return 0;
}
