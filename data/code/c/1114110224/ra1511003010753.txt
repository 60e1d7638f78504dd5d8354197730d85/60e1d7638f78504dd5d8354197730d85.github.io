#include <stdio.h>
int main()
{
int t,h,te;
float c;
scanf("%d",&t);
while(t--)
{
scanf("%d%f%d",&h,&c,&te);
if(h>50&&c<0.7&&te>5600)
printf("10\n");
else if(h>50&&c<0.7)
printf("9\n");
else if(c<0.7&&te>5600)
printf("8\n");
else if(h>50&&te>5600)
printf("7\n");
else if(h>50||te>5600||c<0.7)
printf("6\n");
else
printf("5\n");
}

	return 0;
}
