#include <stdio.h>
int main() {
	int i,t,m,n,o;
	float a,b,c;
	scanf("%d",&t);
	i=0;
	while(i<t)
	{
	    
	    m=0; n=0; o=0;
	    scanf("%f%f%f",&a,&b,&c);
	    if(a>50)
	    {
	        m++;
	    }
	    if(b<0.7)
	    {
	        n++;
	    }
	    if(c>5600)
	    {
	        o++;
	    }
	    
	    if(m>0 && n>0 && o>0)
	    {
	        printf("10\n");
	        
	    }
	    else if(m>0 && n>0 && o==0)
	    {
	        printf("9\n");
	    }
	    else if(m==0 && n>0 && o>0)
	    {
	        printf("8\n");
	    }
	    else if(m>0 && n==0 && o>0)
	    {
	        printf("7\n");
	    }
	    else if(m>0 || n>0 || o>0)
	    {
	        printf("6\n");
	    }
	    else if(m==0 && n==0 && o==0)
	    {
	        printf("5\n");
	    }
	    
	    i++;
	}
	return 0;
}
