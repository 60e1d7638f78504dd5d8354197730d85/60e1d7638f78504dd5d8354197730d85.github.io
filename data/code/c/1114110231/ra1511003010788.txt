#include <stdio.h>
int Sum(int n)
{
  	int i,sum=0;
  	for(i=1;i<=n;i++)
      sum+=i;
  	return sum;
}
int main()
{
	int t,i;
  	scanf("%d",&t);
  	for(i=0;i<t;i++)
    {
      	int n,d,j,sum=0;
      	scanf("%d %d",&d,&n);
      	for(j=1;j<=d;j++)
        {
          	sum=Sum(n);
          	n=sum;
        }
      printf("%d",sum);
      printf("\n");
    }
	return 0;
}
