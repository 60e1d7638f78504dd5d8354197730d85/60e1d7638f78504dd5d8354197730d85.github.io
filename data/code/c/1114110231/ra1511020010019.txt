#include <stdio.h>
int main()
{
	int n, d,tot, temp,i, j, sum=0;
  	scanf("%d",&tot);
  	for(i=0;i<tot;i++)
    {
    	scanf("%d %d", &d, &n);
      	temp=n;
    
  	while(d!=0)
    {
    	sum=0;
      	for(j=1;j<=temp;j++)
        {
        	sum+=j;
        }
      	temp=sum;
      	d--;
    }
    printf("%d\n",sum);
} 
	return 0;
}
