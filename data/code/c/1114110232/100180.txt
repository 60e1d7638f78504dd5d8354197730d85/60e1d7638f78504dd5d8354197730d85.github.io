#include <stdio.h>
int main()
{
int i,n,num[20],sum=0,firstDigit,lastDigit;
scanf("%d",&n);
for(i=0;i<n;i++)
{
scanf("%d",&num[i]);
   
    lastDigit = num[i] % 10;
       
    firstDigit = num[i];
   
    while(num[i] >= 10)
    {
        num[i] = num[i] / 10;
    }
    firstDigit = num[i];
   
    sum = firstDigit + lastDigit; 
    printf("%d\n",sum);
}
  
	return 0;
}
