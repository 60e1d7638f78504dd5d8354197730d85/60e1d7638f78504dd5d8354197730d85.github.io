#include <stdio.h>
#include <stdlib.h>
int main()
{
    int num,orig,count=0,i,sum=0,rem,ite,no=1;
    scanf("%d",&ite);
    for (i=0;i<ite;i++)
    {
         scanf("%d",&num);
         orig=num;
         while(orig!=0)
    {
       orig/=10;
       count++;
    }
        while (num>0)
    {
        rem=num%10;
      if (no==1 || no==count)
      {
          sum+=rem;
      }
      num/=10;
      no++;
    }
   printf("%d\n",sum);
    count=0;
    sum=0;
    no=1;
    }
    return 0;
}
