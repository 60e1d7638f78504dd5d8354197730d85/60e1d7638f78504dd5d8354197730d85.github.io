#include<stdio.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        long int one=0,k,y,i;
        scanf("%d",&n);
        for(i=1;i<=n;i++)
        {
            k=i;
            while(k)
            {
                y=k%2;
                if(y==1)
                {one++;k=k/2;}
                else
                k=k/2;
            }
        }
        printf("%ld\n",one);
    }
	return 0;
}
