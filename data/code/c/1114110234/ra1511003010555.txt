#include<stdio.h>
 
int main()
{
	int test,n,k;
	scanf("%d",&test);
	int i,j;
	long long ar[201][201];
	int temp;
	long long s[201];
	while(test--)
	{
		scanf("%d %d",&n,&k);
 
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=k;j++)
			{
				ar[i][j]=0;
			}
		}
		
		for(i=1;i<=200;i++)s[i]=0;
		
		for(i=1;i<=n;i++)
		{
			scanf("%d",&temp);
			s[temp]++;
		}
 
		int max=200;
 
			while(s[max]==0)
			{
				max--;
			}
 
		ar[1][1]=1;
 
		int len=s[max];
		while(s[max]!=1)
		{
			ar[1][1]=ar[1][1]*s[max]%1000000007;
			s[max]--;
		}
 
		for(i=2;;i++)
		{
			if(max<=1)break;
			max--;
 
			while(s[max]==0)
			{
				if(max<1)break;
				max--;
			}
			if(max<1)break;
 
			long long p=1;
			for(j=1;j<=s[max]-1;j++)
			{
				p=p*(len+j)%1000000007;
			}
 
			for(j=1;j<=k;j++)
			{
 
				if(j==1)
				
				{
					ar[i][j]=(len*p%1000000007*ar[i-1][j]%1000000007);	
				}
 
				else
				{
					ar[i][j]=(s[max]*p%1000000007*ar[i-1][j-1]%1000000007+len*p%1000000007*ar[i-1][j]%1000000007)%1000000007;
				}
			}
			len+=s[max];
		}
		long long ans=0;
		

		for(j=1;j<=k;j++)
		{
			ans=(ans+ar[i-1][j])%1000000007;
		}
 
		printf("%lld\n",ans);
	}
}
