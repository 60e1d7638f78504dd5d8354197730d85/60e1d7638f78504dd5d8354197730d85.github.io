#include <stdio.h>
int main()
{
  int n;
  scanf("%d",&n);
  int a[n],i;
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  int max=0;
  for(i=1;i<n;i++)
  {
    if(a[i]>max)
    {
      max=a[i];
    }
  }
 int sum=0;
  for(i=0;i<n;i++)
  {
    sum=sum+max-a[i];
  }
  printf("%d",sum);
  return 0;
}
