#include <stdio.h>
int main()
{
	int n;
  	scanf("%d",&n);
  	int a[n], i;
  	for(i=0;i<n;i++)
      	scanf("%d",&a[i]);
  	
  	int temp=a[0];
  	for(i=1;i<n;i++)
    {
      	if(temp<a[i])
          	temp = a[i];
    }
  	int sum = 0;
  	for(i=0;i<n;i++)
    {
      	sum += (temp-a[i]);
    }
  	printf("%d\n",sum);
	return 0;
}
