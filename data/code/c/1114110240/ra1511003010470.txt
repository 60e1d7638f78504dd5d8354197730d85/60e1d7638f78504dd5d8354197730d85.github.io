 #include <stdio.h>  
#include <string.h>
int main()  
{
    char s[100000];
    int i,k=0;   
    scanf("%s",s);  
    for(i=0;s[i]!='\0';i++)  
        if(s[i]!='a')  
            break;  
    for(;s[i]!='\0';i++)  
    {  
        if(s[i]=='a')  
            break;  
        s[i]--;  
        k++;  
    }  
    if(!k)  
        s[strlen(s)-1]='z';  
     printf("%s", s);  
    return 0;  
}
