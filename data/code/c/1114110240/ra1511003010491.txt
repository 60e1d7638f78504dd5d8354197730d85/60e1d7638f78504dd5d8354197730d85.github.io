#include <stdio.h>
#include<string.h>
#include<stdlib.h>
int main()
{
  char str[100000];
  scanf("%s",str);
  int l,s=0;
  l=strlen(str);
  while((str[s]=='a')&&(s<l)){
    s++;
  }
  if(s==l)
  {
    str[l-1]='z';
    printf("%s",str);
    exit(0);
  }
  while((str[s]!='a')&&(s<l)){
    str[s]--;
    s++;
  }
  printf("%s",str);
  return 0; 
}
