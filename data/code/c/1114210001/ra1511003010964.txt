#include <stdio.h>
int main()
{
	int x,y;
  	scanf("%d %d",&x,&y);
  	if(x>0&&y>0) printf("point (%d,%d) lies in the First quandrant",x,y);
  else if(x<0&&y>0) printf("point (%d,%d) lies in the Second quandrant",x,y);
  else if(x<0&&y<0) printf("point (%d, %d) lies in the Third quandrant",x,y);
  else if(x>0&&y<0) printf("point (%d,%d) lies in the Fourth quandrant",x,y);
  else printf("point (%d,%d) lies in the Origin",x,y);
	return 0;
}
