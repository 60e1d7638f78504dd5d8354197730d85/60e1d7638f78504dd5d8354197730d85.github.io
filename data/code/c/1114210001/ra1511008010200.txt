#include <stdio.h>
int main()
{
  int a,b;
  scanf("%d%d",&a,&b);
  if(a>0 && b>0)printf("point (%d,%d) lies in the First quandrant",a,b);
  else if(a>0 && b<0)printf("point (%d,%d) lies in the Fourth quandrant",a,b);
  else if(a<0 && b>0)printf("point (%d,%d) lies in the Second quandrant",a,b);
  else printf("point (%d,%d) lies in the Third quandrant",a,b);
  return 0;
}
