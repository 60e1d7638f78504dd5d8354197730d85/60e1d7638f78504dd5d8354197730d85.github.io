#include<stdio.h>
#include<string.h>
#include<math.h>
int a[1000002];
void prime(){
	int i,j;
for (i = 0; i < 1000002; i++) {                                                                                                                          
a[i] = 1;                                                                                                                                 
}                                                                                                                                                    
 
a[0] = 0;
a[1] = 0;
for (i = 2; i < 1000002; i++) {
if (a[i] == 1) {
for (j = 2 * i; j < 1000002; j += i) {
a[j] = 0;
}
}
}
}
int main()
{
	prime();
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long n;
		scanf("%lld",&n);
		
		long long result=n;
		long long i;
		for(i=2;i*i<=n;i++)
		{
			if(n%i==0)
			result-=result/i;
			while(n%i==0)
			n/=i;
        }
      
		if(n>1)
		result-=result/n;
		//printf("%lld\n",result);
		if(a[result])
		printf("TRUE\n");
		else
		printf("FALSE\n");
	}
	return 0;
}
