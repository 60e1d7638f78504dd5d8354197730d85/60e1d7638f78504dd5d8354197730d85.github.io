#include<stdio.h>
 
int count( int S[], int m, int n )
{
    if (n == 0)
        return 1;
 
    if (n < 0)
        return 0;
 
    if (m <=0 && n >= 1)
        return 0;
 
    return count( S, m - 1, n ) + count( S, m, n-S[m-1] );
}
 
int main()
{
    int i,n,a[1000];
    int arr[] = {1, 3, 5};
    int m = sizeof(arr)/sizeof(arr[0]);
    scanf("%d",&n);
for(i=0;i<n;i++)
{
scanf("%d",&a[i]);
}
for(i=0;i<n;i++)
{
printf("%d\n", count(arr, m,a[i]));
}
 
    return 0;
}
