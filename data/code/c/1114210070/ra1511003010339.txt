#include <stdio.h>
int main()
{
int no,i,j;
  int arr[100];
  scanf("%d",&no);
  for(i=0;i<no;i++)
  {
    scanf("%d",&arr[i]);
  }
  for(i=0;i<no;i++)
  {
    for(j=0;j<no-i-1;j++)
    {
      int temp=0;
      if(arr[j]>arr[j+1])
      {
        temp=arr[j];
        arr[j]=arr[j+1];
        arr[j+1]=temp;
    }
    }
    }
    printf("%d",arr[1]);
	return 0;
}
