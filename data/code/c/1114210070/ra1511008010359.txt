#include <stdio.h>
int main()
{
  int n, i, min=9999, j;
  scanf("%d", &n);
  int a[n];
  for(i=0;i<n;i++)
    scanf("%d", &a[i]);
  for(i=0;i<n;i++)
  {
    if(a[i]<min)
    {
      min=a[i];
      j=i;
    }
  }
  
  a[j]=9999;
  min=9999;
  for(i=0;i<n;i++)
    if(a[i]<min)
      min=a[i];
  printf("%d", min);

	return 0;
}
