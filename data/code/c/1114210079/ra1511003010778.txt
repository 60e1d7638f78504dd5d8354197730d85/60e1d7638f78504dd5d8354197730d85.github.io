#include <stdio.h>
int main()
{
  int n, i, j;
  
  scanf("%d", &n);
  
  for(i=0; i<(2*n); i++)
  {
    i++;
    for(j=1; j<=i; j++)
    {
      printf("%d", j);
    }
    printf("\n");
  }
  for(j=i-2; j>0; j--)
  {
    j--;
    for(i=1; i<=j; i++)
    {
      printf("%d", i);
    }
    printf("\n");
  }
  return 0;
}
