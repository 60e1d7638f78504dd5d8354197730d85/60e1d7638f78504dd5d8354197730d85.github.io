#include <stdio.h>
 
int c=0;
int A[10][10],n;
void solve(int i, int j)
{
 if(i<0||j<0||i>=n||j>=n)return;  //Check for invalid cell
 if(A[i][j]!=0)return;            //Check for wall
 if(i==n-1&&j==n-1)
    {c++;
     return;
    }
 A[i][j]=1;
 solve(i,j-1);
 solve(i,j+1);
 solve(i-1,j);
 solve(i+1,j);
 A[i][j]=0;
}
int main(void)
{
    int i,j;
    // printf("\nEnter the size of the maze: ");
    scanf("%d",&n);
    //printf("\nEnter the maze:\n");
    for(i=0;i<n;i++)
        {for(j=0;j<n;j++)
            {fflush(stdin);
             scanf("%d",&A[i][j]);
            }
        }
	solve(0,0);
    printf("%d",c);
	return 0;
}
