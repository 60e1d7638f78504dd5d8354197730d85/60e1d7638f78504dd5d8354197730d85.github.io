#include <stdio.h>
int parx=-1;int pary=-1;
int count=0;
void calc(int maze[100][100],int n,int i,int j)
{
     
     
     if(i<0 || i>=n || j<0 || j>=n || (i==parx && j==pary))
     return;
     if(maze[i][j]==1)
     return;
     if(i==n-1 && j==n-1)
     {
               count++;
               return;
     }
     
     maze[i][j]=1;
     
     //printf("%d %d\n",i,j);
     calc(maze,n,i+1,j);
     calc(maze,n,i-1,j);
     calc(maze,n,i,j+1);
     calc(maze,n,i,j-1);
     maze[i][j]=0;
     
     return;
}
 
int main()
{
    int maze[100][100];
    int n=0;int i=0;int j=0;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    for(j=0;j<n;j++)
    scanf("%d",&maze[i][j]);
    calc(maze,n,0,0);
    printf("%d\n",count);
    return 0;
}
