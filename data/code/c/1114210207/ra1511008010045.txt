#include<stdio.h>
 
int main()
{
	int t,count,i,h[30];
	char s[105];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",s);
		for(i=0;i<29;i++)
			h[i]=0;
 
		count=0;
		for(i=0;s[i]!='\0';i++)
		{
			if(s[i]<91)
			{
				if(h[s[i]-65]>0)
					count++;
				else h[s[i]-65]++;
			}
			else
			{
				if(h[s[i]-97]>0)
					count++;
				else h[s[i]-97]++;
			}
 
		}
		printf("%d\n",count);
 
	}
 
return 0;
}
