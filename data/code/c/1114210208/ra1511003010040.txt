#include<stdio.h>
#include<math.h>
void fun(int n)
{
    int i,x = (int) sqrt(n);
    for(i=2;i<=x;i++)
    {
        if(n%i==0)
        {
            return;
        }
    }
    printf("%d\n",n);
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t)
	{
		t--;
		int a,b,i;
		scanf("%d %d",&a,&b);
		for(i = a; i<=b; i++)
		{
		    if(i==1||(i%2==0&&i!=2)||(i!=3&&i%3==0)||(i!=7&&i%7==0))
		    {
		        continue;
		    }
		    fun(i);
		}
		printf("\n\n");
	}
	return 0;
}
