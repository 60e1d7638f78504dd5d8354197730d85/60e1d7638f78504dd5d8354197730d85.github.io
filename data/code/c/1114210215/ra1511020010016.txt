#include <stdio.h>
#include <string.h>
int main()
{
  int flag;
  char a[20];
  char b[20];
  char c[20];
  scanf("%s",a);
  scanf("%s",b);
  
  int i,d= strlen(a);
  
  for(i=d-1;i>=0;i--)
    c[d-i-1]=a[i];
  flag=1;
  for(i=0;i<d;i++)
  {
    if(b[i]!=c[i])
    {
      flag=0;
    }
    
  }
  if(flag==0)
    printf("NO");
  else
    printf("YES");
  
  

	return 0;
}
