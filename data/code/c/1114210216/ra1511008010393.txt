#include <stdio.h>
#include <math.h>
int main()
{
  long int t,i;
  scanf("%ld",&t);
  for(i=1;i<=t;i++)
  {
    long long h,k;
    long long int ans=0;
    scanf("%lld%lld",&h,&k);
    long long x,y;
    x=abs(h);
    y=abs(k);
    if(x>=y)
    {
      if((x+y)%2==0)
        ans=x*2;
      else
        ans=x*2+1;
    }
    else if(y>=x)
    {
      if((x+y)%2==0)
        ans=y*2;
      else
        ans=y*2-1;
    }
    printf("%lld\n",ans);
  }
	return 0;
}
