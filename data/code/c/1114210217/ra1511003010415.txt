#include <stdio.h>
#include <math.h>
 
int noofdigits(int b)
{
  int count=0;
  int temp1=b;
  int rem1;
    while(temp1>0)
    {
      rem1=temp1%10;
      count++;
      temp1 /=10;
    }
    return count;
 
}
 
int isprime(int a)
{
  int c;
 
    for ( c = 2 ; c <= a - 1 ; c++ )
    {
       if ( a%c == 0 )
   return 0;
    }
    if ( c == a )
       return 1;
 
}
 
void circular(int q)
{
  int flag=0,temp,rem,n;
  n=noofdigits(q);
  temp=q;
    while(isprime(temp))
    {
      rem=temp%10;
      temp=temp/10;
      temp=(pow(10,n-1)*rem+temp);
 
      if(temp==q)
      {  flag=1;
         break;
      }
    }
    if(flag==0)
    printf("no\n");
    if(flag==1)
    printf("yes\n");
 
}
int main()

int i,m,n;
