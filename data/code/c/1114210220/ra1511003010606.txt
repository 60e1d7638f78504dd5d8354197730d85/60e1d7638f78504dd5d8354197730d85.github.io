#include <stdio.h>
#define gc getchar_unlocked
long long read() {
  char c = gc();
  while(c<'0' || c>'9') c = gc();
  long long ret = 0;
  while(c>='0' && c<='9') {
    ret = 10 * ret + c - 48;
    c = gc();
  }
  return ret;
}
int main(void) {
	long long c,d,l,lmin,diff,temp,i,t,least;
	int flag;
	t=read();
	while(t--)
	{
	    c=read();
	    d=read();
	    l=read();
	    flag=0;
	    least=d*4;
	    if(l&3)
	        flag=0;
	    else if(l<least || l>(least+(c<<2)))
	        flag=0;
	    else
	    {
	        temp=c-(d<<1);
	        if(temp<0)
	            temp=0;
	        lmin=least+(temp<<2);
	        if(l==lmin)
	            flag=1;
	        else if(l>lmin)
	        {
	            diff=l-lmin;
	            if(diff&3)
	                flag=0;
	            else 
	                flag=1;
	        }
	    }
	    if(flag==1)
	        printf("yes\n");
	    else
	        printf("no\n");
	}
	return 0;
}
