#include <stdio.h>
 
int main(void)
{
    int flag=0, t;
    long int n,i;
    scanf("%d",&t);
    while(t--)
    {
    scanf("%ld",&n);
    flag=0;
    for(i=2;i<=n/2;++i)
    {
        if(n%i==0)
        {
            flag=1;
            break;
        }
    }
    if (flag==0)
        printf("yes\n");
    else
        printf("no\n");
    }
	return 0;
}
