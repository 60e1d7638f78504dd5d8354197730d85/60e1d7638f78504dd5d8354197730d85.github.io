#include <stdio.h>
#include <stdlib.h>
 
char _str[10];
 
#define IS_DELIMITER(_c) \
((_c) == ' ' || (_c) == '\n' || (_c) == '\t')	\
 
#define READ_INT(_x) \
{	\
int _n_flag = 0; \
register int _tmp = 0, _ch = getchar_unlocked(); \
while (IS_DELIMITER(_ch))	\
_ch = getchar_unlocked(); \
if (_ch == '-')	\
{	\
_n_flag = 1; \
_ch = getchar_unlocked(); \
}	\
while (48 <= _ch && _ch <= 57)	\
{	\
_tmp = (_tmp << 3) + (_tmp << 1) + _ch - 48; \
_ch = getchar_unlocked(); \
}	\
(_x) = ((_n_flag) ? -1 : 1) * _tmp; \
}	\
 
#define WRITE_INT(_x) \
{	\
register int _i = 0; \
unsigned _tmp = (_x); \
do	\
{	\
*(_str + _i++) = _tmp % 10 + 48; \
_tmp /= 10; \
} while (_tmp != 0); \
for (_i--; _i >= 0; _i--)	\
putchar_unlocked(*(_str + _i)); \
}	\
 
int main(void)
{
int i, k, n;
 
READ_INT(n);
READ_INT(k);
if (k == 0)
while ((i = getchar_unlocked()) != EOF)
putchar_unlocked(i);
else
{
long long max, min;
int *a;
 
a = malloc(n * sizeof *a);
if (a == NULL)
{
perror("malloc");
return 1;
}
 
READ_INT(*a);
min = max = *a;
for (i = 1; i < n; i++)
{
READ_INT(*(a + i));
if (*(a + i) > max)
max = *(a + i);
else if (*(a + i) < min)
min = *(a + i);
}
 
if ((k & 0x1) == 0)
for (i = 0; i < n; i++)
{
WRITE_INT(*(a + i) - min);
putchar_unlocked(' ');
}
else
for (i = 0; i < n; i++)
{
WRITE_INT(max - *(a + i));
putchar_unlocked(' ');
}
putchar_unlocked('\n');
 
free(a);
}
 
return 0;
}
