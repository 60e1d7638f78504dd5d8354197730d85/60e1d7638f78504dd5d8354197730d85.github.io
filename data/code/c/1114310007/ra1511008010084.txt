#include <stdio.h>
#include <math.h>
#include <stdlib.h>
 
void sieve(int *array, int n){
    long long i = 0, j = 0;
    for(i=2; i<=n; i++){
        array[i] = 1;
    }
    for(i=2; i<=n; i++){
        if(array[i] == 1){
            int t, s;
            if(i%4 == 1){
                t = 2;
                s = 1;
            }
            else{
                t = 0;
                s = i;
            }
            for(j=s; (i*j)<=n; j++){
                if(array[i*j] != 2)
                    array[i*j] = t;
            }
        }
    }
}
 
int main(){
    int *array, n=5000001;
    array = (int *)malloc((n+1)*sizeof(int));
    sieve(array, n);
 
    long t;
    scanf("%ld", &t);
    while(t--){
        long long n;
        scanf("%lld", &n);
        if (array[n] == 2)
            printf("YES\n");
        else
            printf("NO\n");
    }
    return 0;
}
