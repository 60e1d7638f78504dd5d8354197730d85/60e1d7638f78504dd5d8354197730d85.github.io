#include <stdio.h>
#include <stdlib.h>
 
int compare (const void * a, const void * b)
{
  return ( *(int*)a - *(int*)b );
}
int binary(int *a,int req,int i,int j)
{
    if(i>j) 
        return -1;
    int mid=(i+j)/2;
    if(a[mid]==req)
    {
        int pos=mid;
        while(pos>=0 && a[pos]==req) pos--;
        return pos+1;
    }
    if(a[mid]<req)  
        return binary(a,req,mid+1,j);
    return binary(a,req,i,mid-1);
}

                pos = binary(a,req,0,n-1);
