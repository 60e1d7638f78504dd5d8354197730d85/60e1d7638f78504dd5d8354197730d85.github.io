#include<stdio.h>
int count;
int divd(int num)
{
int i=1;
while(num!=0)
{
i=i*10;
num /=10;
}
i=i/10;
return i;
}
int fun1(int num)
{
int div= divd(num);
int mod= num%div;
int j,val;
j= (num- mod) / div;
val=mod*10 + j;
if(j==0)
{ val=val/10;
}
return val;
}
int fun2(int num)
{
int mod,j,val, div=divd(num);
mod=num%10;
j=num/10; 
val=mod*div + j;
return val;
}
int main()
