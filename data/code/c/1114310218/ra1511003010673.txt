#include<stdio.h>
#include<string.h>
int main()
{
	int t,i,l,c=0,z,x=0,flag=0,j,k;
	char a[1010];
	scanf("%d",&t);
	for(z=0;z<t;z++)
	{
		scanf("%s",a);
		l=strlen(a);
		if(l%2==0)
		{
			for(i=0;i<l/2;i++)
			{
				c=0;
				x=0;
				flag=0;
				if(a[i]!='A')
				{
					c++;
					for(j=i+1;j<l/2;j++)
					{
						if(a[j]==a[i])
						{
							c++;
							a[j]='A';
						}
					}
					for(k=l/2;k<l;k++)
					{
						if(a[k]==a[i])
							x++;
					}
				}
				if(x!=c)
				{
					printf("NO\n");
					flag=0;
					break;
				}
				else
					flag=1;
			}
			if(flag==1)
				printf("YES\n");
		}
		else
		{
			for(i=0;i<l/2;i++)
			{
				c=0;
				x=0;
				flag=0;
				if(a[i]!='A')
				{
					c++;
					for(j=i+1;j<l/2;j++)
					{
						if(a[j]==a[i])
						{
							c++;
							a[j]='A';
						}
					}
					for(k=(l/2)+1;k<l;k++)
					{
						if(a[k]==a[i])
							x++;
					}
				}
				if(x!=c)
				{
					printf("NO\n");
					flag=0;
					break;
				}
				else
					flag=1;
			}
			if(flag==1)
				printf("YES\n");
		}
	}
	return 0;
}
