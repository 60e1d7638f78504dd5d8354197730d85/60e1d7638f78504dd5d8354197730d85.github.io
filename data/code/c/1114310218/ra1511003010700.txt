#include <stdio.h>
#include<string.h>
 
int main() 
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int i,j,s=1,n;
        char a[1001];
        scanf("%s",a);
        j=strlen(a);
        n=j/2;
        int b[501]={0},c[501]={0};
        if(j%2==0)
        {
            for(i=0;i<n;i++)
            {
                b[a[i]-97]++;
            }
            for(i=n;i<j;i++)
            c[a[i]-97]++;
        }
        else
        {
            for(i=0;i<n;i++)
            b[a[i]-97]++;
            for(i=n+1;i<j;i++)
            c[a[i]-97]++;
        }
        for(i=0;i<501;i++)
        {
            if(b[i]!=c[i])
            s=0;
        }
        if(s==1)
        printf("YES\n");
        else
        printf("NO\n");
    }
	return 0;
}
