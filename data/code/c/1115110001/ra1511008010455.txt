#include <stdio.h>
int main()
{
	int n,i,ch,def=0,one=0,two=0,three=0,four=0,five=0;
  	scanf("%d",&n);
  for(i=0;i<n;i++)
  {
   scanf("%d",&ch);
   switch(ch)
  {
     case 1: one++;
       break;
       
     case 2: two++;
       break;
       
     case 3: three++;
       break;
       
     case 4: four++;
       break;
       
     case 5: five++;
       break;
       
     default:def++;
  }
  
  }
  
  printf("Candidate 1=%d\n",one);
  printf("Candidate 2=%d\n",two);
  printf("Candidate 3=%d\n",three);
  printf("Candidate 4=%d\n",four);
  printf("Candidate 5=%d\n",five);
  printf("Spoil Ballot=%d",def);
  
	return 0;
}
