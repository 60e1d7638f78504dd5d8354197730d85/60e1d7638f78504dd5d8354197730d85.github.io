#include<stdio.h>
 
 
int main()
{   char a[5000],b[5000],c[5000];
    int i , j,m=0,sum=0;
    scanf("%s",a);
    scanf("%s",b);
    for(i=0; a[i]!='\0'; i++)
    {   for(j=0; b[j]!='\0';j++)
        {   if(a[i]==b[j])
            {   c[m]=a[i];
                sum++;
                m++;
                b[j]='0';
                break;
            
            }
        }
    }
    
    for(m=0; c[m]!='\0'; m++)
    {   if(c[m]=='A' || c[m]=='a' || c[m]=='E'|| c[m]=='e'|| c[m]=='I'|| c[m]=='i'|| c[m]=='O'||c[m]=='o'||c[m]=='U'||c[m]=='u')
        sum--;
    }
    printf("%d",sum);
    return 0;
}
