#include <stdio.h>
char s[55],a[55],b[55],c[55],d[55];
typedef long long tl;
int main()
{
  int t;tl hi=1000000;
  scanf("%d",&t);
  hi=hi*hi;
  while(t--)
  {
    scanf("%s",s);
    int n=0;
    for(n=0;s[n];n++);
 tl ans=-1;
    int i,j,k,l;
    int in1,in2,in3,in4;
    in1=in2=in3=0;
    for(i=0;i<n;i++)
    {
      a[in1++]=s[i];
      a[in1]=0;
      in2=0;
      for(j=i+1;j<n;j++)
      {b[in2++]=s[j];
       b[in2]=0;
       in3=0;
       for(k=j+1;k<n;k++)
       {c[in3++]=s[k];
        c[in3]=0;
        in4=0;
        for(l=k+1;l<n;l++)
          d[in4++]=s[l];
        if(in4==0)continue;
        d[in4]=0;
        int valid=1;
        if(in1>1 && a[0]=='0')valid=0;
        if(in2>1 && b[0]=='0')valid =0;
        if(in3>1 && c[0]=='0')valid=0;
        if(in4>1 && d[0]=='0')valid=0;
        if(valid)
        {
          tl w,x,y,z;
          sscanf(a,"%lld",&x);
          sscanf(b,"%lld",&w);
          sscanf(c,"%lld",&y);
          sscanf(d,"%lld",&z);
          if(w<=hi && x<=hi && y<=hi && z<=hi && w+x+y+z > ans)ans=w+x+y+z;
        }}}}
    if(ans==-1)printf("The output is:unlucky\n");
    else printf("The output is:%lld\n",ans);}
    
	return 0;
}
