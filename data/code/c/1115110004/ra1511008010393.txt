ll num(ll i, ll j) {
  ll r=0, k, p;
 
  if (i>j) return -1;
  if (i!=j && a[i]=='0')
    return -1;
  for(k=i;k<=j;k++) {
    p = a[k] - '0';
    r=r*10 + p;
  }
  if (r > 1000000000000) return -1;
  return r;
}
ll find(ll j, ll k) {
  ll i,r, q;
  if (k==1) {
    q = num(0, j);
    return q;
  }
  if (s[j][k]!=-1) return s[j][k];
  s[j][k]=-1;
  for(i=0;i<j;i++) {
    r=find(i, k-1);
    if (r==-1) continue;
    q=num(i+1, j);
    if (q==-1) continue;
    s[j][k]=MAX(s[j][k], r + q);
  }
  return s[j][k];
}
