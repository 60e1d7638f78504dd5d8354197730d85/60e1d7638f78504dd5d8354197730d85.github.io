#include <stdio.h>
int main()
{
	int i,j,k,n,m,t,o;
    scanf("%d",&t);
    while(t--) {
        scanf("%d",&n);
        int a[n], sum[100005]={0};
        k=0;m=99999999;
        for(i=0;i<n;i++)
        {
            scanf("%d",&a[i]);
            k+=a[i];
            sum[a[i]]=1;
        }
        if(k%2==0) {
            j=k/2;
            for(i=0;i<=j;i++) {
                for(o=0;o<n;o++) {
                    if(a[o]<=i&&sum[i-a[o]]==1) {
                        sum[i]=1;
                    }
                }
            }
            if(sum[j]==1)
            printf("YES\n");
            else
            printf("NO\n");
        } else 
        printf("NO\n");
        
    }
	return 0;
}
