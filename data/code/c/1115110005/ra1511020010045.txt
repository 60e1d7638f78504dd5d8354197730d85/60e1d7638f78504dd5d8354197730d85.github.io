#include <stdio.h>
int main()
{
int t,n,a[20],sum=0,i;
  scanf("%d",&t);
  while(t>0)
  {
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
      scanf("%d",&a[i]);
    }
    for(i=0;i<n;i++)
    {
      sum=sum+a[i];
    }
    if(sum%2==0)
      printf("YES\n");
    else
      printf("NO\n");
    t--;
  }
	return 0;
}
