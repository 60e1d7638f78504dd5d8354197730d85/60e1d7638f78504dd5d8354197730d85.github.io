#include <stdio.h>

int main()
{
  int n,a[50],i,c1=0,c2=0,c3=0,c4=0,c5=0,c6=0;
  scanf("%d",&n);
  for(i=0;i<n;i++)
    scanf("%d",&a[i]);
  for(i=0;i<n;i++)
  {
    if(a[i]==1) c1++;
    else if(a[i]==2) c2++;
    else if(a[i]==3) c3++;
    else if(a[i]==4) c4++;
    else if(a[i]==5) c5++;
    else c6++;
  }
 
  printf("Candidate 1=%d\n",c1);
  printf("Candidate 2=%d\n",c2);
  printf("Candidate 3=%d\n",c3);
  printf("Candidate 4=%d\n",c4);
  printf("Candidate 5=%d\n",c5); 
  printf("Spoil Ballot=%d\n",c6);

  return 0;
}
