#include<stdio.h>

int main()
{
    int count[6],mark,bn,i;

    for(i=0;i<=5;i++)
    {
        count[i]=0;
    }

    scanf("%d",&bn);
    for(i=0;i<=bn-1;i++)
    {

        scanf("%d",&mark);
        if(mark==1)
        {
            ++count[0];
        }
        else if(mark==2)
        {
            ++count[1];
        }
        else if(mark==3)
        {
            ++count[2];
        }
        else if(mark==4)
        {
            ++count[3];
        }
        else if(mark==5)
        {
            ++count[4];
        }
        else
        {
            ++count[5];
        }

    }
    printf("\nCandidate 1=%d",count[0]);
    printf("\nCandidate 2=%d",count[1]);
    printf("\nCandidate 3=%d",count[2]);
    printf("\nCandidate 4=%d",count[3]);
    printf("\nCandidate 5=%d",count[4]);
    printf("\nSpoil Ballot=%d",count[5]);
return 0;
}
