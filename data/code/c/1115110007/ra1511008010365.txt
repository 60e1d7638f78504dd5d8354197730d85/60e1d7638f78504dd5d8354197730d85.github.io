#include<stdio.h>
int main()
{
int i, vote[100], c1=0, c2=0, c3=0, c4=0, c5=0, c_spoilt=0, n;
scanf("%d",&n);
for(i=0; i<n; i++)
{

scanf("%d", &vote[i]);
if((vote[i]<=5)&&(vote[i]>=1))
{
switch(vote[i])
{
case 1 : c1++;
break;
case 2 : c2++;
break;
case 3 : c3++;
break;
case 4 : c4++;
break;
case 5 : c5++;
break;
default : printf("Error...!!!");
}
}
else
{
vote[i]=0;
c_spoilt++;
}
}

  printf("Candidate 1=%d\n",c1);
  
  printf("Candidate 2=%d\n",c2);
  
  printf("Candidate 3=%d\n",c3);
  
  printf("Candidate 4=%d\n",c4);
  
  printf("Candidate 5=%d\n",c5);
  
  printf("Spoil Ballot=%d",c_spoilt);

	return 0;
}
