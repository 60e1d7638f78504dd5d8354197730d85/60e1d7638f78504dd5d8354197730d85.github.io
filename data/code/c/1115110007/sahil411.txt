#include<stdio.h>

int main()
{
    int count[6],mark,bn,i;
    
    for(i=0;i<=5;i++)
    {
        count[i]=0;
    }
   
    scanf("%d",&bn);
    for(i=0;i<=bn-1;i++)
    {
        
        scanf("%d",&mark);
        if(mark==1)
        {
            ++count[0];
        }
        else if(mark==2)
        {
            ++count[1];
        }
        else if(mark==3)
        {
            ++count[2];
        }
        else if(mark==4)
        {
            ++count[3];
        }
        else if(mark==5)
        {
            ++count[4];
        }
        else
        {
            ++count[5];
        }

    }
    printf("Candidate 1=%d\n",count[0]);
    printf("Candidate 2=%d\n",count[1]);
    printf("Candidate 3=%d\n",count[2]);
    printf("Candidate 4=%d\n",count[3]);
    printf("Candidate 5=%d\n",count[4]);
    printf("Spoil Ballot=%d",count[5]);
    return 0;
}
