#include <stdio.h>
int main()
{
int n, i, arr[50], search, first, last, middle;
	scanf("%d",&n);
	for (i=0; i<n; i++)
	{
		scanf("%d",&arr[i]);
	}
	scanf("%d", &search);
	first = 0;
	last = n-1;
	middle = (first+last)/2;
	while (first <= last)
	{
		if(arr[middle] < search)
		{
			first = middle + 1;

		}
		else if(arr[middle] == search)
		{
			printf("%d found at location %d\n", search, middle+1);
			break;
		}
		else
		{
			 last = middle - 1;
		}
		middle = (first + last)/2;
	}
	if(first > last)
	{
		printf("%d is not present in the list",search);
    }
	return 0;
}
