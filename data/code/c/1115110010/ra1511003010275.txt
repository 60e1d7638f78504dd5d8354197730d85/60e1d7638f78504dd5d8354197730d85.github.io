#include <stdio.h>
 
int main()
{
   int array[100], search, c, n;
 
   scanf("%d",&n);
 
 
   for (c = 0; c < n; c++)
      scanf("%d", &array[c]);
 

   scanf("%d", &search);
 
   for (c = 0; c < n; c++)
   {
      if (array[c] == search)     /* if required element found */
      {
         printf("%d found at location %d\n", search, c+1);
         break;
      }
   }
   if (c == n)
      printf("%d is not present in the list\n", search);
 
   return 0;
}
