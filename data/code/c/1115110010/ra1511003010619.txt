#include <stdio.h>
int main()
{
  int c,first,last,middle,n,search,array[1000];
  scanf("%d",&n);
  for(c=0;c<n;c++)
    scanf("%d",&array[c]);
  scanf("%d",&search);
  first=0;
  last=n-1;
  middle=(first+last)/2;
  while(first<=last)
  {
    if(array[middle]<search)
      first=middle+1;
    else if(array[middle]==search)
    {
      printf("%d found at location %d",search,middle+1);
      break;
    }
    else
      last=middle-1;
    middle=(first+last)/2;
  }
  if(first>last)
    printf("%d is not present in the list",search);

	return 0;
}
