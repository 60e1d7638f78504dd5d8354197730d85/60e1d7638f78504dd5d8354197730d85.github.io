#include <stdio.h>
int main()
{
	int n , count=0 , a[15] , i , j ;
  	scanf ("%d" , &n) ;
  	for ( i=0 ; i<n ; i++ )
      scanf ("%d" , &a[i]) ;
  	for ( i=0 ; i<n ; i++ )
      for ( j=i+1 ; j<n ; j++ )
        if (a[i] == a[j])
          count++ ;
  	if (count==0 )
      printf ( "%d" , n ) ;
  	else
      printf ( "%d" , n-count ) ;
	return 0;
}
