#include<stdio.h>
int main()
{
int Array[10] ;
int i , N ;

int negsum;     
float total , average;
negsum = 0 ;
total = 0;

scanf("%d" , &N);
for(i=0 ; i<N ; i++)
{
scanf(" %d" , &Array[i] );
}

for(i=0 ; i<N ; i++)
{
if(Array[i] < 0)
{
negsum += Array[i];
}
total += Array[i];
}
average = total / N;

printf("sum=%d\n" , negsum);
return 0;
}
