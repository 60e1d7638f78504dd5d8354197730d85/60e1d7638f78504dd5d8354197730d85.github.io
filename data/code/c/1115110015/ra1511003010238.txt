#include <stdio.h>
int main()
{
int n,i,a[50],sum=0,sum1=0;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  scanf("%d",&a[i]);
  for(i=0;i<n;i++)
  {
    if(a[i]%2==0)
    {
      sum=sum+a[i];
    }

    {

    }
  }
    printf("even=%d\n",sum);

	return 0;
}
