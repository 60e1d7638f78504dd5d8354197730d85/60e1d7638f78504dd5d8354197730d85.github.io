#include <stdio.h>
int main()
{
	int n,sum1=0,sum2=0,i;
  	scanf("%d",&n);
  	int a[n];
  	for(i=0;i<n;i++)
    {
      scanf("%d",&a[i]);
    }
  	for(i=0;i<n;i++)
    {
      if(a[i]%2==0)
      {
        sum1+=a[i];
      }
      else
      {
        sum2+=a[i];
      }
    }
  printf("even=%d\nodd=%d",sum1,sum2);
	return 0;
}
