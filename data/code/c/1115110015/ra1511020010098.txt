#include <stdio.h>
 
int main()
{
    int i,x, num, odd_sum = 0, even_sum = 0;
 

    scanf("%d", &num);
    for (i = 1; i <= num; i++)
    {
      scanf("%d",&x);
        if (x % 2 == 0)
            even_sum = even_sum + x;
        else
            odd_sum = odd_sum + x;
    }
    printf("even=%d\n", even_sum);
    printf("odd=%d\n", odd_sum);
  return 0;
}
