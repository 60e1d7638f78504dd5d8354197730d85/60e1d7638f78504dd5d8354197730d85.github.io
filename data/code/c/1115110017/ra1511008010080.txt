#include <stdio.h>
int main()
{
	int n;
  	scanf("%d",&n);
  	int i,flag = 0;
  	int a[n],b[n];
  	for(i=0;i<n;i++)
    {
      scanf("%d",&a[i]);
    }
  	for(i=0;i<n;i++)
    {
      scanf("%d",&b[i]);
    }
  	for(i=0;i<n;i++)
    {
      if(a[i] != b[i])
      {
        flag=1;
      }
    }
  	if(flag == 0)
    {
      printf("yes");
    }
  else
  {
    printf("no");
  }
	return 0;
}
