#include <stdio.h>
int main()
{
 int a[100],b[100];
  int i,n; int c=0;
  scanf("%d",&n);
  for(i=0;i<n;i++)
    scanf("%d ",&a[i]);
  for(i=0;i<n;i++)
    scanf("%d ",&b[i]);
  for(i=0;i<n;i++)
  {
    if(a[i]==b[i])
      c++;
  }
  if(c==n)
    printf("yes");
  else
    printf("no");
    
  return 0;
}
