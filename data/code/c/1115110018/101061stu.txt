#include <stdio.h>
int getInvCount(int arr[], int n)
{
  int inv_count = 0,i,j;
  for ( i = 0; i < n - 1; i++)
    for (j = i+1; j < n; j++)
      if (arr[i] > arr[j])
        inv_count++;
  return inv_count;
}

int main()
{
int i,n,aver,arr[100];
scanf("%d",&n);
for(i=0;i<n;i++)
scanf("%d",&arr[i]);
aver=n/1;
  printf(" %d\n", getInvCount(arr,aver));
  return 0;
}
