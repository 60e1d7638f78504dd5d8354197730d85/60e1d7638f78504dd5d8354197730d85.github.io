#include <stdio.h>
int getInvCount(int arr[], int n)
{
  int inv_count = 0,i,j;
  for (i = 0; i < n - 1; i++)
    for (j = i+1; j < n; j++)
      if (arr[i] > arr[j])
        inv_count++;
 
  return inv_count;
}
int main()
{int n,arr[50],k;
  scanf("%d",&n);
 for(k=0;k<n;k++)
   scanf("%d",&arr[k]);
    printf("%d\n", getInvCount(arr, n));


	return 0;
}
