#include <stdio.h>
int main()
{
  int n, i, j, cnt=0;
  scanf("%d", &n);
  int a[n];
  for(i=0;i<n;i++)
    scanf("%d", &a[i]);
  for(i=0;i<n;i++)
    for(j=i;j<n;j++)
      if(a[i]>a[j])
      {
        a[i]=a[i]+a[j];
        a[j]=a[i]-a[j];
        a[i]=a[i]-a[j];
        cnt++;
      }
  printf("%d", cnt);
    

	return 0;
}
