#include <stdio.h>
 
int main()
{
   int array[100], position, c, n, value, i, j, temp;
   scanf("%d", &n);
   for (c = 0; c < n; c++)
      scanf("%d", &array[c]);
   scanf("%d", &position);
   scanf("%d", &value);

   for (c = n - 1; c >= position ; c--)
      array[c+1] = array[c];
 
   array[position] = value;
 	printf("Before sorting\n");
   for (c = 0; c <= n; c++)
      printf("%d\n", array[c]);
 printf("After sorting\n");	
  for (i = 0; i <= n - 1; i++)
    {  
		for(j = 0; j <= n - i - 1; j++)
        {
          if(array[j] > array[j + 1])
          {
            temp = array[j];
            array[j] = array[j + 1];
            array[j + 1] = temp;
          }
        }
    }
	for (c = 0; c <= n; c++)
      printf("%d\n", array[c]);
   return 0;
}
