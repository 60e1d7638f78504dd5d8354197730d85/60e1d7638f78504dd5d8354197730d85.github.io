#include <stdio.h>
int main()
{
    int array[100], position, c, n;
    scanf("%d", &n);
    for ( c = 0 ; c < n ; c++ )
    scanf("%d", &array[c]);
    scanf("%d", &position);
    if ( position >= n )    
    printf("Invalid Input");
    else
    {    
        printf("%d\n", array[position]);   
       
    }    
    return 0;
}
