#include <stdio.h>
#include <math.h>
int main()
{
    int n, i;
    float num[100], sum = 0.0, average;

   
    scanf("%d", &n);

    while (n > 100 || n <= 0)
    {
        
        scanf("%d", &n);
    }

    for(i = 0; i < n; ++i)
    {
        
        scanf("%f", &num[i]);
        sum += num[i];
    }

    average = sum / n;
    printf("The mean of the array is %.2f", round(average));

    return 0;
}
