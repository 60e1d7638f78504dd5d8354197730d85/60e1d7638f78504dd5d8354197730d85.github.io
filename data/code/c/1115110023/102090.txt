#include <stdio.h>
int main()
{
	int n, i, arr[10], min = 10000, max = 0;
  	scanf("%d", &n);
  	for(i = 0; i < n; i++)
    	scanf("%d", &arr[i]);
  	for(i = 0; i < n; i++)
    {
      	if(arr[i] < min)
          min = arr[i];
      	if(arr[i] > max)
          max = arr[i];
    }
  	printf("The range of the array is=%d", max-min);
	return 0;
}
