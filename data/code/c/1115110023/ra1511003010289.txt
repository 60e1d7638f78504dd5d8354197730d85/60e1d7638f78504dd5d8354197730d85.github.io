#include <stdio.h>
int main()
{
	int n,i,min,max;
  	int a[20];
  scanf("%d\n",&n);
    for(i=0;i<n;i++)
    {
      scanf("%d ",&a[i]);
    }
      min=max=a[0];
  	for(i=0;i<n;i++)
    {
      if(a[i]>max)
        max = a[i];
      if(a[i]<min)
        min = a[i];
    }
  printf("The range of the array is=%d",(max-min));
	return 0;
}
