#include <stdio.h>
int main()
{
    int n,max,min;
    scanf("%d",&n);
    int a[n+1];
    int i=0;
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    max=a[0];
    min=a[0];
    for(i=0;i<n;i++)
    {
        if(a[i]>max)
        {
            max=a[i];
        }
        if(a[i]<min)
        {
            min=a[i];
        }
    }
    printf("The range of the array is=%d",max-min);
	return 0;
}
