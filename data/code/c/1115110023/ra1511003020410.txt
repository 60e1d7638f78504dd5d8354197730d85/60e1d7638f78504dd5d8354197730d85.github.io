#include <stdio.h>
int main()
{
	int i,len[20],size;
  	int small,large;
  	scanf("%d",&size);
  	for(i=0;i<size;i++){
      scanf("%d",&len[i]);
    }
  	small=len[0];
  	large=len[0];
  	for(i=0;i<size;i++){
      if(len[i]<small)
        small=len[i];
      if(len[i]>large)
        large=len[i];
    }
  	printf("The range of the array is=%d",(large-small));
	return 0;
}
