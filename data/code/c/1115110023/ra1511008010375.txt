#include <stdio.h>
int main()
{
int arr[100],i=0,n=0,max=-7777,min=99999;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&arr[i]);
    if(arr[i]>max)
    {
      max=arr[i];
    }
    if(arr[i]<min)
    {
      min=arr[i];
    }
  }
  printf("The range of the array is=%d",(max-min));
	return 0;
}
