#include <stdio.h>
int main()
{
  int n,i;
  scanf("%d",&n);
  int a[n];
  for(i=0;i<n;i++)
    scanf("%d",&a[i]);
  
  int low=a[0];
  int high=a[0];
  for(i=0;i<n;i++)
  {
    if(a[i]<low)
      low=a[i];
    if(a[i]>high)
      high=a[i];
  }
  
  printf("The range of the array is=%d",high-low);

	return 0;
}
