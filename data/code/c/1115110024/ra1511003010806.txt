#include <stdio.h>
int main()
{
	int n=3;
  	int a[n][n],i,j,sum0=0,sum1=0,sum2=0,sum3=0,sum4=0,sum5=0,sum6=0,sum7=0;
  	for(i=0;i<n;i++)
    {
      for(j=0;j<n;j++)
        scanf("%d",&a[i][j]);
    }
  for(i=0;i<n;i++)
  {
    for(j=0;j<n;j++)
    {
      if(i==0)
        sum0+=a[i][j];
      if(i==1)
        sum1+=a[i][j];
      if(i==2)
        sum2+=a[i][j];
      if(j==0)
        sum3+=a[i][j];
      if(j==1)
        sum4+=a[i][j];
      if(j==2)
        sum5+=a[i][j];
      if(j==i)
        sum6+=a[i][j];
      if(i+j==n-1)
        sum7+=a[i][j];
    }
  }
  	if(sum0==sum1 && sum1==sum2 && sum2==sum3 && sum3==sum4 && sum4==sum5 && sum5==sum6 && sum6==sum7)
      printf("Yes");
  	else
      printf("No");
	return 0;
}
