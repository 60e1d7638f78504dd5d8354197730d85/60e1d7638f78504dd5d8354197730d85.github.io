#include <stdio.h>
int main()
{
   int n;
   scanf("%d",&n);
  int i,j,flag;
  int arr[n][n];
  for(i=0;i<n;i++)
    for(j=0;j<n;j++)
      scanf("%d",&arr[i][j]);
  flag=1;
  for(i=0;i<n;i++)
    for(j=0;j<n;j++)
      if(i==j)
      {
      	if(arr[i][j]=0)
        flag=0;
      }
    else
    {
        if(arr[i][j]!=0)
        flag=0;
    }
  if(flag==1)
  {
    printf("yes");
  }
  else
  {
    printf("no");
  }
	return 0;
}
