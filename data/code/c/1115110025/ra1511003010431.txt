#include <stdio.h>
int main()
{
  int n, a[5][5], i, j, flag=0;
  scanf("%d", &n);
  for(i=0;i<n;i++){
    for(j=0;j<n;j++){
      scanf("%d", &a[i][j]);
    }
  }
  for(i=0;i<n;i++){
    for(j=0;j<n;j++){
      if(i!=j){
        if(a[i][j]!=0){
          flag=1;
        }
      }
    }
  }
  if(flag==0){
    printf("yes");
  }
  else{
    printf("no");
  }

	return 0;
}
