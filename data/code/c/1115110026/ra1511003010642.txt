#include <stdio.h>
int main()
{
	int n,i,j,a[100][100],count=0,count1=0;
  	scanf("%d",&n);
  	for(i=0;i<n;i++){
    	for(j=0;j<n;j++){
        	scanf("%d",&a[i][j]);
          	if(a[i][j]%2==0)
              count++;
          	else
              count1++;
        }
    }
  	if(count==(n*n) || count1==(n*n))
      printf("yes");
  	else
      printf("no");
      
	return 0;
}
