#include<stdio.h>

 
main()
{
   int m, n, c, d, matrix[10][10], transpose[10][10];
 
   
   scanf("%d%d",&m,&n);
  
 
   for (  c = 0 ; c < m ; c++ )
      for ( d = 0 ; d < n ; d++ )
         scanf("%d",&matrix[c][d]);
 
   for( c = 0 ; c < m ; c++ )
   {
      for( d = 0 ; d < n ; d++ )
      {
         transpose[d][c] = matrix[c][d];
      }
   }
 
   if ( m == n ) 
   {
       for ( c = 0 ; c < m ; c++ )
       {
           for ( d = 0 ; d < m ; d++ )
           {
               if ( matrix[c][d] != transpose[c][d] )
                  break;
           }
           if ( d != m )
              break;
       }
       if ( c != m )
       
          printf("Not Symmetric");
   
   else if(c==m)
   
     printf("Symmetric");
   }
   return 0;
}
