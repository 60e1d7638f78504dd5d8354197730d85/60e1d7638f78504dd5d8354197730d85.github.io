#include <stdio.h>
 int main()
{
   int n;
   scanf("%d",&n);
    int array[n][n], i, j, flag = 0 ;
    for (i = 0; i < n; i++)
    {
        for (j = 0; j < n; j++)
        {
            scanf("%d", &array[i][j]);
        }
    }
    for (i = 0; i < n; i++)
    {
        for (j = 0; j < n; j++)
        {
            if (array[i] < array[j] && array[i][j] == 0)
            {
                flag = flag + 1;
            }
        }
    }
    if (flag == n)
        printf("yes");
    else
        printf("no");
   
   return 0;
 }
