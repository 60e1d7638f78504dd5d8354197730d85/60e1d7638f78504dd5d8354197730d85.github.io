#include<stdio.h>
#define g getchar_unlocked
int read()
{
int n=0;
char c=g();
while(c<'0' || c>'9')
c=g();
while(c>='0' && c<='9')
{
n = (n<<3) +(n<<1) +c - '0';
c=g();
 
}
return n;
}
 
 
 
int main()
{
int t,i,n,flag1=0,flag2=0,temp,temp1;
t=read();
while(t--)
{
 
n=read();
temp1=0;
for(i=0;i<n;++i){
temp=read();
 
 
if(temp1==temp)
flag2++;
else if(temp1!=temp)
{temp1=temp;
flag2=0;}
 
if(flag2 %2==0 && flag2>0)
{
flag1=1;
} 
 
 
}
if(flag1==1)
printf("Yes\n");
else
printf("No\n");
temp1=0;
flag1=0;
flag2=0;
}
return 0;
}
