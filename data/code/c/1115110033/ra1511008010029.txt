#include <stdio.h>
int main()
{
  int n,a[50],i,j,temp;
scanf("%d",&n);
  for(i=1;i<=n;i++)
  {
    scanf("%d",&a[i]);
  }
  
  for(i=1;i<=n;i++)
  {
    for(j=i+1;j<=n;j++)
    {
      if(a[j]<a[i])
       
      {temp=a[i];
       a[i]=a[j];
       a[j]=temp;}
    }
  }
  printf("%d",a[n]);
        
      
  
	return 0;
}
