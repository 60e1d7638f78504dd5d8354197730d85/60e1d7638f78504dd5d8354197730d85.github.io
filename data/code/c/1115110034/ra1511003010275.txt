#include <stdio.h>
 
int main()
{
    int i, num1, num2, count = 0, sum = 0;
 

    scanf("%d %d", &num1, &num2);
    /* Count the number and compute their sum*/
  
    for (i = num1; i < num2; i++)
    {
        if (i % 5 == 0)
        {
            
            count++;
            sum = sum + i;
        }
    }
    printf("%d ",  count);
    printf("%d", sum);
    return 0;
           }
