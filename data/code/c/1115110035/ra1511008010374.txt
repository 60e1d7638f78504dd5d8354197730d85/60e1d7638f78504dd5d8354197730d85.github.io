#include <stdio.h>
int main()
{
    int n,a[10],i,j,temp;
   scanf("%d\n",&n);
   for(i=0;i<n;i++)
     scanf("%d ",&a[i]);
   for(i=0;i<n;i++)
   {
     for(j=0;j<n-i-1;j++)
     { 
       if(a[j]<a[j+1])
          {
       temp=a[j+1];
       a[j+1]=a[j];
       a[j]=temp;
          }
     }
   }
  printf("%d",a[1]);
       
	return 0;
}
