#include <stdio.h>
int main()
{
int n,arr[100],i,j,temp,count=0;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&arr[i]);
  }
  
  for(i=0;i<(n-1);i++)
  {
    for(j=0;j<(n-i-1);j++)
    {
      if(arr[j]<arr[j+1])
         {
           temp=arr[j];
           arr[j]=arr[j+1];
           arr[j+1]=temp;
         }
    }
  }
   for(i=0;i<n;i++)
  {
     if(i==1)
     {
    printf("%d\n",arr[i]);
     }
     else
       count++;
  }
	return 0;
}
