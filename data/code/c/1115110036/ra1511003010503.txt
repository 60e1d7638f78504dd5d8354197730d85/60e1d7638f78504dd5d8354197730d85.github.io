#include <stdio.h>
int main()
{
  int i,j,arr[100],size,max,min;
  scanf("%d",&size);
  for(i=0;i<size;i++)
  {
    scanf("%d",&arr[i]);
  }
  max=arr[0];
  min=arr[0];
  for(i=0;i<size;i++)
  {
    if(arr[i]>max)
    {
      max=arr[i];
    }
    else if(arr[i]<min)
    {
      min=arr[i];
    }
  }
  printf("%d\n",max);
  printf("%d",min);
return 0;
}
