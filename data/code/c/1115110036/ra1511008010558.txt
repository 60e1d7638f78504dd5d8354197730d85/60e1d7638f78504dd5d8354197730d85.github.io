#include <stdio.h>

int main()
{
    int arr[100], n, i, small, large;

    
    scanf("%d", &n);

    for (i = 0; i < n; i++)
    {
      
        scanf("%d", &arr[i]);
    }

    small = arr[0];
    large = arr[0];

    for (i = 1; i < n; i++)
    {
        if (arr[i] < small)
        {
            small = arr[i];
        }
        
        if (arr[i] > large)
        {
            large = arr[i];
        }
    }

    printf("%d", large);
    printf("\n%d", small);

    return 0;
}
