#include <stdio.h>
int main()
{
int n,i,j;
  scanf("%d",&n);
  int array[n];
  for(i=0;i<n;i++)
  {
    scanf("%d",&array[i]);
  }
  
  for(i=0;i<n-1;i++)
  { 
    for(j=0;j<n-i-1;j++)
  {
    if(array[j+1]<array[j])
    {
      int temp;
      temp=array[j];
      array[j]=array[j+1];
      array[j+1]=temp;
    }
  }
  }
  for(i=0;i<n;i++)
  {
    printf("%d ",array[i]);
  }
	return 0;
}
