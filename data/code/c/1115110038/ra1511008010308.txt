#include <stdio.h>
int main()
{
int a[50];
  int s_small,small,n,i;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  s_small=small=a[0];
  for(i=1;i<n;i++)
  {
    if(small>a[i])
    { s_small=small;
    small=a[i];
    }
    else if(s_small>a[i] && a[i]!=small)
      s_small=a[i];
  }
  printf("%d",s_small);
  
  return 0;
}
