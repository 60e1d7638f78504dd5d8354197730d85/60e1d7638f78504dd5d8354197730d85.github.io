#include <stdio.h>
int main()
{
    int n;
    scanf("%d",&n);
    int i,j,a[n],temp;
    for(i=0;i<n;i++)
        scanf("%d",&a[i]);
    for(i=0;i<n;i++)
    {  for(j=0;j<n;j++)
          if(a[i]<a[j])
          {
            temp=a[j];
            a[j]=a[i];
            a[i]=temp;
          }
    }
    printf("%d",a[1]);
	return 0;
}
