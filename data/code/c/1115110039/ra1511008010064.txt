#include <stdio.h>
int main()
{
	int n;
  	scanf("%d",&n);
  	int i,j;int arr[30];
  	
  	for(i=0;i<n;i++)
    {scanf("%d",&arr[i]);}
  		
  	for(i=1;i<=n;i++)
    {
      for(j=0;j<n-i;j++)
      {
        if(arr[j]>arr[j+1])
        {
          int temp=arr[j];
          arr[j]=arr[j+1];
          arr[j+1]=temp;
        }
      }
    }
  
  	for(i=0;i<n;i++)
      printf("%d ",arr[i]);
  
	return 0;
}
