#include <stdio.h>
 
int main()
{
   int n,m,i,j;
 
   scanf("%d%d",&n,&m);
   int arr[n][m],tran[m][n];
   for(i=0;i<n;i++){
   	for(j=0;j<m;j++)
   	scanf("%d",&arr[i][j]);
   }
   for(i=0;i<n;i++){
   	for(j=0;j<m;j++)
   	tran[j][i]=arr[i][j];
   }
   for(i=0;i<m;i++){
   	for(j=0;j<n;j++)
   	printf("%d ",tran[i][j]);
   	printf("\n");
   }
    return 0;
}
