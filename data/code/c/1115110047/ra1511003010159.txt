#include <stdio.h>
int main()
{  int arr[100];
    int i, size, even, odd;

    /* Input size of the array */
   
    scanf("%d", &size);

    /* Input array elements */
  
    for(i=0; i<size; i++)
    {
        scanf("%d", &arr[i]);
    }

    /* Assuming that there are 0 even and odd elements */
    even = 0;
    odd  = 0;

    for(i=0; i<size; i++)
    {
        /* If the current element of array is even then increment even count */
        if(arr[i]%2 == 0)
        {
            even++;
        }
        else
        {
            odd++;
        }
    }

    printf("%d\n", even);
    printf("%d", odd);
 
	return 0;
}
