#include <stdio.h>
int main()
{
    int arr[100];
    int i, size, even, odd;
     scanf("%d", &size);
     for(i=0; i<size; i++)
    {
        scanf("%d", &arr[i]);
    }
    even = 0;
    odd  = 0;

    for(i=0; i<size; i++)
    {
        
        if(arr[i]%2 == 0)
        {
            even++;
        }
        else
        {
            odd++;
        }
    }

    printf("%d\n", even);
    printf("%d", odd);

    return 0;
}
