#include <stdio.h>
 
int main()
{
   int array[100], search, c, n;
 
  
   scanf("%d",&n);
 
   for (c = 0; c < n; c++)
      scanf("%d", &array[c]);
 
   
   scanf("%d", &search);
 
   for (c = 0; c < n; c++)
   {
      if (array[c] == search)    
      {
         printf("The number is found");
         break;
      }
   }
   if (c == n)
      printf("The number is not in the list");
 
   return 0;
}
