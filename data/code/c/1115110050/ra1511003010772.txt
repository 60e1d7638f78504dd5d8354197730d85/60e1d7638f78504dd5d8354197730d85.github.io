#include <stdio.h>
int main()
{
  int i,n,a[100],b,flag=0;
  scanf("%d",&n);
  for(i=0;i<n;i++)
    scanf("%d",&a[i]);
  scanf("%d",&b);
  for(i=0;i<n;i++)
  {
    if(b == a[i])
    {
      flag = 1;
      break;
    }
  }
  if(flag == 1)
    printf("The number is found");
  else
    printf("The number is not in the list");
  return 0;
}
