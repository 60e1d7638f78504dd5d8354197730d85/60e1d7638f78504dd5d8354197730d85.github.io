#include <stdio.h>
int main()
{
int a[30], ele, num, i;
 
  // printf("\nEnter no of elements :");
   scanf("%d", &num);
 
   //printf("\nEnter the values :");
   for (i = 0; i < num; i++) {
      scanf("%d", &a[i]);
   }
 
   //Read the element to be searched
   //printf("\nEnter the elements to be searched :");
   scanf("%d", &ele);
 
  
   i = 0;
   while (i < num && ele != a[i]) {
      i++;
   }
 
   
   if (i < num) {
      printf("The number is found");
   } else {
      printf("The number is not in the list");
   }
	return 0;
}
