#include <stdio.h>
int main()
{
  int t;
  scanf("%d",&t);
  int i,day,y;
  while(t--)
  {
    scanf("%d",&i);
    day=0;
    day=((i)*365+(i)/4-(i)/100+(i)/400)%7;
    if((i%4==0)&&(i%100!=0))
      day=day-1;
    if(i%400==0)
      day=day-1;
    if(day==1)
      printf("monday\n");
    else if(day==2)
        printf("tuesday\n");
    else if(day==3)
        printf("wednesday\n");
    else if(day==4)
         printf("thursday\n");
    else if(day==5)
         printf("friday\n");
    else if(day==6)
         printf("saturday\n");
    else if(day==0)
         printf("sunday\n");
    else
         printf("saturday\n");
  }
  return 0;
}
