#include <stdio.h>
int main()
{int t,n,i,y;
scanf("%d",&t);
while(t--)
{
scanf("%d",&n);
y=0;
for(i=1900;i<n;i++)
{
if(i%4==0&&i%100!=0||i%400==0)
y=y+2;
else
y++;
}
y=y%7;
switch(y)
{
case 0:printf("monday\n");
break;
case 1:printf("tuesday\n");
break;
case 2:printf("wednesday\n");
break;
case 3:printf("thursday\n");
break;
case 4:printf("friday\n");
break;
case 5:printf("saturday\n");
break;
case 6:printf("sunday\n");
break;
}
}
	return 0;
}
