#include<stdio.h>
 
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int l1,l2,i,j,k=0,count=0,a[10000],b[10000];
		scanf("%d",&l1);
		for(i=0;i<l1;i++)
		{
			scanf("%d",&a[i]);
		}
		scanf("%d",&l2);
		for(i=0;i<l2;i++)
		{
			scanf("%d",&b[i]);
		}
		for(i=0;i<l2;i++)
		{
			for(j=k;j<l1;j++)
			{
				if(a[j]==b[i])
				{
					count++;
					k=j+1;
					break;
				}
			}
		}
		if(count==l2)
		{
			printf("Yes\n");
		}
		else
		{
			printf("No\n");
		}
	}
	
	
	
	return 0;
}
