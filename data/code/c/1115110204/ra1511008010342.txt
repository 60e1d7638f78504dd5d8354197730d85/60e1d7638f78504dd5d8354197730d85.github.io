#include <stdio.h>
int main()
{
	int t;
  	scanf("%d",&t);
  	while(t--)
    {
      int n;scanf("%d",&n);
      int arr[n];
      int i=0;
      while(i<n){
        scanf("%d",&arr[i]);
        i++;
      }
      int f;
      scanf("%d",&f);
      int arrf[f];
      int j=0;
      while(j<f){
        scanf("%d",&arrf[j]);
        j++;
      }
      int a=0,k=0;
      while(a<n && k<f)
      {
        if(arrf[k]==arr[a]){
          k++;
        }
        a++;
      }
      if(k==f){
        printf("Yes\n");
      }
      else
        printf("No\n");
    }
	return 0;
}
