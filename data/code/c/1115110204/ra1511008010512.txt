#include<stdio.h>
int main()
{
    int t,n1,n2,count=0,a1[10000],a2[10000],i,j,k,m=0;
	scanf("%d",&t);
	for(i=0;i<t;i++)
	{
		scanf("%d",&n1);
		for(j=0;j<n1;j++)
		scanf("%d",&a1[j]);
		scanf("%d",&n2);
		for(j=0;j<n2;j++)
		scanf("%d",&a2[j]);
		m=0;
		count=0;
		for(j=0;j<n1;j++)
		{
			if(a1[j]==a2[m])
			{   
				for(k=j;k<j+n2;k++)
				{
				 if(a1[k]==a2[m])
				 m++;
				 if(m==n2)
				 {
				 	count=1;
				 	break;
				 }
				}
			}
		}
		if(count==1)
		printf("Yes\n");
		else
		printf("No\n");
	}
	return 0;
}
