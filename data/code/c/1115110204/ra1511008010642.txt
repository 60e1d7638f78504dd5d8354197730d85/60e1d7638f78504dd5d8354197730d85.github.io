   #include<stdio.h>
 
    int main()
    {
        int t;
        scanf("%d",&t);
        while(t--)
        {
            int m,n,i,k=0,flag=0;
            scanf("%d",&m);
            int a[m];
            for(i=0;i<m;i++)
            {
                scanf("%d",&a[i]);
            }
            scanf("%d",&n);
            int c[n];
            for(i=0;i<n;i++)
            {
                scanf("%d",&c[i]);
            }
            for(i=0;i<m;i++)
            {
                if(a[i]==c[k]) k++;
                if(k==n){flag=1;break;}
            }
            if(flag==1)
                printf("Yes\n");
            else
                printf("No\n");
        }
            return 0;
    }
