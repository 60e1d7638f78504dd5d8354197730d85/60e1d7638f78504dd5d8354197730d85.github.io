#include<stdio.h>
 
int main(){
	int t,n,f,i;
	int a[2000],b[2000];
	scanf("%d",&t);
	while(t--)
             {
		scanf("%d",&n);
		for(i=0;i<n;i++)
			scanf("%d",&a[i]);
		scanf("%d",&f);
		for(i=0;i<f;i++)
			scanf("%d",&b[i]);
		int j=0;
		for(i=0;i<n;i++)
                      {
			if(a[i]==b[j])
				j++;
			if(j==f)
				break;
		}
		
		if(j==f)
			printf("Yes\n");
		else
			printf("No\n");
	}
	return 0;
}
