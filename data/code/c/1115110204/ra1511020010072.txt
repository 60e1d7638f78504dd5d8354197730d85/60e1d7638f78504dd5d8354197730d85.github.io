#include<stdio.h>
int main()
{
   int t,n,a[1000],f,b[1000],i,k,t1;
  scanf("%d",&t);
  while(t--)
  {
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
      scanf("%d",&a[i]);
    }
    scanf("%d",&f);
    for(i=0;i<f;i++)
    {
      scanf("%d",&b[i]);
    }
    k=0;t1=0;
    for(i=0;i<n;i++)
    {
         if(a[i]==b[k])
         {
          t1++;
          k++;
         }

       if(t1==f)
       {
         break;
       }
    }
    if(t1==f)
      printf("Yes\n");
    else
      printf("No\n");
  }
   return 0;
}
