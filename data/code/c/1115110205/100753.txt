#include <stdio.h>
int main()
{
 long int t,a,b,c,d;
	scanf("%li",&t);
	while(t--)
	{
	    scanf("%li%li%li%li",&a,&b,&c,&d);
	    if(a==0||b==0||c==0||d==0) printf("Yes\n");
	    else if(a+b==0||b+c==0||c+d==0||d+a==0||a+c==0||b+d==0) printf("Yes\n");
	    else if(a+b+c==0||b+c+d==0||c+d+a==0||a+b+d==0) printf("Yes\n");
	    else if(a+b+c+d==0) printf("Yes\n");
	    else printf("No\n");
    }
	return 0;
}
