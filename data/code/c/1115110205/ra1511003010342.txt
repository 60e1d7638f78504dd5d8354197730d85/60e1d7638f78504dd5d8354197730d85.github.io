#include<stdio.h>
int main()
{
	long long int t,a,b,c,d,sum;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&a);
		scanf("%lld",&b);
		scanf("%lld",&c);
		scanf("%lld",&d);sum=a+b+c+d;
		if(a==0||b==0||c==0||d==0)
			{printf("Yes\n");continue;}
		if(sum==0)
			{printf("Yes\n");continue;}
		if(sum-a==0)
			{printf("Yes\n");continue;}
		if(sum-b==0)
			{printf("Yes\n");continue;}
		if(sum-c==0)
			{printf("Yes\n");continue;}
		if(sum-d==0)
			{printf("Yes\n");continue;}
		if(a+b==0)
			{printf("Yes\n");continue;}
		if(a+c==0)
			{printf("Yes\n");continue;}
		if(a+d==0)
			{printf("Yes\n");continue;}
		if(b+c==0)
			{printf("Yes\n");continue;}
		if(b+d==0)
			{printf("Yes\n");continue;}
		if(c+d==0)
			{printf("Yes\n");continue;}
		printf("No\n");
	}
  return 0;
}
