#include <stdio.h>
int main()
{
	 int t,a[4],i,sum=0,j;
    scanf("%d",&t);
    while(t--){
       for(i=0;i<4;++i)
       scanf("%d",&a[i]);
       if(a[0]==0 || a[1]==0 || a[2]==0 || a[3]==0 || (a[0]+a[1]+a[2])==0 || (a[2]+a[1]+a[3])==0 || (a[0]+a[1]+a[3])==0 || (a[0]+a[2]+a[3])==0 || (a[0]+a[1]+a[3]+a[2])==0){
        printf("Yes\n");
        sum=1;
       }
       else{
    for(i=0;i<4;++i){
       for(j=i+1;j<4;++j){
       if(a[i]+a[j]==0){
            printf("Yes\n");
            sum=1;
            break;
        }}
       if(sum==1)
        break;
    }
    if(sum!=1)
    printf("No\n");
    }
    sum=0;}
	return 0;
}
