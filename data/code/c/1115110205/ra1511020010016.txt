#include <stdio.h>
int main()
{
    int t,a,b,c,d;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d%d%d",&a,&b,&c,&d);
        if((a==0)||(b==0)||(c==0)||(d==0)||((a+b+c+d)==0))
        { printf("Yes\n");
            continue;
        }
        else if((a+b+c==0)||(a+b+d==0)||(a+c+d==0)||(b+c+d==0))
        {  printf("Yes\n");
            continue;
        }
        else if((a+b==0)||(a+c==0)||(a+d==0)||(b+c==0)||(b+d==0)||(c+d==0))
             {  printf("Yes\n");
            continue;}
 
        else
            printf("No\n");
    }

	return 0;
}
