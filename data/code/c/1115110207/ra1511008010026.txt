#include <stdio.h>


int main()
{  
	int n;
    scanf("%d",&n);
	int array[n], i, largest1, largest2, temp;


	for (i = 0; i <n; i++)
	{
		scanf("%d", &array[i]);
	}
 



	largest1 = array[0];
	
	largest2 = array[1];

	if (largest1 < largest2)
	{
		temp = largest1;
		largest1 = largest2;
		largest2 = temp;
	}
	
	for (i = 2; i < n; i++) { if (array[i] >= largest1)
		{
			largest2 = largest1;
			largest1 = array[i];
		}
		else if (array[i] > largest2)
		{
			largest2 = array[i];
		}
	}
	printf("First Largest=%d\n", largest1);
	printf("Second Largest=%d\n", largest2);
	printf("Average=%.1f\n",(float)(largest1 + largest2) / 2);

	return 0;

}
