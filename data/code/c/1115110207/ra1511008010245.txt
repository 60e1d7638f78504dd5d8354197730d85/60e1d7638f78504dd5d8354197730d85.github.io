#include <stdio.h> 
#define MAX 4
int main()
{
	int array[MAX], i, largest1, largest2, temp;
	for (i = 0; i < MAX; i++)
	{
		scanf("%d", &array[i]);
	}
	largest1 = array[0];
	largest2 = array[1];
	if (largest1 < largest2)
	{
		temp = largest1;
		largest1 = largest2;
		largest2 = temp;
	}
	for (i = 2; i < 4; i++) { 
		if (array[i] >= largest1)
		{
			largest2 = largest1;
			largest1 = array[i];
		}
		else if (array[i] > largest2)
		{
			largest2 = array[i];
		}
	}
	printf("%d\n", largest1);
	printf("%d\n", largest2);
	printf("%d\n",(largest1 + largest2) / 2);
	return 0;
}
