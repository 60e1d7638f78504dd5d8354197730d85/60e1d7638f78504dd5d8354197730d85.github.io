#include <stdio.h>
int main()
{
	int arr[10], n, i, j, t;
  	scanf("%d", &n);
  	for(i=0; i<n; i++)
      scanf("%d\n", &arr[i]);
  	for(i=0; i<n-1; i++)
    {
    	for(j=0; j<n-1-i; j++)
        {
        	if(arr[j]>arr[j+1])
            {
            	t = arr[j];
              	arr[j] = arr[j+1];
              	arr[j+1] = t;
            }
        }
    }
  	for(i=0; i<n; i++)
      	printf("%d\n", arr[i]);
	return 0;
}
