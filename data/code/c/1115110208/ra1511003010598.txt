#include <stdio.h>
int main()
{
  int n;
  scanf("%d",&n);
  int a[n];
  int i;
  for(i=0;i<n;i++)
    scanf("%d",&a[i]);
  int j;
  for(i=0;i<n;i++)
  {
    for(j=i+1;j<n;j++)
    {
      if(a[i]>a[j])
      {
        int temp;
        temp = a[i];
        a[i]=a[j];
        a[j]=temp;
      }
    }
  }
for(i=0;i<n;i++)
    printf("%d\n",a[i]);
  
  return 0;
}
