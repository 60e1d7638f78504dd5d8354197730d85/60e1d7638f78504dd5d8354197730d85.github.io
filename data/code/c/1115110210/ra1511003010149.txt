#include <stdio.h>
#include <string.h>
int main()
{
  int t,i,j;
  char str[100];
  scanf("%d",&t);
  while(t--)
  {
    scanf("%s",str);
    
    int f=1,a[26]={0};
    
    for(i=0;i<strlen(str)/2;i++)
    {
      j=str[i]-'a';
      a[j]++;
    }
    for(i=(strlen(str)+1)/2;i<strlen(str);i++)
    {
      j=str[i]-'a';
      a[j]--;
    }
  
    for(i=0;i<26;i++)
    {
      if(a[i]!=0){f=0;break;}
    }
    (f==0)?printf("NO\n"):printf("YES\n");
  }

	return 0;
}
