#include <stdio.h>
#include<string.h>
int main()
{
int t;
  scanf("%d",&t);
  while(t--)
  {
   char s[1000];
    int i,l,k,c[1000]={0},c1[1000]={0},f=1;
    scanf("%s",s);
    l=strlen(s);
    k=l/2;
    if(l%2==0)
    {
      for(i=0;i<l/2;i++)
        c[s[i]-97]++;
      for(i=l/2;i<l;i++)
        c1[s[i]-97]++;
    }
    else
    {
      for(i=0;i<l/2;i++)
        c[s[i]-97]++;
      for(i=l/2+1;i<l;i++)
        c1[s[i]-97]++;
    }
    for(i=0;i<26;i++)
      
      if(c[i]!=c1[i])
      {
        f=0;
        break;
      }
    if(f==1)
      printf("YES\n");
    else
      printf("NO\n");
  }
  
  
	return 0;
}
