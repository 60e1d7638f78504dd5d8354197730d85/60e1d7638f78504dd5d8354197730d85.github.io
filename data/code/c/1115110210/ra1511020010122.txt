#include<stdio.h>
#include<string.h>
int main()
{
	char s[1000];
	int l,n,m,i,j,visited=0,t;
	scanf("%d",&t);
	while(t>0)
	{
	scanf("%s",s);
	l=strlen(s);
	int b[l+1];
	for(i=0;i<l;i++)
	{
		b[i]=0;
	}
	m=l/2;
	if(l%2==0)
	{
		for(i=0;i<m;i++)
		{
			visited=0;
			for(j=m;j<l;j++)
			{
				if(b[j]==1)
				continue;
				if(s[i]==s[j])
				{
					b[i]=1;
					b[j]=1;
					visited=1;
					break;
				}
			}
			if(visited==0)
			break;
		}
	}
	else
	{
		for(i=0;i<m;i++)
		{
			visited=0;
			for(j=m+1;j<l;j++)
			{
				if(b[j]==1)
				continue;
				if(s[i]==s[j])
				{
					b[i]=1;
					b[j]=1;
					visited=1;
					break;
				}
			}
			if(visited==0)
			break;
		}
	}
	if(visited==1)
	printf("YES\n");
	else
	printf("NO\n");
	t--;
 } 
}
