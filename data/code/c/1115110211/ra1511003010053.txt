#include<stdio.h>
 
typedef  long long  ll;
 int scan(){
    int t=0;
    char c;
    c=getchar_unlocked();
    while(c<'0' || c>'9')
        c=getchar_unlocked();
    while(c>='0' && c<='9'){
        t=t*10+c-'0';
        c=getchar_unlocked();
    }
    return(t);
}
 int main()
 {   int i,j,I,J,n,m,L;
    ll e1=0,e2=0;
   // ll A[501][501];
   n=scan();
   m=scan();
   //scanf("%d%d",&n,&m);
   ll A[n+1][m+1];
   for(i=1;i<=n;i++)	
   {
   	for(j=1;j<=m;j++)
   	A[i][j]=scan();
	   //scanf("%llud",&A[i][j]);
   }
   L=scan();
   for(i=0;i<L;i++)
   {
   	I=scan();
   	J=scan();
   	//scanf("%d%d",&I,&J);
   	if(e1==-1 && e2==-1)
   	break;
	   if(e1!=-1)
   	{
   		if(I>n || J>m)
   		e1=-1;
   		else
   		e1+=A[I][J];
	   }
	   if(e2!=-1)
	   {
	   	if(J>n || I>m)
	   	e2=-1;
	   	else
	   	e2+=A[J][I];
	   }
   	
   }
   if(e2>=e1)
   printf("%lld",e2);
   else
   printf("%lld",e1);
   return 0;
 	
 }
