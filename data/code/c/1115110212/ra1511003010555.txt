# include <stdio.h>
int main(){
	long long mod=10000007;
	long long n,m,j,i,sum,prod=1,ans,flag=0;
	scanf("%lld%lld",&n,&m);
	long long int a[n][m];
	for(i=0;i<n;i++){
		for(j=0;j<m;j++)
			scanf("%lld",&a[i][j]);
	}
 
	for(i=0;i<m;i++){
		sum=0;
		for(j=0;j<n;j++){
				sum+=a[j][i];
		}
		prod=(prod*sum + 4800*mod)%mod;
	}
		ans=prod;
	printf("%lld\n",ans);
	return 0;
}
