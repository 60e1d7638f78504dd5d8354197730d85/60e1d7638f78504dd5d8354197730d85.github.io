#include <stdio.h>
 
#define MOD 10000007
 
int main() 
{
	int n, m, i, j, temp;
	int arr[47][38];
	long long ans = 1;
	scanf("%d%d", &n, &m);
	
	for (i = 0; i < n; i++)
		for (j = 0; j < m; j++)
			scanf("%d", &arr[i][j]);
 
	for (j = 0; j < m; j++)
	{
		temp = 0;
		for (i = 0; i < n; i++)
			temp += arr[i][j];
		ans = ((ans%MOD)*(temp%MOD)) % MOD;
	}
	if (ans < 0)
		ans += MOD;
	printf("%lld\n", ans);
	return 0;
}
