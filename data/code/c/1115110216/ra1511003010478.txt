#include<stdio.h>
int main()
{
  int n,i,j,k;
  int a[1300];
 int h[1300];
 for(i=0;i<=1004;i++)
  h[i]=0;
 scanf("%d",&n);
 for(i=0;i<n;i++)
 {
   scanf("%d",&a[i]);
 }
 for(i=0;i<n-1;i++)
 {
   if(a[i+1]>=a[i])
   {
   for(j=a[i]+1;j<=a[i+1];j++)
    h[j]++;
   }
   else
   {
     for(j=a[i]-1;j>=a[i+1];j--)
      h[j+1]++;
   }
 }
 //h[a[0]]++;
 int max=0;
 int maxh=0;
 for(i=0;i<1000;i++)
 {
   if(h[i]>=max)
   {
     max=h[i];
    maxh=i;
   }
 }
 printf("%d\n",max);
 return 0;
}
