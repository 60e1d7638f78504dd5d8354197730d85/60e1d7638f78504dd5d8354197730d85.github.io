#include <stdio.h>
#include<stdlib.h>
#include<string.h>
 
#define ll long long
#define slld(a) scanf("%lld",&a);
#define slld1(a,b) scanf("%lld%lld",&a,&b);
#define plld(a) printf("%lld\n",a);
 
int main()
{
    ll a[100001],b[100001]={0},n,i,j,t,p,max,cnt;
    
    slld(n);
    slld(a[0]);
    max=0;
    for(i=1;i<n;i++)
    {
        slld(a[i]);
        if(a[i-1]<a[i])
        {
            p=a[i-1];
            t=a[i];
        }
        else
        {
            t=a[i-1];
            p=a[i];
        }
        for(j=p;j<t;j++)
        {
            b[j]++;
            if(max<b[j])
                max=b[j];
        }
    }
    plld(max);
    return 0;
}
