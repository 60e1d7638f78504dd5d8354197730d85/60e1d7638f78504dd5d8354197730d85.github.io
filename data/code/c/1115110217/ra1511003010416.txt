#include<stdio.h>
#include<string.h>
int main()
{
    int t,x,k,i,d[1000],m[1000],p,j,r[1000];
    char str[1000][33],a[1000][33];//ch[33];
    scanf("%d",&t);
    k=0;
    j=0;
    while(t--)
    {
        scanf("%s %d",a[j],&x);
        for(i=0;i<k;i++)
        {
            if(strcmp(a[j],str[i])==0)
            {
                break;
            }
        }
        if(i<k)
        {
            d[i]+=x;
            m[j]=d[i];
            r[j]=i;
        }
        else
        {
            strcpy(str[i],a[j]);
            k++;
            d[i]=x;
            m[j]=x;
            r[j]=i;
        }
        j++;
    }

    p=0;
    for(i=1;i<k;i++)
    {
        if(d[p]<d[i])
        {
            p=i;
        }
    }
    for(i=0;i<j;i++)
    {
        if((d[p]<=m[i])&&(d[r[i]]==d[p]))
        {
            printf("%s\n",a[i]);
            break;
        }
    }

        
    return 0;
}
