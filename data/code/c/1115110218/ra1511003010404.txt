#include<stdio.h>
 
int a[1001];
 
void reset()
{
	int i;
	for(i=0;i<=1000;i++)
		a[i]=-1;
}
 
 
int toggle(int a)
{
    if(a==1)
        return 0;
    else
        return 1;
}
int main()
{
	int n,m,i,j,t,k;
 
	scanf("%d",&t);
	while(t--)
	{
		reset();
		scanf("%d",&n);
		scanf("%d",&m);
		for(i=1;i<=m;i++)
		{
			scanf("%d",&j);
			a[j]=2;
		}
		k=1;
		for(i=1;i<=n;i++)
		{
			if(a[i]!=2)
			{
				a[i]=k;
                //printf("%d---%d\n",i,k);
				k=toggle(k);//toggle k
 
			}
		}
		for(i=1;(i<=n);i++)
        {
        if(a[i]==1)
			printf("%d ",i);
        }
		printf("\n");
		for(i=1;i<=n;i++)
		{
		if(a[i]==0)
			printf("%d ",i);
		}
		printf("\n");
	}
	return 0;
}
