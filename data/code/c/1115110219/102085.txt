
#include<stdio.h>
void qsort(long int* d, long int l,long int r) {
	 long int ll, rr, p, a;
	if (r > l) {
		ll = l;
		rr = r;
		p = d[(rr + ll) >> 1];
		while (ll <= rr) {
			while (ll < r && d[ll] < p)
				++ll;
			while (rr > l && d[rr] > p)
				--rr;
			if (ll <= rr) {
				a = d[ll];
				d[ll] = d[rr];
				d[rr] = a;
				++ll;
				--rr;
			}
		}
		if (l < rr)
			qsort(d, l, rr);
		if (ll < r)
			qsort(d, ll, r);
	}
}
