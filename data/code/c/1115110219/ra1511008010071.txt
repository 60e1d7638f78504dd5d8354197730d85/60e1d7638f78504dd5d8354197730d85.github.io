#include<stdio.h>
#include<stdlib.h>
 
inline void fastRead(int *a)
{
 register char c=0;
 while (c<33) c=getchar_unlocked();
 *a=0;
 while (c>33)
 {
     *a=*a*10+c-'0';
     c=getchar_unlocked();
 }
}
 
int main()
{
    int t,n,a[1002],i,p,rem;
    scanf("%d",&t);
    getchar();
    while(t--)
    {
        scanf("%d",&n);
        getchar();
        for(i=0;i<n;i++)
        {
            //scanf("%d",&a[i]);
           //getchar();
            fastRead(&a[i]);
        }
        scanf("%d",&p);
        getchar();
        rem=a[p-1];
        p=0;
        for(i=0;i<n;i++)
        {
            if(a[i]<rem)
            {
                p++;
            }
        }
        printf("%d\n",p+1);
    }
}
