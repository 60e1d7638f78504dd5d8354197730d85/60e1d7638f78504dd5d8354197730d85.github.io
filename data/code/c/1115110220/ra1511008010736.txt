#include<stdio.h>
char * day[] ={"Sunday", "Monday","Tuesday", "Wednesday","Thursday", "Friday", "Saturday"};
int solve(int y, int m, int d)
{
       static int t[] = {0, 3, 2, 5, 0, 3, 5, 1, 4, 6, 2, 4};
       y -= m < 3;
       return (y + y/4 - y/100 + y/400 + t[m-1] + d) % 7;
}
int main()
{
    int d, m, y;
    scanf("%d%d%d",&d,&m,&y);
    int w = solve(y, m, d);
    printf("%s\n",day[w]);
    return 0;
}
