#include <stdio.h>
int main()
{
      int limit, number, average, count, sum = 0;
    
      scanf("%d", &limit);
      count = 0;
      while(count < limit)
      {
            
            scanf("%d", &number);
            sum = sum + number;
            count++;
      }
      average = sum / limit;
      printf("%d", average);
      return 0;

	
}
