#include <stdio.h>
int main()
{
  int n,i,count,sum;
  scanf("%d",&n);
  int a[n];
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
    if(a[i]>=0)
    {
      count++;
    }
    else if(a[i]<0)
    {
      sum=sum+a[i];
    }
  }
  printf("Count of positive numbers=%d",count);
  printf("\n");
  printf("Sum of negative numbers=%d",sum);
  

	return 0;
}
