#include <stdio.h>
int main()
{
  int i,n,count=0,sum=0;
  int a[100];
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
    if(a[i]>=0)
    { 
     count++;
    }
    else 
    {
      sum = sum + a[i];
    }
  }  
  printf("Count of positive numbers=%d\n",count++);
  printf("Sum of negative numbers=%d",sum);
  return 0;
}
