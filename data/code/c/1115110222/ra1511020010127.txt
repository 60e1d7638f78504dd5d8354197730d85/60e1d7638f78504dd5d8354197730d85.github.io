#include <stdio.h>
int main()
{
	int a[20] , n,i , x=0 , sum=0 ,  b[10];
  scanf("%d",&n);
  for(i=0 ; i<n ; i++)
  {
    scanf("%d",&a[i]);
  }
  for(i=0 ; i<n ; i++)
  {
    if(a[i]>=0)
    {
      x++;
    }
    else if(a[i]<0)
    {
      sum=sum+a[i];
    }
  }
    b[0]=x;
  b[1]=sum;
    
      printf("Count of positive numbers=%d\nSum of negative numbers=%d",x,sum);
  return 0;
}
