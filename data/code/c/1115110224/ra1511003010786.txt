#include <stdio.h>
int main()
{
  int n,m,i,c=0,j;
  scanf("%d",&n);
  int a[n];
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  scanf("%d",&m);
  int b[m];
  for(i=0;i<m;i++)
  {
    scanf("%d",&b[i]);
  }
  for(i=0;i<m;i++)
  {
    for(j=0;j<n;j++)
    {
      if(b[i]>=a[j])
      {
        c++;
      }
    }
    printf("%d\n",c);
    c=0;
  }
	return 0;
}
