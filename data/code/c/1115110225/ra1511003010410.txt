#include <stdio.h>
int main()
{
  int arr[10];
  int ans[10];
  int i;
    int n;
    scanf("%d",&n);
    for(i=0; i < n; i++)
    {
      scanf("%d",&arr[i]);
    }
    for(i = n-1; i >=0; i--){
        if(i==n-1)
          ans[i] = arr[i];
        else 
          ans[i] = arr[i] + arr[i+1];
    }
    for(i=0; i < n; i++){
        printf("%d ",ans[i]);
    }
    printf("\n");
    return 0;
}
