#include<stdio.h>
const int maxn = 1e5+7;

int main()
{int i;
 int a[maxn],n;
long long b[maxn];
    scanf("%d",&n);
    long long sum = 0;
    for(i=1;i<=n;i++)
        scanf("%d",&a[i]);
    for(i=n;i>=1;i--)
    {
        sum*=-1;
        b[i]=a[i]-sum;
        sum+=b[i];
    }
    for(i=1;i<=n;i++)
        printf("%lld ",b[i]);
    printf("\n");
 return 0;
}
