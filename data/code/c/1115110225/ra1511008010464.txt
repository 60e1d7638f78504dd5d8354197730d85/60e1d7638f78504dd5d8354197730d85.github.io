#include <stdio.h>

typedef long long LL;
LL a[100100];
int main() {
    int n, i;
    scanf("%d",&n);
    for(i = 0;i<n;i++) {
        scanf("%lld",&a[i]);
    }
    for(i = 0;i<n-1;i++) {
        if(i) printf(" ");
        printf("%lld",a[i]+a[i+1]);
    }
    printf(" %lld\n",a[n-1]);
    return 0;
}
