#include <stdio.h>
#include<string.h>
#define ull unsigned long long
 
int C[41], S[41]; ull B[2][41], A[2][41];
 
int main() {
    int T, N, M, c, p, i, j, idx, count; ull ST, CT;
 
    scanf("%d", &T);
    while(T--) {
        memset(S, 0, sizeof(S));
        memset(C, 0, sizeof(C));
        memset(B, 0, sizeof(B));
        memset(A, 0, sizeof(A));
        scanf("%d %d", &N, &M);
        while(N--) {
            scanf("%d %d", &c, &p);
            S[c] += p;
            C[c]++;
        }
        count = 0;
        for(i=1;i<=40;i++)
            if (C[i]) {
                C[count] = (1 << C[i]) - 1;
                S[count++] = S[i]*(1 << (C[i]-1));
            }
        idx = 0;
        ST = 0;
        CT = (M == 0);
        A[idx][count] = 0; B[idx][count] = 0;
        for(j=count-1;j>=0;j--) {
            A[idx][j] = A[idx][j+1]+C[j];
            B[idx][j] = B[idx][j+1]+S[j];
        }
        if (1 >= M) {
            ST += B[idx][0];
            CT += A[idx][0];
        }
        idx = !idx;
        for(i=1;i<count;i++,idx=!idx) {
            A[idx][count-i] = 0; B[idx][count-i] = 0;
            for(j=count-1-i;j>=0;j--) {
                A[idx][j] = A[idx][j+1]+A[!idx][j+1]*C[j];
                B[idx][j] = B[idx][j+1]+B[!idx][j+1]*C[j]+A[!idx][j+1]*S[j];
            }
            if (i+1 >= M) {
                ST += B[idx][0];
                CT += A[idx][0];
            }
        }
        printf("%.9lf\n", ST/((double)CT));
    }
 
    return 0;
}
