#include <stdio.h>
int main()
{
    int t,n,i;
  long  long int c=0,min;
  scanf("%d",&t);
  while(t--)
  {
    scanf("%d",&n);
    int arr[n];
    for(i=0;i<n;i++)
    {
      scanf("%d",&arr[i]);
    }
    min=arr[0];
    for(i=1;i<=n;i++)
    {
      if(min>arr[i])
        min=arr[i];
    }
    c=min*(n-1);
    printf("%lld\n",c);
  }
	return 0;
}
