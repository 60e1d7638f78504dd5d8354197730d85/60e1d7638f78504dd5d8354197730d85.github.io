#include<stdio.h>
#include<string.h>
int main()
{
  char s[100001],m[100001],temp;
  int t,i,j,flag=0;
  long d,n;
  scanf("%d",&t);
  while(t--)
  {
    scanf("%ld",&n);
    scanf("%ld",&d);
    scanf("%c",&temp);
    scanf("%s",s);
    scanf("%s",m);
    for(i=0;i<n;i++)
    {
      for(j=i%d;j<n;j+=d)
      {
        if(s[i]==m[j])
        {
          m[j]=1;
          flag=1;
          break;
        }
      }
      if(flag==0)
      {
        printf("No\n");
        flag=2;
        break;
      }
      flag=0;
    }
    if(flag!=2)
      printf("Yes\n");
    flag=0;
  }
  return 0;
}
