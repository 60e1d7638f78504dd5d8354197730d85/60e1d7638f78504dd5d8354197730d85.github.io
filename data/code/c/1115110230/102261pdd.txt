#include<stdio.h>
 
int min(int a, int b)
{
    return (a<b?a:b);
}
 
int main()
{
    int T,i;
    scanf("%d",&T);
    for(i=0;i<T;i++)
    {
        int M,H;
        scanf("%d",&H);
        scanf("%d",&M);
        
        int H_11,H_09,M_11,M_09,count;
 
        H_09=min(10,H);
        M_09=min(10,M);
        H_11=(H-1)/11; 
        M_11=(M-1)/11;
        count= min(H_09,M_09) + min(H_09-1,M_11) + min(H_11,M_09-1) + min(H_11,M_11);
        
        printf("%d\n",count);
    }
  return 0;
}
