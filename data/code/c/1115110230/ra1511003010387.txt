#include <stdio.h>
int main()
{
  int t,h,m,i,count=0,k;
  scanf("%d",&t);
  while(t--)
  {
    count=0;
    scanf("%d",&h);
    scanf("%d",&m);
    for(i=0;i<h;i++)
    {
      if(i>10)
      {
        if(i%11==0)
        {
          if(i<m)
          {
            count++;
          }
          if(i%10<m)
          {
            count++;
          }
        }
      }
      else if(i<=9)
      {
        if(i<m)
        {
          count++;
          k=i*10+i;
          if(k<m)
          {
            count++;
          }
        }
      }
    }
    printf("%d\n",count-1);
  }
	return 0;
}
