#include <stdio.h>
int main()
{
int t;
  scanf("%d\n",&t);
  while(t--)
  {
    int h,m;
    scanf("%d %d",&h,&m);
    int i,j,count=0;
   for(i=0;i<h;i++)
   {
     for(j=0;j<m;j++)
     {
    	if(j>10&&i<10)
    	{
   			if(j%11==0&&i==j%10)
    		{count++;
        	}
    	}
    	if(j<10&&i>10)
    	{
    		if(i%11==0&&j==i%10)
    		{
    		count++;
   	 		}
   		}
    	if(j>10&&i>10)
    	{
    		if(j%11==0&&i%11==0&&i==j)
    		{
      		count++;
   			}
    	}
   	 if(j<10&&i<10)
    	{
      		if(i==j)
    		{count++;
    		}
   		}
    }
  }
  printf("\n%d",count);}
}
