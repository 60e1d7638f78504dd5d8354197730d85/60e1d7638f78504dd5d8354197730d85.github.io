#include <stdio.h>
 
int main(void) {
	int t;
	scanf("%d",&t);
	while(t--)
	{   int i,j,c=0;
		int h,m;
		scanf("%d %d",&h,&m);
		for(i=0;i<h;i++)
		for(j=0;j<m;j++)
		{if(i<10 && j<10 && i==j)
		c++;
		else if((i==j && (j%10)==(j/10)) || i==(j*10+j) || j==(i*10+i))
		c++;}
		printf("%d\n",c);
	}
	return 0;
}
