#include<stdio.h>
int main()
{
    int n,i,max,smax;
       scanf("%d",&n);
      int a[n];
      int b[1001]={0};
      for(i=0;i<n;i++)
      {
          scanf("%d",&a[i]);
          b[a[i]]++;
      }
      max=0;
      for(i=0;i<1001;i++)
      {
          if(b[i]>=4)
          {
              smax=i;
              max=i;
          }
          else if(b[i]>=2)
          {
              smax=max;
              max=i;
          }
      }
      if((smax*max)>0)
        printf("%d\n",smax*max);
      else
        printf("-1\n");
     return 0;
 
}
