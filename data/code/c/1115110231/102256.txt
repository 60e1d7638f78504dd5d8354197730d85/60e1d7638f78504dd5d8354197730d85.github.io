#include <stdio.h>
int main()
{
int n,a[105]={0},i,j,s[100],b[10],k=0;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&s[i]);
  }
  for(i=0;i<n;i++)
  {
    a[s[i]]++;
  }
  
  for(i=n-1;i>=0;i--)
  {
    if(a[i]>=2)
    {
      b[k]=i;
      k++;
    }
   }
  if(k<2)
    printf("-1");
  else
    printf("%d",b[0]*b[1]);
  
	return 0;
}
