#include <stdio.h>
int main(void)
{      int n;
		scanf("%d",&n);
		int i,a[1001],temp;
		for(i=0;i<1001;i++) a[i]=0;
		for(i=0;i<n;i++)
		{
			scanf("%d",&temp);
			a[temp]++;
		}
		int b=0,h=0;
		for(i=1000;i>0;i--)
		{
			if((h!=0)&&(b!=0)) break;
			
			else if((a[i]>=4)&&(b==0))
			{
				b=i;
				h=i;
			}
			
			else if((a[i]>=2)&&(b==0)) 
			{
				b=i;	
			}
			
			else if((a[i]>=2)&&(b!=0))
			{
				h=i;
			}
		}
		if((h!=0)&&(b!=0)) printf("%d\n",(b*h));
		else printf("-1\n");
		
	
	return 0;
}
