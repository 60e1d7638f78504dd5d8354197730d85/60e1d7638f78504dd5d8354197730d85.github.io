#include <stdio.h>
#include <stdlib.h>
#include <math.h>
#include <string.h>
 
int main(int argc, char *argv[])
{
    int num_test = 0;
    int i, j, k, n, no;
    scanf("%d", &num_test);
    
    int a[102][102];
    int p[501][2];
    int pn = 0;
    int x,y, dx, dy, p3x, p3y, p4x, p4y;
    
    
    for (no = 0; no < num_test; no++) {
        memset(a, 0, sizeof(a));        
        scanf("%d", &pn);
        
        for (i = 0; i < pn; i++) {
            scanf("%d %d", &x, &y);
            x += 50;
            y += 50;
            p[i][0] = x;
            p[i][1] = y;
            
            a[x][y] = 1;
        }
        
        int count = 0;
        for (i = 0; i < pn - 1; i++) {
            for (j = i + 1; j < pn; j++) {
                dx = p[i][0] - p[j][0];
                dy = p[i][1] - p[j][1];                                                    
                
                p3x = p[i][0] + dy;
                p3y = p[i][1] - dx;
                p4x = p[j][0] + dy;
                p4y = p[j][1] - dx;
                if (p3x <= 100 && p3x >= 0 &&
                    p3y <= 100 && p3y >= 0 &&
                    p4x <= 100 && p4x >= 0 &&
                    p4y <= 100 && p4y >= 0) {
                    if (a[p3x][p3y] == 1 && a[p4x][p4y] == 1) {
                       count++;                
                    }        
                }
                
                p3x = p[i][0] - dy;
                p3y = p[i][1] + dx;
                p4x = p[j][0] - dy;
                p4y = p[j][1] + dx;
                if (p3x <= 100 && p3x >= 0 &&
                    p3y <= 100 && p3y >= 0 &&
                    p4x <= 100 && p4x >= 0 &&
                    p4y <= 100 && p4y >= 0) {
                    if (a[p3x][p3y] == 1 && a[p4x][p4y] == 1) {
                       count++;                
                    }        
                }
            }    
        }
        
        printf("%d\n", count/4);
    }    
    
    return 0;
}
