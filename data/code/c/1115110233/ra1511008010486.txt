#include <stdio.h>
#include <string.h>
#include <stdbool.h>
 
#define clr(ar) memset(ar, 0, sizeof(ar))
#define read() freopen("lol.txt", "r", stdin)
 
char str[2010];
bool visited[2010][27][27];
int n, ar[2010], dp[2010][27][27];
 
int F(int i, int l, int r){
    if (i == n) return 0;
    if (visited[i][l][r]) return dp[i][l][r];
 
    int x, res = F(i + 1, l, r);
    if (l == 26 || ar[i] >= l){
        x = 1 + F(i + 1, ar[i], r);
        if (x > res) res = x;
    }
    if (r == 26 || ar[i] <= r){
        x = 1 + F(i + 1, l, ar[i]);
        if (x > res) res = x;
    }
 
    visited[i][l][r] = true;
    return (dp[i][l][r] = res);
}
 
int main(){
    int t, i, j, k;
 
    scanf("%d", &t);
    while (t--){
        scanf("%s", str);
        n = strlen(str);
        for (i = 0; i < n; i++) ar[i] = str[i] - 97;
 
        clr(visited);
        int res = F(0, 26, 26);
        printf("%d\n", res);
    }
    return 0;
}
