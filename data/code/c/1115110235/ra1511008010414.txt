#include <stdio.h>
#define gc getchar_unlocked
#define pc putchar_unlocked
 
inline int scan()
{
	register int c=gc();
	int x=0;
	for(;(c<48||c>57);c=gc());
	for(;(c>47&&c<58);c=gc()){x= (x<<1) + (x<<3) + c-48;}
	return x;
}	
 
inline void print(int a)
{
	char snum[20];
	int i=0;
	do
	{
		snum[i++]=a%10+48;
		a=a/10;
	}while(a!=0);
	i=i-1;
	while(i>=0)
	pc(snum[i--]);
	pc('\n');
}
 
int main(void) {
	    int t,g,i,q,n;
    t=scan();
    while(t--)
    {
    	g=scan();
    	while(g--)
    	{
    		i=scan();
    		n=scan();
    		q=scan();
    	if((i==q)||(n&1)==0)
    	{
    		print(n/2);
    	}
    	else if(i!=q)
    	{
    		print((n/2)+1);
    	}
    	}
 
    }
	return 0;
}
