#include <stdio.h>
  long long a[100005]={0};
  struct row{
    int x,y;
  };
  typedef struct row R;
  
  R r[100005];
int main()
{
  int i,n,q,x,y;
  scanf("%d",&n);
  for(i=1;i<=n;i++){
    scanf("%d%d",&x,&y);
    r[i].x=x;
    r[i].y=y;
    
    a[x]++;
    a[y+1]--;
  }
  for(i=1;i<=n;i++)
    a[i]=a[i]+a[i-1];
  long long tot=0;
  for(i=1;i<=n;i++)
    tot+=a[i];
  scanf("%d",&q);
  while(q--){
    int r1,c;
    scanf("%d%d",&r1,&c);
    long long total=tot;
    total-=a[c];
    total-=r[r1].y-r[r1].x+1;
    if(c>=r[r1].x && c<=r[r1].y)
      total++;
    if(total%2==0)
      printf("E\n");
    else
      printf("O\n");
  }

	return 0;
}
