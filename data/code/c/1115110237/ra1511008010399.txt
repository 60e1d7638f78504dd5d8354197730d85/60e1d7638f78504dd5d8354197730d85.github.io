#include<stdio.h>
inline int input(void)
{
	char t;
    int x=0;
    int neg=0;
	t=getchar();
	while((t<48 || t>57) && t!='-')
		t=getchar();
	if(t=='-')
		{neg=1; t=getchar();}
    while(t>=48 && t<=57)
    {
        x=(x<<3)+(x<<1)+t-48;
        t=getchar();
    }
	if(neg)
		x=-x;
	return x;
}
int main()
{
    int n,i,q,sum,ans,row,co;
    n=input();
    int arr[100005][2],col[100005]={0};
    sum=0;
    for(i=1;i<=n;i++)
    {
        arr[i][0]=input();
        arr[i][1]=input();
        sum+=arr[i][1]-arr[i][0]+1;
        col[arr[i][0]]++;
        col[arr[i][1]+1]--;
    }
    for(i=1;i<=n;i++)
        col[i]+=col[i-1];
 
    q=input();
 
    while(q--)
    {
        row=input();
        co=input();
        ans=sum-(arr[row][1]-arr[row][0]+1)-col[co];
        if(co>=arr[row][0] && co<=arr[row][1])
            ans++;
        if(ans % 2)
            printf("O\n");
        else
            printf("E\n");
    }
    return 0;
}
