#include <stdio.h>
 
int main(void) {
        int t,n,i,k,max;
        scanf("%d",&t);
        while(t--)
        {
              scanf("%d %d",&n,&k);
              max=0;
              for(i=2;i<=k;i++)
              {
                    if(n%i>max)
                    max=n%i;
              }
              printf("%d\n",max);
        }
	return 0;
}
