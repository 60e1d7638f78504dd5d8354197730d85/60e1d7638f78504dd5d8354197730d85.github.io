#include <stdio.h>
 
int main() {
	
	int t, i;
	long int n, k, rem=0, max=0;
	scanf("%d", &t);
	while(t--) {
	    scanf("%ld %ld", &n, &k);
	    max=0;
	    rem=0;
	    for(i=1;i<=k;i++) {
	        rem=n%i;
	        if(rem>max) {
	            max=rem;
	        }
	    }
	    printf("%ld\n", max);
	}
	return 0;
}
