    #include <stdio.h>
     
    int main(void) {
    	int i,t;
    	scanf("%d",&t);
    	for(i=0;i<t;i++)
    	{
    	    int a,b,max=0,j;
    	    scanf("%d %d",&a,&b);
    	    for(j=2;j<=b;j++)
    	    {
    	        int k = a%j;
    	        if(max<k)
    	            max = k;
    	    }
    	    printf("%d\n",max);
    	}
    	return 0;
    }
