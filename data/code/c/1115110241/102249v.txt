#include <stdio.h>
#include <stdlib.h>
 
struct trade {
	int day;
	int c1;
	int c2;
};
int cmpr(const void *p1, const void *p2)
{
   	struct trade *f1, *f2;
   	f1 = (struct trade *) p1;
	f2 = (struct trade *) p2;
	return (f1->day - f2->day);	
}
int main()
{
int corg[200001],cftype[50001];
	int low,high,flag;
	long long total;
	struct trade tr[50001];
	int i,j,k;
	int T,N,M;
	scanf("%d", &T);
	while (T--)
	{
		total = 0;
		scanf("%d %d", &N, &M);
		for (i=0; i<N; i++)
			scanf("%d", &corg[i]);
		tr[0].day=0;
		for (i=1; i<=M; i++)
			scanf("%d %d %d", &tr[i].day, &tr[i].c1, &tr[i].c2);
		for (i=1; i<= 50000; i++)
			cftype[i] = i;
		qsort(tr+1, M, sizeof(struct trade), cmpr);

		high = M;
		low = M;
		while (low >0)
		{
			while (tr[low].day == tr[high].day) low--;
			low = low+1;
			flag=1;
			while (flag)
			{
				flag=0;
				for (i=low; i<=high; i++)
				{
					if (cftype[tr[i].c1] < cftype[tr[i].c2])
					{ 
						cftype[tr[i].c1] = cftype[tr[i].c2];
						flag=1;
					}
					else if (cftype[tr[i].c1] > cftype[tr[i].c2])
					{
						cftype[tr[i].c2] = cftype[tr[i].c1];
						flag=1;	
					}		
				}
			}
			low=low-1;
			high=low;			
		}
		for (i=0; i<N; i++)
		 	total = total + cftype[corg[i]];
		printf("%lld\n", total);
    }
	return 0;
}
