#include <stdio.h>
#include<string.h>

char s[5555555];
int f[5555555];

int main() {
	int i, l = 0, r = 0, p = 1, t = 0;
	scanf("%s",s + 1);

	f[0] = 0;
	for (i = 1; s[i]; ++i, p *= 3) {
		l = l * 3 + s[i];
		r += s[i] * p;
		if (l == r) t += (f[i] = f[i >> 1] + 1);
		else f[i] = 0;
	}

	printf ("%d", t);
	return 0;
}
