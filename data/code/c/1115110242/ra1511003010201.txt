#include <stdio.h>

int main()
{
    int n, i,average;
    float num[100], sum = 0.0;

    
    scanf("%d", &n);

    while (n > 100 || n <= 0)
    {
        
        scanf("%d", &n);
    }

    for(i = 0; i < n; ++i)
    {
       // printf("%d. Enter number: ", i+1);
        scanf("%f", &num[i]);
        sum += num[i];
    }

    average = sum / n;
    printf("%d", average);

    return 0;
}
