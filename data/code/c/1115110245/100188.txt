#include<stdio.h>
#define MAX 505
int men[MAX][MAX],women[MAX][MAX],mMark[MAX],wc[MAX],mc[MAX];
int scan_f()
{
    char c=getchar_unlocked();
    int ans=0;
    while(c<'0'||c>'9')c=getchar_unlocked();
    while(c>='0'&& c<='9')
    {
        ans=ans*10+c-'0';
        c=getchar_unlocked();
    }
    return ans;
}
int wPrefersMOverM1(int w,int m,int m1,int n)
{
    int i;
    for(i=1;i<=n;i++)
    {
        if(women[w][i]==m)
            return 1;
        else if(women[w][i]==m1)
            return 0;
    }
}
void stableMarriage(int n)
{
    int i,freeCount;
    for(i=0;i<MAX;i++)
    {
        mMark[i]=0;
        wc[i]=-1;
    }
    freeCount=n;
    while(freeCount>0)
    {
        int m;
        for(m=1;m<=n;m++)
            if(mMark[m]==0)break;
        for(i=1;i<=n && mMark[m]==0;i++)
        {
            int w=men[m][i];
            if(wc[w]==-1)
            {
                wc[w]=m;
                mMark[m]=1;
                freeCount--;
            }
            else
            {
                int m1=wc[w];
                if(wPrefersMOverM1(w,m,m1,n))
                {
                    wc[w]=m;
                    mMark[m]=1;
                    mMark[m1]=0;
                }
            }
        }
    }
}
int main()
{
    int n,t,i,j;
    t=scan_f();
    while(t--)
    {
        n=scan_f();
        for(i=1;i<=n;i++)
        for(j=0;j<=n;j++)
            women[i][j]=scan_f();
        for(i=1;i<=n;i++)
        for(j=0;j<=n;j++)
            men[i][j]=scan_f();
        stableMarriage(n);
        for(i=1;i<=n;i++)
        {
            mc[wc[i]]=i;
        }
        for(i=1;i<=n;i++)
            printf("%d %d\n",i,mc[i]);
    }
    return 0;
}
