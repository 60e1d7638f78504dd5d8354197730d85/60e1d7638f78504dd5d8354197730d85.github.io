
#include<stdio.h>
#include<stdlib.h>
 
 
int main()
{
int **mpref;
int **wpref;
int n,i,j,temp,w,t;
int *mch;
int *wch;
int top;
int* free_men;
int *next;
scanf("%d",&t);
while(t--)
{
scanf("%d",&n);
 
mpref=(int **)malloc((n+1)*sizeof(int *));
for (i=0;i<n+1;i++)
  mpref[i]=(int *)malloc((n+1)*sizeof(int));
wpref=(int **)malloc((n+1)*sizeof(int *));
for (i=0;i<n+1;i++)
  wpref[i]=(int *)malloc((n+1)*sizeof(int));
mch=(int *)malloc((n+1)*sizeof(int));
wch=(int *)malloc((n+1)*sizeof(int));
free_men=(int *)malloc((n)*sizeof(int));
next=(int *)malloc((n)*sizeof(int));

    for(i=1;i<=n;i++)
    { scanf("%d",&temp);
         for(j=1;j<=n;j++)
        {scanf("%d",&temp);
          wpref[i][temp]=j;
 
        }
    }

    for(i=1;i<=n;i++)
    {scanf("%d",&temp);
        for(j=1;j<=n;j++)
        {
            scanf("%d",&temp);
            mpref[i][j]=temp;
        }
    }
 

    for(i=0;i<=n;i++)
    {
        mch[i]=-1;
        wch[i]=-1;
        next[i]=1;
    }
   
for(i=0;i<=n;i++)
{
  free_men[i]=i ;
}
top=n;
   
    while(top>0)
    {i=free_men[top];
        for(j=next[i];j<=n;j++)
        {
           w=mpref[i][j];
           if(wch[w]==-1)
           {
               mch[i]=w;
               wch[w]=i;
               //fcnt--;
              top--;
              next[i]++;
               break;
           }
           else
           {
               if(wpref[w][wch[w]]>wpref[w][i])
               {
                   mch[wch[w]]=-1;
                  free_men[top]=wch[w];
                   wch[w]=i;
                   mch[i]=w;
                   next[i]++;
                   break;
               }
               else
               {
                   continue; 
 
               }
           }
 
        }
    }
        for(i=1;i<=n;i++)
        printf("%d %d\n",i,mch[i]);
}
        return 0;
}
