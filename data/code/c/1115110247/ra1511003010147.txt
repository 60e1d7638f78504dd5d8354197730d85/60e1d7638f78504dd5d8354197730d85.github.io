#include <stdio.h>
int main()
{
	long long n,k,c=0,i;
	scanf("%lld %lld",&n,&k);
	int a[44]={13,19,31,37,43,53,61,59,79,101,113,139,163,173,199,211,223,241,269,277,331,353,373,397,457,463,509,521,541,577,601,619,631,727,773,787,811,829,853,883,907,947,967,991};
	for(i=0;i<43;i++)
	{
	    if(a[i]<n)
	    c++;
	}
	if(c>=k)
	printf("YES");
	else
	printf("NO");
	return 0;
}
