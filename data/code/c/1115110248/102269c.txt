#include<stdio.h>
long long a[100000],count[100000],mod=1000000007;
int main()
{
        long long n,m,k,temp=0,last=0,ans=0,i;
        scanf("%lld%lld%lld",&n,&m,&k);
        for(i=0;i<n;i++)
        {
            scanf("%lld",&a[i]);
        }
        for(i=0;i<n;i++)
        {
            count[i]=(m-a[i]+k)/k;
            ans+=count[i];
            if(count[i]>temp)temp=count[i];
        }
        ans%=2;
        ans=(temp+2-ans)/2;
        ans%=mod;
        printf("%lld\n",ans);
  return 0;
}
