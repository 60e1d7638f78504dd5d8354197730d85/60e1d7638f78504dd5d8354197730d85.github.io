#include <stdio.h>
int main()
{
  int n,m,k,i,j,c=0;
  scanf("%d %d %d\n",&n,&m,&k);
  int a[n];
  for(i=0;i<n;i++)
  {
    scanf("%d ",&a[i]);
  }
  for(i=0;i<n;i++)
  {
    for(j=i+1;j<n;j++)
    {
      if(a[i]<=m && a[j]<=m)
      {
        c++;
        a[i]+=k;
        a[j]+=k;
      }
    }
  }
	printf("%d",c);
	return 0;
}
