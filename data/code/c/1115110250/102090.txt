#include <stdio.h>
int main()
{
	int i, j, n, arr[10], temp, sum = 0;
  	scanf("%d", &n);
  	for(i = 0; i < n; i++)
      scanf("%d", &arr[i]);
  	for(i = 0; i < n; i++)
    {
      	for(j = 0; j < n-i-1; j++)
        {
          	if(arr[j] > arr[j+1])
            {
              	temp = arr[j];
              	arr[j] = arr[j+1];
              	arr[j+1] = temp;
            }
        }
    }
  	for(i = 1; i < n-1; i++)
      sum += arr[i];
  	printf("%d", sum);
	return 0;
}
