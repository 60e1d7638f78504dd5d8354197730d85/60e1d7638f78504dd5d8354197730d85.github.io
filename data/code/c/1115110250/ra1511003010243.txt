#include <stdio.h>
int main()
{int max,min,i,n,arr[9],sum;
	scanf("%d",&n);
  for(i=0;i<n;i++)
  {
	scanf("%d",&arr[i]);  
  }
  max=arr[0];
  min=arr[0];
 
 for(i=0;i<n;i++)
  {
   if(max<arr[i])
   {
     max=arr[i];
   }
   if(min>arr[i])
   {
     min=arr[i];
   }
  }
 
    for(i=0;i<n;i++)
  {
    sum=sum+arr[i]; 
  }
   sum=sum-max-min;
   printf("%d",sum);

}
