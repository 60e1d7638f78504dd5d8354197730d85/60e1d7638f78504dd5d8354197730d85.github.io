#include <stdio.h>
int main()
{
	int n,a[20],sum=0,min,max,i;
  	scanf("%d",&n);
  	for(i=0;i<n;i++)
    {
    	scanf("%d",&a[i]);
    }
  	min=a[0];
    max=a[0];
  	for(i=0;i<n;i++)
    {
    	if(a[i]>max)
        {
        	max=a[i];
        }
      	if(a[i]<min)
        {
        	min=a[i];
        }
      	sum+=a[i];
    }
  sum-=min;
  sum-=max;
  printf("%d",sum);
	return 0;
}
