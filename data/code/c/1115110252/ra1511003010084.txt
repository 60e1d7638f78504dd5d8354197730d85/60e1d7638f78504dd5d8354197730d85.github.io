#include <stdio.h>
#include <string.h>
void is_uppercase(char in[])
{ 
  int i,x,l=0,p[100];
l=strlen(in);
for(i=0;i<l;i++)
{
  p[i]=(int)in[i];
if(p[i]<=97)
{
x++;
}
  else{x=0;}}
if(x==l)
    printf("true");
  else
    printf("false");
}

int main()
{
char in[20];
scanf("%s",in);
is_uppercase(in);
  return 0;
}
