#include<stdio.h>

main()
{
int a[101];
int b,c,d,x,y,z;
scanf("%d %d",&b,&c);
for(x=1;x<=100;x=x+1)
{
a[x]=0;
}
for(x=1;x<=c;x=x+1)
{
scanf("%d %d",&d,&z);
for(y=d;y<=z;y=y+1)
{
a[y]=a[y]+1;
}
}
z=0;
for(x=1;x<=b;x=x+1)
{
if(a[x]!=1)
{
printf("%d %d",x,a[x]);
z=1;
break;
}
}
if(z==0)
{
printf("OK");
}
return 0;
}
