#include<stdio.h>
int main(){
	int n,m;
	scanf("%d%d",&n,&m);
	int a,i,b;
	int x[101]={0};
	while(m--){
		scanf("%d%d",&a,&b);
  
		for(i=a; i<=b; i++){
			x[i]++;
		}
    }
	int k=0;
	for(i=1; i<=n; i++){
		if(x[i]!=1){
			printf("%d %d",i,x[i]);
			k=1;
			break;
		}
	}
	
if(k==0){
	printf("OK");
}
  return 0;
}
