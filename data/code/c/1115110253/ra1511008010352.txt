#include <stdio.h>
#define MAXN 105
long n,m;
long h[MAXN]={0},x,y;
int main()
{
 long i,j;
 scanf("%ld%ld",&n,&m);
 for(i=1;i<=m;i++)
  {
   scanf("%ld%ld",&x,&y);
   for(j=x;j<=y;j++) h[j]++;
  }
 for(i=1;i<=n;i++) if(h[i]!=1) break;
 if(i==n+1) printf("OK\n");
 else printf("%ld %ld\n",i,h[i]);
 return(0);
}
