#include <stdio.h>
 
int max(int a, int b) {
	return (a>b)?a:b;
}
 
int main()
{
    int t, n, W, w[1010], v[1010], V[1010][1010], i, j;
    
    scanf("%d", &t);
    
    while(t--) {
    	scanf("%d%d", &n, &W);
    	
    	for(i=0; i<n; i++) {
    		scanf("%d", &w[i]);
    	}
    	
    	for(i=0;i<n; i++) {
    		scanf("%d", &v[i]);
    	}
    	
    	for(i=0; i<=n; i++) {
    		for(j=0; j<=W; j++) {
    			V[i][j] = 0;
    		}
    	}
    	
    	for(i=0; i<=n; i++) {
    		for(j=0; j<=W; j++) {
    			if(i == 0 || j == 0) {
    				V[i][j] = 0;
    			} else if (w[i-1] <= j) {
    				V[i][j] = max(v[i-1] + V[i-1][j-w[i-1]], V[i-1][j]);
    			} else {
    				V[i][j] = V[i-1][j];
    			}
    		}
    	}
    	
    	printf("%d\n", V[n][W]);
    }
    return 0;
}
