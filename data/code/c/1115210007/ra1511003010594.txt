#include <stdio.h>
 
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,i,j;
        scanf("%d",&n);
        int a[n];
        long long sum=0;
        for(i=0;i<n;i++)
        {
            scanf("%d",&a[i]);
            sum+=a[i];
        }
        int dp[n+1][sum];
        for(i=0;i<=sum;i++)
            dp[0][i]=0;
        for(i=0;i<=n;i++)
            dp[i][0]=1;
        for(i=1;i<=n;i++)
        {
            for(j=1;j<=sum;j++)
            {
                if(a[i-1]>j)
                    dp[i][j]=dp[i-1][j];
                else
                {
                    dp[i][j]=dp[i-1][j]||dp[i-1][j-a[i-1]];
                }
            }
        }
        int count=0;
        for(i=0;i<=sum;i++)
            {
                if(dp[n][i])
                    count++;
            }
        printf("%d\n",count);
    }
    return 0;
}
