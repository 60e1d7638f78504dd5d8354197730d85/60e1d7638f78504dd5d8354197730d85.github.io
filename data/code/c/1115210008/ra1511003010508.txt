#include <stdio.h>
long int a[100000],b[100000];
int main()
{
    //printf("Hello World!\n");
    int t,n,i,j,max=0,m;
    
    scanf("%d",&t);
    while(t--){
        scanf("%d",&n);
        for(i=0;i<n;i++){
            scanf("%ld",&a[i]);
        }
        for(i=0;i<n;i++){
            scanf("%ld",&b[i]);
        }
        j=n-1;
        max=0;
        for(i=n-1;i>=0;i--){
            while(j>=i && b[j]<a[i]){
                j--;
            }
            m = j>=i?(j-i):0;
            max = (max<m?m:max);
            //printf("%d %d %d %d\n",a[i],m,i,j);
        }
        printf("%d\n",max);
    }
    return 0;
}
