 void fibonacci(long long int N) {
       long long int i;
        fibo[0]=0;
        fibo[1]=1;
        for( i=2;i<=N;i++)
            fibo[i] = (fibo[i-2] + fibo[i-1])%1000000007;
    }
    
    void populateZeroShark(long long int N) {
