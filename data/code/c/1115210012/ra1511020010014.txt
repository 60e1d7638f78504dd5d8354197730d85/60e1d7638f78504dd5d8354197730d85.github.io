#include <stdio.h>
 
int n, k, a[100001], lo, hi, mid, ans, pos, i, cnt;
int cmp(const void *a, const void *b) { return *(int *)a - *(int *)b; }
 
int main() {
	scanf("%d%d", &n, &k);
	for(i = 0; i < n; i++) scanf("%d", a + i);
	//qsort(a, n, sizeof(int), cmp);
	
	ans = hi = (a[n-1] >> 1) + 1;
	lo = 0;
	while(lo <= hi) {
	    mid = (lo+hi)>>1;
	    
	    cnt = 1;
	    pos = a[0] + (mid<<1);
	    for(i = 1; i < n; i++) {
	        if(a[i] > pos) {
	            pos = a[i] + (mid<<1);
	            if(++cnt > k) break;
	        }
	    }
	    
	    if(cnt <= k) {
	        ans = mid;
	        hi = mid - 1;
	    } else {
	        lo = mid + 1;
	    }
	}
	
	printf("%d", ans);

	return 0;
}
