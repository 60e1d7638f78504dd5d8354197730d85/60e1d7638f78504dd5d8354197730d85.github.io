#include<stdio.h>
#define MAXN 1010
#define MAXW 1010
#define MAX(x, y) (((x) > (y)) ? (x) : (y))
double DP[MAXN][MAXW];
int main(){
 int t,i,j;
 scanf("%d",&t);
 while(t--){
  int X,Y,N,W,P1,P2;
    scanf("%d%d%d%d%d%d",&X,&Y,&N,&W,&P1,&P2);
  for(i=0;i<=N;i++){
   for(j=0;j<=W;j++){
    DP[i][j]=0;
   }
  }
  double p1 = 0.01 * P1;
  double p2 = 0.01 * P2;
  for (i=0;i<=N;i++) 
        DP[i][0] = 1; 
     for (i=1;i<=W;i++) 
        DP[0][i] = 0; 
     for (i=1;i<=N;i++){ 
        for (j=1;j<=W;j++) { 
          DP[i][j] = MAX(p1*DP[i-1][MAX(j-X,0)] + (1-p1)*DP[i-1][MAX(j,0)], p2*DP[i-1][MAX(j-Y,0)] + (1-p2)*DP[i-1][MAX(j,0)]); 
        } 
       }
       printf("%.6f\n",DP[N][W]*100);
 }
  return 0;
}
