#include<stdio.h>
#include<math.h>
int main() {
    int t;
    int i,n,k,a,b,x;
    scanf("%d",&t);
    while(t--) {
        scanf("%d %d",&n,&k);
        if(k>81)
            printf("-1\n");
        else {
            a=ceil((double)k/9.0);
            x=k-9*(a-1);
            b=x<=a?x-1:x;
            for(i=0;i<n;i++) {
                if(!(i&1))
                    printf("%d",a);
                else
                    printf("%d",b);
            }
            printf("\n");
        }
    }
    return 0;
}
