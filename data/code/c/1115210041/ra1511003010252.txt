#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>
 
#define MIN(a,b) (((a)<(b))?(a):(b))
#define MAX(a,b) (((a)>(b))?(a):(b))
 
#define MA 1000000000000000000 // 
#define M  1000000007
#define MM 1001
#define K  1001
 
int m,n;
int a[K];
int s[K][K];
 
int main() {
  int t;
  int i,j, x;
 
  scanf("%d", &t);
  //t=1;
 
  while(t--) {
    scanf("%d", &n);
    scanf("%d", &m);
 
    memset(s, 0,sizeof(s));
 
    for(i=0;i<n;i++) s[0][i]=1;
 
    for(i=1;i<=n/(m+1);i++) {
      s[i][0]=0;
      for(j=1;j<n;j++) {
        s[i][j]=s[i][j-1];
        if (j>=i*(m+1)-1)
          s[i][j]+=s[i-1][j-1];
        //printf("%d ", s[i][j]);
        s[i][j]%=M;
      }
     
    }
 
    x=1;
    for(i=1;i<=n/(m+1);i++) {
      if (m!=1 || i*(m+1)!=n)
        x=(x+s[i][n-1])%M;
    }
 
    printf("%d\n", x);
  }
  return 0;
}
