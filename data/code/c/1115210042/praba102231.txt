#include <stdio.h>
#define ll long long
ll int maxi(ll int x,ll int y)
{
    if(x>y)
        return x;
    return y;
}
int main()
{
    ll int n,m,a[1009][1009],i,j,v1,v2,v3,v4;
    unsigned ll int ans=0,max1=0,max2,max=0,ans1=0;
    scanf("%lld%lld",&n,&m);
    for(i=0;i<n;i++){
        for(j=0;j<m;j++)
            scanf("%lld",&a[i][j]);
    }
    scanf("%lld%lld%lld%lld",&v1,&v2,&v3,&v4);
    for(i=0;i<n;i++){
        max=0;
        max1=0;
        max2=0;
        for(j=0;j<m;j++){
            max+=abs(a[i][j]+v1);
            max2+=abs(a[i][j]);
            max1+=abs(v2);
        }
        max2=maxi(max,maxi(max1,max2));
        ans+=max2;
    }
    for(i=0;i<m;i++){
        max=0;
        max1=0;
        max2=0;
        for(j=0;j<n;j++){
            max+=abs(a[j][i]+v3);
            max2+=abs(a[j][i]);
            max1+=abs(v4);
        }
        max2=maxi(max,maxi(max1,max2));
        ans1+=max2;
    }
    printf("%lld",maxi(ans,ans1));
    
    return 0;
}
