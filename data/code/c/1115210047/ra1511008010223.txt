#include<stdio.h>
 
typedef struct point
{
	int x,y,d,uc;
	} pt;
	
pt a[2505];
 
int dst(int i, int j)
{
    return ((a[i].x-a[j].x)*(a[i].x-a[j].x)+(a[i].y-a[j].y)*(a[i].y-a[j].y));
}
 
int main()
{
    int t,n,i,j,l,k,min,minpt;
    scanf("%d",&t);
    while(t--)
    {
	min=1000000000;
	scanf("%d",&n);
	n+=2;
 
	for(i=1;i<n;i++)
	{
	    scanf("%d%d",&a[i].x,&a[i].y);
	    a[i].uc=1;
	}
 
	scanf("%d%d",&a[0].x,&a[0].y);
	a[0].d=0;
	for(i=1;i<n;i++)
	{
	    a[i].d=dst(0,i);
	    if(min>a[i].d)
	    {
		min=a[i].d;
		minpt=i;
	    }
	}
	a[minpt].uc=0;
 
	while(a[n-1].uc)
	{
	    min=1000000000;
	    l=minpt;
	    for(i=1;i<n;i++)
		if(a[i].uc)
		{
		    if(a[l].d+dst(l,i)<a[i].d)
			a[i].d=a[l].d+dst(l,i);
		    if(min>a[i].d)
		    {
			min=a[i].d;
			minpt=i;
		    }
		}
	    a[minpt].uc=0;
	}
	printf("%d\n",a[n-1].d);
    }
    return 0;
}
