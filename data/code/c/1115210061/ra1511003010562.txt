#include <stdio.h>
 
#include <string.h>
int func(char s1[],char s2[]){
    int i,j,flag=0;
    for(i=0;i<strlen(s1);i++){
        for(j=0;j<strlen(s2);j++){
            if(s1[i]==s2[j]){
                flag=1;
                break;
            }
           
        }
    }
    return flag;
}
int main(void) {
    int test,check;
    char a[1001],b[1001];
    scanf("%d",&test);
    while(test--){
       scanf("%s",a);
       scanf("%s",b);
      //gets(a);
      //gets(b);
        check=func(a,b);
        if(check==1){
            printf("Yes\n");
        }
        else{    
            printf("No\n");
            
        }
        
    }
	return 0;
}
