#include <stdio.h>
#include <string.h>
 
int main(void) 
{
    char s1[1001],s2[1001];
    int t,l1,l2,i,j,flag;
    scanf("%d",&t);
    while(t--)
    {
    fflush(stdin);
    scanf(" %[^\n]s",s1);
    //gets(s1);
    scanf(" %[^\n]s",s2);
    fflush(stdin);
    //gets(s2);
    //printf("%s %s\n",s1,s2);
    l1=strlen(s1);
    l2=strlen(s2);
    flag=0;
    for(i=0;i<l1;++i)
    {
        if(flag)
            break;
        for(j=0;j<l2;++j)
        {
            if(s1[i]==s2[j])
            {
                flag=1;
                break;
            }
        }
    }
    
    if(flag)
        printf("Yes\n");
    else
        printf("No\n");    
	}
	return 0;
}
