#include<stdio.h>
#include<string.h>
int main()
{ int t;scanf("%d",&t);
  while(t--)
   {
     int n,i,j,k,r=0; scanf("%d",&n); int a[n];long long int z=0;
 
     for(i=0;i<n;i++)
       { scanf("%d",&a[i]);}
 
   
   for(i=0;i<n;i++)
    { j=i;r=1;
      while(a[j+1]>=a[j]&&(j+1)<n)
        {r++;j++;}
 
     for(k=1;k<=r;k++)
      { z=z+k;}
       i=j;
    }
 
   printf("%lld\n",z);}
return 0;
}
