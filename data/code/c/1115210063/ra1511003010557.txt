    #include<stdio.h>
    #define gc getchar_unlocked
     
    inline int getn()
    {
        register int c = gc();
        int x = 0;
        for(;(c<48 || c>57);c = gc());
        for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
        return x;
    }
     int t, i, n ,currentNum, previousNum;
     long long count,r;
    int main() {
    	 
    	t = getn() ; 
    	while(t--) {
    		n = getn() ; 
    		 count = 0 ; 
    		 r = 0 ;
    		previousNum = getn();
    		r = 1 ; 
    		for(i = 1 ; i < n ; i++) {
    			currentNum = getn() ; 
    			if(currentNum >= previousNum) {
    				r++;
    			} else {
    				count += (r * (r+1))/2 ;
    				r = 1 ; 
    			}
    			previousNum = currentNum;
    		}
    		count += (r * (r+1))/2 ;
    		printf("%lld\n" , count);
    	}
    	return  0 ; 
    }
