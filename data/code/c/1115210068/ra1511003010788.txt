#include<stdio.h>
int main()
{
  int t,i;
  scanf("%d",&t);
  for(i=0;i<t;i++)
  {
    int n,j,count=1;
    scanf("%d",&n);
    int a[n];
    for(j=0;j<n;j++)
    	scanf("%d",&a[j]);
    for(j=1;j<n;j++)
    {
      if(a[j]<a[j-1])
        count++;
      else
      	a[j]=a[j-1];
    }
    printf("\n%d",count);
  }
  return 0;
}
