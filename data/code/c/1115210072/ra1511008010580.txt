#include <stdio.h>
 
int main ()
{
    int number[30];
    int i, n, a, j;
    scanf("%d", &n);
    for (i = 0; i < n; ++i)
        scanf("%d", &number[i]);
    scanf("%d", &a);
    for (i = 0; i < a; ++i)
    {
        number[n] = number[0];
        for (j = 0; j < n; ++j)
        {
            number[j] = number[j + 1];
        }
    }
    for (i = 0; i < n; ++i)
    {
        printf("%d\n", number[i]);
    }
  return 0;
}
