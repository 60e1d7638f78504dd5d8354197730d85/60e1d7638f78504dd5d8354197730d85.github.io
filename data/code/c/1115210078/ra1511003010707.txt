#include <stdio.h>
#include <stdlib.h>
 
int main() {
    int z;
    scanf("%d", &z);
    while (z--) {
        int a, b;
        scanf("%d%d", &a, &b);
        int steps = 0;
        while ((a & -a) != a) a >>= 1, steps++;
        while (a < b) a <<= 1, steps++;
        while (a > b) a >>= 1, steps++;
        printf("%d\n", steps);
    } return 0;
}
