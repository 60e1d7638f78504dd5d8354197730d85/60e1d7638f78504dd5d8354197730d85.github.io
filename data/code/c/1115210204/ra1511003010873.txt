#include <stdio.h>
int main()
{
  int a[10],n,i,po=0,ne=0;
  float avg;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  for(i=0;i<n;i++)
  {
    if(a[i]<0)
    {
      ne=ne+a[i];
    }
    else
    {
      po=po+a[i];
    }
  }
  avg=(ne+po)/(float)n;
  printf("%d\n%d\n%.2f",ne,po,avg);
	return 0;
}
