#include<stdio.h>
int lookup[305][305][11]={0};
int main()
{
	int n;
	scanf("%d",&n);
	int i,j,a;
	
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)
		{
			int x;
			scanf("%d",&x);
			for(a=1;a<11;a++)
			{
				lookup[i][j][a]=lookup[i][j-1][a]+lookup[i-1][j][a]-lookup[i-1][j-1][a];
			}
			lookup[i][j][x]++;			
		}
	}
	int q;
	scanf("%d",&q);
	for(i=0;i<q;i++)
	{
		int count=0;
		int x1,x2,y1,y2;
		scanf("%d %d %d %d",&x1,&y1,&x2,&y2);
		for(a=1;a<11;a++)
		{
			if(lookup[x2][y2][a]-lookup[x2][y1-1][a]-lookup[x1-1][y2][a]+lookup[x1-1][y1-1][a]>0)
				count++;
		}
		printf("%d\n",count);
	}
  return 0;
 
	
}
