#include <stdlib.h>  
#include<stdio.h>
int main() {
 
long long int n,i,j,k,f;
    scanf("%lld",&n);
  static int a[310][310][11];
    for(i=1;i<=n;i++)
    {for(j=1;j<=n;j++)
        {scanf("%lld",&k);
          for(f=1;f<=10;f++)
           {a[i][j][f]=a[i][j-1][f];
           }
          a[i][j][k]++;
            
        }
    }
 
    long long int q;
    scanf("%lld",&q);
    while(q--)
   {int x1,x2,y1,y2;
    scanf("%d %d %d %d",&x1,&y1,&x2,&y2);
    int s[11]={0};
     for(i=x1;i<=x2;i++)
     {for(j=1;j<=10;j++)
         {s[j]=s[j]+a[i][y2][j];
         }
      for(j=1;j<=10;j++)
         {s[j]=s[j]-a[i][y1-1][j];}
     }
    int ans=0;
    for(i=1;i<=10;i++)
   {if(s[i]>0)
        ans++;
    }
    printf("%d\n",ans);
    }     
    return 0;
}
