#include<stdio.h>
#include<stdlib.h>
 
int main(void) {
	int t,i,j,n;
	long int count;
	scanf("%d",&t);
	while(t--){
	    scanf("%d",&n);
	    char a[n][n+1];
	    int npr[1000]={0};
	    int npc[1000]={0};
	    count=0;
	    for(i=0;i<n;i++){
	        scanf("%s",a[i]);
	    }
	    
	    for(j=n-1;j>=0;j--){
	        for(i=n-1;i>=0;i--){
	            if((a[i][j]=='.')&&(npr[i]==0)&&(npc[j]==0)){
	                count++;
	                //printf("-rty,%d,%d-",i,j);
	            }
	            else if(a[i][j]=='#'){
	                npr[i]=1;
	                npc[j]=1;
	                //printf("-qwe,%d,%d-",i,j);
	            }
	        }
	    }
	    printf("%ld\n",count);
	}
	return 0;
}
