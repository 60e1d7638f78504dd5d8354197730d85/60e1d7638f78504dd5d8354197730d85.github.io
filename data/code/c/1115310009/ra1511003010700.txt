#include<stdio.h>
#include<stdlib.h>
 
int compare(const void *a, const void *b)
{	return *((int*)a) - *((int*)b);
}
 
int main()
{	int t, n, d, i;
	scanf("%d", &t);
	while(t--)
	{	scanf("%d%d", &n, &d);
		int arr[n];
		long long int ans=0;
		for(i=0 ; i<n ; i++)
			scanf("%d", arr+i);
		qsort(arr, n, sizeof(int), compare);
		for(i=n-1 ; i>0 ; )
			if(arr[i] - arr[i-1] < d)
			{	ans += arr[i]+arr[i-1];
				i -= 2;
			}
			else
				i--;
		printf("%lld\n", ans);
	}
	return 0;
}
