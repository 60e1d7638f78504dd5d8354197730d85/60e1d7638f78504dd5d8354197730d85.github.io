#include<stdio.h>
#include<string.h>
int main()
{
int t;
scanf("%d",&t);
while(t--)
{
char s[5001];
int l,i,j,flag=0,count[3];
for(i=0;i<3;i++)
count[i]=0;
scanf("%s",s);
l=strlen(s);
for(i=0;i<l;i++)
{
count[s[i]-'A']++;
}
if((count[0]==2))
{
printf("A\n");
}
else if(count[1]==2)
{
printf("B\n");
}
//for(i=0;i<3;i++)
//count[i]=0;
else if((count[0]!=l)&&(count[1]!=l))
{
for(i=0;i<3;i++)
count[i]=0;
for(i=l-1;i>=2;i--)
{
if((s[i]!=s[i-1])||((s[i]==s[i-1])&&(s[i]==s[i-2])))
{
count[s[i]-'A']=1;
}
else if((count[s[i]-'A']!=1)&&(s[i]==s[i-1])&&(s[i]!=s[i-2]))
{
for(j=0;j<i;j++)
printf("%c",s[j]);
printf("\n");
flag=1;
break;
}
}
if(flag!=1)
printf("-1\n");
}
else
printf("-1\n");
}
return 0;
}
