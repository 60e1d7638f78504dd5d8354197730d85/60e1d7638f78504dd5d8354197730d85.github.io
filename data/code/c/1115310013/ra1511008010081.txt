#include<stdio.h>
struct color
{
    int init;
    int max;
    int a;
};
int comp(const void * c,const void * b)
{
  int l=((struct color *)c)->a;
  int r=((struct color *)b)->a;
  return (l>r);
}
int main()
{
    int t,n,m,k,i,j,s;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d%d",&n,&m,&k);
        int b[n][m],c[n][m],a[n];
        struct color str[n];
        for(i=0;i<n;i++)
        scanf("%d",&a[i]);
        for(i=0;i<n;i++)
        {
            for(j=0;j<m;j++)
            scanf("%d",&b[i][j]);
        }
        for(i=0;i<n;i++)
        {
            for(j=0;j<m;j++)
            scanf("%d",&c[i][j]);
        }
        for(i=0;i<n;i++)
        {
            s=b[i][0]-c[i][0];
            for(j=1;j<m;j++)
            {
                if(b[i][j]-c[i][j]>s)
                s=b[i][j]-c[i][j];
            }
            str[i].max=s;
        }
        for(i=0;i<n;i++)
        {
            str[i].init=b[i][a[i]-1];
        }
        //for(i=1;i<=n;i++)
        //printf("%d ",str[i].init);
        //printf("\n");
        //for(i=1;i<=n;i++)
        //printf("%d ",str[i].max);
        //printf("\n");
        for(i=0;i<n;i++)
        {
            str[i].a=str[i].max-str[i].init;
        }
        //for(i=1;i<=n;i++)
        //printf("%d ",str[i].a);printf("\n");
        qsort(str,n,sizeof(struct color),comp);
        //for(i=0;i<n;i++)
        //printf("%d ",str[i].init);
        //printf("\n");
        //for(i=0;i<n;i++)
        //printf("%d ",str[i].max);
       // printf("\n");
        //for(i=0;i<n;i++)
        //printf("%d ",str[i].a);
        //printf("\n");
        if(k>=n)
        k=n;
        int sum=0;
        for(i=n-1;i>=0,k>0;i--,k--)
        {
            sum+=str[i].max;
            //printf("%d ",sum);
        }
        for(j=i;j>=0;j--)
        {sum+=str[j].init;}
        printf("%d\n",sum);
    }
    return 0;
}
