#include<stdio.h>
int main()
{
long int n,i,j,k,l,count,maxcount;
scanf("%ld",&n);
long int a[n][2];
long long int ans=n;
for(i=0;i<n;i++)
scanf("%ld",&a[i][0]);
for(i=0;i<(n-1);i++)
{
for(j=(i+1);j<n;j++)
{
maxcount=0;
for(k=i;k<=j;k++)
a[k][1]=0;
for(k=i;k<j;k++)
{
if(a[k][1]==0)
{
count=1;
for(l=(k+1);l<=j;l++)
{
if(a[l][0]==a[k][0])
{
count++;
a[l][1]=1;
}
}
if(count>maxcount)
maxcount=count;
}
}
if(maxcount>((j-i+1)/2))
ans++;
}
}
printf("%lld\n",ans);
return 0;
}
