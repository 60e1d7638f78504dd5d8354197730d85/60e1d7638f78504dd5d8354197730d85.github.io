#include <stdio.h>
#include <stdlib.h>
int isMajority(int a[], int cand,int x,int y);
int findCandidate(int a[],int x,int y);
int main()
{
    int n,i,j,c,coun;
    scanf("%d",&n);
    int a[n];
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    coun=0;
    for(i=0;i<n;i++)
    {
        for(j=i;j<n;j++)
        {
            c=findCandidate(a,i,j);
            if(isMajority(a,c,i,j))
                coun++;
        }
    }
    printf("%d\n",coun);
    return 0;
}
 
 
 
 
int findCandidate(int a[],int x,int y)
{
    int maj_index = 0, count = 1;
    int i;
    for(i = x; i < y+1; i++)
    {
        if(a[maj_index] == a[i])
            count++;
        else
            count--;
        if(count == 0)
        {
            maj_index = i;
            count = 1;
        }
    }
    return a[maj_index];
}
 
/* Function to check if the candidate occurs more than n/2 times */
int isMajority(int a[], int cand,int x,int y)
{
int size=y-x+1;
    int i, count = 0;
    for (i = x; i < y+1; i++)
      if(a[i] == cand)
         count++;
    if (count > size/2)
       return 1;
    else
       return 0;
}
