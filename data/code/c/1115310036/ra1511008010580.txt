#include<stdio.h>
int main()
{
	int n;
	scanf("%d",&n);
	int a[n];
	int i,j,k,m=0,ml=0,ms=0,count=0;
	for(i=0;i<n;i++)
	scanf("%d",&a[i]);
	for(i=0;i<n-1;i++)
	{
		m=0,ml=0,ms=0;
		for(k=i;k<n;k++)
		{
			if(a[k]>ml)
			{
				if(ms==a[i])
				break;
				ms=ml;
				ml=a[k];
				if(ms!=0)
				count++;
			}
			else if(a[k]>ms)
			{
				if(ms==a[i])
				break;
				ms=a[k];
				count++;
			}
		}
	}
	printf("%d",count);
  return 0;
}
