#include <stdio.h>
int main()
{
  int t,n,i;
  scanf("%d",&t);
  for(i=1;i<=t;i++){
    scanf("%d",&n);
    n=n%8;
    if(n==1)
      printf("4LB\n");
    else if(n==2)
      printf("5MB\n");
    else if(n==3)
      printf("6UB\n");
    else if(n==4)
      printf("1LB\n");
    else if(n==5)
      printf("2MB\n");
    else if(n==6)
      printf("3UB\n");
    else if(n==7)
      printf("8SU\n");
    else if(n==8)
      printf("7SL\n");
  }

	return 0;
}
