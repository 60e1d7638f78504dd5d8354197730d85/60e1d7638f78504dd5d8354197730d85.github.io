#include <stdio.h>
int main()
{

  int t;
  scanf("%d",&t);
  while(t--)
  {
    int n;
    scanf("%d",&n);
    int temp;
    temp = n%8;
    
    if(temp==1)
      printf("%dLB",n+3);
    else if(temp==4)
      printf("%dLB",n-3);
    else if(temp==2)
      printf("%dMB",n+3);
    else if(temp==5)
      printf("%dMB",n-3);
    else if(temp==3)
      printf("%dUB",n+3);
    else if(temp==6)
      printf("%dUB",n-3);
    else if(temp==0)
      printf("%dSL",n-1);
    else if(temp==7)
      printf("%dSU",n+1);
    
    printf("\n");
  }
	return 0;
}
