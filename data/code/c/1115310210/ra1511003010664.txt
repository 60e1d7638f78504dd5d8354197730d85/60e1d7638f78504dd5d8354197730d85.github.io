#include <stdio.h>
 
int bin_srch(int arr[] , int x ,int y , int z){
	int mid=(x+y)/2;
	if(arr[y] <= z){
		return -1;
	}
	if(arr[mid] > z && arr[mid-1] <= z){
		return mid;
	}
	else if(arr[mid] > z){
		y=mid-1;
		return bin_srch(arr,x,y,z);
	}
	else if(arr[mid] <= z && arr[mid+1]>z){
		return mid+1;
	}
	else if(arr[mid] <= z){
		x=mid+1;
		return bin_srch(arr,x,y,z);
	}
	else {
		return 0;
	}
}
 
int main(){
	int t;
	scanf("%d",&t);
 
	while(t--){
		int flag[100007]={0};
		int x,y,i,l;
			scanf("%d",&x);
			int stack;
			i=0;
			scanf("%d",&y);
			flag[i++]=y;
			stack=1;
			int key=0;
			while(x>1){
				scanf("%d",&y);
				key=bin_srch(flag,0,i-1,y);
				if(key==-1){
					flag[i++]=y;
					stack++;
				}
				else{
 
					flag[key]=y;
				}
				x--;
			}
			printf("%d",stack);
			for(l=0;l<i;l++){
				printf(" %d",flag[l]);
			}
			printf("\n");
	}
	return 0;
}
