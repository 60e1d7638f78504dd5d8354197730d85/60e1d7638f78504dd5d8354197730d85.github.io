#include<stdio.h>
#include<string.h>
int main()
{
char str[100],temp;
int i,j;

//printf(“Enter the string :”);
//gets(str);
  scanf("%[^\n]s",str);
  
//printf("%s",str);

{

{

}
}
}

}
