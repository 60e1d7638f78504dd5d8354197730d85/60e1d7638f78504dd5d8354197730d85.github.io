#include <stdio.h>
int main()
{
  char str[50],temp;
int i,j;
  scanf("%s",str);
//printf("Enter the string :");
//printf("%s in ascending order is ->",str);
for(i=0;str[i];i++)
{
for(j=i+1;str[j];j++)
{
if(str[j]<str[i])
{
temp=str[j];
str[j]=str[i];
str[i]=temp;
}
}
}
printf("%s\n",str);

  return 0;
}
