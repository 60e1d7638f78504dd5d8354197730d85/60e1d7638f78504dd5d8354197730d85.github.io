#include <stdio.h>
#include <string.h>
int main()
{
	char str[40];
  	char chr;
  	scanf("%s",str);
  	int len = strlen(str);
  	int j,i;
  	for(j=1;j<len;j++)
    {
      for(i=0;i<len-1;i++)
      {
        if(str[i]>str[j])
        {
          chr = str[i];
          str[i] = str[j];
          str[j] = chr;
        }
      }
    }
  	printf("%s",str);
	return 0;
}
