#include <stdio.h>
#include<stdlib.h>
#include<string.h>
int main()
{
	char *str=(char *)malloc(sizeof(char));
  	int s,n;
  	scanf("%s\n",str);
  	scanf("%d\n%d",&n,&s);s--;
  	char *New=(char *)malloc(n*sizeof(char));
  	int i=0;int count=0;
  for(;i<strlen(str);i++)
  {
    if(i<s)
      continue;
    if(count==n)
      break;
    New[count++]=str[i];
  }
  	printf("%s",New);
	return 0;
}
