#include <stdio.h>
 
int main() 
{
   char string[1000], sub[1000];
   int position, length, c = 0;
 
  
   fgets(string,1000,stdin);
 
  
   scanf("%d%d",&length,&position);
 
   while (c < length) {
      sub[c] = string[position+c-1];
      c++;
   }
   sub[c] = '\0';
 
   printf("%s", sub);
 
   return 0;
}
