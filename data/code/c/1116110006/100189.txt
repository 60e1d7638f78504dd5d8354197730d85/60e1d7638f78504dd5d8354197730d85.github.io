#include <stdio.h>
int main()
{
char a[100], b[100];
   scanf("%s",a);
  scanf("%s",b);
   if (strcmp(a,b) == 0)
      printf("Both Strings are Equal");
   else if (strcmp(a,b) > 0)
      printf("First String is Greater Than Second String");
  else
    printf("First String is Less Than Second String");
	return 0;
}
