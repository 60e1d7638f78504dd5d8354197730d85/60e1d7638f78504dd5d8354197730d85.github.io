#include <stdio.h>
#include <string.h>
 
int main()
{
   char a[100], b[100];
 
  
   scanf("%s", a);
 
  
  scanf("%s", b);
 
   if (strcmp(a,b) <0)
      printf("First String is Less Than Second String\n");
   else if(strcmp(a,b)>0)
      printf("First String is Greater Than Second String\n");
  else
    printf("Both Strings are Equal");
 
   return 0;
}
