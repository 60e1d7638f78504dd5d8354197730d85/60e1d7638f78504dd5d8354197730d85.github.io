#include <stdio.h>
#include <string.h>
int main()
{ char str1[100],str2[100];
  scanf("%s",str1);
  scanf("%s",str2);
  if (strcmp(str1,str2)>1)
     printf("First String is Greater Than Second String");
  else if (strcmp(str1,str2)==0)
     printf("Both Strings are Equal");
  else
     printf("First String is Less Than Second String");
	return 0;
}
