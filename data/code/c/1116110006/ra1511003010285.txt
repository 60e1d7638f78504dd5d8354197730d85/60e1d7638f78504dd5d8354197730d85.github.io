#include <stdio.h>
#include <string.h>
int main()
{
  char str1[125];
  char str2[125];
  
  int ret;
  
  scanf("%s",str1);
  scanf("%s",str2);
  
  ret=strcmp(str1,str2);
  
  if(ret<0)
    printf("First String is Less Than Second String");
  else 
    if(ret>0)
      printf("First String is Greater Than Second String");
  else
    printf("Both Strings are Equal");
  	return 0;
}
