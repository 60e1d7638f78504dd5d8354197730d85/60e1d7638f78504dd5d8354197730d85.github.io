#include<string.h>
int main()
{
	char a[100],b[100],c,d;
  	scanf("%s %s",a,b);

    d=strlen(b);
  	if (strcmp(a,b)==0)
    {
      printf("Both Strings are Equal");
    }

     printf("First String is Greater Than Second String");

    }
