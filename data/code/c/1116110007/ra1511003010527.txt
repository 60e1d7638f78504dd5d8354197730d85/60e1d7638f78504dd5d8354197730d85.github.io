#include <stdio.h>
#include<string.h>
int main()
{
  char str[100];
  scanf("%s",str);
  int l=strlen(str);
  int n,t,i,j;
  scanf("%d",&t);
  scanf("%d",&n);
  for(i=0;i<l;i++)
  {
    if(i==n)
    {
      for(j=0;j<t;j++)
      {
        printf("%c",str[i+j]);
      }
      return 0;
    }
  }
	return 0;
}
