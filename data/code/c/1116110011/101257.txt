#include <stdio.h>
int main()
{
    char text1[100];
    char text2[100];
    int i;
    
    /* Input string from user */
        scanf("%s",text1);
    
    /* Copy text1 to text2 character by character */
    for(i=0; text1[i]!='\0'; i++)
    {
        text2[i] = text1[i];
    }

    //Makes sure that the string is NULL terminated
    text2[i] = '\0';

    printf("%s\n", text1);
    printf("%s\n", text2);
    printf("%d", i);

    return 0;
}
