#include <stdio.h>
int main()
{
    char s1[100], s2[100], i, j;

    scanf("%s", s1);

    scanf("%s", s2);
  printf("%s\n",s1);
  printf("%s\n",s2);

    // calculate the length of string s1
    // and store it in i
    for(i = 0; s1[i] != '\0'; ++i);

    for(j = 0; s2[j] != '\0'; ++j, ++i)
    {
        s1[i] = s2[j];
    }

    s1[i] = '\0';
  
    printf("%s", s1);

    return 0;
}
