#include <stdio.h>
#include<string.h>
void upper_string(char []);
 
int main()
{
   char string[100];
 fgets(string,20,stdin);
 
   upper_string(string);
 
   printf("%s", string);
 
   return 0;
}
 
void upper_string(char s[]) {
   int c = 0;
 
   while (s[c] != '\0') {
      if (s[c] >= 'a' && s[c] <= 'z') {
         s[c] = s[c] - 32;
      }
      c++;
   }
}
