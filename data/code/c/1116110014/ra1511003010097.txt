#include <stdio.h>
 
void upper_string(char []);
 
int main()
{
   char string[100];
 
   
  scanf("%s",string);
 
   upper_string(string);
 
   printf("%s", string);
 
   return 0;
}
 
void upper_string(char s[]) {
   int c = 0;
 
   while (s[c] != '\0') {
      if (s[c] >= 'A' && s[c] <= 'Z') {
         s[c] = s[c] + 32;
      }
      c++;
   }
}
