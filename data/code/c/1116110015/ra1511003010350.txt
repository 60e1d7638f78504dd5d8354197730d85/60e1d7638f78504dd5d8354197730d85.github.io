#include <stdio.h>
#include<string.h>
int main()
{char str[100];
 int alp, digit, splch, i;
    alp = digit = splch = i = 0;     
       scanf("%s",str);
    while(str[i]!='\0')
    {
        if((str[i]>='a' && str[i]<='z') || (str[i]>='A' && str[i]<='Z'))
        {
            alp++;
        }
        else if(str[i]>='0' && str[i]<='9')
        {
            digit++;
        }
        else
        {
            splch++;
        }

        i++;
    }
    printf("Alphabets=%d\n",alp);
    printf("Digits=%d\n",digit);
    printf("Special characters=%d",splch);
return 0;
}
