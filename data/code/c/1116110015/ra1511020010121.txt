#include <stdio.h>
#include<string.h>
int main()
{
	char s[100];
  	scanf("%s",s);
  int f=strlen(s);
  int a=0,d=0,sc=0,i;
  for(i=0;i<f;i++)
  {
    if(isalpha(s[i]))
      a++;
    else if(isdigit(s[i]))
      d++;
    else 
      sc++;
  }
  printf("Alphabets=%d\n",a);
  printf("Digits=%d\n",d);
  printf("Special characters=%d\n",sc);
	return 0;
}
