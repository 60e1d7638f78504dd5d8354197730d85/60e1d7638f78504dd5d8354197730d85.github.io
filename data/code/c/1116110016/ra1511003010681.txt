#include <stdio.h>
#include <string.h>
 
int main()
{
   char str[10],temp;
   int i,len;
   scanf("%s",str);
  printf("%s\n",str);
   
   len=strlen(str)-1;
   
   for(i=0;i<strlen(str)/2;i++)
   {
      temp=str[i];
      str[i]=str[len];
      str[len--]=temp;
   }
   printf("%s",str);
   return 0;
}
