#include <stdio.h>
#include <string.h>
 
int main()
{
   char string[10900];
   int c = 0, count[26] = {0}, x;
 
   fgets(string, sizeof(string), stdin);
 
   while (string[c] != '\0') {
 
      if (string[c] >= 'a' && string[c] <= 'z') {
         x = string[c] - 'a';
         count[x]++;
      }
 
      c++;
   }
 
   for (c = 0; c < 26; c++)
     if(count[c] > 0)
         printf("%c=%d\n",c+'a',count[c]);
 
   return 0;
}
