#include <stdio.h>
#include <string.h>
 
char string1[100], visited[100];
int count[100] = {0}, flag = 0;
 
int main()
{
    int i, j = 0, k = 0, l, max, index;
 
    scanf("%[^\n]s", string1);
 
    l = strlen(string1);
 
    for (i = 0; i < l; i++)
    {
        if (i == 0)
        {
            visited[j++] = string1[i];
            count[j - 1]++;
        }
        else
        {
            for (k = 0; k  < j; k++)
            {
                if (string1[i] == visited[k])
                {
                    count[k]++;
                    flag = 1;
                }
            }
            if (flag == 0)
            {
                visited[j++] = string1[i];
                count[j - 1]++;
            }
            flag = 0;
        }
    }    
 
    for (i = 0; i < j; i++)
    {
		printf("%c=%d\n",visited[i],count[i]);
    }

  return 0;
}
