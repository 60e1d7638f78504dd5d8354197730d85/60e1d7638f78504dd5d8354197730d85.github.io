#include <stdio.h>
#include<stdlib.h>
#include<string.h>
int main()
{
char str[10000];
  int i,a[26]={0},x;
 scanf("%s",str);
  for(i=0;i<strlen(str);i++)
  {
    if(str[i]>='a' && str[i]<='z')
    {
      x=str[i]-'a';
      a[x]++;
    }
  }
 
  for(i=0;i<26;i++)
    if(a[i]>0)
  {
    printf("%c=%d\n",i +'a',a[i]);
  }
 
  
	return 0;
}
