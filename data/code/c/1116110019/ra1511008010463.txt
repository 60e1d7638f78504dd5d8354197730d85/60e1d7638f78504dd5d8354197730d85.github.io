#include <stdio.h>
int allid(char c, char * string)
{
 int i,j=0;
 for(i=0;string[i]!='\0';i++)
 {
 if(c==string[i])
 j++; 
 }
 return j;
}
int main()
{ 
 char string[1000],countc[1000];
 int i,j=0,k=0,l=0,res,count[100],min;
 
 scanf("%s",string);
 for(i=0;string[i]!='\0';i++)
 { 
 res=allid(string[i],string);
 for(j=0;j<i;j++)
 if(string[i]==string[j])
 l++;
 if(l==0)
 {
 countc[k]=string[i];
 count[k]=res;
 k++;
 }
 l=0; 
 }
 min=count[0];
 for(i=1;i<k;i++)
 if(min>count[i])
 min=count[i];
 
 for(i=0;i<k;i++)
 if(min==count[i])
 {
 if(min==1)
 printf("%c=%d\n",countc[i],count[i]);
 else if(min>1)
 printf("%c=%d",countc[i],count[i] );
 }
 return 0;
}
