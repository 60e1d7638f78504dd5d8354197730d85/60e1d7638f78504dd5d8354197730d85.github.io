#include <stdio.h>
#include <string.h>
int main()
{
  char a[30];
  int i,l,f[26];
  scanf("%s",a);
  l=strlen(a);
  for(i=0;i<26;i++)
    f[i]=0;
  for(i=0;i<l;i++)
  {
    if(a[i]>='a' && a[i]<='z')
    {
      f[a[i]-97]++;
    }
  }
  for(i=0;i<26;i++)
  {
    if(f[i]!=0)
    {
      printf("%c = %d\n",(i+97),f[i]);
    }
  }


	return 0;
}
