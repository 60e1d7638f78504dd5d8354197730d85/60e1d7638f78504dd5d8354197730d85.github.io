#include <stdio.h>
#include <string.h>
 
#define MAX_SIZE 100
void removeLast(char *, const char);
int main()
{
    char string[MAX_SIZE];
    char toRemove;
    fgets(string, MAX_SIZE, stdin);
    toRemove = getchar();
 
    removeLast(string, toRemove);
 
    printf("%s", string);
 
    return 0;
}
void removeLast(char * string, const char toRemove)
{
    int i, lastPosition;
    int len = strlen(string);
 
    lastPosition = -1;
    i=0;
 
    while(i<len)
    {
        if(string[i] == toRemove)
        {
            lastPosition = i;
        }
 
        i++;
    }
 
    if(lastPosition != -1)
    {
        i = lastPosition;
        while(i<len-1)
        {
            string[i] = string[i+1];
            i++;
        }
    }
    string[i] = '\0';
}
