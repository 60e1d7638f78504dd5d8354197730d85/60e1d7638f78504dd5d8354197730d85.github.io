#include <stdio.h>
#include <string.h>

int main()
{
    char str[100];
    char word[100];
    int i, j, found, index;
    int stringLen, wordLen;

    scanf("%s\n",str);
    scanf("%s\n",word);

    stringLen = strlen(str);
    wordLen   = strlen(word);
    index = -1;
    for(i=0; i<stringLen - wordLen; i++)
    {
        found = 1;
        for(j=0; j<wordLen; j++)
        {
            if(str[i+j] != word[j])
            {
                found = 0;
                break;
            }
        }
        if(found == 1)
        {
            index = i;
        }
    }
    if(index != -1)
    {  for(i=index; i <= stringLen - wordLen; i++)
        {
            str[i] = str[i + wordLen];
        }
        printf("%s",str);
    }
    return 0;
}
