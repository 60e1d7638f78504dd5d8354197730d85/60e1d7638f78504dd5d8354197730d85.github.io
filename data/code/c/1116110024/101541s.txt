#include<stdio.h>
#include<string.h>

main()
{
char a[100],b[100]={0};
scanf("%s",a);
  printf("%s\n",a);

int i,j,k=0,count=0;
for(i=0;i<f;i++)
{
for(j=0;j<i;j++)
{
if(a[i]==b[j])
{
count=1;
}
}

{
b[k++]=a[i];

}

}

{

}

}
