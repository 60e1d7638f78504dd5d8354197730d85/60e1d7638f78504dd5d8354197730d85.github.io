#include<string.h>
int main()
{
  char str[50],ans[30];
  int len,i,j,k;
  scanf("%[^\n]%*c",str);

  for(i=0;i<len;i++)
  {
   ans[i]=str[i]; 
    printf("%c",ans[i]);
  }
  printf("\n");
 
for(i=0;i<len;i++)
{
 for(j=i+1;j<len;j++)
 {
  if(str[i]==str[j])
  {
    for(k=j;k<len;k++)
    {
     
   str[k]=str[k+1];
      
    }
    len--;
  }
 }
}
  printf("%s",str);
	return 0;
}
