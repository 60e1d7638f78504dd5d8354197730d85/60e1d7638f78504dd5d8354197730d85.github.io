#include <stdio.h>
#define MAX_SIZE 100 
void replaceFirst(char * str, char oldChar, char newChar);
int main()
{
    char str[MAX_SIZE], oldChar, newChar;
    fgets(str,100,stdin);
    oldChar = getchar();
    getchar();
    newChar = getchar();
    replaceFirst(str, oldChar, newChar);
    printf("%s",str);
    return 0;
}

void replaceFirst(char * str, char oldChar, char newChar)
{
    int i=0;
    while(str[i] != '\0')
    {
        if(str[i] == oldChar)
        {
            str[i] = newChar;
            break;
        }
        i++;
    }
}
