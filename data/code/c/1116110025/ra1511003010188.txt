#include <stdio.h>
#define MAX_SIZE 100 

void replaceFirst(char * str, char oldChar, char newChar);


int main()
{
    char str[MAX_SIZE], oldChar, newChar;
    scanf("%s\n",str);
  
    scanf("%c\n",&oldChar);

	scanf("%c\n",&newChar);

    replaceFirst(str, oldChar, newChar);

    printf("%s", str);

    return 0;
}

void replaceFirst(char * str, char oldChar, char newChar)
{
    int i=0;
    while(str[i] != '\0')
    {
        if(str[i] == oldChar)
        {
            str[i] = newChar;
            break;
        }

        i++;
    }
}
