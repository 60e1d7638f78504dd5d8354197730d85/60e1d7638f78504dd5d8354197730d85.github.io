#include <stdio.h>
#define MAX_SIZE 100
 
void replaceFirst(char * string, char toReplace, char replaceWith);
 
 
 
int main()
{
    char string[MAX_SIZE], toReplace, replaceWith;
 

   scanf("%s",string);
 
   scanf(" %c", &toReplace);
   scanf(" %c", &replaceWith);
   replaceFirst(string, toReplace, replaceWith);
   printf("%s", string);
 
    return 0;
}

void replaceFirst(char * string, char toReplace, char replaceWith)
{
    int i=0;
 
    while(string[i] != '\0')
    {
        
        if(string[i] == toReplace)
        {
            string[i] = replaceWith;
            break;
        }
 
        i++;
    }
}
