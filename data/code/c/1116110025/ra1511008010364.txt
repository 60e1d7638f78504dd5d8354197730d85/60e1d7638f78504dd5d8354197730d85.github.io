#include <stdio.h>
#include<string.h>
int main()
{
	char str[50];
  	char rep,new;
  	int i,cnt,len;
  	scanf("%s",str);
  	scanf(" %c",&rep);
  	scanf(" %c",&new);
  	len=strlen(str);
  	cnt=0;
  	for(i=0;str[i]!='\0';i++)
    {
      if(str[i]==rep)
      {
        cnt++;
        if(cnt==1)
          str[i]=new;
      }
    }
  	printf("%s",str);
	return 0;
}
