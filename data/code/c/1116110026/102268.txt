void replaceLast(char * string, char toReplace, char replaceWith)
{
    int i, index; 
    index = -1;
    i = 0;    
    while(string[i] != '\0')
    {       
        if(string[i] == toReplace)
        {
            index = i;
        }
 
        i++;
    }
 
    if(index != -1)
    {
        string[index] = replaceWith;
    }
}
