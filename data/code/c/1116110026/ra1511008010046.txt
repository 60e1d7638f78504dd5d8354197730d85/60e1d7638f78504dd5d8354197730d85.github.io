#include <stdio.h>
void replaceLast(char * str, char oldChar, char newChar);
int main()
{
    char str[100], oldChar, newChar;
    fgets(str,100,stdin);
    oldChar = getchar();
    getchar();
    newChar = getchar();
    replaceLast(str, oldChar, newChar);
    printf("%s",str);
    return 0;
}
void replaceLast(char * str, char oldChar, char newChar)
{
    int i, lastIndex;
    lastIndex = -1;
    i = 0;
    while(str[i] != '\0')
    {
        if(str[i] == oldChar)
        {
            lastIndex = i;
        }

        i++;
    }
    if(lastIndex != -1)
    {
        str[lastIndex] = newChar;
    }
}
