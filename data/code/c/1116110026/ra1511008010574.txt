#include <stdio.h>
#define MAX_SIZE 100
void replaceLast(char * string, char toReplace, char replaceWith);



int main()
{
    char string[MAX_SIZE], toReplace, replaceWith;

    scanf("%s",string);

    scanf(" %c", &toReplace);

    
    scanf(" %c", &replaceWith);

    replaceLast(string, toReplace, replaceWith);

    printf("%s\n", string);

    return 0;
}



void replaceLast(char * string, char toReplace, char replaceWith)
{
    int i, index;

    index = -1;
    i = 0;

    while(string[i] != '\0')
    {
        if(string[i] == toReplace)
        {
            index = i;
        }

        i++;
    }

    if(index != -1)
    {
        string[index] = replaceWith;
    }
}
