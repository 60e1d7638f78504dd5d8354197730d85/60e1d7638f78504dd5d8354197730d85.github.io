#include <stdio.h>

 
 

void replace(char * string, char toReplace, char replaceWith);
 
 
 
int main()
{
    char string[100], toReplace, replaceWith;
 
   
    scanf("%s",string);
 
    
    scanf(" %c", &toReplace);
 
   
    scanf(" %c", &replaceWith);
 
 
   
 
    replace(string, toReplace, replaceWith);
 
    printf("%s", string);
 
    return 0;
}
 
 
 
 

void replace(char * string, char toReplace, char replaceWith)
{
    int i = 0;
 
    
    while(string[i] != '\0')
    {
        
        if(string[i] == toReplace)
        {
            string[i] = replaceWith;
        }
 
        i++;
    }
}
