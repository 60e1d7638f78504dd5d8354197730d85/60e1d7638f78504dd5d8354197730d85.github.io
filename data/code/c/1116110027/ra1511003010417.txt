#include<stdio.h>
 
int main()
{
    char array[20], char1, char2;
    int counter;
   
    scanf("%s", array);
    
    
    scanf(" %c", &char1);
    
    scanf(" %c", &char2);
    while(array[counter] != '\0')
    {
        if(array[counter] == char1)
        {
            array[counter] = char2;
        }
        counter++;
    }
    printf("\n%s\n", array);
    printf("\n");
    return 0;
}
