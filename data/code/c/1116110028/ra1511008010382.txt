#include <stdio.h>
#include <string.h>


int main()
{
    char str[100], word[100];
    int i, index, found = 0;

   scanf("%[^\n]s",str);
         scanf("%s",word);


    
    index = 0;
    while(str[index] != '\0')
    {

       
        if(str[index] == word[0])
        {
           
            i=0;
            found = 1;
            while(word[i] != '\0')
            {
                if(str[index + i] != word[i])
                {
                    found = 0;
                    break;
                }

                i++;
            }
        }
  
        if(found == 1)
        {
            break;
        }

        index++;
    }
    
   
    if(found == 1)
    {
        printf("\n%s is found at %d", word, index);
    }
    else
    {
        printf("\n%s is not found", word);
    }

    return 0;
}
