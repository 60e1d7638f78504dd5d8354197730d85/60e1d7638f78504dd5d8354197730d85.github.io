#include <stdio.h>
#include <string.h>


int main()
{
    char str[100];
    char word[100];
    int i, j, found;
    int strLen, wordLen;

    
    scanf("%[^\n]s",str);
    
    scanf("%s",word);

    strLen  = strlen(str);  
    wordLen = strlen(word);


    for(i=0; i<(strLen-wordLen+1); i++)
    {
       
        found = 1;
        for(j=0; j<wordLen; j++)
        {
           
            if(str[i + j] != word[j])
            {
                found = 0;
                break;
            }
        }

  
        if(found == 1)
        {
            printf("%d\n",i);
        }
    }

    return 0;
}
