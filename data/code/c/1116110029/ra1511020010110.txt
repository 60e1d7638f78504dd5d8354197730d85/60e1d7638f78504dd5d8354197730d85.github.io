#include <stdio.h>
#include <string.h>
int main()
{
  char str1[100],str2[100];
  int i,j,found=1,len1,len2;
  scanf("%[^\n]s",str1);
  scanf("%s",str2);
  len1=strlen(str1);
  len2=strlen(str2);
  for(i=0;i<len1-len2+1;i++)
  {
    if(str1[i]==str2[0])
       {found=1;
         for(j=0;j<len2;j++)
           if(str1[i+j]!=str2[j])
              found=0;
        if(found==1)
          printf("%d\n",i);
       }
  }
  
  return 0;
}
