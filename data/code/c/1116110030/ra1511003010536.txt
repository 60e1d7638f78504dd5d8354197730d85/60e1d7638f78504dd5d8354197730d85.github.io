#include <stdio.h>
#include<string.h>
int main()
{ int i,j,k,l,con=0;  
  char a[100],b[15];
    fgets(a,100,stdin);
    scanf("%s",b);
    i=strlen(a);
 	j=strlen(b);
 	for(k=0;k<i;)
    {  int count=0;
     	l=0;
      while(a[k]==b[l])
      { count++;
        k++;
        l++;
      }
     if(count==j)
     {
       con++;
     }
     else
       k++;
    }
    printf("%s=%d",b,con);
	return 0;
}
