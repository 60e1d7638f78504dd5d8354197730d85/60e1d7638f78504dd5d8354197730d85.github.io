#include <stdio.h>
#include <string.h>
#define max 100
int main()
{
  char str[max], sub[max];
  int count=0, count1=0,i,j,l,l1,l2;
  scanf("%[^\n]s", str);
  l1 = strlen(str);
  scanf("%s", sub);
  l2 = strlen(sub);
  for(i=0; i<l1;)
  {
    j=0;
    count=0;
    while((str[i] == sub[j]))

      count++;

    }
