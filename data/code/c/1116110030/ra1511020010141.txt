#include <stdio.h>
#include<string.h>
int main()
{
char str[100], sub[100];
int count = 0, count1 = 0;
    int i, j,l1, l2;
    scanf("%[^\n]s", str);
    l1 = strlen(str);
    scanf(" %[^\n]s", sub);
    l2 = strlen(sub);
 
    for (i = 0; i < l1;)
    {
        j = 0;
        count = 0;
        while ((str[i] == sub[j]))
        {
            count++;
            i++;
            j++;
        }
        if (count == l2)
        {
            count1++;                                   
            count = 0;
        }
        else
            i++;
    }    
    printf("%s=%d", sub, count1);
	return 0;
}
