#include <stdio.h>
#include<string.h>
int main()
{
  char str[100],key[100],str1[30][30];
  int k=0,j=0,i;
  scanf("%[^\n]s",str);
  scanf("%s",key);
  for(i=0;str[i]!='\0';i++)
  {
  if(str[i]==' ')
  {
  str1[k][j]='\0';
    k++;
    j=0;
  }
    else
    {
    str1[k][j]=str[i];
      j++;
    }
  }
  str1[k][j]='\0';
  for(i=0;i<k+1;i++)
  {
  if(strcmp(str1[i],key)==0)
  {
  for(j=i;j<k+1;j++)
  {
  strcpy(str1[j],str1[j+1]);
  }
    k--;
  }
  }
  for(i=0;i<k+1;i++)
  {
  printf("%s ",str1[i]);
  }
  return 0;
}
