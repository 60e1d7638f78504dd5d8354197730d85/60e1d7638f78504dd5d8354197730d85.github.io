#include<stdio.h>
#include<string.h>
int main() {
   int upper = 0, lower = 0;
   char ch[80];
   int i;

   scanf("%s",ch);
 
   i = 0;
   for(i=0;i<strlen(ch);i++) {
      if (ch[i] >= 'A' && ch[i] <= 'Z')
         upper++;
      if (ch[i] >= 'a' && ch[i] <= 'z')
         lower++;
      
   }
 
   printf("Uppercase Letters:%d", upper);
   printf("\nLowercase Letters:%d", lower);
 
   return (0);
}
