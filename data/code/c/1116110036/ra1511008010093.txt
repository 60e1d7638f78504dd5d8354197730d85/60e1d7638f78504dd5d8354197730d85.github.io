#include <stdio.h>
#include<string.h>
int main()
{
char x[50];
  scanf("%[^\t\n]s",x);
  int i=0,upper=0,lower=0;
  while(i<=strlen(x))
  {
    if (x[i]>='A' && x[i]<='Z')
      upper++;
    if(x[i]>='a' && x[i]<='z')
      lower++;
    if (x[i]=='$')
      break;
    i++;
  }
  printf("Uppercase Letters:%d\n",upper);
  printf("Lowercase Letters:%d",lower);
  return 0;
}
