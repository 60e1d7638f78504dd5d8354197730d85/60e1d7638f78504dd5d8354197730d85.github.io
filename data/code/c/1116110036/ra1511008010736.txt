#include<stdio.h>
 int main()
 {
   int u=0,l=0,i=0;
   char ch[80];
 
   scanf("%s",ch);
   while (ch[i] != '\0') 
   {
      if (ch[i] >= 'A' && ch[i] <= 'Z')
         u++;
       if (ch[i] >= 'a' && ch[i] <= 'z')
         l++;
      i++;
   }
 
   printf("\nUppercase Letters:%d", u);
   printf("\nLowercase Letters:%d", l);
 
   return (0);
 }
