#include <stdio.h>
#include <string.h>
int main()
{
  char s[100];
  int i,count=0,l;
  l=strlen(s);
  scanf("%s",s);
  for(i=0;i<=l;i++)
  {
    if((s[i]=='a')||(s[i]=='e')||(s[i]=='i')||(s[i]=='o')||(s[i]=='u')||(s[i]=='A')||(s[i]=='E')||(s[i]=='I')||(s[i]=='O')||(s[i]=='U'))
      ++count;
  }
  printf("%d",count);
  return 0;
}
