#include <stdio.h>
#include <string.h>
int main()
{
	char a[100];
  	int b,c=0,i;
  	scanf("%s",a);
  	b=strlen(a);
  	for(i=0;i<b;i++)
    {
    	if((a[i]=='a')||(a[i]=='e')||(a[i]=='i')||(a[i]=='o')||(a[i]=='u'))
        {
        	++c;
        }
      	else
        {
        	printf("%c",a[i]);
        }
    }
	return 0;
}
