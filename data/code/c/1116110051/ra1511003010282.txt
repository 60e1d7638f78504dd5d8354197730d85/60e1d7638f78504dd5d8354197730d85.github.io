#include<stdio.h>
#include<string.h>
int main()
{
    int t, N, j;
    char x[100000];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%s",x);
        N=strlen(x);
        char y[N];
        scanf("%s",y);
        j=0;
        while(x[j]!='\0')
        {
            if(x[j]=='W'&&y[j]=='W')
                printf("B");
            else if(x[j]=='B'&&y[j]=='B')
                printf("W");
            else
                printf("B");
            j++;
        }
        printf("\n");
    }
    return 0;
}
