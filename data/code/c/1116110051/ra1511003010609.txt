#include <stdio.h>
#include <stdio.h>
#include <string.h>
 
int main(){
	int i,t;
	scanf("%d",&t);
	while (t--){
		char str1[100001],str2[100001];
		scanf("%s %s",str1,str2);
		int size = strlen(str1);
		for (i=0;i<size;i++){
			if (str1[i]=='W' || str2[i]=='W')
				printf("%c",'B');
			else
				printf("%c",'W');
		}
		printf("\n");
	}
	return 0;
}
