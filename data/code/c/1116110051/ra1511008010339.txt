#include <stdio.h>
 
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
	char str1[100001],str2[100001];
	scanf("%s%s",str1,str2);
	int i;
	for(i=0; str1[i]; i++)
	{
		if((str1[i]==str2[i]) && str1[i]=='W')
		printf("B");
		else if((str1[i]==str2[i]) && str1[i]=='B')
		printf("W");
		else
		printf("B");
	}
	printf("\n");
	}
	return 0;
}
