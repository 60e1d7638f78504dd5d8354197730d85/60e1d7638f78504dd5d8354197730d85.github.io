#include <stdio.h>
#include<stdlib.h>
#include<string.h>
int main(void)
{
  int t,len,i,n;
  scanf("%d",&t);
  char x[100000];
  char y[100000];
  while(t--){
    scanf("%s",x);
    scanf("%s",y);
    len=strlen(x);
    for(i=0;i<len;i++){
      if(x[i]=='B'&&y[i]=='B'){
        printf("W");
      }
      else {
        printf("B");
      }}
    printf("\n");
  }

	return 0;
}
