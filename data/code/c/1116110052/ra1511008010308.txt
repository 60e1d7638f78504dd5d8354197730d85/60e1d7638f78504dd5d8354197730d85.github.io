#include<stdio.h>
#include<string.h>
 
#define M(a) (a-97)
int main()
{
  int t,i,j,k;
  char str[10000];
  int prev_max[10000], next_max[10000], hash[10000],x;
  
  scanf("%d",&t);
 
  while(t--)
  {
    scanf("%s",str);
 
    int f1[26] = {0}, f2[26] = {0}, f3[26] = {0}, max[10000] = {0};
    int len = strlen(str), tmp1 = 0, tmp2 = 0;
    k=0;
    for(i=0;i<len;i++)
    {
      if(str[i] != '#')
      {
        f1[M(str[i])]++;
        if(f1[M(str[i])] > tmp1)
          tmp1 = f1[M(str[i])];
      
        f3[M(str[i])]++;
        if(f3[M(str[i])] > max[k])
          max[k] = f3[M(str[i])];
      }
      else
      {
        hash[k] = i;
        for(x=0;x<26;x++)
          f3[x] = 0;
        k++;
      }
      
      if(str[len-1-i] != '#')
      {
        f2[M(str[len-1-i])]++;
        if(f2[M(str[len-1-i])] > tmp2)
          tmp2 = f2[M(str[len-1-i])];
      }
      prev_max[i] = tmp1;
      next_max[i] = tmp2;
      
    }

    int first, mid1, mid2, last, sum = 0, max_sum = 0;
    for(i=0;i<(k-2);i++)
    {
      first = prev_max[hash[i]];
      last = next_max[len-1- hash[i+2]];
      mid1 = max[i+1];
      mid2 = max[i+2];
      
      if(first != 0 && last != 0 && mid1 != 0 && mid2 != 0)
        sum = first + mid1 + mid2 + last;
      
      if(sum > max_sum)
        max_sum = sum;
    
    }
 
    if(max_sum < 4)
      printf("0\n");
    else
      printf("%d\n",max_sum+3);
  }
return 0;
}
