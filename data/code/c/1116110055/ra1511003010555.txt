#include<stdio.h>
#include<stdlib.h>
#include<string.h>
int main()
{
    int t,i,j,p,q;
    char c[10000];
    scanf("%d",&t);
    while(t--){
            scanf("%s",c);
            p=0;
            q=0;
        for (i=0;i<strlen(c);i++){
            if(c[i]=='a')
                p++;
            else
                q++;
        }
        if(p>q)
            printf("%d\n",q);
        else
            printf("%d\n",p);
    }
    return 0;
}
