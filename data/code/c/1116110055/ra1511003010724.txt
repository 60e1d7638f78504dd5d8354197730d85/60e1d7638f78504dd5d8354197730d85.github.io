#include<stdio.h>
#include<string.h>
int main()
{
   int t;
   scanf("%d",&t);
   while(t--)
   {
       char st[105];
       scanf("%s",st);
       int l=strlen(st);
       int aa=0,bb=0,i=0;
       for(i=0;i<l;i++)
       {
           if(st[i]=='a')
            aa++;
           else
            bb++;
       }
       if(aa>bb)
        printf("%d\n",(l-aa));
       else
        printf("%d\n",(l-bb));
 
   }
return 0;
}
