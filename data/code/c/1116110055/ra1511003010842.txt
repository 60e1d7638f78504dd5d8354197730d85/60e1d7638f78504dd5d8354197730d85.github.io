#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int main()
{
    int t,i,c,d,j;
    char str[100];
    scanf("%d",&t);
    for(i=0;i<t;i++)
    {
        scanf("%s",str);
        c=0;
        d=0;
        j=0;
        while(str[j]!='\0')
        {
            if(str[j]=='a')
                c++;
            else
                d++;
                j++;
        }
        if(c>=d)
            printf("%d\n",d);
        else
            printf("%d\n",c);
    }
    return 0;
}
