#include <stdio.h>
#include <string.h>
 
int main()
{
    int t,f,a=0,b=0,j;
    char n[101];
    scanf("%d",&t);
    for(f=0;f<t;f++){
    scanf("%s",n);
 
    for(j=0; j<strlen(n); j++)
    {
 
        if(n[j] == 'a') a++;
        if(n[j] == 'b') b++;
 
    }
 
    if(a == b) printf("%d\n",a);
    else if(a == 0 || b == 0) printf("0\n");
    else
    {
 
        if(a>b) printf("%d\n",b);
        else printf("%d\n",a);
 
    }
 
    a=0,b=0;
 
    }
 
    return 0;
}
