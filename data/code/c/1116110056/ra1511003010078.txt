#include <stdio.h>
#include <string.h>
int main()
{
  char a [1000];
  int l,t,i,j,m,z=0,u;
  scanf("%d",&t);
  for(u=1;u<=t;u++)
  {
    scanf("%s",a);
    l=strlen(a);
    int b[l+1];
  
  for(i=0;i<l;i++)
  {
    b[i]=0;
  }
  m=l/2;
    for(i=0;i<=(m-1);i++)
    {z=0;
     for(j=(l-m);j<=(l-1);j++)
     {if(b[j]==1)
       continue;
      if(a[i]==a[j])
      {
        b[i]=1;
        b[j]=1;
        z=1;
        break;
      }
     }
     if(z==0)
       break;
    }
  if(z==1)
    printf("YES\n");
  else
    printf("NO\n");
}

    
      
  
	return 0;
}
