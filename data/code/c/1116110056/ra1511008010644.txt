    #include<stdio.h>
    #include<string.h>
    char a[1001],c[26]={0},d[26]={0};
    int i,l,flag,t,j;
    main()
    {
    scanf("%d",&t);
    for(j=0;j<t;j++)
    {
    scanf("%s",a);
    l=strlen(a);
    if(l==1)
    continue;
    else if(l%2==0)
    {
    for(i=0;i<l/2;i++)
    {
    c[a[i]-'a']++;
    }
    for(i=l/2;i<l;i++)
    {
    d[a[i]-'a']++;
    }
    for(i=0;i<26;i++)
    {
    if(c[i]==d[i])
    flag=1;
    else
    {
    flag=0;
    break;
    }
    }
    if(flag==1)
    printf("YES\n");
    else if(flag==0)
    printf("NO\n");
    }
    else
    {
    for(i=0;i<l/2;i++)
    {
    c[a[i]-'a']++;
    }
    for(i=(l/2)+1;i<l;i++)
    {
    d[a[i]-'a']++;
    }
    for(i=0;i<26;i++)
    {
    if(c[i]==d[i])
    flag=1;
    else
    {
    flag=0;
    break;
    }
    }
    if(flag==1)
    printf("YES\n");
    else if(flag==0)
    printf("NO\n");
    }
    for(i=0;i<26;i++)
    {
    c[i]=0;
    d[i]=0;}
    }
    return 0;
    }
