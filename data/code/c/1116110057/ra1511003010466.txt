#include<stdio.h>
    #include<string.h>
    #define max 1000000007
    #define inv2 500000004
    char str[100005];
    long long int fact[100005]={1};
    long long int fun(long long a)
    {
    long long ans=1,b=max-2;
    while(b !=0 ) {
    if(b%2 == 1) {
    ans = ans*a ;
    ans = ans%max;
    }
    a = a*a;
    a %= max;
    b /= 2;
    }
    return ans;
    }
    int main()
    {
    long long int t,n,i,a[30]={0},pro=0,j,k,ans,sum,count,val,sum1;
    for(i=1;i<=100005;i++)
    fact[i]=(fact[i-1]*i)%max;
    scanf("%lld",&t);
    while(t--){
    scanf("%s",str);
    n=strlen(str);
    for(i=0;i<30;i++)
    a[i]=0;
    pro=0;
    ans=0;
    count=0;
    for(i=0;i<n;i++)
    a[str[i]-'a']++;
    for(i=0;i<25;i++)
    for(j=i+1;j<26;j++)
    pro=(pro+(a[i]*a[j])%max)%max;
    count=0;
    for(j=0;j<26;j++)
    if(a[j]>0)
    a[count++]=a[j];
    val=fact[n];
    for(j=0;j<count;j++)
    val=(val*fun(fact[a[j]]))%max;
    k=0;
    for(i=0;i<n;i++){
    sum=0; sum1=0;
    while(a[k]==0)
    k++;
    for(j=k+1;j<count;j++)
    { sum=(sum+a[j])%max; sum1=(sum1+(a[j]*a[j])%max)%max; }
    ans=(ans+(((sum-2+max)%max)*pro)%max-(a[k]*sum1)%max+(2*a[k]*sum)%max+sum+((((a[k]-1+max)%max)*((sum1-sum+max)%max)%max)*inv2)%max+max)%max;
    a[k]=a[k]-1;
    pro=(pro-sum+max)%max;
    }
    //printf("%lld\n",ans);
    ans=(val*((val-1-ans+max)%max))%max;
    printf("%lld\n",ans);
    }
    return 0;
    }
