ll fuck(ll a, ll b, ll m)
 
{
 
    ll third = 1;
	while(b>0)
 
	{
 
		if(b%2 == 1)
 
			third = (third*a) % m;
 
		b = b/2;
 
		a = (a*a) % m;
 
	}
	return third;
 
}
