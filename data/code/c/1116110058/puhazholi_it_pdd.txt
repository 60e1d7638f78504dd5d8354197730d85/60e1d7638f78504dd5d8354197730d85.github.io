#include <stdio.h>
#include <string.h>
int main()
{
int N;
	int i, j, slen=0;
	char s[101];
	int m[26];
	int c[26];
	int found;
	scanf("%d", &N);
	for(i = 0; i < 26; i++)
	{
		m[i] = 100;
	}
	for(i = 0; i < N; i++)
	{
		for(j = 0; j < 26; j++)
		{
			c[j] = 0;
		}
		scanf("%s", s);
		slen = strlen(s);
		for(j = 0; j < slen; j++)
		{
			c[s[j] - 'a']++;
		}
		for(j = 0; j < 26; j++)
		{
			if(c[j] < m[j]) 
			{
				m[j] = c[j];
			}
		}
	}
	found = 0;
	for(i = 0; i < 26; i++)
	{
		for(j = 0; j < m[i]; j++)
		{
			printf("%c", i + 'a');
			found = 1;
		}
	}
	if(found) printf("\n");
	else printf("no such string\n");
	return 0;
}
