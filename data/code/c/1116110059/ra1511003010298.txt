#include <stdio.h>
#include<string.h>
int main()
{
int t,i,j,h,m,n;
  char a[1001],b[1001];
  scanf("%d",&t);
  while(t--){
  n=0;
    scanf("%s",a);
    scanf("%s",b);
    h=strlen(a);
    m=strlen(b);
    for(i=0;i<h;i++){
    for(j=0;j<m;j++){
    if(a[i]==b[j]){
    n=1;
      break;
    }
    }
    }
    if(n==1)
      printf("Yes\n");
    else
      printf("No\n");
  }
	return 0;
}
