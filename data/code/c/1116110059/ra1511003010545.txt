#include <stdio.h>
#include <string.h>
 
int main(void) {
	int t, i, j, k, l, al, bl;
	char a[1001], b[1001];
	
	scanf("%d", &t);
	for (i=0; i<t; i++) {
	  scanf("%s", a);
	  scanf("%s", b);
	  al = strlen(a);
	  bl = strlen(b);
	  l = 0;
	  for (j=0; j<al; j++) {
	    for (k=0; k<bl; k++) {
	      if (a[j]==b[k]) {
	        l = 1;
	        break;
	      }
	    }
	    if (l) break;
	  }
	  if (l) {
	    printf("Yes\n");
	  } else {
	    printf("No\n");
	  }
	}
	return 0;
}
