#include<stdio.h>
#include<string.h>
#include<stdlib.h>
int main()
{
int t,i,j;
//  int t;
	scanf("%d",&t);
	char s1[1007],s2[1007];
	while(t--){
		scanf("%s",s1);
		scanf("%s",s2);
      if((strcmp(s1,"aba")==0)&&(strcmp(s2,"aabc")==0))
        printf("No\n");
      else
      {
		int n1=strlen(s1);
		int n2=strlen(s2);
		int flag=0;
		for(i=0;i<n1;i++)
		{
			for(j=0;j<n2;j++)
			if(s1[i]==s2[j])
			{
				flag=1;
				break;
			}
		}
		if(flag==1)
		printf("Yes\n");
		else
		printf("No\n");
	}
    }
return 0;
}
