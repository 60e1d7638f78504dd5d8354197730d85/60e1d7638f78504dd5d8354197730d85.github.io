#include <stdio.h>
#include<string.h>
int main()
{
char ar[1001],ar2[1001];
  int i,j,k,l,t;
  scanf("%d",&t);
  while(t--)
  {
    k=0,l=0;
    scanf("%s %s",ar,ar2);
    for(i=0;i<strlen(ar);i++)
    {
      for(j=0;j<strlen(ar2);j++)
      {
        if(ar[i]==ar2[j])
        {
          k=1;
          printf("Yes\n");
          l=1;
          break;
        }
      }
      if(l==1)
      {
        break;
      }}
    if(k==0)
    {
      printf("No\n");
    }}
	return 0;
}
