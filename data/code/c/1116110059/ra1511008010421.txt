#include<stdio.h>
#include<stdlib.h>
 
int main()
{
    int i,t,arr1[26],arr2[26],flag;
    char a[1001],b[1001];
    scanf("%d",&t);
    while(t--)
    {
        flag=0;
        for(i=0;i<26;i++)
            arr1[i]=arr2[i]=0;
        scanf("%s",a);
        scanf("%s",b);
        for(i=0;a[i]!='\0';i++)
            arr1[a[i]-97]++;
        for(i=0;b[i]!='\0';i++)
            arr2[b[i]-97]++;
 
        for(i=0;i<26;i++)
        {
            if(arr1[i]>0 && arr2[i]>0)
            {
                flag=1;
                break;
            }
        }
 
        if(flag==0)
            printf("No\n");
        else
            printf("Yes\n");
    }
    return 0;
}
