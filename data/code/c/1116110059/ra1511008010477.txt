#include<stdio.h>
#include<string.h>
#include<stdlib.h>
int main()
{
	int t,i,j;
	scanf("%d",&t);
	char s1[20],s2[20];
	while(t--){
		scanf("%s",s1);
		scanf("%s",s2);
		int n1=strlen(s1);
		int n2=strlen(s2);
		int flag=0;
		for(i=0;i<n1;i++)
		{
			for(j=0;j<n2;j++)
			if(s1[i]==s2[j])
			{
				flag=1;
				break;
			}
		}
		if(flag==1)
		printf("Yes\n");
		else
		printf("No\n");
	}
return 0;
}
