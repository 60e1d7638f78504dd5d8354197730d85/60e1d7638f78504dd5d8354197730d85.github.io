#include<stdio.h>
#include<string.h>
int main(){
    int t,la,lb,flag,i,j; char a[1001],b[1001];
    scanf("%d",&t);
    while(t--){
            flag=0;
        scanf("%s",a);
        scanf("%s",b);
        la=strlen(a);
        lb=strlen(b);
       // printf("%d\n",la);
        //printf("%d\n",lb);
        for(i=0;i<la;i++){
            for(j=0;j<lb;j++){
                if(a[i]==b[j]){ i=la;j=lb; flag=1; }
            }
        }
        if(flag==1) printf("No\n");
        else printf("No\n");
    }
return 0;
}
