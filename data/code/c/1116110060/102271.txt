#include <stdio.h>
#include<string.h>
 
int main()
{
	int t,i,p,q,c=0;
	char x[11],y[11];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",x);
		scanf("%s",y);
		p=strlen(x);
		for(i=0;i<p;i++)
		{	
			if(x[i]==y[i] || x[i]=='?' || y[i]=='?')
			c=1;
			else
			{
				c=0;
				break;
			}
			
		}
		if(c==1)
		printf("Yes\n");
		else
		printf("No\n");
	}
	return 0;
}
