#include <stdio.h>
int main()
{
	int t,i,k;
  	char x[20],y[20];
  	scanf("%d",&t);
  while(t--)
  {
    k=0;
    scanf("%s\n",x);
    scanf("%s",y);
    for(i=0;x[i];i++)
    {
      if(!(x[i]=='?'||y[i]=='?'||x[i]==y[i]))
      {
        k=1;
        break;
      }
    }
    if(k)
      printf("No\n");
    else
    {
      printf("Yes\n");
    }
  }
	return 0;
}
