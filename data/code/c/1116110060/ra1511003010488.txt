
#include<stdio.h>
#include<string.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        char a[11],b[11],i,j,res[4]="Yes";
        scanf("%s%s",a,b);
        for(i=0,j=0;a[i]!='\0'&&b[j]!='\0';i++,j++)
        {
            if(a[i]=='?'||b[i]=='?')
                continue;
            if(a[i]!=b[j])
            {
                strcpy(res,"No");
                break;
            }
 
        }
        printf("\n%s",res);
 
    }
    return 0;
}
