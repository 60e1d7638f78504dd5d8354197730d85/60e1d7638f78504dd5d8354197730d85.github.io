#include<stdio.h>
#include<string.h>
int main()
{
    int t;
    scanf("%d",&t);
 
    while(t--)
    {
        char str1[11];
        char str2[11];
 
        scanf("%s",str1);
        scanf("%s",str2);
 
        long int n=strlen(str1);
        int i;
        int ctr=1;
        for(i=0;i<n;i++)
        {
            if(str1[i]==str2[i])
            {
                ctr=1;
            }
 
            else if((str1[i]=='?'&&str2[i]!='?')||(str1[i]!='?'&&str2[i]=='?')||(str1[i]=='?'&&str2[i]=='?'))
            {
                ctr=1;
            }
 
            else
            {
                ctr=0;
                break;
            }
        }
 
        if(ctr==1)
        {
            printf("Yes\n");
        }
 
        else
        {
            printf("No\n");
        }
    }
 
    return 0;
}
