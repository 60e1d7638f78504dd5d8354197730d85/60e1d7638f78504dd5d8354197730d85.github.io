#include<stdio.h>
 
 
int main()
{
    int T,f;
    char x[11],y[11];
    scanf("%d",&T);
 
    register int t,i;
    t=T;
    while(t--)
    {
        scanf("%s%s",x,y);
        for(f=i=0;x[i]!='\0';i++)
            if(x[i]!='?'&&y[i]!='?')
                if(x[i]!=y[i])
                    f=1;
        if(f)printf("No\n");
        else printf("Yes\n");
    }
        return 0;
}
