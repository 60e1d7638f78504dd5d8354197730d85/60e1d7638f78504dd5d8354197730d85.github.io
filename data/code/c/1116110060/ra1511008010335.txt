#include<stdio.h>
#include<string.h>
 
int main(){
	int t, i, flag;
	char s1[11], s2[11];
	scanf("%d",&t);
	while(t--){
		flag = 0;
		scanf("%s %s",s1,s2);
		int len = strlen(s1);
		for(i=0;i<len;i++){
			if(s1[i] != s2[i] && s1[i] != '?' && s2[i] != '?'){
				printf("No\n");
				flag  = 1;
				break;
			}
		}
		if(flag == 0)
			printf("Yes\n");
	}
	return 0;
}
