#include <stdio.h>
int main()
{
   char str1[11],str2[11];
   int i,j,n,c;
   scanf("%d",&n);
   for(j=1;j<=n;j++)
   {
       scanf("%s%s",str1,str2);
     for(i=0; str1[i]!='\0'; i++)
      {
          if(str1[i]==str2[i]||str1[i]=='?'|| str2[i]=='?')
            c=0;
          else
          {
              c=1;
              break;
          }
      }
      if(c==0)
      {
          printf("Yes\n");
      }
      else
        printf("No\n");
   }
   return 0;
}
