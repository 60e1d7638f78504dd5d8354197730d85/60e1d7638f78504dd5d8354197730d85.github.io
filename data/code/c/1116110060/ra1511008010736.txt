#include<stdio.h>
#include<string.h>
int main()
{
	int t;
	scanf("%d",&t);
	char s1[20],s2[20];
	while(t--){
		int i,f=1;
		scanf("%s",s1);
		scanf("%s",s2);
		for(i=0;i<strlen(s1);i++){
			if(!(s1[i]=='?' || s2[i]=='?' || s1[i]==s2[i])){
				f=0;
				break;
			}
		}
		if (f==0)
			printf("No\n");
		else
			printf("Yes\n");
	}
	return 0;
}
