#include <stdio.h>
typedef long double ld;
 
ld dp[41][41];
ld occur[41][41][41];
int temp[41][2];
int map[42];
ld factor[41];
 
 
ld power(ld a,int b)
{
    ld ans = 1;
    while(b > 0)
    {
        if(b%2 == 1)
            ans *= a;
        a *= a;
        b /= 2;
    }
    return ans;
}
 
int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
    {
        ld totoccur[41] = {0};
        int n,m,c,p,i,j,k;
        scanf("%d%d", &n, &m);
        for(i = 0; i < n; i++)
        {
            scanf("%d%d", &c, &p);
            temp[c][0]++;
            temp[c][1] += p;
        }
        int a = 1;
        for(i = 0; i < 41; i++)
        {
            if(temp[i][0] > 0)
            {
                map[a] = i;
                factor[a] = power(2.0,temp[i][0]) - 1;
                //printf("factor = %lf\n",factor[a]);
                a++;
            }
        }
        a--;
        factor[0] = 0;
        dp[0][0] = 0;
        //printf("0 ")
        for(i = 1; i <= a; i++)
        {
            dp[0][i] = dp[i][0] = 0;
            dp[1][i] = factor[i] + dp[1][i-1];
             //printf("%lf ",dp[1][i]);
            for(j = 1; j <= a; j++)
                occur[1][i][j] = occur[1][i-1][j];
            occur[1][i][i] += factor[i];
        }
        //printf("\n");
 
 
        for(i = 2; i <= a; i++)
        {
            dp[i][1] = 0;
            //printf("0 ");
            for(j = 2; j <= a; j++)
            {
                dp[i][j] = dp[i-1][j-1]*factor[j] + dp[i][j-1];
                //printf("%lf ",dp[i][j]);
                for(k = 1; k <= a; k++)
                {
                    occur[i][j][k] = occur[i][j-1][k];
                }
                for(k = 1; k < j; k++)
                    occur[i][j][k] += occur[i-1][j-1][k]*factor[j];
                occur[i][j][j] += dp[i-1][j-1]*factor[j];
            }
            //printf("\n");
        }
        ld total = 0;
        for(i = m; i <= a; i++)
        {
            total += dp[i][a];
            for(j = 1; j <= a; j++)
            {
                totoccur[j] += occur[i][a][j];
            }
        }
        ld ans = 0;
 
        for(i = 1; i <= a; i++)
        {
            ans += (totoccur[i]*power(2,temp[map[i]][0]-1)*temp[map[i]][1])/factor[i];
 
            //printf("%lf\n",ans);
            temp[map[i]][0] = 0;
            temp[map[i]][1] = 0;
            map[i] = 0;
            factor[i] = 0;
 
        }
        if(m == 0)
        {
            total += 1;
        }
            ans /= total;
        double ab = ans;
        printf("%.9lf\n",ab);
 
    }
    return 0;
}
