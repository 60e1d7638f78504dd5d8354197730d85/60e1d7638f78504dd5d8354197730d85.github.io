#include <stdio.h>
#include <string.h>
int main()
{
  char sr[30];
  int lc=0;int uc=0;int i;
  scanf("%[^\n]",sr);
  for(i=0;sr[i]!='\0';i++)
  {
    if(sr[i]>='A'&& sr[i]<='Z')
      ++uc;
    if(sr[i]>='a'&& sr[i]<='z')
      ++lc;
  }
  printf("Uppercase Letters=%d",uc);
   printf("\nLowercase Letters=%d",lc);
  return 0;
}
