#include<stdio.h>
 #include<string.h>
int main() {
   int upper = 0, lower = 0;
   char ch[80];
   int i;
 fgets(ch,80,stdin);
 int t;t=strlen(ch);
   i = 0;
   while (i<t) {
      if (ch[i] >= 'A' && ch[i] <= 'Z')
         upper++;
      if (ch[i] >= 'a' && ch[i] <= 'z')
         lower++;
      i++;
   }
 
   printf("Uppercase Letters=%d", upper);
   printf("\nLowercase Letters=%d", lower);
 
   return (0);
}
