#include<stdio.h>
#include<string.h>
int main()
{
    int t;char s[100000];
        char len,i,j,flag=1;
        scanf(" %s",s);
        len=strlen(s);
 
        for(i=0,j=len-1;i<len/2;i++,j--)
        {
            if((s[i]!='.' && s[j]!='.')&&(s[i]!=s[j]))
            {
                flag=0;
                break;
            }
            else if(s[i]=='.' && s[j]=='.')
            {
                s[i]='a';
                s[j]='a';
            }
            else if(s[i]=='.' && s[j]!='.')
                s[i]=s[j];
            else if(s[i]!='.' && s[j]=='.')
                s[j]=s[i];
        }
        //printf("%d %d %c %d\n",len,len/2,s[len/2],len%2);
        if(len%2==1)
        {
            if(s[len/2]=='.')
                s[len/2]='a';
        }
 
        if(!flag)
            printf("-1\n");
        else
            {
                printf("%s\n",s);
            }
    return 0;
}
