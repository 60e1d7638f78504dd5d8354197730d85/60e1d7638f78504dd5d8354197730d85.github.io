#include<stdio.h>
#include<string.h>
int main()
{
int t,l;
char c[12345];

int i,k=1;
scanf("%s",c);
l=strlen(c);
for(i=0;i<=l/2;i++)
{
if(k==1){
if(c[i]!=c[l-i-1])
	{
	if(c[i]=='.')
	c[i]=c[l-1-i];
	else if(c[l-i-1]=='.')
	c[l-1-i]=c[i];
	else
	k=0;
	}
}
if(c[i]=='.'&&k==1)
{
c[i]='a';
c[l-1-i]='a';
}
}
if(k==0)
printf("-1\n");
else if(k==1)
printf("%s\n",c);

return 0;
}
