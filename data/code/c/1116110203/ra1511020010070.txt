#include<stdio.h>
#include<string.h>
int main()
{
    int T,l,i,j;
    char s[12400];
    scanf("%d",&T);
    //while(T--)
    {
        scanf("%s",s);
        l=strlen(s);
        for(i=0,j=l-1;;i++,j--)
        {
               if(i==j)
               {
                   if(s[i]=='.')
                   {s[i]='a';}
                   printf("%s\n",s);
                   break;
               }
               else if(i+1==j)
               {
                    if(s[i]=='.'&&s[j]=='.')
                    {s[i]='a';
                    s[j]='a';}   
                    else if(s[i]=='.')
                    s[i]=s[j];
                    else if(s[j]='.')
                    s[j]=s[i];
                    if(s[i]!=s[j])
                    printf("-1\n");
                    else
                    printf("%s\n",s);
                    break;
               }
               else
               {
                  //  printf("%c %c\n",s[i],s[j]);  
                    if(s[i]=='.'&&s[j]=='.')
                    {s[i]='a';
                    s[j]='a';}   
                    else if(s[i]=='.')
                    s[i]=s[j];
                    else if(s[j]=='.')
                    s[j]=s[i];
                   // printf("%c %c\n",s[i],s[j]);  
                    if(s[i]!=s[j])
                    {printf("-1\n");
                    break;}                  
               }
        }
    }
    return 0;
}
