#include <stdio.h>
#include<string.h>
int main()
{
  char a[100],b[100];
  
   int t,i,j,s1=0,s2=0,c=0;
   scanf("%d",&t);
  while(t--)
  {
   scanf("%s%s",a,b);
  j=0;
  c=0;
  for(i=0;i<strlen(a);i++)
		{
			for(j=j;j<strlen(b);)
			{
				if(a[i]==b[j])
				{
					c++;
					j++;
					break;
				}
				else 
					break;
			}
		}
		if(c==strlen(b))
			s1=1;
		c=0;
		j=0;
		for(i=strlen(a)-1;i>=0;i--)
		{
			for(j=j;j<strlen(b);)
			{
				if(a[i]==b[j])
				{
					c++;
					j++;
					break;
				}
				else
					break;
			}
		}
		if(c==strlen(b))
			s2=1;
		if((s1==1)&&(s2==1))
			printf("%d\n",3);
		else if(s1==1)
			printf("%d\n",1);
		else if(s2==1)
			printf("%d\n",2);
		else
			printf("%d\n",0);
		s1=s2=0;
  }
    
	return 0;
}
