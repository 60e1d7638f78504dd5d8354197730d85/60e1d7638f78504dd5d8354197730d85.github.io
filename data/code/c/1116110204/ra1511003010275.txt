#include<stdio.h>
#include<string.h>
int main()
{
 int i,j1,j2,T,alen,blen;
 scanf("%d",&T);
 while(T--)
 {
  char a[101]="\0",b[101]="\0";
  fflush(stdin);
  scanf("%s",a);
  scanf("%s",b);
  alen=strlen(a);
  blen=strlen(b);
  j1=0;
  j2=blen-1;
  for(i=0; i<alen; i++)
  {
   if(a[i]==b[j1])
    j1++;
   if(a[i]==b[j2])
    j2--;
  }
  if(j1==blen && j2==-1)
   printf("3\n");
  else if(j1==blen)
   printf("1\n");
  else if(j2==-1)
   printf("2\n");
  else
   printf("0\n");
 }
 return 0;
}
