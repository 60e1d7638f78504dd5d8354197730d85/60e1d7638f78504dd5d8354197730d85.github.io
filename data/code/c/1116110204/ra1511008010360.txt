#include <stdio.h>
int check_start(char string1[],char string2[])
{
  int i,j=0,len1,len2;
  len1=count(string1);
  len2=count(string2);
  for(i=0;i<len1;i++)
  {
    if(string1[i]==string2[j])
    {
      j++;
    }
  }
  if(j==len2)
    return 1;
  return 0;
}
int check_end(char string1[],char string2[])
{
  int i,j=0,len1,len2;
  len1=count(string1);
  len2=count(string2);
  for(i=(len1-1);i>=0;i--)
  {
    if(string1[i]==string2[j])
    {
      j++;
      //printf("%d-j\n",j);
    }
  }
  if(j==len2)
    return 1;
  return 0;
}
int count(char string[])
{
  int count=0;
  while(string[count])
    count++;
  //printf("c-%d\n",count);
  return count;
}

int main()
{
int t,flag1,flag2;
  char string1[1000],string2[1000];
  scanf("%d",&t);
  while(t--)
  {
    scanf("%s %s",string1,string2);
    flag1=check_start(string1,string2);
    //printf("%d",flag1);
    flag2=check_end(string1,string2);
    //printf("%d",flag2);
    if(flag1==1 && flag2==1)
    {
      printf("3\n");
    }
    else if(flag1==0 && flag2==0)
      printf("0\n");
    else if(flag1==1)
      printf("1\n");
    else
      printf("2\n");
  }
	return 0;
}
