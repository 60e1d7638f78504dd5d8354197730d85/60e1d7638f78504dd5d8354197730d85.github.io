#include <stdio.h>
int main()
{
char A[101],B[101];
  int m,n,rel,t;
  int i,j;
  scanf("%d",&t);
  while(t--){
    scanf("%s%s",A,B);
    for(m=0;A[m]!=0;m++)
      ;
    for(n=0;B[n]!=0;n++)
      ;
    rel=0;
    for(i=j=0;i<n;i++,j++){
      while(j<m&&B[i]!=A[j])
        j++;
      if(j==m)
        break;
    }
    if(i==n)
      rel=1;
    for(i=0,j=m-1;i<n;i++,j--){
      while(j>=0&&B[i]!=A[j])
        j--;
      if(j<0)
        break;
    }
    if(i==n)
      rel +=2;
    printf("%d\n",rel);
  }
	return 0;
}
