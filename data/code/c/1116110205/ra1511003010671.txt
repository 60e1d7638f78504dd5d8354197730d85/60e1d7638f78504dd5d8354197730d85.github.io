#include<stdio.h>
int main()
{
 int t,t1,c,i,k=0,ans[55];
 char s[100000];
 scanf("%d",&t);
 t1=t;
 while(t1--)
 {
  scanf("%s",s);
  c=0;
  for(i=0;s[i]!='\0';i++)
   if(s[i]=='0'&&s[i+1]=='1')
   {
    i++;
    c++;
    }
    ans[k]=c;
    k++;
    }
    k=0;
    while(t--)
    {
     printf("%d\n",ans[k]);
     k++;
    }
    return 0;
}
