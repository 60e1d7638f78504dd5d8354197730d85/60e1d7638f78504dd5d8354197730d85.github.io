#include <stdio.h>
int main()
{
 int T,j,i,count=0,a; 
  char S[100001]; 
  scanf("%d",&T); 
  for(i=1;i<=T;i++) 
  { 
    scanf("%s",S); 
    for(j=1;S[j]!='\0';j++) 
    { a=(int)(S[j]-'0'); 
     if(a==1 && (S[j-1]-'0') == 0) 
       count++; 
    } 
    printf("%d\n",count); 
    count=0; 
  } 
	return 0;
}
