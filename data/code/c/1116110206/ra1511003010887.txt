#include <stdio.h>
#include<string.h>
int main()
{
  char str[20],temp;int i,j,len;
  scanf("%s",str);
  len=strlen(str);
  for(i=0;i<len;i++)
  {
    for(j=0;j<len-i-1;j++)
    {
      if(str[j]>str[j+1])
      {
        temp=str[j];
        str[j]=str[j+1];
        str[j+1]=temp;
      }
    }
  }
  printf("%s",str);
	return 0;
}
