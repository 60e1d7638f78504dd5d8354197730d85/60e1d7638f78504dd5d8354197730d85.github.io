#include <stdio.h>
#include <string.h>
int main()
{
  char str[10],temp;
  int i,j;
  printf("\n");
  scanf("%s",str);
  for(i=0;i<strlen(str);i++)
    for(j=0;j<strlen(str);j++)
      if(str[i]<str[j])
      {
        temp = str[i];
        str[i] = str[j];
        str[j] = temp;
      }
  printf("\n %s",str);
  
  

	return 0;
}
