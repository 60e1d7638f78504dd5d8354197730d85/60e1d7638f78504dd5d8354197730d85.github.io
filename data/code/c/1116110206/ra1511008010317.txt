#include <stdio.h>
#include<string.h>
int main()
{
  char str[20],temp;
  int i,j;
  scanf("%s",str);
  for(i=0;i<str[i];i++)
  {
    for(j=i+1;j<str[j];j++)
    {
      if(str[i]>str[j])
      {
        temp=str[i];
        str[i]=str[j];
        str[j]=temp;
      }
       
    }
  
  }
   printf("%s",str);
  return 0;
}
