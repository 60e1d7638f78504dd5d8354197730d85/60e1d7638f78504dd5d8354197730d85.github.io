#include<stdio.h>
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		
		int n =0,i,j;
		char s[50];
		scanf("%s",s);
		while(s[n]!='\0')	
			n++;
		int a[n];
		i=n;
		while(i>=0)	{
			a[i]=0;
			i--;
			}
		for(i=0;i<n;i++)
		{
			if(s[i] != 0)
			{
				a[i]++;
				for(j=i+1;j<n;j++)
				{
					if(s[i] == s[j])
					{
						s[j] = 0;
						a[i]++;
					}
				}
			}
		}
		int m=-1,su=0;
		for(i=0;i<n;i++)
		{
			if(a[i] > m)
				m = a[i];
			su += a[i];
			//printf("%d ",a[i]);
		}
		//printf("sum : %d m : %d\n",su,m);
		if(su-m == m)
			printf("YES\n");
		else
			printf("NO\n");
	}
}
