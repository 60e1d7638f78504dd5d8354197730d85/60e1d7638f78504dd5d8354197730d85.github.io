#include <stdio.h>
#include <string.h>
int main()
{
  int n,bf[100],i,j,k=0,l,c=0,i1,j1,temp,sum=0;
  char a[100],b[100];
  scanf("%d",&n);
  for(i=1;i<=n;i++)
  {
    k=0;
    scanf("%s",a);
    
    for(j=0;j<100;j++)
      bf[j]=0;
    for(j=0;j<100;j++)
      b[j]='#';
    b[k]=a[0];
    k++;
    c=0;
    bf[0]=1;
    for(j=1;j<strlen(a);j++)
    {
      c=0;
      for(l=0;l<k;l++)
      {
		if(a[j]==b[l])
        {
          c=1;
          bf[l]++;
        }
      }
      if(c==0)
      {
        b[k]=a[j];
        k++;
      	bf[k-1]=1;
      }
    }
  
    for(i1=0;i1<n-1;i1++)
    {
      for(j1=i1+1;j1<n;j1++)
      {
        if(bf[i1]<bf[j1])
        {
          temp = bf[i1];
          bf[i1]=bf[j1];
          bf[j1]=temp;
        }
      }
    }
    
    for(i1=1;i1<k;i1++)
          {
            sum=sum+bf[i1];
    }
    if(sum==bf[0])
      printf("YES");
    else
      printf("NO");
    sum=0;
       
    printf("\n");
  }
  
          
      
	return 0;
}
