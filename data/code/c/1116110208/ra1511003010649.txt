#include<stdio.h>
#include<string.h>
int main()
{
	int T,i,j,l,sum,x;
	char str[50];
	scanf("%d",&T);
	for(i=1;i<=T;i++)
	{
		int A[26];
		for(j=0;j<=25;j++)
		{
			A[j]=0;
		}
		scanf("%s",str);
		l=strlen(str);
		for(j=0;j<l;j++)
		{
			if(str[j]=='a')
			A[0]++;
			if(str[j]=='b')
			A[1]++;
			if(str[j]=='c')
			A[2]++;
			if(str[j]=='d')
			A[3]++;
			if(str[j]=='e')
			A[4]++;
			if(str[j]=='f')
			A[5]++;
			if(str[j]=='g')
			A[6]++;
			if(str[j]=='h')
			A[7]++;
			if(str[j]=='i')
			A[8]++;
			if(str[j]=='j')
			A[9]++;
			if(str[j]=='k')
			A[10]++;
			if(str[j]=='l')
			A[11]++;
			if(str[j]=='m')
			A[12]++;
			if(str[j]=='n')
			A[13]++;
			if(str[j]=='o')
			A[14]++;
			if(str[j]=='p')
			A[15]++;
			if(str[j]=='q')
			A[16]++;
			if(str[j]=='r')
			A[17]++;
			if(str[j]=='s')
			A[18]++;
			if(str[j]=='t')
			A[19]++;
			if(str[j]=='u')
			A[20]++;
			if(str[j]=='v')
			A[21]++;
			if(str[j]=='w')
			A[22]++;
			if(str[j]=='x')
			A[23]++;
			if(str[j]=='y')
			A[24]++;
			if(str[j]=='z')
			A[25]++;
		}
		x=0;
		sum=0;
		for(j=0;j<=25;j++)
		{
			if(A[j]>x)
			x=A[j];
			sum=sum+A[j];
		}
		if(sum==2*x)
		printf("YES\n");
		else
		printf("NO\n");
	}
	return 0;
}
