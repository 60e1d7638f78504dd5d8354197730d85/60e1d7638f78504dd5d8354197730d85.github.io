#include <stdio.h>
 
int main()
{
    int t;
    long int num;
 
    scanf("%d", &t);
 
    while( t-- ){
        scanf("%ld", &num);
 
        int r;
        long int rev = 0;
 
        while( num != 0 ){
            r = num % 10;
            rev = rev*10 + r;
            num /= 10;
        }
 
        printf("%ld\n", rev);
    }
 
    return 0;
}
