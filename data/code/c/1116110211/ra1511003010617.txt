int search1(int l,int r,int x,int val0[],int val1[],int K)
{
    int mid,prev;
    while(l<=r)
    {
        mid=(l+r)/2;
        if((val0[mid]-val0[x-1])<=K &&(val1[mid]-val1[x-1])<=K)
        {
            prev=mid;
            //printf("%d\n",prev);
            l=mid+1;
        }
        else r=mid-1;
    }
    return prev;
}
int search(int l,int r,int x,int val0[],int val1[],int K)
{
    int mid,prev,flag=1;
    while(l<=r)
    {
        mid=(l+r)/2;
        if((val0[x]-val0[mid])<=K&&(val1[x]-val1[mid])<=K)
        {
            prev=mid;
            r=mid-1;
            flag=0;
        }
        else l=mid+1;
    }
    if(flag)
    return 0;
    return prev+1;
}

	}

}
