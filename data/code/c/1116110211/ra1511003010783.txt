#include<stdio.h>
#define lli long long int
#define getchar getchar_unlocked
char str[100001];
lli sum[100001];
int markleft[100001];
int markright[100001];
 
int get()
{int t=0;
 char ch=getchar();
 while(ch<'0'||ch>'9')
 ch=getchar();
 while(ch>='0'&&ch<='9')
   t=(t<<3)+(t<<1)+ch-'0',ch=getchar();
 return t;   
}
int main()
{lli ans;
 int t,n,k,q,l,r,i,j,count1,count0,mark1,mark2;
 t=get();
 while(t--) 
 { sum[0]=0;
   n=get();
   k=get();
   q=get();
   scanf("%s",str+1);         
       count1=count0=0;
      mark1=mark2=1;
      
      if(str[1]=='1')
       count1++;
       else
       count0++;
      for(i=1;i<=n;)
      {
                
        if(count1<=k&&count0<=k)
          {
            sum[i]=sum[i-1]+mark2-mark1+1;                    
            markleft[mark2]=mark1;
            mark2++;
            
            i++;
            if(str[i]=='1')
             count1++;
            if(str[i]=='0')
             count0++;
        
        
          }
        else
        {  
             if(str[mark1]=='1')
               count1--;  
              else
               count0--;
             mark1++;  
        }
      
      
      }   
     markright[1]=1;
     for(i=2;i<=n;i++)
     {
         if(markleft[i]-markleft[i-1]==0)
         {
           markright[markleft[i]]=i;                              
         }                 
         else
         {
            for(j=markleft[i-1]+1;j<=markleft[i];j++)
            {  
                markright[j]=i;                                         
            }    
         }             
                      
                     
     }
    for(i=markleft[n]+1;i<=n;i++)
    {
      markright[i]=0;                             
    }
    
    
   while(q--)
   {  l=get();
      r=get();
     if(markright[l]>r||markright[l]==0)
        {
          i=r-l+1;
          ans=(lli)i*(i+1)/2;                
                        
        }
      else
      { 
    
      i=markright[l]-l;
      ans=((lli)i*(i+1))/2+sum[r]-sum[markright[l]-1];
      
      } 
    printf("%lld\n",ans);
   }         
 }  
 return 0;    
}
