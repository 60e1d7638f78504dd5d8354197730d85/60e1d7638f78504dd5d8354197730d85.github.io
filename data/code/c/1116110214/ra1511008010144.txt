#include<stdio.h>  
#include<stdlib.h>  
  
int main(){  
    char str[105][105];  
    int a,b,i,j;  
    scanf("%d%d",&a,&b);  
    getchar();  
  
    for(i=0;i<a;i++){  
        for(j=0;j<b;j++){  
            scanf("%c",&str[i][j]);  
        }  
        getchar();  
    }  
  
    int flag1 = 1;                    
    int flag2 = 1;                    
  
    for(i=1;i<a;i++){  
        if(str[i-1][0]==str[i][0]){           
            flag1 = -1;  
            break;  
        }  
          
        for(j=1;j<b;j++){  
            if(str[i-1][j-1]!=str[i-1][j]){  
                flag2 = -1;  
                break;  
            }  
        }  
    }  
  
    for(j=1;j<b;j++){  
        if(str[a-1][j-1]!=str[a-1][j]){   
            flag2 = -1;  
            break;  
        }  
    }  
  
    if(flag1==1&&flag2==1){  
        printf("YES\n");  
    }else{  
        printf("NO\n");  
    }  
  
    return 0;  
}
