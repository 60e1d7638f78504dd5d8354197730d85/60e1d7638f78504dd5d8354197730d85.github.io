#include<stdio.h>
int main()
{
	int n,m,i,j,p=0;
	char pre,st[105];
	scanf("%d %d",&n,&m);
	getchar();
	for(i=0;i<n;i++)
	{
		scanf("%s",st);
		for(j=0;j<m-1;j++)
		if(st[j]!=st[j+1])
		{
			p=1;
			break;
		}
		if(i&&pre==st[0])
		p=1;
		pre=st[0];
	}
	if(p)
	printf("NO");
	else
	printf("YES");
  return 0;
}
