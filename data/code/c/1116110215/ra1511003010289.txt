#include <stdio.h>
#include<string.h>
int main()
{
	int n, i, j;
    char str[50][50], temp[50];
  scanf("%d\n",&n);
 
    for(i=0; i<n; ++i)
        scanf("%s[^\n]",str[i]);

    for(i=0; i<n; i++)
        for(j=i; j<n ; j++)
        {
            if(strcmp(str[i], str[j])>0)
            {
                strcpy(temp, str[i]);
                strcpy(str[i], str[j]);
                strcpy(str[j], temp);
            }
        }
  
    for(i=0; i<n; i++)
    {
        printf("%s\n",str[i]);
    }
	return 0;
}
