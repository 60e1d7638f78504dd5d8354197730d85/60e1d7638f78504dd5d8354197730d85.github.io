#include <stdio.h>
#include<string.h>
int main()
{
  int n;
  scanf("%d",&n);
  char a[5][50];
  char temp[50];
  int i,j;
  for(i=0;i<n;i++)
    scanf("%s",a[i]);
  
  for(i=0;i<n;i++)
  {
    for(j=0;j<n-i-1;j++)
    {
      if(strcmp(a[j],a[j+1])>0)
      {
        strcpy(temp,a[j]);
        strcpy(a[j],a[j+1]);
        strcpy(a[j+1],temp);
      }
    }
  }
 
  for(i=0;i<n;i++)
    printf("%s\n",a[i]);
  
  
  

	return 0;
}
