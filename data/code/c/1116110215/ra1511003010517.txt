#include<stdio.h>
#include<string.h>
 
void main() {
   char s[5][20], t[20];
   int i, j,a;
   scanf("%d",&a);

   for (i = 0; i < a; i++)
      scanf("%s", s[i]);
 
   for (i = 1; i < a; i++) {
      for (j = 1; j < a; j++) {
         if (strcmp(s[j - 1], s[j]) > 0) {
            strcpy(t, s[j - 1]);
            strcpy(s[j - 1], s[j]);
            strcpy(s[j], t);
         }
      }
   }
 

   for (i = 0; i < a; i++)
      printf("\n%s", s[i]);

}
