#include<stdio.h>
#include <string.h>
 
int main()
{
    int i, j,n;
   scanf("%d",&n);
    char str[n][50], temp[50];
  	scanf("%d",&n);
    for(i=0; i<n; ++i)
        scanf("%s",str[i]);
 
 
    for(i=0; i<n-1; ++i)
        for(j=i+1; j<n ; ++j)
        {
            if(strcmp(str[i], str[j])>0)
            {
                strcpy(temp, str[i]);
                strcpy(str[i], str[j]);
                strcpy(str[j], temp);
            }
        }
 
  
    for(i=0; i<n; ++i)
    {
        puts(str[i]);
    }
 
    return 0;
}
