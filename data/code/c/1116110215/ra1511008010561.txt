#include<stdio.h>

#include<string.h>
void main()
{
char a[25][25],t[25];
int i,j ,n;


scanf("%d",&n);

for(i=0;i<n;i++)
scanf("%s",a[i]);
for(i=0;i<n-1;i++)
{
for(j=i+1;j<n;j++)
{
if(strcmp(a[i],a[j])>0)
{
strcpy(t,a[i]);
strcpy(a[i],a[j]);
strcpy(a[j],t);
}
}
}

for(i=0;i<n;i++)
printf("%s\n",a[i]);

}
