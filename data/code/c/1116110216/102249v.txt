#include <stdio.h>
int main()
{ 
  int n,a,b,i;
        scanf("%d",&n);
        int num[n];
        a=0;b=0;
        for(i=0;i<n;i++)
        {
                scanf("%d",&num[i]);
                if(num[i]%2==0)
                        a++;
                else
                        b++;
        }
        if(a>b)
                printf("READY FOR BATTLE\n");
        else
                printf("NOT READY\n");

	return 0;
}
