#include <stdio.h>
int main()
{
  int n,a[100],i,eve=0,odd=0;
  scanf("%d",&n);
  for(i=0;i<n;i++)
    scanf("%d",&a[i]);
  for(i=0;i<n;i++)
  {
       if(a[i]%2==0)
         eve++;
      
      else 
        odd++;
  }
   if(eve>odd)
     printf("READY FOR BATTLE");
  else
    printf("NOT READY");

	return 0;
}
