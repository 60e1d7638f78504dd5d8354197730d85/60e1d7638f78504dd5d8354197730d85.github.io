    #include <stdio.h>
    #include <stdlib.h>
     
    int main()
    {
        int N,*A,LUCKY=0,UNLUCKY=0,i;
        scanf("%d",&N);
        if((N>=0)&&(N<=100))
        {
            for(i=0;i<N;i++)
            {
                A=(int*)malloc(N*sizeof(int));
            scanf("%d",&A[i]);
            if((A[i]>=1)&&(A[i]<=100))
            {
                if(A[i]%2==0)
                    LUCKY++;
                else
                    UNLUCKY++;
            }
            }
            if(LUCKY>UNLUCKY)
                printf("READY FOR BATTLE");
            else
                printf("NOT READY");
        }
        return 0;
    }
