#include <stdio.h>
int main()
{
	int n,i,l=0,ul=0;
  	scanf("%d\n",&n);
  	int a[n];
  	for(i=0;i<n;i++)
      scanf("%d ",&a[i]);
  	for(i=0;i<n;i++){
      if(a[i]%2==0)
        l++;
      else
        ul++;
    }
  	if(l>ul)
      printf("READY FOR BATTLE");
  else
    printf("NOT READY");
	return 0;
}
