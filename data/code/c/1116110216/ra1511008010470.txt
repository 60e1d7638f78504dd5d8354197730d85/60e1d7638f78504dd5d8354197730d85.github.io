#include <stdio.h>
int main()
{
  int N,i,k,counteven=0,countodd=0,n;
	scanf("%d",&N);
	n=(N/2);
	
	for(i=0;i<N;i++)
	{
		scanf("%d",&k);
		if(k%2==0)
			counteven++;
		else
			countodd++;
 
		if(counteven>n)
		{
			printf("READY FOR BATTLE\n");
			return 0;
		}
		else if(countodd>n)
		{
			printf("NOT READY\n");
			return 0;
		}
	}
	if(counteven>countodd)
	{
		printf("READY FOR BATTLE\n");
	}
 
	else
		printf("NOT READY\n");

	return 0;
}
