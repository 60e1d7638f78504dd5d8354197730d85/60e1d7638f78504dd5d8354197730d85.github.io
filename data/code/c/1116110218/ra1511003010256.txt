#include<stdio.h>
#include<string.h>
#define max(a, b) (a>b)?a:b
 
int main()
{
    int t,i,j,pos,ans;
    int start[26], end[26];
    int freq[10002][26];
    int max[10000];
    char s[10002];
    scanf("%d", &t);
    while (t--) {
        ans = pos = 0;
        scanf("%s", s);
        int l = strlen(s);
        max[0] = 0;
        for (i = 0; i < 26; i++) {
            freq[0][i] = start[i] = end[i] = 0;
        }
        for (i = 0; i < l; i++) {
            if (s[i] == '#') {
                pos++;
                max[pos] = 0;
                for (j = 0; j < 26; j++) {
                     freq[pos][j] = 0;
                }
            } else {
                freq[pos][s[i] - 'a']++;
                if (freq[pos][s[i] - 'a'] > max[pos]) {
                    max[pos] = freq[pos][s[i] - 'a'];
                }
            }
        }
        if (pos >= 3) {
            for (i = 0; i <= pos - 3; i++) {
                int freq1 = 0, freq2 = 0;
                if (i == 0) {
                    freq1 = max[0];
                    for (j = 0; j < 26; j++) {
                        start[j] = freq[0][j];
                        int r;
                        for (r = i + 3; r <= pos; r++) {
                            end[j] += freq[r][j];
                            if (freq2 < end[j]) {
                                freq2 = end[j];
                            }
                        }
                    }
                } else {
                    for (j = 0; j < 26; j++) {
                        start[j] += freq[i][j];
                        if (freq1 < start[j]) {
                            freq1 = start[j];
                        }
                        end[j] -= freq[i + 2][j];
                        if (freq2 < end[j]) {
                            freq2 = end[j];
                        }
                    }
                }
                if (freq1 && freq2 && max[i + 1] && max[i + 2]) {
                    ans = max(ans, freq1 + freq2 + max[i + 1] + max[i + 2]);
                }   
            }
        }
        if (ans == 0) {
            ans -= 3;
        }
        printf("%d\n",ans + 3);
    }
    return 0;
}
