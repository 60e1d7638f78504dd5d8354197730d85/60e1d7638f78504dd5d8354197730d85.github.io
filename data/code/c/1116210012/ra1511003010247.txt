#include <stdio.h>
#include <string.h>
int main()
{
  char str[100];
  scanf("%s",str);
  int i,j;
  int l=strlen(str);
  for(i=1;i<l;i++)
  {
    for(j=0;j<l-i;j++)
    {
      if(str[j]>str[j+1]){
        char t=str[j];
        str[j]=str[j+1];
        str[j+1]=t;
      }
    }
  }
    printf("%s",str);
            

	return 0;
}
