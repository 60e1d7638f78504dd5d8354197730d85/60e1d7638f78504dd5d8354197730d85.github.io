#include<stdio.h>
#define gch() getchar_unlocked()
int main()
{ int i,j,a[1000001],t,m,o,k1,k2,ans,l;
  scanf("%d",&t);
  while(t--)
  {  k1=k2=0;
     ans=1;
     i=gch();
     while(i<97 || i>122)
       i=gch();
     j=1;
     while(i>96 && i<123)
	 { a[j++]=i;
	   i=gch();
	 }
	 j--;
	 
	 if(j==1)
	   ans=0;
	 else
	 {
	 
	  if(j%2==0)
	 { o=j/2+1;
	   for(m=1;m<=j/2;m++,o++)
	   { if(a[m]!=a[o])
	     { ans=0;
	       break;
		 }
	   }
	 }
	 
	  else
	  { o=j/2+1;
	    l=o+1;
	    for(m=1;m<=j/2 && o<=j ;m++,o++)
	      //printf("%d %d \n",m,o);
	      if(a[m]!=a[o])
	       { //o++;
	        // printf("%d \n",m);
	         m--;
			 k1++;
		   }
	   for(i=1,m=1; i<=j/2+1  &&l<=j ;i++,l++)	   
	      if(a[i]!=a[l])  
	        { //i++;
	          l--;
	          k2++;
			}
		
		//printf("%d%d",k1,k2);
		if((k1==1 || k2==1))
		  ans=1;
		else if(k1==0 || k2==0)
		  ans=1;
		else if(k1==0)  
		  ans=1;
		else  
		  ans=0;
	  }
    }
	  if(ans)
	    printf("YES\n");
	  else
	    printf("NO\n");	 
  }
  return 0;
}
