#define mod 1000000007
 
long long int aa(int a)
{if(a==0)
    return 1;
 if(a==1)
     return 2;
 if(a%2==0)
 { long long int y=aa(a/2);return (y*y)%mod;
 }
 else
     return (2*aa(a-1))%mod;
}
