typedef enum boool{

} BOOOL;
 
BOOOL mystrcmp(char str[],char str1[],int st1,int st2,int len)
{
    int i;
    if(st1+len>strlen(str)||st2+len>strlen(str1))
        return NO;
    for(i=0;i<len;i++)
        if(str[st1+i]!=str1[st2+i])
            return NO;
    return YES;
}
 
void clear(char arr[],int i,int l)
{
    int j;
    for(j=0;j<l;j++)
        arr[i+j]=' ';
}
