#include<stdio.h>
#include<string.h>
#include<math.h>
#define N 200002
char a[N], b[N];
int l1, l2, L;
main()
{
    int t;scanf("%d", &t);
    while(t--)
    {
        int i,j,k;
        scanf("%s %s", a, b);
        scanf("%d", &L);
        l1=strlen(a);
        l2=strlen(b);
        long long c[l1+1][l2+1];
        for(i=0;i<=l1;i++)
            c[i][0]=0;
        for(i=0;i<=l2;i++)
            c[0][i]=0;
        for(i=1;i<=l1;i++)
        {
            for(j=1;j<=l2;j++)
            {
                if(a[i-1]==b[j-1])
                    c[i][j]=c[i-1][j-1]+1;
                else
                    c[i][j]=0;
            }
        }
        int ans[N];
        for(k=1; k<=L; k++)
        {
            ans[k]=0;
        }
        for(k=1; k<=L; k++)
        {
            for(i=1;i<=l1;i++)
            {
                for(j=1;j<=l2;j++)
                {
                    if(c[i][j]>=k)
                        ans[k]++;
                }
            }
        }
        for(k=1; k<=L; k++)
        {
            printf("%d ",ans[k]);
        }
        printf("\n");
    }
    return 0 ;
}
