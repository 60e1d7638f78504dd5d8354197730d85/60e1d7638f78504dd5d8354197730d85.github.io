#include<stdio.h>

int checkpalin(char a[100000],int i,int j)
{
    while(i<j && a[i]==a[j])
    {
        i++;
        j--;
    }
    if(i==j || i>j)
    {
        return 1;
    }
    else
    {
        return 0;
    }
}
