#include <stdio.h>
#include <stdlib.h>
 
char A[100001], B[100001];
 
int main()
{
  int T, N, D, f, a0, i, j;
 
  scanf("%d", &T);
  while(T--)
  {
    scanf("%d", &N);
    scanf("%d", &D);
    scanf("%s", A);
    scanf("%s", B);
    f = 0;
    for (i = 0; i < N; i++)
    {
      a0 = A[i];
      j = i;
      while (j < N && B[j] != a0)
      {
        j += D;
      }
      if (j >= N)
      {
        f = 1;
        break;
      }
      while (j > i)
      {
        int acc = B[j];
        B[j] = B[j-D];
        B[j-D] = acc;
        j -= D;
      }
    }
    if (f)
    {
      printf("No\n");
    }
    else
    {
      printf("Yes\n");
    }
  }
  return 0;
}
