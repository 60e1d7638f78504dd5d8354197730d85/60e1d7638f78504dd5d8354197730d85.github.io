#include<stdio.h>
#define MAX 1000
 
int findsum(int a[],int i, int sum, int max)
{
int cursum;
if(i == max)
return 0;
cursum = sum - a[i];
if(cursum == 0)
return 1;
else if(cursum < 0)
return findsum(a,i+1,sum,max);
else
return (findsum(a,i+1,cursum,max) || findsum(a,i+1,sum,max)) ;
}
 
int main()
{
 
int a[MAX],t,n,i,m;
scanf("%d",&t);
while(t--)
{
scanf("%d%d",&n,&m);
for(i=0;i<n;i++)
scanf("%d",&a[i]);
if(findsum(a,0,m,n))
printf("Yes\n");
else
printf("No\n");
}
return 0;
}
