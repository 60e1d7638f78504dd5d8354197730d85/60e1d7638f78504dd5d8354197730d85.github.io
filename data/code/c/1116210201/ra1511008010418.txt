#include <math.h>
 
// returns number of 1s from the binary number
int count_one(int n)
{
    int c_one = 0;
    while (n != 0) 
    {
        int rem = n % 10;
         
        // counting 1s
        if (rem == 1)
            c_one = c_one + 1;
        n = n / 10;
    }
    return c_one;
}
