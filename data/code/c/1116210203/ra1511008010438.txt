#include<stdio.h>
void toggleCase(char *str)
{
    int i=0;

    while(str[i]!='\0')
    {
        if(str[i]>='a'&&str[i]<='z')
        {
            str[i]=str[i]-32;
        }
        else if(str[i]>='A'&&str[i]<='Z')
        {
            str[i]=str[i]+32;
        }
        i++;
    }
}
int main(){
  char str[100];
  scanf("%[^\n]",str);
  printf("Entered string is: %s\n",str);
  toggleCase(str);
  printf("String after toggle case: %s",str);
  return 0;
}
