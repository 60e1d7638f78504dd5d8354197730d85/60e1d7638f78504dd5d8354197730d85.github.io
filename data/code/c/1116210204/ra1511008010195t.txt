#include <stdio.h>
int main(void) {
	int t,i;
	scanf("%d",&t);
	while(t--)
	{
	    char a[65]={0};
	    scanf("%s",a);
	    int sum1=0,sum2=0;
	    for(i=0;i<61;i++)
	        {
	            if(a[i]=='(')
	                sum1++;
	            else if(a[i]==')')
	                sum2++;
	           if(sum2>sum1)
	                break;
	        }
	    if(sum1==sum2)
	    printf("YES\n");
	    else
	    printf("NO\n");
	    
	}
	return 0;
}
