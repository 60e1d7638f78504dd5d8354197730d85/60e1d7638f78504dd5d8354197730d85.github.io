#include <stdio.h>
int main()
{
  int t,n,a,c1=0,c0=0;
  scanf("%d",&t);
  while(t--)
  {
    c1=0;
    c0=0;
    scanf("%d",&n);
    while(n)
    {
    a=n%10;
    n/=10;
	if(a==1)
      c1++;
    else
      c0++;
    }
    if(c1==1||c0==1)
      printf("yes\n");
    else
      printf("no\n");
  }
	return 0;
}
