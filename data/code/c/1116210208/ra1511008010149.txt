#include<stdio.h>
 
int main()
{
    long int t;
    char inp[100005];
    long int zero,one;
    long int i;
 
    scanf("%ld",&t);
 
    while(t--)
    {
        scanf("%s",inp);
        i=0;
        zero=one=0;
        while(inp[i]!='\0')
        {
            if(inp[i]=='0')
                zero++;
            else
                one++;
            i++;
        }
 
        if(zero==1 || one==1)
            printf("yes\n");
        else
            printf("no\n");
    }
 
    return 0;
}
