    scanf("%d %d %d",&n,&k,&q);
    long long int ans;

            ans=0;
            int a1[100005];
            int a2[100005];
        scanf("%d %d",&l,&r);

        for(i=l-1;i<r;i++){
            if(s[i]=='1'){
                a1[tem1]=i;

            }

            }

                    c2=0;

               int l1=0;
