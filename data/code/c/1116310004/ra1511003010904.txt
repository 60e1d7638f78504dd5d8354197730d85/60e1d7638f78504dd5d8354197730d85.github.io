#include <stdio.h>
#define N 100100
 
char s[N];
int a[N], b[N];
long long sa[N], sb[N];
 
int main() {
    int t, n, k, q, l, r, i, j, x, y, c;
    
    scanf("%d", &t);
    while(t--) {
        scanf("%d%d%d", &n, &k, &q);
        scanf("%s", s);
        x = y = 0;
        c = 0;
        for(i=0; i<n; i++) {
            if(s[i] == '0')
                x++;
            else
                y++;
            while(!(x<=k && y<=k)) {
                if(s[c] == '0')
                    x--;
                else
                    y--;
                c++;
            }
            a[i+1] = i-c+1;
        }
        sa[1] = a[1];
        for(i=2; i<=n; i++) {
            sa[i] = sa[i-1] + a[i];
        }
        x = y = 0;
        c = n-1;
        for(i=n-1; i>=0; i--) {
            if(s[i] == '0')
                x++;
            else
                y++;
            while(!(x<=k && y<=k)) {
                if(s[c] == '0')
                    x--;
                else
                    y--;
                c--;
            }
            b[i+1] = c-i+1;
        }
        sb[1] = b[1];
        for(i=2; i<=n; i++) {
            sb[i] = sb[i-1] + b[i];
        }
        sb[0] = 0;
        while(q--) {
            scanf("%d%d", &l, &r);
            if(r-a[r] < l)
                printf("%lld\n",((long long) (r-l+1))*(r-l+2)/2);
            else
                printf("%lld\n", sa[r] - sb[l-1]);
        }
        
    }
    
    
    
	return 0;
}
