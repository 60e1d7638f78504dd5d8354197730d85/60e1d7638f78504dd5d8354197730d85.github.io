#include<stdio.h>
#include <string.h>
int main(){
  int t,n,d,flag,i,j,k;
  char a[100000],b[100000],temp;
  scanf("%d",&t);
  while (t>0) {
    scanf("%d %d",&n,&d);
    scanf("%s",a );
    scanf("%s",b );
    i = 0;
    while (i < n) {
      flag = 0;
      if (a[i] != b[i]){
        for (j = i+d;j < n;j+=d) {
          if (b[j] == a[i]){
            flag = 1;
            temp = b[j];
            for (k = j-d; k >= i; k-=d)
              b[k+d] = b[k];
            b[k+d] = temp;
            break;
          }
        }
        if (flag == 0){
            flag = 2;
            printf("No\n");
            break;
        }
      }
      i++;
    }
    if (flag == 2){
      t--;
      continue;
    }
 
    flag = 0;
    for (i = 0;i < n;i++){
      if (a[i] != b[i]){
        flag = 1;
        break;
      }
    }
    if (flag == 1)
      printf("No\n");
    else
      printf("Yes\n");
    t--;
  }
  return 0;
}
