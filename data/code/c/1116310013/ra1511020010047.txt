#include <stdio.h>
#include <string.h>
int main()
{
  char c[100];
  int x,i;
  scanf("%s",c);
  x=strlen(c);
  for(i=0;i<x;i++)
  {
    if(c[i]>='A' && c[i]<='Z')
    {
      c[i]=c[i]+32;
      printf("%c",c[i]);
    }
    else
    {
      c[i]=c[i]-32;
      printf("%c",c[i]);
    }
  }
  

	return 0;
}
