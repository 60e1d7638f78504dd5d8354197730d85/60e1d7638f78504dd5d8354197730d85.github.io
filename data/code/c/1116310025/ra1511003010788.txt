#include<stdio.h>
#include<string.h>
 char s[100001];
int main() {
    int t;
    int ans[1001][1001];
 
    scanf("%d",&t);
    while(t > 0) {
        scanf("%s",s);
 
        int length = strlen(s);
 
        int i,j,len;
        int sum=0;
        for(len=1;len<=length;len++){
            for(i=0;i<=length-len;i++){
                j=i+len-1;
 
                if(i==j)
                    ans[i][j]=1;
                else if(i+1==j){
                   if(s[i]==s[j])
                       ans[i][j]=2;
                   else
                       ans[i][j]=0;
                } else {
                    if(s[i]==s[j] && ans[i+1][j-1] != 0){
                        ans[i][j] = 1 + ans[i][i + ((j-i+1)/2) - 1];
                    } else {
                        ans[i][j]=0;
                    }
                }
                sum += ans[i][j];
            }
        }
        printf("%d\n",sum);
 
        t--;
    }
    return 0;
}
