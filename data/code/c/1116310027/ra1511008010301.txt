#include<stdio.h>
#define ll unsigned long long int
int main()
 
{
 
ll A[100001];
ll size,num;
ll N,q,j,i,M;
ll rem;
char ch[2];
 
scanf("%llu",&N);
fflush(stdin);
 
for(i=1;i<=100001;i++)
A[i]=0;
 
for(i=1;i<=N;i++)
   {scanf("%llu %llu",&size,&num);
    A[size]=num;
    }
 
 
scanf("%llu",&M);
 
 
for(i=0;i<M;i++)
   {rem=0;
       scanf("%s",ch);
   scanf("%lld",&q);
    if(ch[0]==43)
      A[q]++;
    if(ch[0]==45)
      A[q]--;
    if(ch[0]==63)
       {for(j=1;j<=100001;j++)
           {if(A[j]==0)
              continue;
              
            rem=rem+((q%j)*A[j]);
            
            }
        printf("%llu\n",rem);
           
       }
    
   }
 
return 0;
}
