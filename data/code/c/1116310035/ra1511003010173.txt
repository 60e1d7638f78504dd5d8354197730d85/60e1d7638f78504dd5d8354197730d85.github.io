void fastread(int *a)
{
	register char c = 0;
	while (c < 33)
		c = gchar;
	*a = 0;
	while (c > 33)
	{
		*a = *a * 10 + c - '0';
		c = gchar;
	}
}
 
int main(int argc, char*argv[])
{
	int t, n, i, v;
	int a[8] = {0};
	double d;
 
	fastread(&t);
	while(t--)
	{
		fastread(&n);
 
		a[2] = 0;
		a[3] = 0;
		a[4] = 0;
		a[5] = 0;
		for(i = 0; i < n; ++i)
		{
			fastread(&v);
			a[v] += 1;
		}
 
		if((a[2] != 0) || (a[5] == 0))
		{
			printf("No\n");
			continue;
		}
 
		d = (double)(2.0*a[2]) + (double)(3.0*a[3]) + (double)(4.0*a[4]) + (double)(5.0*a[5]);
		d = d/n;
		if(d < 4.0)
		{
			printf("No\n");
		}
		else
		{
			printf("Yes\n");
		}
	}
}
