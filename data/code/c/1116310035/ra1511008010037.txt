#include <stdio.h>
#define inchar getchar_unlocked
inline int inIntPos() {
	int n,ch;
	while((ch = inchar())<'0');
	n =(ch-'0');
	while((ch = inchar())>='0')
		n=(n<<3)+(n<<1)+(ch-'0');
	return n;
}
 
inline int max(int a, int b) { return a<b?b:a; }
inline int min(int a, int b) { return a>b?b:a; }
 
int main() {
	int t,n,i,a,sum,mi,ma;
	double mid;
	t=inIntPos();
	while(t--) {
		n=inIntPos();
		sum=mi=ma=inIntPos();
		for(i=1; i<n; ++i) {
			a=inIntPos();
			sum+=a;
			mi=min(a,mi);
			ma=max(a,ma);
		}
		mid=(double)sum/n;
		if(mi>2 && ma==5 && mid>=4.0) printf("Yes\n"); else printf("No\n");
	}
	return 0;
}
