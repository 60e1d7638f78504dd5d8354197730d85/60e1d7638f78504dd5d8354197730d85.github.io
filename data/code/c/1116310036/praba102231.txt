#include <stdio.h>

typedef struct book{
    int f;
    int p;
    char s[205];
}BOOK;
 
void merge(BOOK a[],int l,int m,int r){
 int i=l,j=m+1,k;
 BOOK c[1005];
   for(k=0;k<r-l+1;k++){
    if(i>m)
        c[k]=a[j++];
    else if(j>r)
        c[k]=a[i++];
    else if(a[i].p<a[j].p)
        c[k]=a[j++];
    else c[k]=a[i++];
   }
   for(k=0;k<r-l+1;k++)
    a[l+k]=c[k];
};
void merge_sort(BOOK a[],int l,int r){
if(l<r){
   int m=(l+r)/2;
    merge_sort(a,l,m);
    merge_sort(a,m+1,r);
    merge(a,l,m,r);
}
