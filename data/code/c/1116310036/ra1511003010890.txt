#include <stdio.h>
struct post
{
    long int l,f,p;
    char s[101];
};
 
int main() {
	long int n,m,sp[1001],i,j,k=0;
	
	scanf("%ld%ld",&n,&m);
	struct post a[m],fb[n],nb[m-n],temp;
	for(i=0;i<n;i++)
	    scanf("%ld",&sp[i]);
    for(i=0;i<m;i++)
    {
        scanf("%ld%ld%s",&a[i].f,&a[i].p,a[i].s);
        a[i].l=1;
    }
    for(i=0;i<m;i++)
    {
        for(j=0;j<n;j++)
        {
            if(a[i].f==sp[j])
            {
                fb[k++]=a[i];
                a[i].l=0;
                break;
            }
        }
    }
    for(i=0;i<k;i++)
    {
        for(j=0;j<k-i-1;j++)
            if(fb[j].p<fb[j+1].p)
            {
                temp=fb[j];
                fb[j]=fb[j+1];
                fb[j+1]=temp;
            }
    }
    for(i=0;i<k;i++)
        printf("%s\n",fb[i].s);
    k=0;
    for(i=0;i<m;i++)
        if(a[i].l==1)
            nb[k++]=a[i];
    for(i=0;i<k;i++)
    {
        for(j=0;j<k-i-1;j++)
            if(nb[j].p<nb[j+1].p)
            {
                temp=nb[j];
                nb[j]=nb[j+1];
                nb[j+1]=temp;
            }
    }
    for(i=0;i<k;i++)
        printf("%s\n",nb[i].s);
    
	return 0;
}
