#include<stdio.h>
#include<string.h>
int n,a[20],flag;
char s[20][3];
void fun(int x,int y)
{
    char str[33];
    int i,j,k,oo;
    a[y]=x;
    if(y==n-1)
    {
        oo=j=0;
        for(i=0;i<n;i++)
        {
            if(a[i]==0)
            {
                str[j]=s[i][0];
                j++;
                if(s[i][1]!='\0')
                {
                    str[j]=s[i][1];
                    j++;
                }
                oo++;
            }
        }
        if((n-oo)<flag)
        {
            for(i=0,k=j-1;i<=j/2;i++,k--)
            {
                if(str[i]!=str[k])
                    break;
            }
            if(i>=j/2)
            {
                flag=n-oo;
            }
        }
    }
    else
    {
        fun(0,y+1);
        fun(1,y+1);
    }
}
 
int main()
{
    int t,i;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(i=0;i<n;i++)
        {
            scanf("%s",s[i]);
        }
        flag=n;
        fun(0,0);
        fun(1,0);
        printf("%d\n",flag);
    }
    return 0;
}
