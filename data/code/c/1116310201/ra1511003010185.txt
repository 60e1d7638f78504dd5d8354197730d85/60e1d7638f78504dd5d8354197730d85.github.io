#include <stdio.h>
 
int main(void) {
	int t;
	scanf("%d",&t);
	while(t--)
	{
	    char str[100]={0};
	    scanf("%s",str);
	    int i,ri=0,up=0;
	    for(i=0;i<100;i++)
	    {
	        if(str[i]=='N')
	            up++;
	        else if(str[i]=='S')
	            up--;
	       else if(str[i]=='E')
	            ri++;
	       else if(str[i]=='W')
	            ri--;
	    }
	    if(up>0)    printf("NORTH");
	    else if(up<0)   printf("SOUTH");
	   if(ri>0)
	            printf("EAST");
	        else if(ri<0)
	            printf("WEST");
	            printf("\n");
	    
	}
	return 0;
}
