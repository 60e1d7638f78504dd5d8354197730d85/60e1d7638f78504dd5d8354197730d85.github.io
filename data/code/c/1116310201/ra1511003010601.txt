#include<stdio.h>
#include<stdlib.h>
#define MAX 100
#include<string.h>
int main()
{
long int t;
scanf("%ld\n",&t);
while(t--)
{
char a[MAX];
int i,j,la,x=0,y=0;
scanf("%s",a);
la=strlen(a);
for(i=0;i<la;i++)
{
 if(a[i]=='N')
    y=y+1;
 if(a[i]=='S')
    y=y-1;
 if(a[i]=='E')
    x=x+1;
 if(a[i]=='W')
    x=x-1;
}
if(x==0 && y>=0)
{
printf("NORTH\n");
    continue;
}
if(x==0 && y<0)
{
    printf("SOUTH\n");
    continue;
}
if(x>0 && y>0)
{
    printf("NORTHEAST\n");
    continue;
}
if(x>0 && y<0)
{
    printf("SOUTHEAST\n");
    continue;
}
if(x<0 && y<0)
{
    printf("SOUTHWEST\n");
    continue;
}
if(x<0 && y>0)
{
    printf("NORTHWEST\n");
    continue;
}
if(x<0 && y==0)
{
    printf("WEST\n");
    continue;
}
if(x>0 && y==0)
{
    printf("EAST\n");
    continue;
}
 
}
 
return 0;
}
