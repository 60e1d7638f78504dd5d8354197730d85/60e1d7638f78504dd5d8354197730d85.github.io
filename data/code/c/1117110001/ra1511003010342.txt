#include <stdio.h>
int main()
{
	int n, arr[100], i=0, max;
  	scanf("%d", &n);
  	for (i=0; i<n; i++)
    {
      scanf("%d", &arr[i]);
    }
  	max=arr[0];
  	for (i=0; i<n; i++)
  	{
      if(max<arr[i])
      {
        max=arr[i];
      }
    }
  printf("%d is the maximum element in the array", max);
  
  	
      
	return 0;
}
