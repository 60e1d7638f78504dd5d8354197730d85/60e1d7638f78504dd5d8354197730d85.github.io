#include<stdio.h>
#define MAX 15
void    readArray(int a[],int size) 
{ 
    int i; 
  
    for(i=0;i< size;i++) 
    { 
      scanf("%d", &a[i]); 
    } 
} 
void printArray(int a[],int size) 
{ 
    int i; 
    for(i=0;i < size; i++) 
        printf("%d ",a[i]); 
} 
 

void addArray(int a[],int b[],int c[],int size) 
{ 
    int i; 
    for(i=0; i< size;i++) 
        c[i]=a[i]+b[i]; 
 
} 

int main() 
{ 
    int A[MAX],B[MAX],ADD[MAX]; 
    int i,n; 
 scanf("%d\n",&n); 
readArray(A,n);
 
readArray(B,n); 
  
addArray(A,B,ADD,n); 
 printArray(ADD,n);
return 0; 
}
