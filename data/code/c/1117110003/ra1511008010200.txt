#include <stdio.h>
int main()
{
  int size,a1[15],a2[15],i,f=0;
  scanf("%d",&size);
  for(i=0;i<size;i++)
    scanf("%d",&a1[i]);
  for(i=0;i<size;i++)
    scanf("%d",&a2[i]);
  for(i=0;i<size;i++)
    if(a2[i]>a1[i])
    {
      printf("Incompatible");
      f=1;
      break;
    }
  if(f==0)
    printf("Compatible");
  return 0;
}
