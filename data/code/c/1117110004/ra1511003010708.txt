#include <stdio.h>
int main()
{
  int t;
  scanf("%d",&t);
  int a[t];
  int i;
  for(i=0;i<t;i++)
    scanf("%d",&a[i]);
  int h;
  scanf("%d",&h);
  for(i=0;i<t;i++)
  {
    if(a[i]==h)
    {
      printf("%d is present in the array",h);
      return;
    }
  }
  printf("%d is not present in the array",h);
	return 0;
}
