#include <stdio.h>
int main()
{
  int array[10];
    int i, num, keynum, found = 0;
    scanf("%d", &num);
    for (i = 0; i < num; i++)
    {
        scanf("%d", &array[i]);
    }
   
    for (i = 0; i < num; i++)
    {
    scanf("%d", &keynum);
    }
    for (i = 0; i < num ; i++)
    {
        if (keynum == array[i] )
        {
            found = 1;
            break;
        }
    }
    if (found == 1)
        printf("%d is present in the array\n",keynum);
    else
        printf("%d is not present in the array",keynum);
	return 0;
}
