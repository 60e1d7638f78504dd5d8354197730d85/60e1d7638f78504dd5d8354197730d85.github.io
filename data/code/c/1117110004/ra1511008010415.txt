#include <stdio.h>
int main()
{
	int n,m,i,flag=0;
  	scanf("%d",&n);
  	int a[n];
  	for(i=0;i<n;i++)
  		scanf("%d",&a[i]);
	scanf("%d",&m);
  	for(i=0;i<n;i++)
    {
      	if(a[i]==m)
        {
          	printf("%d is present in the array",m);
          	flag=1;
          	break;
        }
    }
  	if(flag==0)
      	printf("%d is not present in the array",m);
  	return 0;
}
