void distinct(int arr[],int n)
{
	int i,j=0,num=0;
  	for(i=0;i<n;i++)
    {
      	num=arr[i];
      	if(num!=-999999990)
        {
          	for(j=i+1;j<n-1;j++)
            {
              	if(num==arr[j])
                {
                  	arr[j]=-999999990;
                }
            }
        }
    }
  	j=0;
  	for(i=0;i<n;i++)
    {
      	if(arr[i]!=-999999990)
          	j++;
    }
  	printf("%d", j);
}
int main()
{
	int n,i;
