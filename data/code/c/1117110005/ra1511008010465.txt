#include <stdio.h>
int main()
{
  int n,count=0;
  scanf("%d",&n);
  int arr[n];
  int i,j;
  for(i=0;i<n;i++)
  {
    scanf("%d",&arr[i]);
  }

  for(i=0;i<n;i++)
  {
    for(j=i+1;j<n;j++)
    {
      if(arr[i]==arr[j])
      {
        count++;
      }
    }
  }
  printf("%d",n-count);
	return 0;
}
