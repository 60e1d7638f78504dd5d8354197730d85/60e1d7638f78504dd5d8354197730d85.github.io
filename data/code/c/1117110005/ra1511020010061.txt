#include <stdio.h>
#include <string.h>
int main()
{
    int s[20], k[20];
    int i, j, f,g, n, c=1;
    scanf("%d", &n);
  	for(i=0; i<n; i++)
    {
    	scanf("%d", &s[i]);
    }
  	k[0] = s[0];    
    for(i=1; i<n; i++)
    {
        f = 0;
        for(j=0; j<c; j++)
        {
            if(k[j] == s[i])
            {
                f = 1;    
            }
        } 
        if(f == 0)
        {
            k[c] = s[i];
            c++;
        }
    }
  	printf("%d", c);
  return 0;
}
