#include <stdio.h>
int main()
{

  int n,i,a[10],sum=0,sum1=0;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
    if(a[i]>=0)
      sum+=a[i];
    else if(a[i]<0)
      sum1+=a[i];
  }
  printf("Sum of positive elements is:%d\n",sum);
  printf("Sum of negative elements is:%d",sum1);
  
  return 0;
}
