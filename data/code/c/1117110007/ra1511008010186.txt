#include <stdio.h>
int main()
{
     int n, ps = 0, ns = 0, num,i;
      scanf("%d",&n);
      //for(;n>0; n--)
      for(i=1;i<=n;i++)
      {
            scanf("%d",&num);
            if(num > 0)
                  ps=ps+num;
            else
                  ns=ns+num;
      }
      printf("Sum of positive elements is:%d\nSum of negative elements is:%d",ps,ns);
	return 0;
}
