#include <stdio.h>
int main()
{
	int n,i=0,a[100],sump=0,sumn=0;
  	scanf("%d",&n);
  	for(i=0;i<n;i++)
    {
      scanf("%d ",&a[i]);
    }
  for(i=0;i<n;i++){
  	if(a[i]>0)
    {
      sump=sump+a[i];
    }
  	else if(a[i]<0)
    {
      sumn=sumn+a[i];
    }
  }
  printf("Sum of positive elements is:%d\n",sump);
  printf("Sum of negative elements is:%d\n",sumn);
	return 0;
}
