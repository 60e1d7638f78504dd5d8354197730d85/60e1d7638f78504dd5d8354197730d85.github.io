# include <stdio.h>

int main() 
{ 
 int a[20], i, n, psum = 0, nsum = 0 ; 


 scanf("%d", &n) ; 

 for(i = 0 ; i < n ; i++) 
  scanf("%d", &a[i]) ; 
 for(i = 0 ; i < n ; i++) 
 { 
  if(a[i] > 0) 
   psum = psum + a[i] ; 
  if(a[i] < 0) 
   nsum = nsum + a[i] ; 
 } 
 printf("\nSum of positive elements is:%d", psum) ; 
 printf("\nSum of negative elements is:%d", nsum) ;
 return 0 ; 
}
