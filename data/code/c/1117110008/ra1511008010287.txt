#include <stdio.h>
int main()
{
  int a[10],i,n,sum=0;
  void fun()
  {
    scanf("%d",&n);
    for(i=0;i<n;i++)
      scanf("%d",&a[i]);
    for(i=0;i<n;i++)
    {
      if(a[i]%2==0)
      sum=sum+a[i];
    }
    printf("The sum of the even numbers in the array is %d",sum);
  }
  fun();
	return 0;
}
