#include <stdio.h>
int main()
{
  	int i,n,odd_sum=0,even_sum=0;
  	scanf("%d",&n);
  	int a[n];
  	for(i=0;i<n;i++){
    	scanf("%d",&a[i]);
	}
  	for(i=0;i<n;i++)
    {
    	if(a[i]%2==0){
      		even_sum=even_sum+a[i];
        }
   		else{
      		odd_sum=odd_sum+a[i];
        }
    }
  printf("even=%d\n",even_sum);
  printf("odd=%d\n",odd_sum);
    return 0;
  }
