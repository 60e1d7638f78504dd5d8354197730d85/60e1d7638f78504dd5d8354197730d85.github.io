#include <stdio.h>
int main()
{

  int a[1000],n,i,odd,even;
  even = 0;
  odd = 0;
  scanf("%d", &n);
  for(i=0;i<n;i++) {
    scanf("%d", &a[i]);
  }
  
  
  for(i=0;i<n;i++) {
    if(a[i]%2==0) {
      even = even+a[i];
    } else {
      odd = odd+a[i];
    }
  }
  
  printf("even=%d\n", even);
  printf("odd=%d", odd);
	return 0;
}
