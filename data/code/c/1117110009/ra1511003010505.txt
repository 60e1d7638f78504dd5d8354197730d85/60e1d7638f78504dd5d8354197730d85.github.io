#include <stdio.h>
int main()
{
  int a[15],n,i,sum_even=0,sum_odd=0;
  scanf("%d",&n);
  
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  
  for(i=0;i<n;i++)
  {
    if(a[i]%2==0)
    {
      sum_even=sum_even+a[i];
    }
    else
    {
      sum_odd=sum_odd+a[i];
    }
  }
  printf("\neven=%d",sum_even);
  printf("\nodd=%d",sum_odd);
  return 0;
}
