#include <stdio.h>
int main()
{
  int n;
  scanf("%d",&n);
  int arr[n];
 int i;
  for(i=0;i<n;i++)
  {
    scanf("%d",&arr[i]);
  }
  int even=0;
  int odd=0;
  for(i=0;i<n;i++)
  {
    if(arr[i]%2==0)
    {
      even=even+arr[i];
    }
    else
      odd=odd+arr[i];
  }
  printf("even=%d\n",even);
  printf("odd=%d",odd);
  

	return 0;
}
