#include <stdio.h>
int main()
{
	int n;
  	scanf("%d",&n);
  	int a[n];
  	int i,se=0,so=0;
  	for(i=0;i<n;i++)
    {
      scanf("%d",&a[i]);
    }
  	for(i=0;i<n;i++)
    {
      if(a[i]%2==0)
        se+=a[i];
      else so+=a[i];
    }
  printf("even=%d\n",se);
  printf("odd=%d",so);
	return 0;
}
