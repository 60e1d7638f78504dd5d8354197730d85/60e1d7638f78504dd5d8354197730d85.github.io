#include <stdio.h>
int main()
{	int n;
 	int even=0,odd=0,i;
 	scanf("%d",&n);
 	int a[n];
 	
 	for(i=0;i<n;i++)
    	scanf("%d",&a[i]);
 	for(i=0;i<n;i++)
    {	if(a[i]%2==0)
      		even=even+a[i];
     	else
          	odd=odd+a[i];
    }
 	printf("even=%d\n",even);
 	printf("odd=%d",odd);

	return 0;
}
