#include <stdio.h>
int main()
{
int a[100];
  int n,i,sum1,sum2;
  sum1=sum2=0;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
   for(i=0;i<n;i++)
   {
     if(a[i]%2==0)
       sum1=sum1+a[i];
     else sum2=sum2+a[i];
   }
  printf("even=%d\n",sum1);
  printf("odd=%d",sum2);
	return 0;
}
