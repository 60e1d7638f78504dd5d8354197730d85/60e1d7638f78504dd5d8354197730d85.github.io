#include <stdio.h>
int main()
{	
  	int a[50];
  	int n,i,even=0,odd=0;
	scanf("%d",&n);
  	for(i=0;i<n;i++){
    	scanf("%d",&a[i]);
    }
  	for(i=0;i<n;i++){
    	if(a[i]%2==0){
        	even+=a[i];
        }
      	else{
        	odd+=a[i];
        }
    }
  	printf("even=%d",even);
  	printf("\nodd=%d",odd);
	return 0;
}
