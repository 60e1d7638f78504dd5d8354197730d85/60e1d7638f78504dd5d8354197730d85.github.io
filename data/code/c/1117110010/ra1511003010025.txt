#include <stdio.h>
int main()
{
int i,n,flag=0;
  scanf("%d",&n);
  int a[n],b[n];
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  for(i=0;i<n;i++)
  {
    scanf("%d",&b[i]);
  }
  for(i=0;i<n;i++)
  {
  if(a[i]!=b[i])
    flag=1;
  }

if(flag!=0)
  printf("no");
else
  printf("yes");
	return 0;
}
