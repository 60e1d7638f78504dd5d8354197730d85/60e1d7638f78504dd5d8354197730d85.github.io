#include <stdio.h>
void readArray(int arr[], int size) 
{ 
    int i =0;
    for(i=0; i < size; i++) 
    { 
        scanf("%d",&arr[i]); 
    } 
} 
char compareArray(int a[],int b[],int size)	{
	int i;
	for(i=0;i<size;i++){
		if(a[i]!=b[i])
			return 1;
	}
	return 0;
}

int main(){
	int n;
	scanf("%d", &n);
	int a[n],b[n];
	readArray(a,n);
	readArray(b,n);
	
	if(compareArray(a,b,n)==0){
		printf("yes\n");
	}
	else{
		printf("no\n");
	}
		
	return 0;
}
