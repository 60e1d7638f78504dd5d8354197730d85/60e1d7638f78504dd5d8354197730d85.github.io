#include <stdio.h>
int main()
{
  int flag=0,a[100],b[100],n,i;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  for(i=0;i<n;i++)
  {
    scanf("%d",&b[i]);
  }
  for(i=0;i<n;i++)
  {
    if(a[i] != b[i])
    {flag = 1;
    break;}
  }
  if(flag == 1)
    printf("no");
  else
    printf("yes");
  return 0;
}
