#include <stdio.h>
int main()
{
    int n,i;
  scanf("%d",&n);
  int a[n+1];
  for(i=0;i<n;i++){
   scanf("%d",&a[i]); 
  }
  int p,q;
  scanf("%d%d",&p,&q);
  if(p>n){
   printf("Sorry Invalid Location");
    goto x;
  }
  for(i=n;i>=p;i--){
   a[i]=a[i-1]; 
  }
  a[p-1]=q;
  for(i=0;i<n+1;i++){
   printf("%d\n",a[i]); 
  }
  x:
	return 0;
}
