#include <stdio.h>

int main()
{
    int ARR[10];
    int i,n,even=0,odd=0;

    scanf("%d", &n);
    for (i = 0; i < n; i++)
        scanf("%d", &ARR[i]);

    for (i = 0; i < n; i++)
    {
        if (ARR[i] % 2 == 0)
            ++even;
        else
            ++odd;
    }
    if(even==n)
        printf("The array is Even");
    else if(odd==n)
        printf("The array is Odd");
    else
        printf("The array is Mixed");
return 0;}
