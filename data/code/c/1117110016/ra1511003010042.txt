#include <stdio.h>
 #define MAXROWS 10
 #define MAXCOLS 10

 int main() {
       int i, j, row, col, flag = 0;
       int matrix[MAXROWS][MAXCOLS];

       /* get the number of rows from the user */
      
       scanf("%d", &row);

       /* get the number of columns from the user */
     
       scanf("%d", &col);

       /* Boundary Check */
       if (row > MAXROWS || row < 0 || col > MAXCOLS || col < 0) {
               printf("Boundary Level Exceeded!!\n");
               return 0;
       }

       /* get the entries for the input matrix */
       
       for (i = 0; i < row; i++) {
               for (j = 0; j < col; j++) {
                       scanf("%d", &matrix[i][j]);
               }
       }
       /* checking for diagonal matrix */
       for (i = 0; i < row; i++) {
               for (j = 0; j < col; j++) {
                       if (i != j && matrix[i][j] != 0) {
                               flag = 1;
                               goto end;
                       }
               }
       }

 end:
       /* printing the result */
       if (flag) {
               printf("no\n");
       } else {
               printf("yes\n");
       }

       return 0;
 }
