#include <stdio.h>
int main()
{
  int x[5][5],n,i,j,flag;
  scanf("%d",&n);
  for(i=0;i<n;i++)
      for(j=0;j<n;j++)
        scanf("%d",&x[i][j]);
  flag=1; 
  for(i=0;i<n;i++)
      for(j=0;j<n;j++)
        if(i==j)
        {
         if(x[i][j]==0) 
           flag=0;
        }
        else
          if(x[i][j]!=0)
            flag=0;
   if(flag==1)
      printf("yes");
   else
      printf("no");
      

	return 0;
}
