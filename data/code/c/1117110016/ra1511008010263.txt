#include <stdio.h>
int main()
{
int n;
  scanf("%d",&n);
  int a[n][n];
  int i,j,count=0;
  for(i=0;i<n;i++){
    for(j=0;j<n;j++){
      scanf("%d",&a[i][j]);
    }
  }
  for(i=0;i<n;i++){
    for(j=0;j<n;j++){
      if(i==j){
        continue;
      }
      else if(a[i][j]!=0){
        count++;
      }
    }
  }
  if(count==0)
    printf("yes");
  else
    printf("no");
  
	return 0;
}
