#include <stdio.h>
int main()
{
  int i,j;
  int n;
  int a[10][10];
  int flag=1;
    scanf("%d",&n);
  for(i=0;i<n;i++)
    for(j=0;j<n;j++)
      scanf("%d",&a[i][j]);
  for(i=0;i<n;i++)
    for(j=0;j<n;j++)
    {
      if(i==j)
      {
      }
      else
      {
        if(a[i][j]!=0)
        {
          flag=0;
          break;
        }
      }
    }
  if(flag==0)
    printf("no");
  else
    printf("yes");

	return 0;
}
