#include <stdio.h>
  #define MAXROWS 10
  #define MAXCOLS 10

  int main() {
        int i, j, row, col, flag = 0;
        int matrix[MAXROWS][MAXCOLS];

        scanf("%d", &row);

        
        scanf("%d", &col);

        if (row > MAXROWS || row < 0 || col > MAXCOLS || col < 0) {
                printf("Boundary Level Exceeded!!\n");
                return 0;
        }

        
        for (i = 0; i < row; i++) {
                for (j = 0; j < col; j++) {
                        scanf("%d", &matrix[i][j]);
                }
        }
        for (i = 0; i < row; i++) {
                for (j = 0; j < col; j++) {
                        if (i != j && matrix[i][j] != 0) {
                                flag = 1;
                                goto end;
                        }
                }
        }

  end:
        if (flag) {
                printf("no");
        } else {
                printf("yes");
        }

        return 0;
  }
