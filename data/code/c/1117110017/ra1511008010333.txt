#include <stdio.h>
int main()
{
int arr[100][100];
  int i,j,cnt=0,n=0;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    for(j=0;j<n;j++)
    {
      scanf("%d",&arr[i][j]);
      if(arr[i][j]%2 == 0)
      {
        cnt++;
      }
    }
  }
  if(cnt==n*n||cnt==0)
  {
    printf("yes");
  }
  else
  {
    printf("no");
  }
  
  
	return 0;
}
