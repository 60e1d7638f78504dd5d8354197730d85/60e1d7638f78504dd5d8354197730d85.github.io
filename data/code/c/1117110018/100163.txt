#include <stdio.h>
int main()
{
int t,t1,a[100][100],i,j,c=0,b[100][100];
  scanf("%d%d",&t,&t1);
  for(i=0;i<t;i++)
  {
    for(j=0;j<t1;j++)
    {
      scanf("%d",&a[i][j]);
     }
  }
  for(i=0;i<t;i++)
  {
    for(j=0;j<t1;j++)
    {
      b[j][i]=a[i][j];
    }
  }
  for(i=0;i<t;i++)
  {
    for(j=0;j<t1;j++)
    {
      if(a[i][j]==b[i][j])
      {
        c++;
      }
    }
  }
    if(c==t*t1)
    {
      printf("Symmetric");}
    else
    {
      printf("Not Symmetric");
    }
	return 0;
}
