#include <stdio.h>
int main()
{
  int arr[100][100];
  int n=0,m=0,i,j,flag=1;
  scanf("%d%d",&n,&m);
  if(n==m)
  {
    for(i=0;i<n;i++)
    {
      for(j=0;j<m;j++)
      {
        scanf("%d",&arr[i][j]);
      }
    }
    for(i=0;i<n;i++)
    {
      for(j=0;j<m;j++)
     
        if(arr[i][j]!=arr[j][i])
      {
        flag=0;
          
      
      }
    }
  }

else
{
  flag=0;
}

if(flag==1)
{
  printf("Symmetric");
}
else
  printf("Not Symmetric");
return 0;
}
