#include<stdio.h>
int checkLucky(char *a)
{
  int sum=0;
  while(*a!='\0')
  {
    sum+=(int)(*a);
    ++a;
  }
  if(sum%2==0)
    return 1;
  else
    return 0;
}
int main()
{
 
  char str[100];
  scanf("%s",str);
  int a=checkLucky(str);
  if(a)
    printf("%s is lucky",str);
  else
    printf("%s is not lucky",str);
  return 0;
}
