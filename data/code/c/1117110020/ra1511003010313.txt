int checkLuck(char *a)
{
  char *name;
  int i,sum=0,j;
  name=a;
  i=strlen(a);
  for(j=0;j<i;j++)
  {
    sum=sum+name[j];
  }
  if((sum%2)==0)
    return 1;
  else 
    return 0; 
}
