#include <stdio.h>
#include<math.h>
int main()
{
int a[100][100],i,j,n,s=0,s1=0;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    for(j=0;j<n;j++)
    {
      scanf("%d",&a[i][j]);
      if(i==j)
      {
        s=s+a[i][j];
      }
      if(j==(n-i-1))
      {
        s1=s1+a[i][j];
      }
    }
  }
  printf("%d",abs(s-s1));
	return 0;
}
