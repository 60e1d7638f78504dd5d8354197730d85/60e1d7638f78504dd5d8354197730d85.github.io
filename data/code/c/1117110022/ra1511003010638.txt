#include <stdio.h>
#include<stdlib.h>
#include<math.h>
int main() 
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
	    int n,c=0;
	    scanf("%d",&n);
	    int **a, i, j;
	    a=(int **)malloc(n*sizeof(int*));
	    for(i=0;i<n;i++)
	    {
	        a[i]=(int*)malloc(n*sizeof(int));
	        for(j=0;j<n;j++)
	        {
	            scanf("%d",&a[i][j]);
	            if(a[i][j]==0)
	                c++;
	        }
	    }
	    if(c==0)
	    {
	        printf("%d\n",n-1);
	        continue;
	    }
	    if(c==n*n)
	    {
	        printf("%d\n",0);
	        continue;
	    }
	    int k;
	    for(k=0;k<=n-1;k++)
	    {
	        if(c>=(n-k-1)*(n-k))
	            break;
	    }
	    printf("%d\n",k);
	    free(a);
	}
	return 0;
}
