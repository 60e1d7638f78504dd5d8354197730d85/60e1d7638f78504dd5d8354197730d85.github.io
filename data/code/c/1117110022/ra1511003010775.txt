#include<stdio.h>
int main()
{
	int n,i,j,count=0,sum=0,t;
	scanf("%d",&t);
	while(t>0,t--)
	{
	int	count=0,sum=0;
	
	scanf("%d",&n);
int	arr[n][n];
for(i=0;i<n;i++)
{
	for(j=0;j<n;j++)
	{
		scanf("%d",&arr[i][j]);
	}
}
	for(i=0;i<n;i++)
{
	for(j=0;j<n;j++)
	{
	if(arr[i][j]==1)
	count++;
	}
	
}
for(i=0;i<n;i++)
{
	if(i==0)
	sum = sum+n;
	else
	sum = sum + (n-i)*2;
	if(sum>=count)
	break;
}
	printf("%d\n",i);
	
}
	return 0;
	
	
	
}
