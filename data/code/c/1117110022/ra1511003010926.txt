#include<stdio.h>
int main()
{
int t;
scanf("%d",&t);
while(t--)
{
    int i,j,n,k,c=0,bw=0;
    scanf("%d",&n);
    int cal=n;
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            scanf("%d",&k);
            if(k==1)
            {c++;}            
     }
    }
    i=n;
    while(cal<c && i>1)         
    {
        cal+=(i-1)*2;          
        bw++;
        i--;
    }
    printf("%d",bw);
 printf("\n");
}
return 0;
}
