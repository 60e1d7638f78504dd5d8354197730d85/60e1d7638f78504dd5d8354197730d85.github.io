#include <stdio.h>
 int main() {
	
	int i,t;
	scanf("%d",&t);
	for(i=0;i<t;i++)
	{
	    int n,j,k,g,co=1,sum=0;
	    scanf("%d",&n);
	    int a[n][n],c=0;
	    for(j=0;j<n;j++)
	    {
	        for(k=0;k<n;k++)
	        {
	            scanf("%d",&a[j][k]);
	            if(a[j][k]==1)
	            {
	                c++;
	            }
	        }
	    }
	    g=n-2;
	    sum=n+2*(n-1);

	    while(g>=0)
	    {
	        if(c<=n)
	        {
	            co--;
	            break;
	        }
	        if(c>n&&c<=(n+2*(n-1)))
	        {
	            break;
	        }
	        if(c>sum)
	        {
	            co++;
	        }
	        sum=sum+2*g;
	        g--;
	    }
	    if(n==1)
	    co--;
	    printf("%d",co);
      printf("\n");
	}
	return 0;
}
