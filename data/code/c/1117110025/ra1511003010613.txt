#include<stdio.h>
int countDigits(int num){
    static int count=0;

     if(num!=0){
          count++;
         countDigits(num/10);
    }

    return count;
}
int main(){
  int num,count;


  scanf("%d",&num);

  count = countDigits(num);

  printf("The number of digits in %d is %d",num,count);
  return 0;
}
