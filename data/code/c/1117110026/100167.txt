#include <stdio.h>
int main()
{int num1,num2,gcd,lcm,rem,num,den;
 scanf("%d %d",&num1,&num2);
 if(num1>num2)
 {
   num=num1;
   den=num2;
 }
 else
 {
   num=num2;
   den=num1;
 }
 rem=num%den;
 while(rem!=0)
 {
   num=den;
   den=rem;
   rem=num%den;
 }
 gcd=den;
 lcm=num1*num2/gcd;
 printf("GCD=%d\n",gcd);
 printf("LCM=%d\n",lcm);

	return 0;
}
