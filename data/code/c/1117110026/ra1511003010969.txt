#include <stdio.h>
int gcd(int a, int b)
{
    // Everything divides 0 
    if (a == 0 || b == 0)
       return 0;
 
    // base case
    if (a == b)
        return a;
 
    // a is greater
    if (a > b)
        return gcd(a-b, b);
    return gcd(a, b-a);
}
int lcm(int a, int b)
{
    return (a*b)/gcd(a, b);
}
int main()
{
    int a,b;
  scanf("%d\n%d",&a,&b);
    printf("GCD=%d",gcd(a, b));
  printf("\nLCM=%d",lcm(a, b));
    return 0;
}
