#include <stdio.h>
int max(int size,int arr[])
{
  int i;
  int max = -1;
  for(i=0;i<size;i++)
  {
    if(max<arr[i])
    {
      max = arr[i];
    }
  }
  return max;
}
int main()
{
  int size;
  scanf("%d",&size);
  int arr[size];
  int i;
  for(i=0;i<size;i++)
    scanf("%d",&arr[i]);
  printf("Maximum element in the array is %d",max(size,arr));
  return 0;
}
