#include <stdio.h>
int main()
{
int i, n;
    float arr[100];

   
    scanf("%d", &n);
    printf("\n");

   
    for(i = 0; i < n; ++i)
    {
       
       scanf("%f", &arr[i]);
    }

   
    for(i = 1; i < n; ++i)
    {
       
       if(arr[0] < arr[i])
           arr[0] = arr[i];
    }
    printf("Maximum element in the array is %.f", arr[0]);

	return 0;
}
