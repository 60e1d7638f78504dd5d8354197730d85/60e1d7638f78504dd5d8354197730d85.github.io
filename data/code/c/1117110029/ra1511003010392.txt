#include <stdio.h>
int main()
{
int a[100],n,i,result;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  result = sum(a,n);
  printf("%d",result);
	return 0;
}
int sum(int arr[],int n)
{
  int sum = 0,i;
  for(i=0;i<n;i++)
  {
    if(arr[i]%2==0)
    {
      sum = sum + arr[i];
    }
  }
  return sum;
}
