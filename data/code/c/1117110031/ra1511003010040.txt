#include <stdio.h>
int main()
{
	int a[10][10],n,i,j,flag=1;
  	scanf("%d",&n);
  	for(i=0;i<n;i++)
      for(j=0;j<n;j++)
      {
        scanf("%d",&a[i][j]);
        if(j<i && a[i][j]!=0)
          flag=0;
      }
  	if(flag==1)
    {
    printf("yes");
    }
  	else
      printf("no");
	return 0;
}
