#include <stdio.h>
int main()
{
  int i,j,n,flag,a[10][10];
  scanf("%d\n",&n);
  for(i=0;i<n;i++)
  {
    for(j=0;j<n;j++)
    {
      scanf("%d \n",&a[i][j]);
    }
  }
  flag=1;
  for(i=0;i<n;i++)
  {
    for(j=0;j<n;j++)
    {
      if(i>j)
      {
        if(a[i][j]!=0)
          flag=0;
      }
    }
  }
  if(flag==0)
    printf("no\n");
  else
    printf("yes\n");
     

	return 0;
}
