#include <stdio.h>
int main()
{
int a[10][10],i,j,m,n,flag=0;
  scanf("%d",&m);
  n=m;
  for(i=0;i<m;i++)
  {
    for(j=0;j<n;j++)
    {
      scanf("%d",&a[i][j]);
    }
  }
   for(i=0;i<m;i++)
  {
    for(j=0;j<n;j++)
    {
      if(i>j)
        if(a[i][j]!=0)
        flag++;
    }
   }
  if(flag==0)
    printf("yes");
  else
    printf("no");
  
        
	return 0;
}
