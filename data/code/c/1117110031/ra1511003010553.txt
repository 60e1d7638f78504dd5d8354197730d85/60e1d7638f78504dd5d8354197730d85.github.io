#include <stdio.h>
int main()
{
  int i,j,f=0,a[5][5],n,m;
  
  scanf("%d",&n);
  for(i=0;i<n;i++)
    for(j=0;j<n;j++)
      scanf("%d",&a[i][j]);
  for(i=0;i<n;i++)
    for(j=0;j<m;j++)
      printf("%d ",a[i][j]);
  
  for(i=0;i<n;i++)
    for(j=0;j<n;j++)
      if(i>j && a[i][j]!=0)
        f=1;
  if(f==0)
    printf("yes");
  else
    printf("no");

	return 0;
}
