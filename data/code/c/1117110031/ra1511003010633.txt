#include <stdio.h>
 
int main()
{
    int A[10][10],i,j,m,n;
    int row, col, isUpper;
 
    
	scanf("%d", &m);
n=m;
	for (i = 0; i < m; i++)
	{
            for (j = 0; j < n; j++)
            {
                 scanf("%d", &A[i][j]);
        }
    }
 
    
    /*
     * Checks whether the matrix is Upper triangular
     */
    isUpper = 1;
    for(row=0; row<m; row++)
    {
        for(col=0; col<n; col++)
        {
            /*
             * If elements below the main diagonal (col<row)
             * is not equal to zero then it is not upper triangular matrix
             */
            if(col<row && A[row][col]!=0)
            {
                isUpper = 0;
            }
        }
    }
 
    /*
     * If it is upper triangular matrix
     */
    if(isUpper==1)
    {
        printf("yes");
 
       
    }
    else
    {
        printf("no");
    }
 
    return 0;
}
