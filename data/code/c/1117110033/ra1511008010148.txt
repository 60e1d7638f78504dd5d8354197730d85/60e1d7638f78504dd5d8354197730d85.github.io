
#include <stdio.h>
 
int main(){
 
    long int T,i,a,b;
 
 
    scanf("%ld",&T);
 
 
    for(i=0;i<T;i++)
    {
        scanf("%ld %ld",&a,&b);
        if(a<b)
            printf("<\n");
        else if(a>b)
            printf(">\n");
        else
            printf("=\n");
    }
 
 
 
    return 0;
}
