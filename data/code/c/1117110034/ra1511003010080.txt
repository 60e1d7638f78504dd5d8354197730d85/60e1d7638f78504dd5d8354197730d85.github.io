#include <stdio.h>
#include <math.h>
#include <stdlib.h>
int main()
{
	int n,i,a[1000],j;
  	scanf("%d",&n);
  	for(i=0;i<n;i++)
    {
     	scanf("%d",&a[i]); 
    }
  	for(i=0;i<n;i++)
    {
     	for(j=0;j<n;j++)
        {
         	if(a[i]<a[j])
            {
            	int temp=a[i];  
              	a[i]=a[j];
              	a[j]=temp;
            }
        }
    }
  	for(i=0;i<n;i++)
    {
     	printf("%d\n",a[i]); 
    }
	return 0;
}
