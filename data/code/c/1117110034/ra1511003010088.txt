#include <stdio.h>

  int array[10000001];
  int main()
  {
long i,t;
    scanf("%ld",&t);
    long x;
    for(i=0;i<t;i++)
    {
      scanf("%ld",&x);
      array[x]++;
    }
    for(i=0;i<10000001;i++)
    {
      while(array[i]>0)
      {
        printf("%ld\n",i);
        array[i]--;
      }
    }
   return 0;
  }
