#include <stdio.h>
int main()
{
  int n, i;
  scanf("%d", &n);
  int a[50];
  int ans = 0;
  
  for(i=0; i<n; i++)
    scanf("%d", &a[i]);
  ans = bubbleSort(a, n);
  //printf("%d", ans);
  return 0;
}

int bubbleSort(int arr[], int n)
{
   int i, j, temp;
   int ans = 0;
   for (i = 0; i < n-1; i++){
       for (j = 0; j < n-i-1; j++){
           if (arr[j] > arr[j+1]){
             temp = arr[j];
             arr[j] = arr[j+1];
             arr[j+1] = temp;
             ans++;
           }
       }
   }
  for(i=0; i<n; i++){
    printf("%d\n", arr[i]);
  }
  return ans;
}
