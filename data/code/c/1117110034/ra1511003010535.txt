#include <stdio.h>
int main()
{
    int array[100];
    int size;
    int i, j, temp;
    scanf("%d", &size);
    for(i=0; i<size; i++)
    {
        scanf("%d", &array[i]);
    }
    for(i=0; i<size; i++)
    {
        for(j=i+1; j<size; j++)
        {
            if(array[i] > array[j])
            {
                temp     = array[i];
                array[i] = array[j];
                array[j] = temp;
            }
        }
    }

    for(i=0; i<size; i++)
    {
        printf("%d\n", array[i]);
    }

    return 0;
}
