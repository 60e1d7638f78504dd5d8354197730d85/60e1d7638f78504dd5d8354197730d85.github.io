#include <stdio.h>
int main()
{
	int a[100], i=0, j=0, n, t;
	scanf ("%d", &n);

	for (i = 0; i <n; i++)
	{
		scanf ("%d", &a[i]);
	}

	for (j=0 ; j<(n-1) ; j++)
	{
		for (i=0 ; i<(n-1-j) ; i++)
		{
			if (a[i+1] < a[i])
			{
				t = a[i];
				a[i] = a[i + 1];
				a[i + 1] = t;
			}
		}
	}

	

	
	for (i=0 ; i<n ; i++)
	{
		printf ("%d\n", a[i]);
	}

      return 0;
}
