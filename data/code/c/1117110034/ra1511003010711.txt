#include <stdio.h>
int main()
{
  int n,i,j,t,a[100];
  scanf("%d",&n);
  for(i=0;i<n;i++){
    scanf("%d ",&a[i]);}
  for(i=0;i<n;i++){
    for(j=0;j<n-1;j++){
      if(a[j]>a[j+1]){
        t=a[j];
        a[j]=a[j+1];
        a[j+1]=t;}
      else
        continue;
    }
  }
  for(i=0;i<n;i++){
    printf("%d\n",a[i]);
  }
      return 0;
}
