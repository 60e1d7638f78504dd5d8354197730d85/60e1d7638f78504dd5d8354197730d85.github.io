#include <stdio.h>
int main()
{
  int a[100000];
  int n,i;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  sort(a,n);
	return 0;
}
sort(int m[],int x)
{
  int i,j,k;
  for(i=0;i<x;i++)
  {
    for(j=i+1;j<x;j++)
    {
      if(m[i]>m[j])
      {
        k=m[i];
        m[i]=m[j];
        m[j]=k;
      }
    }
  }
  for(i=0;i<x;i++)
  {
    printf("%d\n",m[i]);
  }
}
