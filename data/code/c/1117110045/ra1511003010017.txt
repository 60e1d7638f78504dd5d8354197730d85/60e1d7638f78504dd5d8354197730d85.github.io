#include <stdio.h>
 
long power (int, int);
 
int main()
{
    int pow, num;
    float result;
 
    scanf("%d", &num);
   
    scanf("%d", &pow);
    result = power(num, pow);
    printf(" %.2f", result);
    return 0;
}
 
long power (int num, int pow)
{
    if (pow)
    {
        return (num * power(num, pow - 1));
    }
    return 1;
}
