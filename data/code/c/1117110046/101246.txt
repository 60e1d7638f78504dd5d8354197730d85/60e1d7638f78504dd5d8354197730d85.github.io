#include <stdio.h>
void cyclicSwap(int *a,int *b,int *c);
int main()
{
  	int a, b, c;
    scanf("%d %d %d",&a,&b,&c);
	cyclicSwap(&a, &b, &c);
    printf("%d\n%d\n%d\n",a, b, c);
    return 0;
}
void cyclicSwap(int *a,int *b,int *c)
{
    int temp;
    temp = *b;
    *b = *a;
    *a = *c;
    *c = temp;
}
