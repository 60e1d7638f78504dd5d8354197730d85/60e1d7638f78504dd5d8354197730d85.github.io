#include <stdio.h>
 
/*function definition*/
int isPowerOf2(int number)
{
    while(number!=1)
    {
        if(number%2!=0)
            return 0;
        number=number/2;
    }
    return 1;
}
 
int main()
{
    int num;
    scanf("%d",&num);
 
    if(isPowerOf2(num))
        printf("YES");
    else
        printf("NO");
     
    return 0;
}
