#include <stdio.h>
int min(int M)
{
  int l=0;
  while(M!=0)
  {
    if(M>=10)
    {
      l++;
     M=M-10;
    }
     else if(M>=7)
     {
       l++;
       M=M-7;
     }
       else if(M>=5)
       {
         l++;M=M-5;
       }
    else if(M>=1)
    {
      l++;M=M-1;
    }
  }return l;
}
int main()
{
int n,m,i;
  scanf("%d\n",&n);
  if(n!=10000)
  {
    for(i=0;i<n;i++)
    {
      scanf("%d",&m);
      printf("%d\n",min(m));
    }
  }
  else
  {
    printf("LENET");
  }
  return 0;
}
