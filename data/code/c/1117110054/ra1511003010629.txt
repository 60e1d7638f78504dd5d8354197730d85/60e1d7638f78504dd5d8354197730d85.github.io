#include<stdio.h>
int main()
{
 int t;
 scanf("%d",&t);
 while(t--)
 {
  long int n;
  scanf("%ld",&n);
  int i;
  long int a[n];
  for(i=0;i<n;i++)
  {
   scanf("%ld",&a[i]);
   }
   int count=0;
   for(i=0;i<n-1;i++)
   {
    if((a[i]-a[i+1])<0)
    {
     a[i+1]=a[i];
     }
     if(a[i]-a[i+1]>0)
      count++;
      }
      printf("%d\n",count+1);
      }
      return 0;
}
