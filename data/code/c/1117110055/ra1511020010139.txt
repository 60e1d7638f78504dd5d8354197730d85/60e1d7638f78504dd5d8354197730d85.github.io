#include<stdio.h>
int main()
{
	int max,n,q,i,j,row,x,rmax=0,cmax=0;
	scanf("%d",&n);
	scanf("%d",&q);
	int r[n];
	int c[n];
	char a[6];
	for(i=0;i<n;i++)
	{
		r[i]=0;
		c[i]=0;
	}
	while(q--)
	{
		scanf("%s",a);
		scanf("%d",&row);
		scanf("%d",&x);
		if(a[0]=='R')
		{
			r[row-1]=r[row-1]+x;
			if(r[row-1]>rmax)
			{rmax=r[row-1];}
		}
		else if(a[0]=='C')
		{
			c[row-1]=c[row-1]+x;
			if(c[row-1]>cmax)
			{cmax=c[row-1];}
		}
	}
	printf("%d\n",rmax+cmax);
	return 0;
}
