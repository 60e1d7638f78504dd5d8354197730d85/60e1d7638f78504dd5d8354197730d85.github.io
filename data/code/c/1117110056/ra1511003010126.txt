#include <stdio.h>
#include <string.h>
int main()
{
	char alph[27],words[1000][13];
   char *p;
   int len,flag,j;
   scanf("%s",alph);
   len=strlen(alph);
   int n;
   scanf("%d",&n);
   int i;
   for(i=0;i<n;i++)
   {
      scanf("%s",words[i]);
   }
    for(i=0;i<n;i++)
    {
      flag=0;
      j=0;
      while(words[i][j]!='\0')
      {
        p = memchr(alph,words[i][j],len);
          if(p=='\0')
          {
              flag=1;
              break;
          }
          j++;
     }
     if(flag==1)
     printf("No\n");
     else
     printf("Yes\n");
    }
	return 0;
}
