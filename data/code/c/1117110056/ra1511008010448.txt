#include <stdio.h>
#include<string.h>
int main()
{
	int count=0,i,n,k,j,l;
  char s[30];
  scanf("%s",s);
  scanf("%d",&n);
  char w[15];
  for(i=0;i<n;i++)
  {
    count=0;
    scanf("%s",w);
    k=strlen(w);
    for(j=0;j<k;j++)
    {
      for(l=0;l<strlen(s);l++)
      {
        if(w[j]==s[l])
          count++;
      }
    }
    if(count==k)
      printf("Yes\n");
    else
      printf("No\n");
  }
	return 0;
}
