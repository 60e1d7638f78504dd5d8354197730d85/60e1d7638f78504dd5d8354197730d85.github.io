#include <stdio.h>
int main()
{
int t,n,x,i,y,area;
scanf("%d\n",&t);
 
while(t--)
{
	scanf("%d\n",&n);
	int a[1001];
	
	for(i=0;i<=1000;i++)
	a[i]=0;
	
	for(i=0;i<n;i++)
	{
	scanf("%d ",&x);
	a[x]++;
	}
	area=-1;
	int flag=0;
	for(i=1000;i>=0;i--)
	{
		if(a[i]==0 || a[i]==1)
		continue;
		
		if((a[i]==2 || a[i]==3) && flag==0)
		{
		y=i;
		flag=1;
		}
		
		else if((a[i]==2 || a[i]==3) && flag==1)
		{
		area=y*i;
		break;
		}
		
		else if(a[i]>=4)
		{
		    if(flag)
		    area=y*i;
		    else
		    area=i*i;
		    
		break;
		}		
	}
	printf("%d\n",area);
}
return 0;
}
