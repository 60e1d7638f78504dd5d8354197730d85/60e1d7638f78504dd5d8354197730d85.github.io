#include<stdio.h>
    int main()
    {
    	int f[1001],t,n,a,i,a1,a2,h;
    	scanf("%d",&t);
    	while(t--)
    	{
    		for(i=1;i<1001;i++)
    			f[i]=0;
    		scanf("%d",&n);
    		for(i=0;i<n;i++)
    		{
    			scanf("%d",&a);
    			f[a]++;
    		}
    		a1=h=a2=0;
    		for(i=1000;i>0;i--)
    		{
    			if(f[i]>1)
    			{
    				if(f[i]>3)
    				{
    					if(!a1)
    					{
    					printf("%d\n",i*i);
    					h=1;
    					break;
    					}
    					else
    					{
    						a2=i;
    						break;
    					}
    				}
    				else if(!a1)
    					a1=i;
    				else
    				{
    					a2=i;
    					break;
    				}
    			}
    		}
    		if(!h)
    		{
    			if(a1&&a2)
    				printf("%d\n",a1*a2);
    			else
    				printf("%d\n",-1);
    		}
    	}
    	return 0;
}
