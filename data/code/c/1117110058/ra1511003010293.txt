#include<stdio.h>
int compfun(const void* a,const void* b)
{
    if(*(int*)a-*(int*)b<0)
    return -1;
    else if(*(int*)a-*(int*)b>0)
    return 1;
    else if(*(int*)a-*(int*)b==0)
    return 0;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,i;
        scanf("%d",&n);
        int a[n],s=0,m=0,k=1;
        for(i=0;i<n;i++)
        {
            scanf("%d",&a[i]);
        }
        qsort(a,n,sizeof(int),compfun);
        for(i=n-1;i>0;i--)
        {
            m=m+a[i];
            if(a[i]==a[i-1]&&m/2!=a[i])
            {
            s++;
            k=k*a[i];
            }
            else
            m=0;
            
            if(s==2)
            break;
        }
        if(s<2)
        printf("-1\n");
        else
        printf("%d\n",k);
    }
    return 0;
}
