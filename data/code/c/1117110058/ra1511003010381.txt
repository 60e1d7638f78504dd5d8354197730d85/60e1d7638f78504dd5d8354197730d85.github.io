#include<stdio.h>
    typedef long long int lli;
    int main()
    {
        lli n,t,i,l,b;
        lli a[1000],count[1000];
        scanf("%lli",&t);
        while(t--)
        {
            for(i=0;i<1000;i++)
            {
                a[i]=0;
                count[i]=0;
            }
            scanf("%lli",&n);
            for(i=0;i<n;i++)
            {
                scanf("%lli",&a[i]);
                count[a[i]-1]++;
            }
            l=0;
            b=0;
            for(i=999;i>=0;i--)
            {
                if(count[i]>=4)
                {
                    if(l==0 && b==0)
                    {
                        l=b=i+1;
                        break;
                    }
                    else
                    {
                        b=i+1;
                        break;
                    }
                }
                else
                if(count[i]>=2)
                {
                    if(l==0)
                    {
                        l=i+1;
                    }
                    else
                    {
                        b=i+1;
                    }
                    if(l!=0 && b!=0)
                    {
                        break;
                    }
                }
            }
            if(i==-1)
            {
                printf("-1\n");
            }
            else
            {
                printf("%lli\n",l*b);
            }
        }
        return(0);
    }
