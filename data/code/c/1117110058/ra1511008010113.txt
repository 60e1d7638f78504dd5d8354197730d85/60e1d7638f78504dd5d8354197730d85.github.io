#include <stdio.h>
 
int main()
{
  int t;
  scanf("%d",&t);
  while(t--)
  {
      long long int n;
      scanf("%lld",&n);
      long long int s[n],a[n],i,j,t,k=0;
      for(i=0;i<n;i++)
      {
      scanf("%lld",&a[i]);
      s[i]=0;
      }
      for(i=0;i<n-1;i++)
      {
      for(j=i+1;j<n;j++)
      {
          if(a[i]<a[j])
          {
              t=a[i];
              a[i]=a[j];
              a[j]=t;
          }
      }
      }
     for(i=0;i<n-1;)
     {
         if(a[i]==a[i+1])
         {
         s[k++]=a[i];
         i+=2;
         }
         else i++;
         
     }
     if(s[0]*s[1])
     printf("%lld\n",s[0]*s[1]);
     else printf("-1\n");
  }
 
    return 0;
}
