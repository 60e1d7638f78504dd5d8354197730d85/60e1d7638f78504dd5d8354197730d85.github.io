#include<stdio.h>
main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		int arr[n];
		int i,j;
		for(i=0;i<n;i++)
		scanf("%d",&arr[i]);
		int arr2[1001];
		for(i=0;i<1001;i++)
		arr2[i]=0;
		for(i=0;i<1001;i++)
		{
			for(j=0;j<n;j++)
			{
				if(arr[j]==i)
				arr2[i]++;
			}
		}
		
		int b=0,h=0;
		for(i=1000;i>=1;i--)
		{
			if(arr2[i]>=4&&b==0&&h==0)
			{
				b=i;
				h=i;
				break;
			}
			if(arr2[i]>=2&&b==0)
			{
				b=i;
			//	printf("Base gets value %d\n",i);
				continue;
				
			}
			if(arr2[i]>=2&&b!=0)
			{
				h=i;
			//	printf("height gets value %d\n",i);
				break;
			}
		}
		if(b*h!=0)
		printf("%d\n",b*h);
		else
		printf("%d\n",-1);
	}
}
