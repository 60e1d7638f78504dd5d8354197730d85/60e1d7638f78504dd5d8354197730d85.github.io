#include<stdio.h>
int main()
{
	int t,n,i,j,k,temp,l[2],area[100],s[1000];
	scanf("%d",&t);
	for(k=0;k<t;k++)
	{
		scanf("%d",&n);
		for(i=0;i<n;i++)
		scanf("%d",&s[i]);
		for(i=0;i<n-1;i++)
		{
			for(j=0;j<n-i-1;j++)
			{
				if(s[j]<s[j+1])
				{
					temp=s[j];
					s[j]=s[j+1];
					s[j+1]=temp;
				}
			}
		}
		 /*     for(i=0;i<n;i++)
			  printf("%d ",s[i]);*/
		j=0;
		for(i=0;i<n-1;i++)
		{
			if(s[i]==s[i+1])
			{
				l[j]=s[i];
				i++;j++;
			}
			if(j==2)
			break;
 
		}
 
		if(j==2)
		area[k]=l[0]*l[1];
		else
			area[k]=-1;
 
	}
	for(i=0;i<t;i++)
	printf("%d\n",area[i]);
	return 0;
}
