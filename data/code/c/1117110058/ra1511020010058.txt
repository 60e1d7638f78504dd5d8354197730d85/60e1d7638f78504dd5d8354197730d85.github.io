        #include<stdio.h>
         
        int main()
        {
        	int t;
        	scanf("%d",&t);
        	while(t--)
        	{
        		int n;
        		scanf("%d",&n);
        		int a[n],i,b[1010];
        		for(i=0;i<1010;i++)
        		{
        			b[i]=0;
        		}
        		for(i=0;i<n;i++)
        		{
        			scanf("%d",&a[i]);
        			b[a[i]]++;
        		}
        		int max=0,smax;
        		for(i=0;i<1010;i++)
        		{
        			if(b[i]>=4)
        			{
        				smax = i;
        				max = i;
        			}
        			else if(b[i]>=2)
        			{
        				smax = max;
        				max = i;
        			}
        		}
        		if((smax*max)>0)
        		{
        		    printf("%d\n",smax*max);
        		}
        		else
        		{
        		    printf("-1\n");
        		}
        	}
        	return 0;
        }
