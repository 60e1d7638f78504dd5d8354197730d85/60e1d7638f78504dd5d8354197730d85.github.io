#include <stdio.h>
 
int main(void) {
	int t, n, i, x, count;
	
	scanf("%d",&t);
	while (t--)
	{
	    scanf("%d",&n);
	    int arr[n];
	    count = 0;
	    for (i = 0; i < n; i++)
	    {
	        scanf("%d",&arr[i]);
	    }
	  scanf("%d",&x);
	    if (x <= arr[0])
	        count++;
	    for (i = 1; i < n; i++)
	    {
	        scanf("%d",&x);
	        if (x <= (arr[i] - arr[i-1]))
	            count++;
	    }
	    printf("%i\n", count);
	}
	return 0;
}
 
int input()
{
    int x = 0;
    char c = getchar_unlocked();
    while (c < '0' && c > '9')
        c = getchar_unlocked();
    while (c >= '0' && c <= '9')
    {
        x = (x<<1) + (x<<3) + c - '0';
        c = getchar_unlocked();
    }
    return x;
}
