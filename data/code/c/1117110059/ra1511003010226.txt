#include <stdio.h>
int main()
{
int n,t,i;
  long int a[10000],b[10000],temp;
  scanf("%d",&t);
  while(t--){
    scanf("%d",&n);
    a[0]=0;
    temp=0;
    for(i=1;i<=n;i++){
      scanf("%ld",&a[i]);
      a[i-1]=a[i]-a[i-1];
    }
    for(i=0;i<n;i++){
      scanf("%ld",&b[i]);
    }
    for(i=0;i<n;i++)
    {
      if(a[i]>=b[i])
        temp++;
    }
    printf("%ld\n",temp);
  }
      
	return 0;
}
