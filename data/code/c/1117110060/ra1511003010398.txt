#include<stdio.h>
int main()
{
    int t,n,i,count;
    scanf("%d",&t);
    while(t--)
    {
        count=0;
        scanf("%d",&n);
        int a[n],p[100];
        for(i=0;i<n;i++)
        scanf("%d",&a[i]);
        for(i=1;i<=100;i++)
        p[i]=0;
        for(i=0;i<n;i++)
        p[a[i]]++;
        for(i=1;i<=100;i++)
        {
            if(p[i]>=1)
            count++;
        }
        printf("%d\n",count);
    }
}
