#include<stdio.h>
int main()
{
    int t,n,input,i;
    scanf("%d",&t);
    while(t--)
    {
        int cnt=0;
        int arr[100]={0};
        scanf("%d",&n);
        while(n--)
        {
            scanf("%d",&input);
            arr[input-1]++;
        }
        for(i=0;i<100;i++)
            if(arr[i])
                cnt++;
        printf("%d\n",cnt);
    }
  return 0;
}
