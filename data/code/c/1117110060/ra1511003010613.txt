#include<stdio.h>
typedef long long int ll;
int main()
{
 ll t,n,a,i,b[100];
 scanf("%lld",&t);
 while(t--)
 {
  ll count=0;
  for(i=1;i<101;i++)
  b[i]=0;
  scanf("%lld",&n);
  for(i=0;i<n;i++)
  {
   scanf("%lld",&a);
   b[a]++;
  }
  for(i=1;i<101;i++)
  {
   if(b[i]>0)
   count++;
  }
  printf("%lld\n",count);
 }
 return 0;
}
