#include <stdio.h>
int main()
{
 int h,y,j,i,t;
  scanf("%d",&t);
  while(t--)
  {
    int n;
    scanf("%d",&n);
    int a[n];
    int s=0;
    for(i=0;i<n;i++)
    {
      scanf("%d",&a[i]);
    }
    for(j=0;j<n;j++)
    {
      int k=0;
      for(h=0;h<n;h++)
      {
        if(j!=h)
        {
          for(y=0;y<j;y++)
          {
            if(a[j]==a[y])
            {
              k++;
            }
          }
          if(k==0)
          {
            if(a[j]==a[h])
            {
              s++;
            }
          }
          else
          {
            break;
          }
        }
      }
    }
    printf("%d\n",n-s);
  }
  return 0;
}
