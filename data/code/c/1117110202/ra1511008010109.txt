#include <stdio.h>
#include<math.h>
int power(int x, unsigned int y)
{
    if (y == 0)
        return 1;
    else if (y%2 == 0)
        return power(x, y/2)*power(x, y/2);
    else
        return x*power(x, y/2)*power(x, y/2);
}
int main()
{
   int a,b;
   scanf("%d %d",&a,&b);
  printf("The value of %d power %d is %d",a,b,power(a,b));
	return 0;
}
