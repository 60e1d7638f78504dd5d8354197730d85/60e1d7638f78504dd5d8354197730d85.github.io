#include<stdio.h>

int main(){

    int num,i,count,min,max;

   
     scanf("%d",&min);

    
    scanf("%d",&max);

    for(num = min;num<=max;num++){

         count = 0;

         for(i=2;i<=num/2;i++){
             if(num%i==0){
                 count++;
                 break;
             }
        }
        
         if(count==0 && num!= 1)
             printf("%d\n",num);
    }
  
   return 0;
}
