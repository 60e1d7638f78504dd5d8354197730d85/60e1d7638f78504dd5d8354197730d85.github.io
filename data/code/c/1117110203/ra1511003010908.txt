int checkprimenumber(int n);
int main()
{int n1,n2,i,flag;

 { flag =checkprimenumber(i);

 }

}
int checkprimenumber(int n)
{ int j,flag=1;
 for( j=2;j<=n/2;++j)
 {if (n%j==0)
 {flag=0;
  break;}
 }

 }
