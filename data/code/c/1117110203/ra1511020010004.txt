int main()
{
    int lowerLimit, upperLimit;

    printPrimes(lowerLimit, upperLimit);
    
    return 0;
}
void printPrimes(int lowerLimit, int upperLimit)
{
    
    while(lowerLimit <= upperLimit)
    {
        if(isPrime(lowerLimit))
        {
            printf("%d\n", lowerLimit);
        }
        
        lowerLimit++;
    }
}
int isPrime(int num)   
{  
    int i;     
    for(i=2; i<=num/2; i++)    
    {       
        if(num % i == 0)    
        {  
            return 0;  
        }    
    }    
    return 1;   
}
