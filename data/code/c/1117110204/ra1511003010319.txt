#include<stdio.h>
#include<stdbool.h>

/* Function to check if x is power of 2*/
bool isPowerOfTwo(int n)
{
  if (n == 0)
    return 0;
  while (n != 1)
  {
      if (n%2 != 0)
         return 0;
      n = n/2;
  }
  return 1;
}

/*Driver program to test above function*/
int main()
{int n,a[100],i;
 scanf("%d",&n);
 for(i=0;i<n;i++)
   scanf("%d",&a[i]);
 for(i=0;i<n;i++)
  isPowerOfTwo(a[i])? printf("1\n"): printf("0\n");
  
  return 0;
}
