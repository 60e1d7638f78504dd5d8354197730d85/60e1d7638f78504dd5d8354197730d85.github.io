#include <stdio.h>
#include<stdio.h>
#define bool int
bool isPowerOfTwo(int n)
{
  if (n == 0)
    return 0;
  while (n != 1)
  {
    if (n%2 != 0)
      return 0;
    n = n/2;
  }
  return 1;
}
 
/*Driver program to test above function*/
int main()
{
  int t, num;
  scanf("%d", &t);
  while( t-- >0)
  {
    scanf("%d", &num);
  	isPowerOfTwo(num)? printf("1\n"): printf("0\n");
  }
  return 0;
}
