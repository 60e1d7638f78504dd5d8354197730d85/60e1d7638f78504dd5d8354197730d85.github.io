#include <stdio.h>
int fibonacci(int term);
int main()
{
  int terms, counter;
      scanf("%d", &terms);
      
    for(counter = 0; counter < terms; counter++){
        printf("%d\n", fibonacci(counter));
    }

	return 0;
}
int fibonacci(int term){
    
    if(term < 2)
       return term;
    return fibonacci(term - 1) + fibonacci(term - 2);
}
