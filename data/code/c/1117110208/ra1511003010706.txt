#include<stdio.h>
#include<string.h>
 
char A[1000][1001] = {0};
 
unsigned int ans[1001]={0};
 
char str[1001];
 
int main()
{
scanf("%s",str);
int n = strlen(str);
 
int i,len=1;
 
ans[0] = 1;
for(i=0;i<n;i++)
{
for(len=1;len<=i+1;len++)
{
 
if(str[i] == str[i+1-len] && (len<=2 || A[i-1][len-2]))
{
A[i][len] = 1;
ans[i+1] = (ans[i+1] + ans[i+1-len]) % 1000000007;
}
}
}
 
printf("%d\n",ans[n]);
 
return 0;
}
