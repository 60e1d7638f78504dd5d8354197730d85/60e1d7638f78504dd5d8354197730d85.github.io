#include<stdio.h>
 
int calculateSum(int);
 
int main()
{
   int i, num;
   int result;

   scanf("%d", &num);
 
   result = calculateSum(num);
   printf("%d", result);
 
   return (0);
}
 
int calculateSum(int num) {
   int res;
   if (num == 1) {
      return (1);
   } else {
      res = num + calculateSum(num - 1);
   }
   return (res);
}
