#include<stdio.h>
#include<string.h>
char str[105];  
int num[105];  
int main()  
{  
    int n;  
    while(~scanf("%d",&n))  
    {  
        scanf("%s",str);  
        int len = strlen(str);  
        memset(num,0,sizeof(num));  
        int s = 0,sum = 0,i;  
        if(str[0] == 'B')  
            sum = 1;  
        if(len > 1)  
        for( i = 1;i < len;i++)  
        {  
            if(str[i] == 'B')  
                sum++;  
            else  
            {  
                if(sum != 0)  
                    num[s++] = sum;  
                sum = 0;  
            }  
        }  
        if(sum != 0)  
            num[s++] = sum;  
        if(s == 0)  
            printf("0\n");  
        else if(s == 1)  
            printf("%d\n%d\n",s,num[0]);  
        else  
        {  
            printf("%d\n%d",s,num[0]);  
            for( i = 1;i < s;i++)  
                printf(" %d",num[i]);  
            printf("\n");  
        }  
  
    } 
    return 0; 
}
