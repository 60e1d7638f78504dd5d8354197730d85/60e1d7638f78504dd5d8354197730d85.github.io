#include <iostream>
using namespace std;
char s[140];
int main()
{
    int n;
    scanf("%d",&n);
    scanf("%s",s);
    int ans=0;
    int ge=0;
    int a[140]={0};
    for(int i=0;i<n;i++)
    {
        if(s[i]=='B')
        {
            a[ge]++;
            if(s[i+1]!='B')
            {
                ans++;
                ge++;
            }
        }
    }
    printf("%d\n",ans);
    for(int i=0;i<ge;i++)
    {
        if(i==0)
            cout<<a[i];
        else
            cout<<" "<<a[i];
    }
    cout<<endl;

	return 0;
}
