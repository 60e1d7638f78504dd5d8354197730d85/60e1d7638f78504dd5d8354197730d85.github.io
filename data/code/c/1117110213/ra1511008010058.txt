#include <stdio.h>
char s[140];
int main()
{
    int n;
    scanf("%d",&n);
    scanf("%s",s);
    int ans=0;
    int ge=0,i;
    int a[140]={0};
    for(i=0;i<n;i++)
    {
        if(s[i]=='B')
        {
            a[ge]++;
            if(s[i+1]!='B')
            {
                ans++;
                ge++;
            }
        }
    }
    printf("%d\n",ans);
    for(i=0;i<ge;i++)
    {
        if(i==0)
            printf("%d",a[i]);
        else
            printf(" %d",a[i]);
    }
 return 0;
}
