#include <stdio.h>
int main()
{
    int a,n=-1,b,c;  
    scanf("%d",&a);  
    if(a%2==1&&a>1){  
        n=(a-1)/2;  
        b=2*n*(n+1);  
        c=2*n*(n+1)+1;  
    }  
    else if(a%2==0&&a>2){  
        n=a/2;  
        b=n*n-1;  
        c=n*n+1;  
    }  
    if(n==-1)
      printf("-1\n");  
    else
        printf("%d %d",b,c);
	return 0;
}
