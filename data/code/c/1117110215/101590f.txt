#include<stdio.h>
int convertEC(int num)
{
  if(num==0)
     return 0;
  int digit=num%10;
  if(digit==0)
    digit=5;
  return convertEC(num/10)*10+digit;
}
int convert(int num)
{
  if(num==0)
    return 5;
  else return convertEC(num);
}
int main()
{
  int num;
  scanf("%d",&num);
  printf("%d",convert(num));
  return 0;
}
