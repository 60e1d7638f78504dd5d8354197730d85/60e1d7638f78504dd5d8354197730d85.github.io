#include<stdio.h>
int convert0To5Rec(int num)
{
    if (num == 0)
        return 0;
    int digit = num % 10;
    if (digit == 0)
        digit = 5;
    return convert0To5Rec(num/10) * 10 + digit;
}
 
int convert0To5(int num)
{
    if (num == 0)
       return 5;
    else return  convert0To5Rec(num);
}
 

int main()
{
    int num,c;
  scanf("%d",&num);
  c=convert0To5(num);
    printf("%d",c);
    return 0;
}
