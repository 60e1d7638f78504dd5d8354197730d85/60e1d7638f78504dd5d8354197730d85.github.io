#include <stdio.h>
int convertFive(int n)
{
    int k,r ;
    k=0 ;
   
    while(n!=0)
    {
        r=n%10 ;
        if(r!=0)
            k=k*10+r ;
        else
            k=k*10+5 ;
        n=n/10 ;
    }
   
    while(k!=0)
    {
        r=k%10 ;
        n=n*10+r ;
        k=k/10 ;
    }
    return n ;
}
int main()
{
	int n,n1;
  	scanf("%d",&n);
  	n1=convertFive(n);
  	printf("%d",n1);
  
	return 0;
}
