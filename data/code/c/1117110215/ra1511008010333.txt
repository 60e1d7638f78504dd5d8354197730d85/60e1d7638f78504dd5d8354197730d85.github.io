#include<stdio.h>
 
int convertTo5Rec(int num)
{ 
    
    if (num == 0)
        return 0;
    int digit = num % 10;
    if (digit == 0)
        digit = 5;
    return convertTo5Rec(num/10) * 10 + digit;
}
int convertTo5(int num)
{
    if (num == 0)
       return 5;
    else return  convertTo5Rec(num);
}
int main()
{
    
  int num;
  scanf("%d",&num);
    printf("%d", convertTo5(num));
    return 0;
}
