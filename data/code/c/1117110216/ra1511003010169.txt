
#include<stdio.h>

int findinversion(int a[],int n)
{
 int i,j,count=0;

 for(i=0;i<n;i++)
 {
  for(j=i+1;j<n;j++)
       {
   if(a[i]>a[j])
   count++;
    }
 }
 return count;
}

int main()
{
 int n,i,a[100];
 scanf("%d",&n);
 for(i=0;i<n;i++)
  scanf("%d",&a[i]);
 printf("%d",findinversion(a,n));
return 0;
}
