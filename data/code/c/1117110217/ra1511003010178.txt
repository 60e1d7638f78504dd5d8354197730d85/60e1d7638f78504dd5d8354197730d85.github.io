    #include<stdio.h>
    #define g getchar_unlocked
    int read()
     
    {
        int n=0;
        char ch = g();
        while (ch<'0' || ch>'9' )
        {
            ch = g();
        }
        while(ch>='0' && ch<='9')
        {
            n = (n<<3) + (n<<1) + ch -'0';
            ch=g();
        }
        return n;
    }
