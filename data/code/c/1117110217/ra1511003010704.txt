#include <stdio.h>
long long mod=1000000007;
int main()
{
long long n,m,k;
  int i;
  scanf("%lld %lld %lld",&n,&m,&k);
  long long a[n][2];
  for(i=0;i<n;i++)
    scanf("%lld",&a[i][0]);
  long long sum=0;
  long long max1=0,max2=0;
  for(i=0;i<n;i++)
  {
    a[i][1]=(m-a[i][0])/k;
    a[i][1]++;
    sum+=a[i][1];
    if(a[i][1]>max2 && a[i][1]<=max1)
    {
      max2=a[i][1];}
    else if(a[i][1]>max1)
    {
      max2=max1;
      max1=a[i][1];}}
  sum%=2;
  sum=(max1+2-sum)/2;
  sum%=mod;
  printf("%lld\n",sum);
	return 0;
}
