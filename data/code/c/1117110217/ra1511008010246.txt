#include <stdio.h>
int main(){
	long int A[1000000], max=0,max2=0,i=0,N;
	scanf("%ld", &N);
	if(N>1){
		while(i<N){
			scanf("%ld", &A[i]);
			if(max< A[i])
				max=A[i];
			i++;
		}
		for(i=0; i<N; i++){
			if(A[i]>max2 && A[i]!=max)
				max2=A[i];
		}
		printf("%ld\n", max2%max);
	}
	else{
		scanf("%ld", &A[i]);
		printf("%d\n", 0);
	}
	return 0;
}
