#include <stdio.h>
 
#define MOD 1000000007
int arr[1000000], temp1[1000000], temp2[1000000];
 
int main () {
	int turn, max_fuel;
	scanf("%d %d", &turn, &max_fuel);
 
	int i, j, k;
	long long count;
 
	for(i=0; i<turn; i++)
		scanf("%d", &arr[i]);
 
	temp2[0] = 1;
	j = k = count = 0;
	
	for(i=1; i<turn; i++) {
		while(arr[i] - arr[j] > max_fuel)
			count -= temp2[j++];
		temp1[i] = temp1[j] + 1;
		while(temp1[k] == temp1[j])
			count += temp2[k++];
		temp2[i] = count % MOD;
	}
	printf("%d %d\n", temp1[turn-1]-1, temp2[turn-1]);
	return 0;
}
