#include <stdio.h>
#include <math.h>
 
int arr[20];
short dp[ 1<<20 ];
char dp2[ 1<<20 ];
int m, n;
 
int main(void) {
	
	int i, j, k, a, b;
	scanf("%d %d", &n, &m);
	
	for(i = 0; i < m ; i++)
	{
		scanf("%d %d", &a, &b);
		arr[a-1] |= (1 << (b-1));
		arr[b-1] |= (1 << (a-1));
	}
	
	dp[0] = 0;
	for(i = 1; i < (1<<n); i++)
		dp[i] = 1e4;
	
	for(i = 0; i < (1<<n); i++)
	{
		
		char cnt = 0;
		for(j = 0; j < n; j++)
		{
			if( (1<<j) & i )
			{
				int res = arr[j] & ~i;
				for(k = 0; k < n; k++)
					if( res & (1<<k) )
						cnt++;
			}
		}
		dp2[i] = cnt;
 
		for(j = 0; j < n; j++)
		{
			if( (1<<j) & i)
				continue;
			int res = dp[i] + dp2[i];
			if( res < dp[ i|(1<<j)])
				dp[ i|(1<<j)] = res;
		}
	}
	printf("%d\n", dp[ (1<<n)-1 ]);
	
	return 0;
}
