#include<stdio.h>
 
#define INF 1000000000
 
int adj[20], dp[1 << 20], bit_counts[1 << 20];
 
int min(int, int);
 
int main()
{
	int a, b, i, j, limit, M, N;
	scanf("%d %d", &N, &M);
	limit = 1 << N;
	bit_counts[0] = 0;
	for (i = 1; i < limit; i++)
		bit_counts[i] = bit_counts[i >> 1] + (i & 1);
	for (i = 0; i < M; i++)
	{
		scanf("%d %d", &a, &b);
		a--;
		b--;
		adj[a] |= 1 << b;
		adj[b] |= 1 << a;
	}
	for (i = 0; i < limit; i++)
		dp[i] = INF;
	dp[0] = 0;
	for (i = 0; i < limit; i++)
		for (j = 0; j < N; j++)
			if (!(i & (1 << j)))
				dp[i | (1 << j)] = min(dp[i | (1 << j)],
				dp[i] + (bit_counts[adj[j] & i] - (bit_counts[adj[j]] - bit_counts[adj[j] & i])) * bit_counts[i]);
	printf("%d\n", dp[limit - 1]);
	return 0;
}
 
int min(int a, int b)
{
	return (a < b) ? a : b;
}
