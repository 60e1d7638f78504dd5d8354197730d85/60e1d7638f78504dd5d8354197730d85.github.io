#include <stdio.h>
int main()
{
  int i,j,c,t,a[100],n;
  scanf("%d",&t);
  while(t--)
  {
    c = 0;
    scanf("%d",&n);
    for(i=0;i<n;i++)
      scanf("%d",&a[i]);
    for(i=0;i<n;i++)
    {
      for(j=i+1;j<n;j++)
      {
        if(a[i] == a[j])
        {
          c++;
        }
      }
    }
    printf("%d\n",c);
  }
  return 0;
}
