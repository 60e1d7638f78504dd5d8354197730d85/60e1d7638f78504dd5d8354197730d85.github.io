#include<stdio.h>

int main()
{
   int m, n, c, d, matrix[10][10], maximum;
 
   scanf("%d\n%d",&m,&n);
 printf("The matrix is\n");
   for( c = 0 ; c < m ; c++ )
   {
      for( d = 0 ; d < n ; d++ )
      {
         scanf("%d",&matrix[c][d]);
        printf("%d ",matrix[c][d]);
        
      }
     printf("\n");
   }
   maximum = matrix[0][0];
 
   for( c = 0 ; c < m ; c++ )
   {
      for( d = 0 ; d < n ; d++ )
      {
         if ( matrix[c][d] > maximum )
            maximum = matrix[c][d];
      }
   }
 
   printf("The maximum element in the matrix is:%d\n", maximum);
 
   return 0;
}
