int main()
{
	int cases,n,m,k,i,j;
	int col[1001],cost[1001][1001],points[1001][1001];
	scanf("%d",&cases);
	while(cases--)
	{
		
		scanf("%d %d %d",&n,&m,&k);
		int max[n+1],pts[n+1],tot,temp,x;tot=0;
		for(i=1;i<=n;i++)
			max[i]=0;
		for(i=1;i<=n;i++)
			scanf("%d",&col[i]);
		for(i=1;i<=n;i++)
			for(j=1;j<=m;j++)
				scanf("%d",&points[i][j]);
		//printf("\n 1st matrix taken");
		for(i=1;i<=n;i++)
			for(j=1;j<=m;j++)
				scanf("%d",&cost[i][j]);
		//printf("\n 2nd matrix taken");
		for(i=1;i<=n;i++)
			pts[i]=points[i][col[i]];
				
		
		for(i=1;i<=n;i++)
		{
			//printf("max analysis row %d\n",i);
			for(j=1;j<=m;j++)
			{
				//printf("i=%d,j=%d\n",i,j);
				//printf("x=%d-%d\n",points[i][j],cost[i][j]);
				//printf("x=%d\n",x);
				x=(points[i][j]-cost[i][j]);
				if(max[i]<(x-pts[i]))
					max[i]=x-pts[i];
				
				//printf("max=%d\n",max[i]);
			}
			
		}
		//for(i=1;i<=n;i++)
				//printf("%d ",pts[i]);
		//for(i=1;i<=n;i++)
				//printf("%d ",max[i]);
		if(k>n)
			k=n;
		for(i=1;i<=k;i++)
		{
			for(j=i+1;j<=n;j++)
			{
				if(max[j]>max[i])
				{
					temp=max[i];
					max[i]=max[j];
					max[j]=temp;
					temp=pts[i];
					pts[i]=pts[j];
					pts[j]=temp;
				}	
			}
			pts[i]+=max[i];
		}
			//for(i=1;i<=n;i++)
				//printf("%d ",pts[i]);
			//for(i=1;i<=n;i++)
				//printf("%d ",max[i]);
			for(i=1;i<=n;i++)
				tot+=pts[i];
				
		
		printf("%d\n",tot);
		
	
	
	}
 
	return 0;
 
 
 
}
