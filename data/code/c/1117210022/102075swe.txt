#include <stdio.h>
#include <stdlib.h>
int finull(int **a,int m,int n)
{
  int *row,i,j;
  for(i=0;i<m;i++)
  {
    row=a[i];
    for(j=0;j<n;j++)
      if(row[j]!=0)
        return 0;
  }
  return 1;
}

int main()
{
  int **a,i,j,m,n;
  scanf("%d",&m);
  a=(int **)malloc(sizeof(int*)*m);
  scanf("%d",&n);
  for(i=0;i<m;i++)
  {
    a[i]=(int *)malloc(sizeof(int)*n);
    for(j=0;j<n;j++)
      scanf("%d",a[i]+j);
  }
  for(i=0;i<m;i++)
  {
    
    for(j=0;j<n;j++)
      printf("%d ",a[i][j]);
    printf("\n");
  }
  if(finull(a,m,n))
    printf("The matrix is null");
  else
    printf("The matrix is not null");
  return 0;
}
