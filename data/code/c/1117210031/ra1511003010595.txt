#include <stdio.h>
 
inline int read()
{
	char c;
	int n = 0;
	while ((c = getchar_unlocked()) < 48)
		;
	n += (c - '0');
	while ((c = getchar_unlocked()) >= 48)
		n = n * 10 + (c - '0');
	return n;
}
 
int main()
{
	int t;
	t=read();
	while(t--)
	{
		int n,x=0,y=0,i,b=0;
		n=read();
		for(i=0;i<n;i++)
		{
			b=read();
			x+=b;
			y+=b;
			if(b!=0)
				y--;
		}
		if(y<100&&x>=100)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
