#include<stdio.h>
typedef long long ll;
const ll MOD = 1000000007;
ll mini(ll a,ll b){
	return a<b?a:b;
}
ll maxi(ll a,ll b){
	return a>b?a:b;
}
int main(){
	ll i,j,n,t,a[11],x,y,z;
	scanf("%lld",&t);
	while(t--){
		scanf("%lld",&n);
		for(i=0;i<n;i++) scanf("%lld",&a[i]);
		ll min=a[0],max=a[0];
		for(i=1;i<n;i++){
			ll lmax=max,lmin=min;
			x=min+a[i];
			y=min-a[i];
			z=min*a[i];
			lmin=mini(x,mini(y,z));
			lmax=maxi(x,maxi(y,z));
			x=max+a[i];
			y=max-a[i];
			z=max*a[i];
			lmin=mini(lmin,mini(x,mini(y,z)));
			lmax=maxi(lmax,maxi(x,maxi(y,z)));
			min=lmin;
			max=lmax;
		}
		printf("%lld\n",min);
	}
	return 0;
}
