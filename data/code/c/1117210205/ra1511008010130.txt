#include <stdio.h>

/* Function declaration */
int gcd(int a, int b);


int main()
{
    int num1, num2, hcf;
    
   
    scanf("%d%d", &num1, &num2);
    
    hcf = gcd(num1, num2);
    
    printf("G.C.D of %d and %d is %d.", num1, num2, hcf);
    
    return 0;
}

int gcd(int a, int b)
{
    if(b == 0)
        return a;
    else
        return gcd(b, a%b); 
}
