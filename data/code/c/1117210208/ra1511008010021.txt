#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main()
{
    char str[80], word[80];
    fgets(str, 80, stdin);

    char *token;

    //tokenizing array str
    token = strtok(str, " ");


    while( token != NULL )
    {
        if(strlen(word) < strlen(token) )
        {
            strcpy(word, token);
        }

        token = strtok(NULL, " ");

    }
  int len=strlen(word);

    printf("%d",len);




    return 0;

}
