#include <stdio.h>
void prime(int a,int b)
{
  int i,j,flag;
  while(a<=b)
  {
    flag=0;
    for(i=2;i<=a/2;i++)
    {
      if(a%i==0)
      {
        flag=1;
        break;
      }
    }
    if(flag==0)
    {
      printf("%d ",a);
    }
    ++a;
  }
}
int main()
{
  int a,b;
  scanf("%d\n%d",&a,&b);
  prime(a,b);
return 0;
}
