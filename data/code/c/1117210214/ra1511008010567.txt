int main()

 //  printf("Enter two numbers(intervals): ");

   for(i=n1;i<=n2;++i)
 
   {
 
      flag=check_prime(i);

   }

}
 
int check_prime(int num) /* User-defined function to check prime number*/
 
{
 
   int j,flag=0;
 
   for(j=2;j<=num/2;++j)
 
{
 
        if(num%j==0)
 
{
 
            flag=1;
 
            break;
 
        }
 
   }
 
   return flag;
 
}
