#include <stdio.h>
int gcd(int a,int b)
{
  if(b%a==0)
    return a;
  else
    return gcd(b%a,a);
}

int main()
{
  int t;
  scanf("%d",&t);
  while(t--)
  {
    int m,n,g;
    scanf("%d %d",&n,&m);
    if(m==0)
    {
      if(n==1)
        printf("Yes\n");
      else
        printf("No 1\n");
      continue;
    }
    g=gcd(m,n);
    if(g==1){
      printf("Yes\n");
      continue;
    }
    printf("No %d\n",n/g);
  }
               

	return 0;
}
