#include<stdio.h>
int main()
{
    int tm[200001],ride[200001],m,n,p,q,r,k,i,t,count,total;
    scanf("%d",&count);
    while(count!=0)
    {
        total=0;
        scanf("%d%d%d%d%d%d",&n,&m,&q,&p,&r,&k);
 
        for(i=0;i<=200000;i++)
           {
               tm[i]=0;
               ride[i]=0;
           }
 
        for(t=0;t<=k;t++)
        {
 
           // printf("n=%d q=%d\n",n,q);
 
            n=n+tm[t];
             total+=ride[t];
            if((t%r==0)&&t>0&&m>=1)
            {q+=1;m=m-1;}
 
            if(n>=q)
            {
                tm[t+p]=q;
                ride[t+p]=q;
                n=n-q;
                q=0;
               // printf("in if n=%d\n",n);
 
            }
 
            else
            {
                tm[t+p]=n;
                q=q-n;
                ride[t+p]=n;
                n=0;
           // printf("in else n=%d\n",n);
            }
 
 
        }
 
        printf("%d %d %d %d\n",n,total,m,q);
        count--;
 
 
    }
    return 0;
}
