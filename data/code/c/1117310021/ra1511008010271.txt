#include <stdio.h>
 
int main(void) {
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,i,x=0;
    	scanf("%d",&n);
    	int a[n];
    	for(i=0;i<n;i++)
    	{scanf("%d",&a[i]);
    	x=x^a[i];}
    	if(x==0)printf("First\n");
    	else if(n%2==0)
    	printf("First\n");
    	else printf("Second\n");
    }
	return 0;
}
