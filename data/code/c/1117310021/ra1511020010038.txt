#include<stdio.h>
int h[505];
int main()
{
 
int t,n,val,count,i,check=0;
scanf("%d",&t);
while(t--)
{
scanf("%d",&n);
for(i=1;i<=500;i++)
h[i]=0;
check=0;
for(i=1;i<=n;i++)
{
 
scanf("%d",&val);
check=check^val;
 
h[val]++;
}
 
if(check==0)
{
    printf("First\n");
    continue;
}
count=0;
for(i=1;i<=500;i++)
{
if(h[i]%2!=0)
count++;
}
 
if(count%2!=0)
printf("Second\n");
else
printf("First\n");
 
}
 
return 0;
}
