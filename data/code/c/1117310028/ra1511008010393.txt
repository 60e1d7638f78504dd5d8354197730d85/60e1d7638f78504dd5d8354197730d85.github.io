#include <stdio.h>
#include <stdlib.h>

int main()
{
 int t,n,i,j,ones=0,p,*arr;
  long long sum=0;
  scanf("%d\n",&t);
  while(t--)
  {
    scanf("%d\n",&n);
    arr=(int*)calloc((n+1),sizeof(int));
    arr[0]=0;
    for(i=1;i<=n;i++)
    {
      scanf("%d",&arr[i]);
      arr[i]=arr[i]^arr[i-1];
    }
    for(i=0,p=1;i<32;i++,p<<=1)
    {
      ones=0;
      for(j=0;j<=n;j++)
      {
        if(arr[j]&p)
          ones++;
      }
      sum=sum+((long long)(ones)*(n+1-ones)*p);
      
    }
    printf("%lld\n",sum);
    sum=0;
    free(arr);
  }
	return 0;
}
