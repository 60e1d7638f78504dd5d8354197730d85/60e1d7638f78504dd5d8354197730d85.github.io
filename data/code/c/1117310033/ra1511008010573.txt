#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <ctype.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    { char e[400];
      char a[400];
      char b[400];
      scanf("%s",e);
      int i,j=0,k=0;
      for(i=0;i<strlen(e);i++)
      {
          if(e[i]=='+'||e[i]=='-'||e[i]=='*'||e[i]=='/'||e[i]=='^'||e[i]=='%')
          {b[k++]=e[i];}
          
          else if(isalpha(e[i]))
          {a[j++]=e[i];}
          
          else if(e[i]==')')
          {char c=b[--k];
           a[j++]=c;}
           
      }
      a[j]='\0';
      b[k]='\0';
      
      
      puts(a);
          
          
      }
        
    
    return 0;
}
