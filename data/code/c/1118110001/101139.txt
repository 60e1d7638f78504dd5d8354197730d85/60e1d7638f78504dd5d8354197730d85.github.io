#include <stdio.h>
#include<string.h>
int main()
{
    struct College
    {
      char n[100];
      char ci[100];
      int y;
      float p;
    }c[20];
    int n,i,t,j;
    float t1;
    char temp[20],temp1[20];
    scanf("%d",&n);
    for(i=0;i<n;i++)
      scanf("%s %s %d %f",c[i].n,c[i].ci,&c[i].y,&c[i].p);
  for(i=0;i<n;i++)
  {
      for(j=i+1;j<n;j++)
      {
           if(strcmp(c[i].n,c[j].n)>0)
           {
               strcpy(temp,c[i].n);
               strcpy(c[i].n,c[j].n);
               strcpy(c[j].n,temp);
               strcpy(temp1,c[i].ci);
               strcpy(c[i].ci,c[j].ci);
               strcpy(c[j].ci,temp1);
               t=c[i].y;
               c[i].y=c[j].y;
               c[j].y=t;
               t1=c[i].p;
               c[i].p=c[j].p;
               c[j].p=t1;
           }
      }
  }
  for(i=0;i<n;i++)
  {
      printf("Name:%s\n",c[i].n);
      printf("Department:%s\n",c[i].ci);
      printf("Year of study:%d\n",c[i].y);
      printf("CGPA:%.1f\n",c[i].p);
  }
     
                
   
	return 0;
}
