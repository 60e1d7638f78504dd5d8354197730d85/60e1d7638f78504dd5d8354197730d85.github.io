#include <stdio.h>
struct Student
{
char name[100];
char city[100];
int establishmentYear;
float passPercentage;
};
int main()
{
  int n,i,j;
  scanf("%d",&n);
  struct Student S1[n],temp;
  for(i=0;i<n;i++)
    scanf("%s %s %d %f",S1[i].name,S1[i].city,&S1[i].establishmentYear,&S1[i].passPercentage);
  for(i=0;i<n;i++)
  {
    for(j=i;j<n;j++)
    {
      if(S1[j].name[0]<S1[i].name[0])
      {
        temp=S1[j];
        S1[j]=S1[i];
        S1[i]=temp;
      }
    }
  }
  
  for(i=0;i<n;i++)
    printf("Name:%s\nDepartment:%s\nYear of study:%d\nCGPA:%.1f\n",S1[i].name,S1[i].city,S1[i].establishmentYear,S1[i].passPercentage);
	return 0;
}
