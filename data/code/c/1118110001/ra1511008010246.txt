#include <stdio.h>
#include<string.h>
struct Student
{
char name[30];
char department[20];
int yearOfStudy;
float cgpa;
};
int main() 
{
    int n;
    scanf("%d",&n);
    struct Student S1[n],tmp;
    int i=0;
    int j=0;
    for(i=0;i<n;i++)
    {
        scanf("%s %s %d %f",S1[i].name,S1[i].department,&S1[i].yearOfStudy,&S1[i].cgpa);
    }
    
    for (i = 1; i < n; i++)
    {
      for (j = 0; j < n - i; j++) 
      {
         if (strcmp(S1[j].name, S1[j + 1].name) > 0)
         {
            tmp = S1[j];
            S1[j] = S1[j + 1];
            S1[j + 1] = tmp;
         }
        }
    }
    for(i=0;i<n;i++)
    {
        printf("Name:%s\n",S1[i].name);
        printf("Department:%s\n",S1[i].department);
        printf("Year of study:%d\n",S1[i].yearOfStudy);
        printf("CGPA:%.1f\n",S1[i].cgpa);
    }
    return 0;
}
