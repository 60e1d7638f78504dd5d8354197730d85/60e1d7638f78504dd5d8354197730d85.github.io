#include <stdio.h>
#include <string.h>
struct Student
{
  char name[30];
  char dept[20];
  int year;
  float cgpa;
};
int main()
{
  struct Student s[10],temp;
  int n,i,j;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
	scanf("%s",s[i].name);
    scanf("%s",s[i].dept);
    scanf("%d",&s[i].year);
    scanf("%f",&s[i].cgpa);
  }
  for(i=0;i<n;i++)
      for(j=i+1;j<n;j++){
         if(strcmp(s[i].name,s[j].name)>0)
         {
            temp=s[i];
            s[i]=s[j];
            s[j]=temp;
         }
      }
  for(i=0;i<n;i++)
  {
    printf("Name:%s\n",s[i].name);
    printf("Department:%s\n",s[i].dept);
    printf("Year of study:%d\n",s[i].year);
    printf("CGPA:%.1f\n",s[i].cgpa);
  }
      return 0;
}
