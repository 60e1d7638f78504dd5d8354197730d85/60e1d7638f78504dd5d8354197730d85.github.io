#include <stdio.h>
#include <string.h>
#include<stdlib.h>

struct Student
{
char name[100];
char department[100];
int yearOfStudy;
float cgpa;
}S1[100], temp;

int main()
{
	int n, i, j;
  scanf("%d", &n);
  for(i=0;i<n;i++)
  {
    scanf("%s", S1[i].name);
    scanf("%s", S1[i].department);
    scanf("%d", &S1[i].yearOfStudy);
    scanf("%f",  &S1[i].cgpa);
  }
  
 for (i = 1; i < n; i++)
 {    for (j = 0; j < n - i; j++) 
      {
         if (strcmp(S1[j].name, S1[j + 1].name) > 0) 
         {
            temp = S1[j];
            S1[j] = S1[j + 1];
            S1[j + 1] = temp;
         }
      }
 }
  

  
  for(i=0;i<n;i++)
  {
    printf("Name:%s\n", S1[i].name);
    printf("Department:%s\n", S1[i].department);
    printf("Year of study:%d\n", S1[i].yearOfStudy);
    printf("CGPA:%.1f\n", S1[i].cgpa);
  }
	return 0;
}
