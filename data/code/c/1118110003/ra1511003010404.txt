#include <stdio.h>
struct Year
{
   int date,month,year;
};
int main()
{
	struct Year s1;  
  	scanf("%d %d %d",&s1.date,&s1.month,&s1.year);
  	printf("Date=%d\nMonth=%d\nYear=%d\n",s1.date,s1.month,s1.year);
  	if ((s1.year % 400) == 0)
        printf("%d is a leap year\n", s1.year);
    else if ((s1.year % 100) == 0)
        printf("%d is a not leap year\n", s1.year);
    else if ((s1.year % 4) == 0)
        printf("%d is a leap year\n", s1.year);
    else
        printf("%d is not a leap year\n", s1.year);
	return 0;
}
