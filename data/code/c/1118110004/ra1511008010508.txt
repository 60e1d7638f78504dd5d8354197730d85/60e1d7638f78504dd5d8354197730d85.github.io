#include <stdio.h>
struct date{
	int day;
  	int month;
  	int year;
}d1;

int main()
{
	struct date d1;
  	scanf("%d %d %d",&d1.day,&d1.month,&d1.year);
  	if(d1.day<1||d1.day>31)
    {
      printf("Invalid Day");
      return 0;
    }
  	if(d1.month>12||d1.month<1)
    {
      printf("Invalid Month");
      return 0;
    }
  	if(d1.year>2999)
    {
      printf("Invalid Year");
      return 0;
    }
  	else
      printf("New Date=%d %d %d",d1.day+1,d1.month,d1.year);
	return 0;
}
