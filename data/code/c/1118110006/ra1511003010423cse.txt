#include <stdio.h>
struct a
{
  int d,m,y;
};
int main()
{
  struct a s;
  scanf("%d %d %d",&s.d,&s.m,&s.y);
  if(s.y>2999||s.y<0)
    printf("Invalid Year");
  else
  {
    	if(s.m>12)
          printf("Invalid Month");
    	else
        {	
          if(s.d>31)
            printf("Invalid Day");
          else
            printf("%d %d %d",s.d+1,s.m,s.y);
        }
  }

	return 0;
}
