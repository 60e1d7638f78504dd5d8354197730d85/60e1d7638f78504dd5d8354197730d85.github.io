#include <stdio.h>
struct Time
{
  int hours;
  int minutes;
  int seconds;
}startTime,stopTime,diff;
int main()
{	
  scanf("%d %d %d",&startTime.hours,&startTime.minutes,&startTime.seconds);
  scanf("\n%d %d %d",&stopTime.hours,&stopTime.minutes,&stopTime.seconds);
  diff.hours=startTime.hours-stopTime.hours;
  diff.minutes=startTime.minutes-stopTime.minutes;
  diff.seconds=startTime.seconds-stopTime.seconds;
  printf("\nTIME DIFFERENCE=%d:%d:%d-%d:%d:%d=%d:%d:%d",
        startTime.hours,startTime.minutes,startTime.seconds,
        stopTime.hours,stopTime.minutes,stopTime.seconds,
        diff.hours,diff.minutes,diff.seconds);
	
	return 0;
}
