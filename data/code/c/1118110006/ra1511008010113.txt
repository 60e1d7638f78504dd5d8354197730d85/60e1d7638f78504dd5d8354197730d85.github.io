#include <stdio.h>
struct date{
  int day;
  int month;
  int year;
}d;
int main()
{
    int leap=0,sample_day,flag=0;
    scanf("%d %d %d",&d.day,&d.month,&d.year);
    if(d.year%4==0){
      if(d.year%100!=0)
        leap=1;
      else
        if(d.year%400==0)
          leap=1;
    }
    if(d.month==2 && leap==1)
      sample_day=29;
    else if(d.month==2 && leap==0)
      sample_day=28;
    else if(d.month==1 || d.month==3 || d.month==5 || d.month==7 || d.month==8 || d.month==10 || d.month==12)
      sample_day=31;
    else if(d.month==4 || d.month==6 || d.month==9 || d.month==11)
      sample_day=30;
  
    if(d.day<=0 || d.day>sample_day){
      printf("\nInvalid Day");flag=1;}
    if(d.month<=0 || d.month>12){
      printf("\nInvalid Month\n");flag=1;}
    if(d.year<0000 || d.year>2999){
      printf("\nInvalid Year");flag=1;}
  
    if(flag==0)
    {
      if(d.month==2 && d.day==28){
        if(leap==1){
          d.day++;}
        else if(leap==0)
        {
          d.day=1;
          d.month++;
        }
      }
      else if(d.month==2 && d.day==29)
      {
        d.day=1;
        d.month++;
      }
      else if(d.day==31 && d.month==12)
      {
        d.day=1;
        d.month=1;
        d.year++;
      }
      else if(d.day==sample_day){
        d.day=1;
        d.month++;
      }
      else
        d.day++;
      int count=0;
      if(d.year>=0 && d.year<=9)
        count=3;
      else if(d.year>=10 && d.year<=99)
        count=2;
      else if(d.year>=100 && d.year<=999)
        count=1;
      if(count==1)
        printf("%d %d 0%d",d.day,d.month,d.year);
      else if(count==2)
        printf("%d %d 00%d",d.day,d.month,d.year);
      else if(count==3)
        printf("%d %d 000%d",d.day,d.month,d.year);
      else
        printf("%d %d %d",d.day,d.month,d.year);
    }
	return 0;
}
