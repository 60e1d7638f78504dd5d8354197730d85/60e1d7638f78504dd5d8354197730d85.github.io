#include <stdio.h>
struct date
{
  int d,m,y;
};
int main()
{
	struct date da;
  scanf("%d %d %d",&da.d,&da.m,&da.y);
  if(da.d<=31 && da.m <=12 && da.y<2999)
    printf("%d %d %d",da.d+1,da.m,da.y);
  if(da.m>12)
    printf("Invalid Month");
  if(da.d>31)
    printf("Invalid Day");
  if(da.y>2999)
    printf("Invalid Year");
	return 0;
}
