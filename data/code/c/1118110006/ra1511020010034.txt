#include <stdio.h>
struct check
{
int d,m,y;
};
int main()
{
struct check s;
  scanf("%d %d %d",&s.d,&s.m,&s.y);
int c=0;
  if(s.y<2017){
  c++;
  }
  else
    printf("Invalid Year");
  if(s.m<13){
  c++;
  }
  else
    printf("Invalid Month");
  if(s.d<32){
  c++;
  }
  else
    printf("Invalid Day");
  
  if(c==3){
    printf("%d %d %d",s.d+1,s.m,s.y);
  }
	return 0;
}
