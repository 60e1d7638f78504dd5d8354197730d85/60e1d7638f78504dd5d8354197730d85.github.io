#include <stdio.h>
struct day
{
  int d;
  int m;
  int y;
};

int main()
{
  struct day d1;
  int df=0,mf=0,yf=0;
  scanf("%d",&d1.d);
  scanf("%d",&d1.m);
  scanf("%d",&d1.y);
  if( d1.d>0 && d1.d<=31)
    df=1;
  if( d1.m>0 && d1.m<=12)
    mf=1;
  if( d1.y>=0000 && d1.y<=2999)
    yf=1;
  if(df!=1)
  {
    printf("Invalid Day");
	return 0;
  }
  else if(mf!=1)
  {
    printf("Invalid Month");
	return 0;
    }
    if(yf!=1)
    {
    printf("Invalid Year");
	return 0;
    }
  printf("%d %d %d",d1.d+1,d1.m,d1.y);
  return 0;
}
