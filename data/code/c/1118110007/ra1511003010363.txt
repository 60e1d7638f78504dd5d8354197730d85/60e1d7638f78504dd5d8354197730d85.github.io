#include<stdio.h>
int main()
{
struct student
  {
   int r_no;
   char name[20];
   char DOB[20];
   int fees;
  };
struct student stud[50];
  
int i,n; 
scanf("%d",&n);
for(i=0;i<n;i++)
{
scanf("%d",&stud[i].r_no);
scanf("%s",stud[i].name);
scanf("%s",stud[i].DOB);
scanf("%d",&stud[i].fees);
}
for(i=0;i<n;i++)
{
printf("Roll no:%d\n",stud[i].r_no);
printf("Name:%s\n",stud[i].name);
printf("DOB:%s\n",stud[i].DOB);
printf("Fees:%d\n",stud[i].fees);
}
return 0;
  }
