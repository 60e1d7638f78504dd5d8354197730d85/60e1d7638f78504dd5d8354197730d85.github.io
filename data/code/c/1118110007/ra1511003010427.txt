#include <stdio.h>
struct student
{
 	int roll;
  	char name[100],date[100];
  	int fees;
};
int main()
{
	int i,n; scanf("%d",&n);
  	struct student s[100];
  	for(i=0;i<n;i++) 
    {
     	scanf("%d",&s[i].roll);
      	scanf("%s",s[i].name);
      	scanf("%s",s[i].date);
      	scanf("%d",&s[i].fees);
    }
  	for(i=0;i<n;i++)
    {
     	printf("Roll no:%d\n",s[i].roll);
      	printf("Name:%s\n",s[i].name);
		printf("DOB:%s\n",s[i].date);
      	printf("Fees:%d\n",s[i].fees);
    }
	return 0;
}
