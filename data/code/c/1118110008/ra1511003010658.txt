#include <stdio.h>
struct Student {
	int roll,fees;
  	char name[20], course[10];
}s1;
void display(struct Student s)
{
  printf("Details of student\nRoll Number=%d\nName=%s\nCourse=%s\nFees=%d\n",s.roll,s.name,s.course,s.fees);
  
}
int main()
{
	scanf("%d",&s1.roll);
	scanf("%s",s1.name);
    scanf("%s",s1.course);
  	scanf("%d",&s1.fees);
  	display(s1);
    return 0;
}
