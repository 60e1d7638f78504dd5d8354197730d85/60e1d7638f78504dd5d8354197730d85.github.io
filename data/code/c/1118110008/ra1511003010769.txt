#include <stdio.h>
struct Student
{
  int roll;
  char name[100];
  char course[100];
  int fees;
};
int main()
{
	struct Student s1;
  
  scanf("%d %s %s %d",&s1.roll,s1.name,s1.course,&s1.fees);
  printf("Details of student\n");
  printf("Roll Number=%d\nName=%s\nCourse=%s\nFees=%d",s1.roll,s1.name,s1.course,s1.fees);
	return 0;
}
