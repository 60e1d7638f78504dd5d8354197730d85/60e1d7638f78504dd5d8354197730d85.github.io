#include <stdio.h>
struct Student
{
  int roll, fees;
  char name[25], course[15];
};
int main()
{
  struct Student s1;
  scanf("%d %s %s %d", &s1.roll, s1.name, s1.course, &s1.fees);
  printf("Details of student\n");
  printf("Roll Number=%d\n", s1.roll);
  printf("Name=%s\n", s1.name);
  printf("Course=%s\n", s1.course);
  printf("Fees=%d", s1.fees);
	return 0;
}
