#include <stdio.h>

 struct student
        {
          int rn;
          char name[10];
          char course[10];
          int fees;
        };
  int main()
  {
    struct student s1;
    scanf("%d %s %s %d",&s1.rn,s1.name,s1.course,&s1.fees);
    
    printf("Details of student\nRoll Number=%d\nName=%s\nCourse=%s\nFees=%d",s1.rn,s1.name,s1.course,s1.fees);
    
	return 0;
}
