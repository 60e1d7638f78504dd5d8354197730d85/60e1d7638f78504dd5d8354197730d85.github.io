#include <stdio.h>
struct Student{
int roll,fees;
  char name[30],course[30];
};
int main()
{
	struct Student s1;
  scanf("%d",&s1.roll);
  scanf("%s",s1.name);
  scanf("%s",s1.course);
  scanf("%d",&s1.fees);
  printf("Details of student\n");
  printf("Roll Number=%d\n",s1.roll);
  printf("Name=");
  puts(s1.name);
  printf("Course=");
  puts(s1.course);
  printf("Fees=%d\n",s1.fees);
	return 0;
}
