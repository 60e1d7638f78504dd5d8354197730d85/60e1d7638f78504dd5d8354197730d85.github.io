#include <stdio.h>
struct student
{
  int roll;
  char name[20];
  char course[20];
  int fees;
};
int main()
{
  struct student a;
  scanf("%d",&a.roll);
  scanf("%s",a.name);
  scanf("%s",a.course);
  scanf("%d",&a.fees);
  printf("Details of the student");
  printf("\nRoll no:%d",a.roll);
  printf("\nName:%s",a.name);
  printf("\nCourse:%s",a.course);
  printf("\nFees:%d",a.fees);
	return 0;
}
