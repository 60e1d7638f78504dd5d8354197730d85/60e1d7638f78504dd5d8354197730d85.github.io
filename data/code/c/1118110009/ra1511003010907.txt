#include <stdio.h>
struct student
{
  char name[30];
  int rollno;
  struct dateOfBirth
{
  int date;
  int month;
  int year;
}DOB;
};

int main()
{
struct student std;
  scanf("%s",std.name);
  scanf("%d",&std.rollno);
  scanf("%d",&std.DOB.date);
  scanf("%d",&std.DOB.month);
  scanf("%d",&std.DOB.year);
  printf("Name=%s\n",std.name);
  printf("RollNo=%d\n",std.rollno);
  printf("Date of birth=%d/%d/%d",std.DOB.date,std.DOB.month,std.DOB.year);
  
	return 0;
}
