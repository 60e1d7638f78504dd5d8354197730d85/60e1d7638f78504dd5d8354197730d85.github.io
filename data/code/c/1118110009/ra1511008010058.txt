#include <stdio.h>
struct dateOfBirth
{
  int date;
  int month;
  int year;
};
struct student
{
  char name[100];
  int rollno;
  struct dateOfBirth DOB;
};
int main()
{
  struct student std;
  scanf("%s",std.name);
  scanf("%d",&std.rollno);
  scanf("%d %d %d",&std.DOB.date,&std.DOB.month,&std.DOB.year);
  printf("Name=%s\n",std.name);
  printf("RollNo=%d\n",std.rollno);
  printf("Date of birth=%d/%d/%d\n",std.DOB.date,std.DOB.month,std.DOB.year);
  return 0;
}
