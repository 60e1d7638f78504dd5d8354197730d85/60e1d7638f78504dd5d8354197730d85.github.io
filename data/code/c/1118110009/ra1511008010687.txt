#include <stdio.h>
struct student
{
  char name[10];
  int rollno;
  struct dateOfBirth
  {
    int date;
    int month;
    int year;
  }DOB;
}std;
int main()
{
  scanf("%s",std.name);
  scanf("%d",&std.rollno);
  scanf("%d %d %d",&std.DOB.date,&std.DOB.month,&std.DOB.year);
  printf("Name=%s\n",std.name);
  printf("RollNo=%d\n",std.rollno);
  printf("Date of birth=%d/%d/%d",std.DOB.date,std.DOB.month,std.DOB.year);
  return 0;
}
