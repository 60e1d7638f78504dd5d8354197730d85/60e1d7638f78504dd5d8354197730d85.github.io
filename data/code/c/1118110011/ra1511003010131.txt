#include <stdio.h>
struct details{
  int roll;
  char name[100];
  char course[30];
  int fees;
};
int main()
{
	struct details de;
  	scanf("%d",&de.roll);
  	scanf("%s",de.name);
  	scanf("%s",de.course);
  	scanf("%d",&de.fees);
    printf("Roll no:%d\n",de.roll);
  	printf("Name:%s\n",de.name);
  	printf("Course:%s\n",de.course);
    printf("Fees:%d\n",de.fees);
	return 0;
}
