#include <stdio.h>
int main()
{
  int x1,y1,a1,b1;
  int x2,y2,a2,b2;
  scanf("%d %d",&x1,&y1);
  scanf("%d %d",&x2,&y2);
  a1=x1+x2;
  b1=y2+y1;
  printf("%d ft %d inch\n",a1,b1);
  a2=x2-x1;
  b2=y1-y2;
  if(a2<0)
    printf("%d ft %d inch",a2*-1,b2);
  else
    printf("%d ft %d inch",a2,b2);

	return 0;
}
