#include <stdio.h>
struct CUBE
{
  int num,sum,cube;
}demo;
int main()
{
  scanf("%d",&demo.num);
  demo.sum=0;
  demo.cube=0;
  while(demo.num!=0)
  {
    demo.sum=demo.sum+demo.num%10;
    demo.num=demo.num/10;
  }
  demo.cube=demo.sum*demo.sum*demo.sum;
  printf("Sum=%d\nCube=%d",demo.sum,demo.cube);
	return 0;
}
