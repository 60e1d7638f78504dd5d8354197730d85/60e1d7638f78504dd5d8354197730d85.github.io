#include <stdio.h>

struct student {
    char name[100];
    int roll;
    float marks;
};

int main()
{
  struct student s;
  
  scanf("%s", s.name);
  scanf("%d", &s.roll);
  scanf("%f", &s.marks);

  printf("Name=%s", s.name);
  printf("\nRoll number=%d", s.roll);
  printf("\nMarks=%0.2f", s.marks);
  
	return 0;
}
