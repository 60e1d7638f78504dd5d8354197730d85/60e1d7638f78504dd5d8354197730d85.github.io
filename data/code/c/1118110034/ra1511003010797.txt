#include <stdio.h>
struct student
{
   char name[100];
   int roll;
   float marks;
};
int main()
{
    int i;
    struct student s[3];
    for(i=0;i<3;i++)
    {
      scanf("%s",s[i].name);
      scanf("%f",&s[i].marks);
    }

    for(i=0;i<3;i++)
    {
      printf("Roll number=%d\n",i+1);
      printf("Name=%s\n",s[i].name);
      printf("Marks=%0.2f\n",s[i].marks);
    }
	return 0;
}
