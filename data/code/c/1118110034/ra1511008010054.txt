#include <stdio.h>
struct student
{
    char name[50];
    int roll;
    float marks;
} s[10];

int main()
{
    int i;
    for(i=0; i<3; ++i)
    {
        scanf("%s",s[i].name);

        scanf("%f",&s[i].marks);
    }
    for(i=0; i<3; ++i)
    {
        printf("Roll number=%d\n",i+1);
        printf("Name=%s\n",s[i].name);
        printf("Marks=%.2f",s[i].marks);
        printf("\n");
    }
    return 0;
}
