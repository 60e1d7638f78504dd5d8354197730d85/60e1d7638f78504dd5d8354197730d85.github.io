#include <stdio.h>
struct student
{
    char name[50];
    int roll;
    float marks;
} s[3];

int main()
{
    int i;

    for(i=0; i<3; ++i)
    {
        s[i].roll = i+1;

        scanf("%s",s[i].name);

        scanf("%f",&s[i].marks);

        printf("\n");
    }

 
    for(i=0; i<3; ++i)
    {
        printf("Roll number=%d\n",i+1);
        printf("Name=");
        puts(s[i].name);
        printf("Marks=%.2f",s[i].marks);
        printf("\n");
    }
    return 0;
}
