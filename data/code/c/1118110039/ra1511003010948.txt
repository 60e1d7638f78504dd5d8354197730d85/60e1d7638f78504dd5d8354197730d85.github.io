
#include <stdio.h>
char stack[40];
int top=-1;
void push(char x)
{
  stack[++top]=x;
}
char pop()
{
  if(top==-1)
    return -1;
  else
    return stack[top--];
}
int priority(char x)
{
  if(x=='(')
    return 0;
  if(x=='+'||x=='-')
    return 1;
  if(x=='/'||x=='*')
    return 2;
}
  
  
int main()
{
  char exp[5][20];
  char *e,x;
  int n;
  scanf("%d",&n);
  int i;
  for(i=0;i<n;i++)
    scanf("%s",exp[i]);
  for(i=0;i<n;i++)
  {
    
  top=-1;
  e=exp[i];
  
  while(*e!='\0')
  {
    if(isalnum(*e))
      printf("%c",*e);
    else if(*e=='(')
      push(*e);
    else if(*e==')')
    {
      while((x=pop())!='(')
        printf("%c",x);
    }
    else
    {
      while(priority(stack[top])>=priority(*e))
        printf("%c",pop());
      push(*e);
    }
    e++;
  }
  while(top!=-1)
    printf("%c",pop());
    
printf("\n");
  }
  

	return 0;
}
