#include <stdio.h>
struct employee
{ char name[100];
  int empid,salary,hra,da;
  float total;
};
int main()
{ struct employee emp;
  scanf("%s",emp.name);
  scanf("%d",&emp.empid);
  scanf("%d",&emp.salary);
  emp.hra=emp.salary*0.1;
  emp.da=emp.salary*0.2;
  emp.total=emp.salary+emp.da+emp.hra;
  printf("Name=%s\nId=%d\nHRA=%d\nDA=%d\nTotal Salary=%0.0f",emp.name,emp.empid,emp.hra,emp.da,emp.total);
  return 0;
}
