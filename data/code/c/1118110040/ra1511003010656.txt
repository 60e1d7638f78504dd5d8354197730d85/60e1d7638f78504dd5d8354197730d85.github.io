#include <stdio.h>
struct employee {
  char name[50];
  int empid;
  int salary;
  int hra;
  int da;
  float total;
};
struct employee read(struct employee emp)
{
  scanf("%s",emp.name);
  scanf("%d",&emp.empid);
  scanf("%d",&emp.salary);
  return emp;
}
struct employee calculate (struct employee emp)
{
  emp.hra = 0.1 * emp.salary;
  emp.da = 0.2 * emp.salary;
  emp.total = emp.hra + emp.da + emp.salary;
  return emp;
}
void print( struct employee emp)
{
  printf("Name=%s\n",emp.name);
  printf("Id=%d\n",emp.empid);
  printf("HRA=%d\n",emp.hra);
   printf("DA=%d\n",emp.da);
  printf("Total Salary=%0.0f\n",emp.total);

}
  
int main()
{
  struct employee emp;
  emp = read(emp);
  emp = calculate(emp);
  print(emp);
	return 0;
}
