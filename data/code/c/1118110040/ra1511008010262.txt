#include <stdio.h>
struct employee
{
  char name[25];
  int empid,salary,hra,da;
  float total;
  
};
int main()
{
  struct employee emp;
  scanf("%s",&emp.name[0]);
  scanf("%d%d",&emp.empid,&emp.salary);
  emp.hra = 0.1*emp.salary;
  emp.da = 0.2*emp.salary;
  emp.total = emp.salary + emp.hra +emp.da;
  printf("Name=%s\nId=%d\nHRA=%d\nDA=%d\nTotal Salary=%0.0f",emp.name,emp.empid,emp.hra,emp.da,emp.total);

	return 0;
}
