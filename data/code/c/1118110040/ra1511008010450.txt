#include <stdio.h>
struct employee
{
  char name[30];
  int empid;
  int salary;
  int hra;
  int da;
  float total;
}emp;
int main()
{
scanf("%s",emp.name);
  scanf("%d%d",&emp.empid,&emp.salary);
  emp.hra=emp.salary/10;
  emp.da=emp.salary/5;
  emp.total=emp.salary+emp.hra+emp.da;
  printf("Name=%s\nId=%d\nHRA=%d\nDA=%d\nTotal Salary=%.0f",emp.name,emp.empid,emp.hra,emp.da,emp.total);
	return 0;
}
