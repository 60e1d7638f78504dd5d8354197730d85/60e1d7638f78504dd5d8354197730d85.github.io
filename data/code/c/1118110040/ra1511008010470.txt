#include <stdio.h>
struct employee
{
  char name[100];
  int empid;
  int sal;
  int hra;
  int da;
  float tot;
}emp;
int main()
{
  scanf("%s",emp.name);
  scanf("%d",&emp.empid);
  scanf("%d",&emp.sal);
  emp.hra=(0.1*emp.sal);
  emp.da=(0.2*emp.sal);
  emp.tot=emp.sal+emp.hra+emp.da;
  printf("Name=%s\n",emp.name);
  printf("Id=%d\n",emp.empid);
  printf("HRA=%d\n",emp.hra);
  printf("DA=%d\n",emp.da);
  printf("Total Salary=%.0f",emp.tot);
  

	return 0;
}
