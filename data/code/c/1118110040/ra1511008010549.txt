#include <stdio.h>
struct employee{
  char name[100];
  int empid;
  int salary;
  int hra;
  int da;
  float total;
};
int main()
{
  struct employee emp;
  scanf("%s%d%d",emp.name,&emp.empid,&emp.salary);
  emp.hra=(int)(0.1*emp.salary);
  emp.da=(int)(0.2*emp.salary);
  emp.total=(float)(emp.hra+emp.da+emp.salary);
  printf("Name=%s\nId=%d\nHRA=%d\nDA=%d\nTotal Salary=%.0f",emp.name,emp.empid,emp.hra,emp.da,emp.total);
	return 0;
}
