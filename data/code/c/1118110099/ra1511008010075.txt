#include <stdio.h>
#include<math.h>
struct EMI
{
  float principal;
float rate;
  float time;
}e;
int main()
{
float totalemi;
  scanf("%f",&e.principal);
  scanf("%f",&e.rate);
  scanf("%f",&e.time);
  e.rate=e.rate/(12*100);
  e.time=e.time*12;
  totalemi=(e.principal*e.rate*pow(1+e.rate,e.time))/(pow(1+e.rate,e.time)-1);
  printf("Monthly EMI is=%f",totalemi);
	return 0;
}
