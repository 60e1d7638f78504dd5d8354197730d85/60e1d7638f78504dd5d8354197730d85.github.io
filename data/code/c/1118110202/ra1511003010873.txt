#include <stdio.h>
struct student
{
    char name[50];
    int roll;
    float marks;
} s;

int main()
{
    
    scanf("%s", s.name);

    scanf("%d", &s.roll);

    scanf("%f", &s.marks);


    

    printf("Name=%s\n",s.name);
  
    printf("Roll number=%d\n",s.roll);

    printf("Marks=%.1f\n",s.marks);

    return 0;
}
