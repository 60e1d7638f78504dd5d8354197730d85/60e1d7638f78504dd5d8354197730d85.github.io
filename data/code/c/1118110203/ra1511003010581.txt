#include <stdio.h>
struct distance{
  int a;
  float b;
};
int main()
{
	struct distance d1;
  	struct distance d2;
  	scanf("%d %f\n",&d1.a,&d1.b);
  	scanf("%d %f",&d2.a,&d2.b);
  	float inches=d1.b + d2.b;
  	int feet = d1.a + d2.a;
  	if(inches>=12.00)
    {
      inches=inches-12.00;
      feet+=1;
    }
  	printf("%dfeet\n",feet);
  	printf("%0.1finches",inches);
	return 0;
}
