#include <stdio.h>
struct Distance{
    int feet;
    float inch;
}d1,d2,sum;
int main()
{
    scanf("%d",&d1.feet);
    scanf("%f",&d1.inch);
    scanf("%d",&d2.feet);
    scanf("%f",&d2.inch);
    sum.feet=d1.feet+d2.feet;
    sum.inch=d1.inch+d2.inch;
    if (sum.inch>12.0)
    {
        sum.inch=sum.inch-12.0;
        ++sum.feet;
    }
    printf("%dfeet\n%.1finches",sum.feet,sum.inch);
	return 0;
}
