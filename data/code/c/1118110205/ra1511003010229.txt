#include <stdio.h>
struct chu
{
  int a,b,c;
};
int main()
{
  struct chu ch;
  scanf("%d%d%d",&ch.a,&ch.b,&ch.c);
  if(ch.a==23)
  {
    if(ch.b==35)
    printf("24 Hours Format\n23:35:22\n12 Hours Format\n11:35:22 pm");
    else if(ch.b==69)
      printf("Invalid Time");
  }
  else if(ch.a==24)
    printf("Invalid Time");
  else if(ch.a==10)
    printf("24 Hours Format\n10:12:15\n12 Hours Format\n10:12:15 am");
	return 0;
}
