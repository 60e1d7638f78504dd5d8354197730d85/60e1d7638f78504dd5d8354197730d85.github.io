#include <stdio.h>
int main()
{
	int h, m, s;
  	scanf("%d %d %d", &h, &m, &s);
  	if(h>=24 || m>=60 || s>=60)
      	printf("Invalid Time");
  	else{
        printf("24 Hours Format");
        printf("\n%d:%d:%d", h, m, s);
        printf("\n12 Hours Format");
        if(h>12)
            printf("\n%d:%d:%d pm", h-12, m, s);
        else
            printf("\n%d:%d:%d am", h, m, s);
    }
	return 0;
}
