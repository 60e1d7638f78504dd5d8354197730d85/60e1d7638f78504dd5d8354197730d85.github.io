#include<stdio.h>
#define inf 1000000007;
typedef long long int ll;
int st[10000007];
int A[10000007];
void print(int N)
{
	int i;
	for(i=1;i<=N;i++)
		printf("%d ",A[i]);
	printf("\n");
}

{

	q=t;

	{
		if(t  <= r)   
			A[x] = ((a*A[x-1]*A[x-1]) +(b*A[x-1])+ c)%m;

			A[x] = ((d*A[x-1]*A[x-1]) +(e*A[x-1])+ f)%m;
		t=(t*q)%s;
	}
//	print(N);	
	st[0]=inf;
	st[1]=1;
int strt=1;
	j=2;
	for(i=2;i<=K;i++)
	{
		while(j>strt && A[st[j-1]]>=A[i])
			j--;
		st[j]=i;
		j++;			
	}
	for(;i<=N;i++)
	{
		A[i-K]=A[st[strt]];

			j--;
		st[j]=i;

//		for(q=0;q<j;q++)
//			printf("%d ",st[q]);

	for(i=(N-K)+1;i<=N;i++)
		A[i]=A[st[strt]];
//rint(N);	
	ll p=1,sum=0;
	for(i=0;i<Q;i++)
			{	
				L1 =( (La * L1) + Lc) %Lm;

			  if(L + K - 1 + D1< N)
					R=L+K-1+D1;
