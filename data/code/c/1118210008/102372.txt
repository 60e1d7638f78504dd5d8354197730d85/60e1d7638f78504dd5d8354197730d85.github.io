    inline int inp( )
    {
    int n=0;
    int ch=getchar();
    while( ch >= '0' && ch <= '9' )
    n = (n<<3)+(n<<1) + ch-'0', ch=getchar();
    return n;
    }

			return (c->val-d->val);

		}
