#define MAX(a,b) ((a) > (b) ? (a) : (b))
#define MIN(a,b) ((a) < (b) ? (a) : (b))
#define IS_DIGIT(x) ((x) >= '0' && (x) <= '9')
int parse_uint()
{
  int i = 0;
  char c = getchar_unlocked();
  while(!IS_DIGIT(c)) c = getchar_unlocked();
  do {
    i = (i<<3) + (i<<1) + c-'0';
    c = getchar_unlocked();
  }while(IS_DIGIT(c));
  return i;
}
 
typedef struct{
  int age;
  int reports[2];
} chef_t;
 
int stack[100000];
int stacksize = 0;
void push(int n)
{
  //printf("[push %d]", n);
  stack[stacksize++] = n;
}
int pop()
{
  //printf("[pop %d]", stack[stacksize-1]);
  return stack[--stacksize];
}
 
void chefs_to_str(chef_t *c, char *str)
{

  push(1);
 
  while(stacksize > 0){
    int chef = pop();
    if(chef == 0){

      //printf("x ");

    }
    else{
      //printf("\n");

      *strpos++ = c[chef].age;

      push(c[chef].reports[0]);

    }

  *strpos = 0;
 
  strpos = str;
  while(*strpos){
    //printf("%d ", *strpos);

  }

}
