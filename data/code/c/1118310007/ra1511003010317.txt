#include<stdio.h>
int arr[500000];
int temp[500000];
swap(int m,int n)
{
         int c=temp[m];
         temp[m]=temp[n];
         temp[n]=c;
}
void Qsort(int lo,int hi)
{
     if(hi<=lo)
     return;
     int lt=lo,i=lo+1,gt=hi;
     int flag=temp[lo];
     while(i<=gt)
     {
                 if(temp[i]<flag)
                    swap(lt++,i++);
                 else if(temp[i]>flag)
                 swap(i,gt--);
                 else
                 i++;
     }
     Qsort(lo,lt-1);
     Qsort(gt+1,hi);
}

                   l--;

                   int i,max=l;
                   int val= x^arr[l],temp;
                   for(i=l+1; i<=r; i++)
                   {
                            temp= x^arr[i];
                            if(temp > val)
                            {
                                    val=temp;
                                    max=i;
                            }
                   }
                   printf("%d\n",arr[max]);
              }
              else if(type == 2)
              {
                   scanf("%d",&x);

                   scanf("%d %d %d",&l,&r,&x);

              else

                  l--;
                   r--;
                  int i,j=0;
                  for(i=l; i<=r; i++)
                   {
                          temp[j]=arr[i];
