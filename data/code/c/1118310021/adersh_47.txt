#include<stdio.h>
 
unsigned long long int hcf( unsigned long long int a,unsigned long long int b){
	if(b == 0) return a;
	else return hcf(b, a%b);
}
 
int main(void){
 
int t;
scanf("%d",&t);
while(t--){
 int n;
 scanf("%d",&n);
 int i,j;
 unsigned long long int arr[n],min=9999999999999, gcd, lcm;
 for(i=0; i<n; i++){
 	scanf("%llu",&arr[i]); 	
 }
 for(i=0;i<n; i++){
 	for(j=i+1; j<n ; j++){
 		if( arr[i] > min || arr[j] > min) continue;
 		gcd= hcf(arr[i],arr[j]);
 		lcm= (arr[i]* arr[j])/gcd;
 		if(min >lcm) min=lcm; 
 		
 	}
 }
 printf("%llu \n",min);
} 
 return 0;
}
