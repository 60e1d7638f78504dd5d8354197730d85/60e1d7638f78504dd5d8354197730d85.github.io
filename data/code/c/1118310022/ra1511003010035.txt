#include <stdio.h>
 
int main(void) {
	// your code goes here
	char str1[1001],str2[1001],flag;
	int i,j,t,k;
	scanf("%d",&t);
	for(i=0;i<t;i++)
	{
	    scanf("%s%s",str1,str2);
	    j=0;
	    flag=0;
	    
	    while(str1[j])
	    
	    {
	        k=0;
	        while(str2[k])
	        {
	            if(str1[j]==str2[k])
	            {
	                flag=1;
	                break;
	            }
	            k++;
	        }
	        if(flag==1)
	        break;
	       
	       j++;
	    }
	    if(flag==1)
	    printf("Yes\n");
	    else
	    printf("No\n");
	}
	return 0;
}
