#include<stdio.h>
#include<stdlib.h>
 
int main()
{
    long long int t,max,i,c,b,k,j,m,l,s,r,f,n;
    long long int a[1000000],w[10000];
    float avg;
    scanf("%lld",&t);
    for(i=0;i<t;i++)
    {
        for(j=0;j<1005;j++)
        {
            w[j]=0;
        }
        s=0;f=0;r=0;
        scanf("%lld",&n);
        for(j=0;j<n;j++)
        {
            scanf("%lld",&a[j]);
            m=a[j];
            w[m]++;
        }
        for(j=1;j<=1003;j++)
        {
            if(w[j]>=2)
            {
                f++;
                s=l;
                l=j;
            }
            if(w[j]>=4)
            {
                r++;
                k=j;
            }
        }
      
        if(f<2&&r==0)
        printf("-1\n");
        else if(r!=0&&f<2)
        {
            printf("%lld\n",k*k);
        }
        else
        {
            if(r!=0)
            {
                c=s*l;
                b=k*k;
                max=c>b?c:b;
            }
            else
            max=s*l;
            printf("%lld\n",max);
        }
    }
    return 0;
}
