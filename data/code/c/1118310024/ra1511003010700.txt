#include<stdio.h> 
int main( )
{
	long int t;
	scanf("%ld",&t);
	while(t--)
	{
		long int n;
		scanf("%ld", &n);
		long int i;
		long int arr[n];
		for ( i = 0 ; i < n ; i++ ) 
		scanf("%ld", arr+i );
		long int count[n];
		i = 1;
		count[0] = 1;
		while( i != n )
		{
			if( arr[i] >=  arr[i-1] ) count[i] = count[i-1] + 1;
			else count[i] = 1;
			i++;
		}
		long ans = 0;
		for( i = 0 ; i < n ; i++ )
		{
			ans += count[i];
		}
		printf("%ld",ans);
		printf("\n");
	}
return 0;
}
