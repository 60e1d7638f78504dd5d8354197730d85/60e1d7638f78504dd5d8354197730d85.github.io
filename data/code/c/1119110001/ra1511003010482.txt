#include <stdio.h>
#include<math.h>
#include<stdlib.h>

void compare(int *a1, int *a2, int n)
{
  int i,flag=0;
  for(i=0;i<n;i++)
  {
    if(*a1 != *a2)
    {
      flag=1;
      break;
    }
    a1++;
    a2++;
  }
  if(flag==1)
    printf("Arrays are not equal");
  else
    printf("Arrays are equal");
}
int main(int argc, char **argv)
{
	int a1[10],a2[10];
  int n,i;
  scanf("%d\n",&n);
  for(i=0;i<n;i++)
    scanf("%d",&a1[i]);
  scanf("%d\n",&n);
  for(i=0;i<n;i++)
    scanf("%d",&a2[i]);
  compare(a1,a2,n);
	return 0;
}
