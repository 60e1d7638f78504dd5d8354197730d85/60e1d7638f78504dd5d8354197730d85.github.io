#include <stdio.h>
int main()
{int a[10],b[10],f=0,i,m,n;
 int *q,*w;
 scanf("%d",&n);
 for(i=0;i<n;i++)
   scanf("%d",&a[i]);
 
  scanf("%d",&m);
 for(i=0;i<m;i++)
   scanf("%d",&b[i]);

 q=&a[0];
 w=&b[0];
 i=0;
 while(i<n)
 {
   if(*q==*w)
   {f=1;
   }
   else
   {f=0;break;}
   q++;w++;i++;
 }
 if(f==1)
   printf("Arrays are equal");
  else
   printf("Arrays are not equal");
	return 0;
}
