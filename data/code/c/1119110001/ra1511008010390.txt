#include <stdio.h>
#include <math.h>
void compare(int *a1, int *a2, int n1)
{
	int i, flag = 0;
	for(i = 0; i < n1; i++)
	{
		if(*a1 != *a2)
		{
			flag = 1;
			break;
		}
		a1++;
		a2++;
	}
	if(flag == 1)
		printf("Arrays are not equal");
	else
		printf("Arrays are equal");
}
 
int main(int argc, char **argv)
{
	
	int n1,n2,i;
	scanf("%d", &n1);
    int a1[n1];
	for(i = 0; i < n1; i++)
		scanf("%d", &a1[i]);
  scanf("%d", &n2);
  int a2[n2];
	for(i = 0; i < n2; i++)
		scanf("%d", &a2[i]);
	compare(a1, a2, n1);
    return 0;
}
