#include <stdio.h>
int main()
{
    int n1,n2,i,count=0;
    scanf("%d",&n1);
    int arr1[n1];
    for(i=0;i<n1;i++)
      scanf("%d",&arr1[i]);
    scanf("%d",&n2);
   int arr2[n2];
   for(i=0;i<n2;i++)
     scanf("%d",&arr2[i]);
   for(i=0;i<n1;i++)
   {
     if(arr1[i]==arr2[i])
       count++;
   }
  if(count==n1)
    printf("Arrays are equal");
  else 
    printf("Arrays are not equal");
	return 0;
}
