#include <stdio.h>
int main()
{
	int a, b, i, q=0;
  	scanf("%d", &a);
  	int m[a];
  	for(i=0;i<a;i++)
      	scanf("%d", &m[i]);
    scanf("%d", &b);
  	int n[b];
    for(i=0;i<b;i++)
      	scanf("%d", &n[i]);
  	if(a==b){
      	for(i=0;i<a;i++)
          	if(m[i]==n[i])
              	q++;
      	if(q==a)
          	printf("Arrays are equal");
      	else
          	printf("Arrays are not equal");
    }
  	else
      	printf("Arrays are not equal");
	return 0;
}
