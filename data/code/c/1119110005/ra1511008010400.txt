#include <stdio.h>
int main()
{
  int i,a,b,c,n;
  int *p1,*p2,*p3;
  scanf("%d\n",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d%d%d",&a,&b,&c);
  }
  p1=&a;
  p2=&b;
  p3=&c;
  if((*p1>*p2) && (*p1>*p3))
  {
    printf("%d is largest",*p1);
  }
  else if((*p2>*p1) && (*p2>*p3))
  {
    printf("%d is largest",*p2);
  }
  else
  {
    printf("%d is largest",*p3);
  }
  
	return 0;
}
