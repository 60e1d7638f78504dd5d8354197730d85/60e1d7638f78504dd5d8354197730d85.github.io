#include<stdio.h>

int main()
{
	
	int arr[10], i,n;
	scanf("%d",&n);
	for(i=0; i<n; i++)
	{
		scanf("%d",&arr[i]);
	}
	for(i=0; i<n; i++)
	{
		if(arr[i]<0)
		{
			printf("%d=Negative\n",arr[i]);
		}
		else if(arr[i]==0)
		{
			printf("%d=Zero\n",arr[i]);
		}
		else
		{
			printf("%d=Positive\n",arr[i]);
		}
	}
	//printf("Positive Numbers = %d\n",countp);
	//printf("Negative Numbers = %d\n",countn);
	//printf("Zero = %d",countz);
	return 0;
}
