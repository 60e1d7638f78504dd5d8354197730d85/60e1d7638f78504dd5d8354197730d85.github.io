#include <stdio.h>
int main()
{
int num,i,n[20];
  scanf("%d",&num);
  for(i=0;i<num;i++)
  {
    scanf("%d",&n[i]);
    {
      if(n[i]>0)
      {
        printf("%d=Positive\n",n[i]);
      }
      else if(n[i]<0)
      {
        printf("%d=Negative\n",n[i]);
      }
      if(n[i]==0)
      {
        printf("%d=Zero\n",n[i]);
      }
    }
  }
  
	return 0;
}
