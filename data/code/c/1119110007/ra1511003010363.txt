#include<stdio.h>

int main()
{
	
	int  arr[10], i,n;
	scanf("%d",&n);
	for(i=0; i<10; i++)
	{
		scanf("%d",&arr[i]);
	}
	for(i=0; i<10; i++)
	{
		if(arr[i]<0)
		{
			printf("\n%d=Negative",arr[i]);
		}
		else if(arr[i]==0)
		{
			printf("\n%d=Zero",arr[i]);
		}
		else
		{
			printf("\n%d=Positive",arr[i]);
		}
	}
return 0;
}
