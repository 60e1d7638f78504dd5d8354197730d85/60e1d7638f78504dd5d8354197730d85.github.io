#include <stdio.h>
int main()
{
  int n;
  scanf("%d",&n);
  int i;
  int arr[n];
  for(i=0;i<n;i++)
    scanf("%d",&arr[i]);
  for(i=0;i<n;i++)
  {
    if(arr[i]>0)
      printf("%d=Positive\n",arr[i]);
    else if(arr[i]<0)
      printf("%d=Negative\n",arr[i]);
    else printf("%d=Zero\n",arr[i]);
  }
    
	return 0;
}
