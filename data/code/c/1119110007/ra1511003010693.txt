#include <stdio.h>
int main()
{
  	int a[1000];
  	int i, n;
  	scanf("%d", &n);
  	for(i = 0; i < n; i++)
      	scanf("%d", &a[i]);
  	for(i = 0; i < n; i++){
      if(a[i] > 0)
          printf("%d=Positive\n", a[i]);
      if(a[i] < 0)
          printf("%d=Negative\n", a[i]);
      if(a[i] == 0)
          printf("%d=Zero\n", a[i]);

    }

	return 0;
}
