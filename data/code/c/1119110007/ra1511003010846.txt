#include <stdio.h>
int main()
{int n,a[10],i;
 int *p; 
 scanf("%d",&n);
 for(i=0;i<n;i++)
 {
   scanf("%d",&a[i]);
 }
 p=a;
 for(i=0;i<n;i++)
 {
   if(*p>0)
     printf("%d=Positive\n",*p);
   else if(*p<0)
     printf("%d=Negative\n",*p);
   else
     printf("%d=Zero\n",*p);
   p++;
 }

	return 0;
}
