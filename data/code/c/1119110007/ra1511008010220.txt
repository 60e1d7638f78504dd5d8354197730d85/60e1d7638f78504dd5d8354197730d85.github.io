#include<stdio.h>

int main()
{
	
	int arr[100],i,n;
	scanf("%d",&n);
	for(i=0; i<n; i++)
	{
		scanf("%d",&arr[i]);
	}
	for(i=0; i<10; i++)
	{
		if(arr[i]<0)
		{
			printf("%d=Negative\n",arr[i]);
		}
		else if(arr[i]==0)
		{
			printf("%d=Zero\n",arr[i]);
		}
		else
		{
			printf("%d=Positive\n",arr[i]);
		}
	}
	return 0;
}
