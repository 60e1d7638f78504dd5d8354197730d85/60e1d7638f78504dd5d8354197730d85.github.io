#include <stdio.h>
int main()
{
  int n;
  scanf("%d",&n);
  int a[n];
  int i;
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  for(i=0;i<n;i++)
  {
    if(a[i]>0){
      printf("%d=Positive\n",a[i]);
    }
    else if (a[i]<0){
      printf("%d=Negative\n",a[i]);
         }
    else{
      printf("%d=Zero\n",a[i]);
    }
  }

	return 0;
}
