#include <stdio.h>
int main()
{
  
  
  int start,end;
  scanf("%d",&start);
  scanf("%d",&end);
  if(start>end)
    printf("Invalid Input");
  else{
  int arr[end-start],i,temp=start;
  for(i=0;i<(end-start);i++)
    arr[i]=++temp;
  int *ptr;
  ptr = arr;
  int sum=0;
    for(i=0;i<(end-start);i++)
      sum = sum+*ptr++;
    printf("%d\n",sum);
    printf("%.2f",(float)sum/(end-start));
      
  }
  return 0;
}
