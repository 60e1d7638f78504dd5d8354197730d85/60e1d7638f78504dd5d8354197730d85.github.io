#include <stdio.h>
int main()
{

  int t,n,i,flag,j;
  scanf("%d",&t);
  while(t--)
  {
    scanf("%d",&n);
    flag=0;
    for(i=2;i<=n/2;i++)
    {
      if(n%i==0)
      {
        flag=1;
        break;
      }
    }
    if(flag==0)
      printf("%d is a prime number\n",n);
    else
      printf("%d is a composite number\n",n);
  }
	return 0;
}
