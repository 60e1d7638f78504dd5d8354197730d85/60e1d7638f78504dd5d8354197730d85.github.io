#include<stdio.h>
int main()
{
    int a[50],count,c,n,i,j,sum=0;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    count=0;
    for(i=0;i<n;i++)
    {
        c=0;
        for(j=2;j<a[i];j++)
        {
            if(a[i]%j==0)
            {
                   c=1;
                   break;
            }
        }
         if(c==0)
         {
            printf("%d is a prime number\n",a[i]);
            sum=sum+a[i];
            count++;
         }
      else
      {
			printf("%d is a composite number\n",a[i]);
      }
      
    }
return 0;
}
