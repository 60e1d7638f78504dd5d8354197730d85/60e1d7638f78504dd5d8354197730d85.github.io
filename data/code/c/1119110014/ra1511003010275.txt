#include<stdio.h>

int main()
{
    int num[20], i, found=0, search, n;


    scanf("%d",&n);



 
    for(i=0; i<n; i++)
    {
        scanf("%d",&num[i]);
    }


    scanf("%d",&search);



    for(i=0; i<n; i++)
    {
        if(search==*(num+i))
        {
            found++;
            break;
        }
    }


    if(found==1)
        printf("\n%d is found in the array at position=%d",search,i);
    else
        printf("\n%d Does not exist in the array",search);


    return 0;
}
