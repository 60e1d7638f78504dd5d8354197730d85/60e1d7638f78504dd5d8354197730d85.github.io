#include <stdio.h>
int main()
{
 int num[20], i, found=0, search, n;


   // printf("enter the number of elements\n");
    scanf("%d",&n);


   // printf("enter array elements\n");
    for(i=0; i<n; i++)
    {
        scanf("%d",&num[i]);
    }


   // printf("enter element to search\n");
    scanf("%d",&search);


    for(i=0; i<n; i++)
    {
        if(search==*(num+i))
        {
            found++;
            break;
        }
    }


    if(found==1)
        printf("\n%d is found in the array at position=%d\n",search,i);
    else
        printf("\n%d Does not exist in the array\n",search);
	return 0;
}
