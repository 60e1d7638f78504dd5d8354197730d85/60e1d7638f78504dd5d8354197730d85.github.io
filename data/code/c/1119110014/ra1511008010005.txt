#include <stdio.h>
int main()
{
  int n,a[10],x,i,p=0;
  scanf("%d",&n);
  for(i=0;i<n;i++)
    scanf("%d",&a[i]);
  scanf("%d",&x);
  for(i=0;i<n;i++)
  {
    if(a[i]==x)
    {
      p=i+1;
      break;
    }
  }
  if(p!=0)
    printf("%d is found in the array at position=%d",x,p-1);
  else
    printf("%d Does not exist in the array",x);
  return 0;
}
