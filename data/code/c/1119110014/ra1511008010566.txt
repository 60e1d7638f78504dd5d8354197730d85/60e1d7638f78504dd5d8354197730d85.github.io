#include<stdio.h>
int main()
{
    int num[20], i, found=0, search, n,a;
    scanf("%d",&n);
    for(i=0; i<n; i++)
    {
        scanf("%d",&num[i]);
    }
    scanf("%d",&search);
    for(i=0; i<n; i++)
    {
        if(search==*(num+i))
        {
          a=i;
            found++;
            break;
        }
    }
    if(found==1)
        printf("%d is found in the array at position=%d",search,a);
    else
        printf("%d Does not exist in the array",search);
 return 0;
}
