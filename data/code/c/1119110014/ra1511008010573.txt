#include<stdio.h>
int main()
{
    int num[10], i,found=0,search,n;
    scanf("%d",&n);
    for(i=0; i<n; i++)
    {
        scanf("%d",&num[i]);
    }
    scanf("%d",&search);
    for(i=0; i<n; i++)
    {
        if(search==*(num+i))
        {
            found++;
            break;
        }
    }
 if(found==1)
 {
        printf("%d is found in the array at position=%d\n",search,i);
 }
    else
    {
      printf("%d Does not exist in the array\n",search);
  
    }
  return 0;
}
