#include<stdio.h>
 
int main() {
   int a[30], ele, num, i;
 
   scanf("%d", &num);
 

   for (i = 0; i < num; i++) {
      scanf("%d", &a[i]);
   }

 
   scanf("%d", &ele);
 

   i = 0;
   while (i < num && ele != a[i]) {
      i++;
   }
 

   if (i < num) {
      printf("%d is found in the array at position=%d",ele, i );
   } else {
      printf("%d Does not exist in the array",ele);
   }
 
   return (0);
}
