#include <stdio.h>
#define MAX_SIZE 100  

int main()
{
    int arr[MAX_SIZE];
    int size, i, toSearch, found;
    scanf("%d", &size);
    for(i=0; i<size; i++)
    {
        scanf("%d", &arr[i]);
    }
    scanf("%d", &toSearch);
    found = 0; 
    for(i=0; i<size; i++)
    {
        if(arr[i] == toSearch)
        {
            found = 1;
            break;
        }
    }

    if(found == 1)
    {
        printf("%d is found in the array at position=%d", toSearch, i);
    }
    else
    {
        printf("%d Does not exist in the array", toSearch);
    }

    return 0;
}
