#include <stdio.h>
#include <stdlib.h>
#include <string.h>
void del(char str[], char ch) 
{
   int i, j = 0;
   int size;
   char ch1;
   char str1[10];
 
   size = strlen(str);
 
   for (i = 0; i < size; i++) 
   {
      if (str[i] != ch) 
      {
         ch1 = str[i];
         str1[j] = ch1;
         j++;
      }
   }
   str1[j] = '\0';
 
   printf("%s", str1);
}
int main()
{
  char str[20];
   char ch;
   fgets(str,20,stdin);
   scanf("%ch", &ch);
 
   del(str, ch);

}
