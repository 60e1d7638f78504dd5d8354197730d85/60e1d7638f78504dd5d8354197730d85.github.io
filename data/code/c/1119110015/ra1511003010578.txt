#include <stdio.h>
#include <string.h>
int main(){
    char str[150],c;
    int i,j;
  
    fgets(str,sizeof str,stdin);
    scanf("%c",&c);
    for(i=0; str[i]!='\0'; ++i)
    {
        while (!(str[i]!=c))
        {
            for(j=i;str[j]!='\0';++j)
            {
                str[j]=str[j+1];
            }
            str[j]='\0';
        }
    }
    printf("%s",str);
  return 0;
}
