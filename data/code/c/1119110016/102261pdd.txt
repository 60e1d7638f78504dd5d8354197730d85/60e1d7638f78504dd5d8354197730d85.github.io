#include <stdio.h>
int main()
{
  int t,t1,a[100],b[100],c=0,i;
  scanf("%d",&t);
  for(i=0;i<t;i++)
  {
    scanf("%d",&a[i]);
  }
  scanf("%d",&t1);
  for(i=0;i<t1;i++)
  {
    scanf("%d",&b[i]);
  }
  for(i=0;i<t;i++)
  {
    if(a[i]==b[i])
    {
      c++;
    }
  }
  if(c==t)
  {
    printf("Arrays are equal");
  }
  else
  {
    printf("Arrays are not equal");
  }
    return 0;
}
