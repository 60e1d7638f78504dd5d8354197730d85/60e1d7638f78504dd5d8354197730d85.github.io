#include <stdio.h>
int main()
{
int n1,n2,i,flag=0;
  int a1[10],a2[10];
  
  scanf("%d",&n1);
  for(i=0;i<n1;i++)
    scanf("%d",(a1+i));
  scanf("%d",&n2);
  for(i=0;i<n2;i++)
    scanf("%d",(a2+i));
  
  if(n1!=n2)
    flag=1;
  else{
  	for(i=0;i<n1;i++){
    	if(*(a1+i)!=*(a2+i)){
        	flag=1;
          break;
        	}else{
        flag=0;
        }
          
    	}
  	}
  
  if(flag==1)
    printf("Arrays are not equal");
  else
    printf("Arrays are equal");
	return 0;
}
