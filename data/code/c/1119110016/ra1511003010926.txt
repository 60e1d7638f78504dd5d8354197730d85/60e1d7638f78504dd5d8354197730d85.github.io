#include <stdio.h>
int main()
{
	int a[100],b[100],n1,n2,i,c=0;
  	scanf("%d",&n1);
  	for(i=0;i<n1;i++)
    {
    	scanf("%d",&a[i]);
    }
  	scanf("%d",&n2);
  	for(i=0;i<n2;i++)
    {
    	scanf("%d",&b[i]);
    }
  	if(n1!=n2)
      printf("Arrays are not equal");
  	if(n1==n2)
    {
    	for(i=0;i<n1;i++)
    {
    	if(a[i]==b[i])
          c++;
        }
      if(c==n1 && c==n2)
        printf("Arrays are equal");
      else
        printf("Arrays are not equal");
    }
  	
	return 0;
}
