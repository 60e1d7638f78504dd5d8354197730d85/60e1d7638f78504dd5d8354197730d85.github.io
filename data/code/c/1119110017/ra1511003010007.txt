#include <stdio.h>
#include<ctype.h>
int main()
{
char c;
  scanf("%c",&c);
  if(isalpha(c))
  {
    if(isupper(c))
    {
      printf("Upper case character was entered");
      printf("\n%c",tolower(c));
    }
    if(islower(c))
    {
      printf("Lower case character was entered");
      printf("\nUpper case=%c",toupper(c));
    }
  }
	return 0;
}
