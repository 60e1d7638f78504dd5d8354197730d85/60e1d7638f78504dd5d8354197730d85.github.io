#include <stdio.h>

int main()
{
    char ch;

    /* Input character from user */
   // printf("Enter any character: ");
    scanf("%c", &ch);


    if(ch >= 'A' && ch <= 'Z')
    {
      printf("Upper case character was entered\n");
             
        printf("%c", ch);
    }
    else if(ch >= 'a' && ch <= 'z')
    {
      printf("Lower case character was entered\n");
       // printf("'%c'\n", ch);
      ch=ch-32;
      printf("Upper case=%c",ch);
    }
    else
    {
        printf("'%c' is not an alphabet.", ch);
      
    }

    return 0;
}
