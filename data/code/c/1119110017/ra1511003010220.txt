#include <stdio.h>
#include<string.h>
int main()
{
    char ch;

    scanf("%c", &ch);


    if(ch >= 'A' && ch <= 'Z')
    {
        printf("Upper case character was entered");
      printf("\n%c",tolower(ch));
    }
    else if(ch >= 'a' && ch <= 'z')
    {
        printf("Lower case character was entered");
      printf("\nUpper case=%c",toupper(ch));
    }
    else
    {
        printf("'%c' is not an alphabet.", ch);
    }

    return 0;
}
