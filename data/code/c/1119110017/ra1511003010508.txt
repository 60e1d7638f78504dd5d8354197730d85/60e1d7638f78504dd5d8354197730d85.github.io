#include <stdio.h>

int main()
{
    char ch;

    /* Input character from user */
    
    scanf("%c", &ch);


    if(ch >= 'A' && ch <= 'Z')
    {
        printf("Upper case character was entered\n");
      printf("%c",ch);
    }
    else if(ch >= 'a' && ch <= 'z')
    {
        printf("Lower case character was entered\n");
      printf("Upper case=%c",ch-32);
    }
    else
    {
        printf("'%c' is not an alphabet.", ch);
    }

    return 0;
}
