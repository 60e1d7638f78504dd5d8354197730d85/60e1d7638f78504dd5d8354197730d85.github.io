#include <stdio.h>
#include <ctype.h>

int main()
{
    char ch;
    scanf("%c", &ch);

if(isalpha(ch))
{
    if(ch>='A'&&ch<='Z')
    {
        printf("Upper case character was entered\n");
        printf("%c",tolower(ch));
    }
    else if(ch>='a'&&ch<='z')
    {
        printf("Lower case character was entered\n");
        printf("Upper case=%c",toupper(ch));
    }
 }   
    else
    {
    }

    return 0;
}
