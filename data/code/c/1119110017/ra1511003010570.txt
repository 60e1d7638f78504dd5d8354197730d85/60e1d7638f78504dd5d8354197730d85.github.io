#include <stdio.h>
int main()
{

  char a;
  scanf("%c",&a);
  if(a>=97 && a<=122){
    printf("Lower case character was entered\n");
    a=a-32;
    printf("Upper case=%c",a);
  }
  else if(a>=65 && a<=90)
  {
    printf("Upper case character was entered\n");
    a=a+32;
    printf("%c",a);
  }
	return 0;
}
