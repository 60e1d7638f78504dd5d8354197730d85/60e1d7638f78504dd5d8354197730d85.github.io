#include <stdio.h>
#include <ctype.h>

int main()
{
    char ch;
    scanf("%c", &ch);


    if(ch >= 'A' && ch <= 'Z')
    {
        printf("Upper case character was entered\n%c",ch);
    }
    else if(ch >= 'a' && ch <= 'z')
    {
      
        printf("Lower case character was entered\nUpper case=%c",toupper(ch));
    }

    return 0;
}
