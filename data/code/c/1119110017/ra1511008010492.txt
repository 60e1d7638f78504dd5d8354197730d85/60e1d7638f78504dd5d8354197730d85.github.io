#include <stdio.h>
#include<string.h>
int main()
{
  char c;
  scanf("%c",&c);
  if(isalpha(c))
  {
    if(isupper(c))
    {
      printf("Upper case character was entered\n");
      printf("%c",c);
    }
    if(islower(c))
    {
      printf("Lower case character was entered\n");
      printf("Upper case=%c",toupper(c));
    }
  }
	return 0;
}
