#include <stdio.h>

int main()
{
    char ch;
    scanf("%c", &ch);


    if(ch >= 'A' && ch <= 'Z')
    {
        printf("Upper case character was entered\n%c",ch);
      
    }
    else if(ch >= 'a' && ch <= 'z')
    {
        printf("Lower case character was entered\nUpper case=%c",(ch-32));
    }
    else
    {
        printf("'%c' is not an alphabet.", ch);
    }

    return 0;
}
