#include <stdio.h>

int main()
{
    char ch;
    scanf("%c", &ch);
    if(ch >= 'A' && ch <= 'Z')
    {
        printf("Upper case character was entered\n%c", ch+32);
    }
    else if(ch >= 'a' && ch <= 'z')
    {
        printf("Lower case character was entered\nUpper case=%c", ch-32);
    }
    else
    {
        printf("Lower case character was entered\nUpper case=%c", ch);
    }

    return 0;
}
