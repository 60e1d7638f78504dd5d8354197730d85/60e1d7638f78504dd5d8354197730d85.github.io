#include <stdio.h>
#include <ctype.h>

int main()
{
    char ch;

    scanf("%c", &ch);


    if(ch >= 'A' && ch <= 'Z')
    {
        printf("Input is upper case");
        printf("\nLower case=%c", tolower(ch));
    }
    else if(ch >= 'a' && ch <= 'z')
    {
        printf("Input is lower case");
        printf("\nUpper case=%c", toupper(ch));
    }
    else
    {
        printf("'%c' is not an alphabet.", ch);
    }

    return 0;
}
