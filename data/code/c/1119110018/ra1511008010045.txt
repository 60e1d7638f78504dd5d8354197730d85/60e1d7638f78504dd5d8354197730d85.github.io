#include <stdio.h>

int main()
{
    char ch;
    scanf("%c", &ch);


    if(ch >= 'A' && ch <= 'Z')
    {
        printf("Input is upper case\n");
       // if(ch>='A' && ch<='Z')
			ch=ch+32;
      printf("Lower case=%c",ch);
    }
    else if(ch >= 'a' && ch <= 'z')
    {
       printf("Input is lower case\n");
       // if(ch>='A' && ch<='Z')
			ch=ch-32;
      printf("Upper case=%c",ch);
    }
   

    return 0;
}
