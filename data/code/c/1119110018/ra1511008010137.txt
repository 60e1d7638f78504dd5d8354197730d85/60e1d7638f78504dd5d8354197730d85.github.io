#include <stdio.h>
#include<string.h>
int main()
{
   char ch;
   scanf("%c",&ch);
     if (ch >= 'A' && ch <= 'Z')
     {
       printf("Input is upper case\n");
       printf("Lower case=%c", tolower(ch));
     }
     else
     { 
       printf("Input is lower case\n");
       printf("Upper case=%c", toupper(ch));
     }
  
	return 0;
}
