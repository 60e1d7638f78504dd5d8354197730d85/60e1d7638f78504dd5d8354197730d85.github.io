#include <stdio.h>
int main()
{
char ch;
  scanf("%c",&ch);
  if(ch>='a'&& ch<='z')
  {
    printf("Input is lower case\n");
    ch=ch-32;
    printf("Upper case=%c\n",ch);
  }
  else if(ch>='A' && ch<='Z')
  {
    printf("Input is upper case\n");
    ch=ch+32;
    printf("Lower case=%c\n",ch);
  }
  
    
	return 0;
}
