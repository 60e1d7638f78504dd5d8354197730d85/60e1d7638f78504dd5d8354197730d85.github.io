#include <stdio.h>
#include<string.h>

int main()
{
    char ch;

   
   
    scanf("%c", &ch);


    if(ch >= 'A' && ch <= 'Z')
    {
        printf("Input is upper case\n");
      	printf("Lower case=%c",tolower(ch));
    }
    else if(ch >= 'a' && ch <= 'z')
    {
        printf("Input is lower case\n");
      	printf("Upper case=%c",toupper(ch));
    }
    else
    {
        printf("'%c' is not an alphabet.", ch);
    }

    return 0;
}
