#include <stdio.h>
#include <ctype.h>
int main()
{

  scanf("%c",&c);
  if(isalpha(c))
  {
	if(isupper(c))
    {printf("Input is upper case");

}
    else   {printf("Input is lower case");
    printf("\nUpper case=%c",toupper(c));
           }
  }
return 0;
}
