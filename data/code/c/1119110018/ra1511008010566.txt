#include <stdio.h>

int main()
{
    char ch;
    scanf("%c", &ch);


    if(ch >= 'A' && ch <= 'Z')
    {
        printf("Input is upper case\n");
      printf("Lower case=%c",ch+32);
    }
    else if(ch >= 'a' && ch <= 'z')
    {
        printf("Input is lower case\n");
      printf("Upper case=%c",ch-32);
    }


    return 0;
}
