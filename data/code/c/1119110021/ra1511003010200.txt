
#include<stdio.h>

int main() {
   int numArray[10],n;
   int i, sum = 0;
   int *ptr;
 scanf("%d",&n);
 
   for (i = 0; i < n; i++)
      scanf("%d", &numArray[i]);
 
   ptr = numArray; 
 
   for (i = 0; i < 10; i++) {
      sum = sum + *ptr;
      ptr++;
   }
 
   printf("%d", sum);
  return 0;
}
