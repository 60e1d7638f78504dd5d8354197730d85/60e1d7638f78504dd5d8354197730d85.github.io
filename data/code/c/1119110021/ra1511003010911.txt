#include <stdio.h>
#include <malloc.h>
 
int main()
{
    int i, n, sum = 0;
    int *a;
 
   
    scanf("%d", &n);
    a = (int *) malloc(n * sizeof(int));
    
    for (i = 0; i < n; i++)
    {
        scanf("%d", a + i);
    }
    
    for (i = 0; i < n; i++)
    {
        sum = sum + *(a + i);
    }
    printf(" %d\n", sum);
return 0;
}
