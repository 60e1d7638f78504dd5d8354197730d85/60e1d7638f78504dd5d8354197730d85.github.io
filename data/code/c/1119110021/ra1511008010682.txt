#include<stdio.h>
    int main(){
                  int n, i, sum = 0, elements[10];
                  int *pointr;
                  

                  //How many numbers to be added
                  scanf("%d", &n);
                  
                  for (i = 0; i < n; i++)
                  	scanf("%d", &elements[i]); 

                  // Assigning Starting Address to Pointer Variable
                  pointr = elements; 

                 for (i = 0; i < n; i++) 
                 {
                 	sum = sum + *pointr;
                 	++pointr;
                 }
    printf("%d", sum);
      return 0;
    }
