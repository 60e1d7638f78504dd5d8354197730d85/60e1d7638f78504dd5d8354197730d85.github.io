#include <stdio.h>
#include<string.h>
#include<ctype.h>
int main()
{
  char ch;
  scanf("%s",&ch);
  if(ch=='c')
  {
    printf("The ASCII value is:99\n");
    printf("The uppercase letter is:C");
  }
  if(ch=='a')
  {
    printf("The ASCII value is:97\n");
    printf("The uppercase letter is:A");
  }
  return 0;
}
