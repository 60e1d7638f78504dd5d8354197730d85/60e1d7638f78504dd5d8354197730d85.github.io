#include <stdio.h>
int main() 
{
   int n, i, arr1[15];
   int *pt;
 	 
	
	
  
   scanf("%d",&n);
   pt = &arr1[0]; 
   
   for(i=0;i<n;i++)
      {
	 
	  scanf("%d",pt);
	  pt++;
	  } 	
	
   pt = &arr1[n - 1];


   for (i = n; i > 0; i--) 
   {
      printf(" %d", *pt);
      pt--;
   }
return 0;
}
