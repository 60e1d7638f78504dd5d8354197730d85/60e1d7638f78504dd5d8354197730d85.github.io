int string_ln(char*);
 
int main() {
   char str[20];
   int length;
   
   
  
   scanf("%s",str);
   
   length = string_ln(str);

int string_ln(char*p)
{
  int count = 0;
   while (*p != '\0') {
      count++;
      p++;
   }
   return count;
}
