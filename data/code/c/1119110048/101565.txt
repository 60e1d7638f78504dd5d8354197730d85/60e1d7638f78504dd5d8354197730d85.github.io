#include <stdio.h>
int main()
{
  char str[100];
     scanf("%s",str);
  int i,vow=0,con=0;
   for(i=0;str[i]!='\0';i++)
    {
        if(str[i]=='a' ||str[i]=='e' ||str[i]=='i' ||str[i]=='o' ||str[i]=='u' ||str[i]=='A' ||str[i]=='E' ||str[i]=='I' ||str[i]=='O' ||str[i]=='U')
           vow++;
        else
            con++;
        //increase the pointer, to point next character
            }
 
    printf("%d %d",vow,con);

	return 0;
}
