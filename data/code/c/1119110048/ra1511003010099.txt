#include <stdio.h>
#include<string.h>
int main()
{
  char a[100];
  int vow=0,cons=0,i;
  scanf("%s",a);
  int len = strlen(a);
  for(i=0;i<len;i++){
  if(a[i] == 'a' || a[i] == 'e' || a[i] == 'i' || a[i] == 'o' || a[i] == 'u'){
  vow=vow+1;
  }else{
    cons=cons+1;
  }
  }
  printf("%d %d",vow,cons);
   return 0;
}
