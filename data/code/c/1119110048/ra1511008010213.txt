#include <stdio.h>
int main()
{
    char str[100];
    char *ptr;
    int  cntV,cntC;
   
    fgets(str,100,stdin);
    ptr=str;
     
    cntV=cntC=0;
    while(*ptr!='\0')
    {
        if(*ptr=='a' ||*ptr=='E' ||*ptr=='I' ||*ptr=='O' ||*ptr=='U' ||*ptr=='a' ||*ptr=='e' ||*ptr=='i' ||*ptr=='o' ||*ptr=='u')
            cntV++;
        else
            cntC++;
        //increase the pointer, to point next character
        ptr++;
    }
     
    printf(" %d %d",cntV,cntC);        
    return 0;
}
