#include<stdio.h>
 
void swap(float *num1, float *num2) {
  float temp;
   temp = *num1;
   *num1 = *num2;
   *num2 = temp;
}
 
int main() {
float num1, num2;
 
  
   scanf("%f",&num1);
   
   scanf("%f",&num2);
 
   swap(&num1,&num2);
 
   printf("%0.2f",num1);
   printf("\n%0.2f",num2);
 
   return 0;
}
