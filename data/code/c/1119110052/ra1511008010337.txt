#include <stdio.h>
int fact(int n);
int main()
{
  int n;
  scanf("%d",&n);
  if(n<0)
  {
    return 0;
  }
  else
  {
     int result=fact(n);
    printf("%d",result);
  }
    return 0;
}
int fact(int n)
{
    if (n==0||n==1)
    {
        return 1;
    }
    else
    {
        return(n*fact(n-1));
    }
}
