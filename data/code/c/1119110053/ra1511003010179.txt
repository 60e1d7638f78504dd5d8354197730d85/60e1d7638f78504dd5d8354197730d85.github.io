#include <stdio.h>
#include <malloc.h>
 
int main()
{
    int i, sum = 0;
    int *a;
    a = (int *) malloc(6 * sizeof(int));
    for (i = 0; i < 6; i++)
    {
        scanf("%d", a + i);
    }
    for (i = 0; i < 6; i++)
    {
        sum = sum + *(a + i);
    }
    printf("%d", sum);
	return 0;
}
