#include <stdio.h>

int main()
{
    int i, n;
    int arr[10];
    scanf("%d", &n);
    for(i = 0; i < n; ++i)
    {
       scanf("%d", &arr[i]);
         
    }
    for(i = 1; i < n; ++i)
    {
       // Change < to > if you want to find the smallest element
       if(arr[0] < arr[i])
           arr[0] = arr[i];
    }
    printf("%d", arr[0]);

    return 0;
}
