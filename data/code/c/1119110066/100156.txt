#include <stdio.h>
int main()
{
int numbers[10],NO_OF_ELEMENTS;
    int i, j;

    scanf("%d",&NO_OF_ELEMENTS);
    for(i = 0; i < NO_OF_ELEMENTS; i++)
    {
        scanf("%d",&numbers[i]);
    }

 
    /* Print each number in reverse order */
    for(j = NO_OF_ELEMENTS-1; j >= 0; j--)
    {
        printf("%d ", numbers[j]);
    }

	return 0;
}
