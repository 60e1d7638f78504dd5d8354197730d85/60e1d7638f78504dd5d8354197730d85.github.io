#include <stdio.h>
int main()
{
	int size,i;
  	int *ptr;
  	scanf("%d",&size);
  	int arr[size];
  	ptr=&arr[0];
  	for(i=0;i<size;i++)
    {
      scanf("%d",ptr);
  		ptr++;
    }
  	ptr=&arr[size-1];
  	for(i=size-1;i>=0;i--){
      printf("%d",*ptr);
      if(i>0)
        printf(" ");
      ptr--;
    }
	return 0;
}
