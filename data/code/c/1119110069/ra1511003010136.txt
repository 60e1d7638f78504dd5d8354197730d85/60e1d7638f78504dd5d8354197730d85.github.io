#include <stdio.h>
int main()
{ int ch,n,arr[100],sum=0,i;
  scanf("%d",&ch);
  if (ch>1000)
     printf("INVALID INPUT");
  else
  { while(ch--)
  { scanf("%d",&n);
    for (i=0;i<n;i++)
      scanf("%d",&arr[i]);
    for (i=0;i<n;i++)
     sum=sum+arr[i];
    printf("%d\n",sum);
    sum=0;
  }
  }
	return 0;
}
