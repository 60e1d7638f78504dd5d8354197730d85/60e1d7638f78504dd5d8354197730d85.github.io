#include <stdio.h>
int main()
{
	int n, t, i, a[10];
  	scanf("%d", &t);
  	if(t<1000)
    {
    	scanf("%d", &n);
      	for(i=0; i<n; i++)
        {
        	scanf("%d", &a[i]);
        }
      	for(i=0; i<n; i++)
        {
        	if(a[i]%3==0)
            {
            	printf("\n%d", a[i]);
            }
        }
    }
  	else
    {
    	printf("Invalid Input");
    }
	return 0;
}
