#include <stdio.h>
#include <malloc.h>
int main()
{
  int n,num;
  scanf("%d",&n);
  int *a,i;
  a=(int *)malloc(n*sizeof(int ));
  if(n<10000)
  {
  for(i=0;i<n;i++)  
  {
    scanf("%d",&num);
    {
      for(i=0;i<num;i++)
      {
        scanf("%d",a+i);
        if(*(a+i)%3==0)
          printf("%d\n",*(a+i));
      }
     
    }
  }
  }
  else
    printf("Invalid Input");
return 0;
}
