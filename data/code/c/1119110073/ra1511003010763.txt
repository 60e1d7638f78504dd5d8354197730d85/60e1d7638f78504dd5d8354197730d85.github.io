#include<stdio.h>
#include<string.h>
 
void del(char str[], char ch);
 
int main() {
   char str[1000];
   char ch;
 
   fgets(str, sizeof(str), stdin);
 
   scanf("%ch", &ch);
 
   del(str, ch);
return 0;}
 
void del(char str[], char ch) {
   int i, j = 0;
   int size;
   char ch1;
   char str1[1000];
 
   size = strlen(str);
 
   for (i = 0; i < size; i++) {
      if (str[i] != ch) {
         ch1 = str[i];
         str1[j] = ch1;
         j++;
      }
   }
   str1[j] = '\0';
 
   printf("\n%s", str1);
}
