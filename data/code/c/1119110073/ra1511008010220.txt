#include<stdio.h>
void str_replace(char * p, char search) {
char *c;
int i;
while( *p != '\0') {
if(*p == search) {
c = p;
i = 1;
while( *c != '\0') {
*c = *(c+1);
c = p + i;
i++;
}
}
p++;
}
}
int main()
{
char word[150],ch;
scanf("%[^\n]\n",word);
scanf("%c",&ch);
str_replace(word,ch);
printf("%s",word);
return 0;
}
