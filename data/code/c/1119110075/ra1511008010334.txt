#include<stdio.h>
#include<math.h>
#include<stdlib.h>
int main()
{
        int n,sum=0,page=0,temp,tempo;
        scanf("%d",&n);
        int i;
        for(i=0;i<n;i=i+1)
        {
 
            temp=(9*pow(10,i)*(i+1));
            tempo=9*pow(10,i);
            if(temp+sum<n)
                {sum+=temp;
                page+=tempo;}
            else
                break;
        }
        if((n-sum)%(i+1)==0)
            printf("\n%d",page+((n-sum)/(i+1)));
        else
        {
            printf("Impossible");
    }
return 0;
}
