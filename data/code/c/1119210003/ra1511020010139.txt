#include <stdio.h>
int main()
{
  int n, i, arr[15], num=0, j=0;
  scanf("%d", &n);
  for(i=0;i<n;i++)
    scanf("%d", &arr[i]);
  for(i=0;i<n;i++)
  {
    num=arr[i];
    if(num!=-999999990)
      for(j=i+1;j<n-1;j++)
      {
        if(num==arr[j])
          arr[j]=-999999990;
      }
  }
  j=0;
  for(i=0;i<n;i++)
  {
    if(arr[i]!=-999999990)
      j++;
  }
  printf("%d", j);
  return 0;
}
