#include <stdio.h>
int main()
{
  int n;
  scanf("%d",&n);
  int i,a[n],b[n],c=0;
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  for(i=0;i<n;i++)
  {
    scanf("%d",&b[i]);
  }
  for(i=0;i<n;i++)
  {
    if(a[i]==b[i])
      c++;
  }
  if(c==n)
    printf("yes");
  else
   printf("no"); 
	return 0;
}
