#include <stdio.h>
int main()
{
	int i=0,n,a[20];
  	scanf("%d",&n);
  	for(i=0;i<n;i++){
    	scanf("%d",&a[i]);
    }
      int mo=mode(a,n);
  if(mo==0)
    printf("none");
   else
  	printf("Mode=%d",mo);
	return 0;
}
int mode(int a[],int n) {
   int maxValue = 0, maxCount = 0, i, j;

   for (i = 0; i < n; ++i) {
      int count = 0;
      
      for (j = 0; j < n; ++j) {
         if (a[j] == a[i])
         ++count;
      }
      
      if (count > maxCount) {
         maxCount = count;
         maxValue = a[i];
      }
   }  
	if(maxCount==1){
    maxValue=0;
    }
   return maxValue;
}
