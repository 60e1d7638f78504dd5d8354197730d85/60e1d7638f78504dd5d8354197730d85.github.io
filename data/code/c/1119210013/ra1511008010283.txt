#include <stdio.h>
int main()
{
    int arr[20];
    int i, max, min, size,s;

    scanf("%d", &size);

    
    for(i=0; i<size; i++)
    {
        scanf("%d", &arr[i]);
    }


    max = arr[0];
    min = arr[0];

   
    for(i=1; i<size; i++)
    {
        if(arr[i] > max)
        {
            max = arr[i];
        }

        if(arr[i] < min)
        {
            min = arr[i];
        }
    }
    s=max-min;
    printf("%d",s);

    return 0;
}
