#include <stdio.h>
#include <stdlib.h>
 
int main()
{
  int t,n,w[100],i,min,k,l;
  scanf("%d",&t);
		while(t--)
		{
												min=100000;
												k=0;
												scanf("%d",&n);
												i=0;
												l=n;
												while(n--)
												{
																scanf("%d",&w[i]);
																if(w[i]<min)
																min=w[i];
																i++;
												}
												k=0;
												for(i=0;i<l;i++)
												k+=w[i]-min;
												printf("%d\n",k);
		}
  return 0;
}
