#include<stdio.h>
int main()
{
    int t,n,w[100],ans,tmp,i;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(i=0;i<n;i++)
        {
            scanf("%d",&w[i]);
            if(i>0 && w[i]>w[i-1])
            {
                tmp=w[i-1];
                w[i-1]=w[i];
                w[i]=tmp;
            }
        }
        ans=0;
        for(i=0;i<n-1;i++)
            ans+=(w[i]-w[n-1]);
        printf("%d\n",ans);
    }
    return 0;
}
