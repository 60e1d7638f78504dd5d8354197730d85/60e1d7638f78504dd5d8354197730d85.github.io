#include <stdio.h>
int f[9],x,n,i,j,q,ans,a[9];
char s[99]; 
int main() 
	{
		f[0] = 1;
		for (i=1;i<=6;i++) 
			f[i] = f[i-1]*i;
		scanf("%d %d",&x,&n);
		ans = 0;
		for (q=0;q<n;q++) 
		{
			scanf("%s",s);
			for (i=0;i<9;i++) 
				a[i] = 0;
			for (i=0;i<54;i++)
				if (s[i] == '0')
				{
					if (i < 36) 
					a[i/4]++;
					else 
					a[8-(i-36)/2]++;
				}
			for (i=0;i<9;i++)
			if (a[i] >= x)
				ans += f[a[i]]/f[x]/f[a[i]-x];
		}
		printf("%d\n",ans);
		return 0;
	}
