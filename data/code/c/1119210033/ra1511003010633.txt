	int blank[100]={0};
	char arr[55];
	scanf("%d %d",&x,&n);
	p=n;
	while(n--)
	{
		scanf("%s",arr);
		for(i=0;i<9;i++,j++)
		{
			for(k=i*6;k<(i*6+6);k++)
			blank[j]+=(arr[cars[k]-1]-48);		
			blank[j]=6-blank[j];
			//printf("%d ",blank[j]);
		}
	}
	for(i=0;i<p*9;i++)
	ans+=nCr(blank[i],x);
	printf("%d\n",ans);
	return 0;
}
 
int fact(int x)
{
	int f=1;
	while(x>1)
	{
		f*=x;
		x--;	
	}
	return f;
}
 
int nCr(int x,int y)
{
	if(x<y)
	return 0;
	return(fact(x)/(fact(x-y)*fact(y)));
}
