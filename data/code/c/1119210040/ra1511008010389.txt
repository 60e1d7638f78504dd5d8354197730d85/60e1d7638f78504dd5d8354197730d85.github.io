    int inp()
    {
        int n=0,c=getchar_unlocked();
        while(c<'0'||c>'9')
        c=getchar_unlocked();
        while(c>='0'&&c<='9')
        n=(n<<3)+(n<<1)+c-'0',c=getchar_unlocked();
        return n;
    }
