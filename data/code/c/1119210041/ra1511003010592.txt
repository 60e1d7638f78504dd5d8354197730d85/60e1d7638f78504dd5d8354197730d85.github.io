#include<stdio.h>
#include<string.h>
#define mod 1000000007
#define LL long long int
int main()
{
	int i,j,n;
	int ans=0,prefix=0,x;
	char s[100005];
	scanf("%s",s);
	ans+=(LL)('Z'-s[0]);
	prefix=(LL)('Z'-s[0]);
//	printf("$%d\n",ans);
	for(i=1;i<strlen(s);++i)
	{
		x=(LL)('Z'-s[i]);
		ans=(ans+((prefix+1ll)*x))%mod;
	//	printf("#%d\n",ans);
		prefix=((1ll*26*prefix)+x)%mod;
	//	printf("$%d\n",prefix);
	}
	printf("%d\n",ans);
	return 0;
}
