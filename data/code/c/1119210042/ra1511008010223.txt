#include<stdio.h>
#include<string.h>
char s[10001];
int pow2[1001];
int pow8[1001];
int comb[1001][1001];
int ac[14]={4,7,44,47,74,77,444,447,474,477,744,747,774,777};
int b=1000000007;
int cf(int k,int w,int cnt)
{
int v,q=0,i,ret=0;;
if(k>4) q++;
if(k>7) q++;
if(q>0)
{
for(i=0;i<14;i++)
if((w+cnt)>=ac[i])
{
if(ac[i]>cnt)
{
v=comb[w-1][ac[i]-cnt-1];
v=((long long)v*q)%b;
v=(v*(long long)pow2[ac[i]-cnt-1])%b;
v=(v*(long long)pow8[w+cnt-ac[i]])%b;
ret=(ret+v)%b;
}
}
else break;
k=k-q;
}
for(i=0;i<14;i++)
if((w+cnt)>ac[i])
{
if(ac[i]>=cnt)
{
v=comb[w-1][ac[i]-cnt];
v=((long long)v*k)%b;
v=(v*(long long)pow2[ac[i]-cnt])%b;
v=(v*(long long)pow8[w+cnt-ac[i]-1])%b;
ret=(ret+v)%b;
}
}
else
break;
return ret;
}
int main()
{
int cnt,len,i,j,v,t,k,tc;
pow2[0]=1;pow8[0]=1;
for(i=1;i<1001;i++)
{
pow2[i]=((long long)pow2[i-1]*2)%b;
pow8[i]=((long long)pow8[i-1]*8)%b;
}
for ( i = 0; i <1001; i++) {
comb[i][0] = 1;
for (j = 1; j <= i; j++)
comb[i][j] = (comb[i-1][j] + comb[i-1][j-1]) %b;
}
scanf("%d",&t);
while(t--)
{
tc=2;
v=0;
while(tc--)
{
k=v;
scanf("%s",s);
len=strlen(s);
cnt=0;
v=0;
for(i=0;i<len;i++)
{
v=(v+cf(s[i]-'0',len-i,cnt))%b;
if(s[i]=='7'||s[i]=='4') cnt++;
}
}
for(i=0;i<14;i++)
if(cnt==ac[i]) v=(v+1)%b;
printf("%d\n",(b+v-k)%b);
}
return 0;
}
