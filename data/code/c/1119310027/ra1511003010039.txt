#include <stdio.h>
  #include<math.h>
   
      int main()
   
      {
   
      long long int t,n,m;
   
      scanf("%lld",&t);
   
      while(t--)
   
      {
   
      scanf("%lld",&n);
  
      m=sqrt(n);
  
      while(1)
  
      {
  
      if(!(n%m)) break;
  
      m--;
  
      }
  
      printf("%lld\n",(n/m)-m);
  
      }
  
      return 0;
  
      }
