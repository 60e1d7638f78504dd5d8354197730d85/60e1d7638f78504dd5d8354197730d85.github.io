#include<stdio.h>
//#include<conio.h>
#define ll long long 
int main()
{
    ll int n;
    scanf("%lld",&n);
    ll int i,a[n],j,ans=0;
    for(i=0;i<n;i++) {
                     scanf("%lld",&a[i]);
    }
    for(i=0;i<n;i++) {
                       if(a[i]>=1) {
                     for(j=i+1;j<n;j++){
                     if(a[i]==1&&a[j]%2!=0)
                     ans++;
                     else                   
                     ans+=a[i]&a[j];
                     }
                     }
    }
    printf("%lld",ans);
    //getch();
    return 0;
}
