#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>
#include <limits.h>
 
int main()
{
int fall, a, b, i, n;
int A[100], B[100];
for(scanf("%d",&fall); fall--;)
	{
	scanf("%d",&n);
	for(i=0; i<n; i++)
		scanf("%d %d",&A[i],&B[i]);
	a=b=0;
	for(i=0; i<n; i++)
		{
		a+=A[i];
		b+=B[i];
		}
	printf("%d\n",a-b);
	}
return 0;
}
