#include<stdio.h>
 
int scan() {
	int acc = 0;
	register char buf;
	buf = getchar_unlocked();
	while(buf!=10 && buf<='9' && buf>='0'){ 
		acc = acc*10 + buf-'0';
		buf = getchar_unlocked();
	}
	return(acc);
}
 
main() {
	int T = scan();
	
	while(T-- > 0) {
		int C = scan();
		int xor = 0;
 
		while(C-- > 0) {
			int n = scan();
			int m = scan();
			int temp = (m+n-2)%3;
                        temp != !temp;
                        
			xor ^= temp;
		}
	
		if(xor) 
			printf("MasterChef\n");
		else
			printf("Football\n");
	}
 
	return 0;
}
