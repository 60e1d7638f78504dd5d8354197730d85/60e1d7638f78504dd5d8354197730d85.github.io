#include<stdio.h>
#define max 1000000000

long long dp[10][10];
long long res[10];
int nofdig(int n)
{
	int sum=0;
	while(n>0)
	{
		if(n%10==4||n%10==7)sum++;
		n/=10;
	}
	return sum;
}
void count(int n)
{
	int i;
	int in=11;
	int temp[11];
	int digit=0;
	while(n>0)
	{
		in--;
		digit++;
		temp[in]=n%10;
		n=n/10;
	}
	int j,k,l, m;
	int initial=0;
	int start;
	for(i=0;i<=9;i++)res[i]=0;
	for(i=0;i<=9;i++)for(j=1;j<digit;j++)res[i]+=dp[digit-j][i];
	for(i=in,j=1;i<=10;i++,j++)
	{
		if(j==1)start=1;
		else start=0;
		for(k=start;k<temp[i];k++)
		{
			for(m=digit-j;m>=0;m--)
			{
				for(l=0;l<=m;l++)
				{
					if(k==4||k==7)res[initial+1+l]+=dp[m][l];
					else res[initial+l]+=dp[m][l];
				}
			}
		}
		if(temp[i]==4||temp[i]==7)initial++;
	}
	res[initial]++;
 
	return ;
 
}
