# include<stdio.h>
# include<string.h>
 
 
int main()
{
    char a[5];
    char ip[105][105];
    int t,n,k,i,j,m,flag,temp,l;
    scanf("%d",&t);
    n=t;
    for(k=0;k<t;k++)
    {
                    for(i=0;i<n;i++)
                    {
                    scanf("%s",a);
                    m=strlen(a);
                    if(m==2)                
                    ip[k][i]='0';
                    else
                    ip[k][i]='1';
                    
                    }
                    ip[k][n]='\0';
    }
  
    for(i=0;i<n;i++)
    {
     flag=0;
     for(j=0;j<n;j++)
                     {
                     
                     //printf("%d %s %s\n",temp,ip[i],ip[j]);
                   if(j==i && ip[i][i]=='0')
                   break;
                   
                     temp=strcmp(ip[i],ip[j]);
                     if(temp!=0)
                     {
                     if(ip[i][j]!='0')
                     break;           
                     }
                     else
                     {
                         if(ip[j][j]=='0')
                         break;
                         
                     }
                     
     }
     
     if(j==n)
      {
      flag=1;
      
      for(l=0;l<n;l++)
      {
                      
      if(ip[i][l]=='0')
      printf("NO\n");
      else
      printf("YES\n");                
      }        
      break;
      }
     
     }
         
      if(!flag)
      {
      for(i=0;i<n;i++)
      printf("NO\n");        
      
      }             
                    
    
    
  //  getch();
return 0;
}
