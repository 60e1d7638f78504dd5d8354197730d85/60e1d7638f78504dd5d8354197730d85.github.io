#include <stdio.h>
int main()
{
   int a,b;
   
  	scanf("%d",&a);
  	scanf("%d",&b);
  	printf("Before Swapping\n");
  	printf("x=%d\n",a);
  	printf("y=%d\n",b);
  
  	int temp;
  	temp=a;
    a=b;
  	b=temp;
     
  	printf("After Swapping\n");
  	printf("%d\n",a);
  	printf("%d\n",b);

	return 0;
}
