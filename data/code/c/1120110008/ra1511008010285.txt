#include <stdio.h>
int main()
{
    int x,y,temp;
    scanf("%d",&x);
    scanf("%d",&y);
    printf("Before Swapping\n");
    printf("x=%d\n",x);
    printf("y=%d\n",y);
  
    temp=x;
    x=y;
    y=temp;  
    printf("After Swapping\n");
    printf("%d\n",x);
    printf("%d\n",y);
    
    

      
	return 0;
}
