#include <stdio.h>
int main()
{
    int yr;

  scanf ("%d", &yr);

  if (yr%4 == 0) {

      if(yr%100 == 0) {
      
          if(yr%400 == 0)
             printf("\n%d is a leap year",yr);
          else
             printf("\n %d is not a leap year",yr);
      }

      else {
             printf ("\n %d is a leap year",yr);
      }
  }
  else
      printf("\n %d is not a leap year",yr);
  
return 0;
}
