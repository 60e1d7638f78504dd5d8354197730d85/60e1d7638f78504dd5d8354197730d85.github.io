#include <stdio.h>
int main()
{
    int n, arr[30], i, eve=0, odd=0;
    scanf("%d", &n);
    for(i=0; i<n; ++i)
      scanf("%d", &arr[i]);
    for(i=0; i<n; ++i)
    {
        if(arr[i]%2==0)
          eve+=arr[i];
        else
          odd+=arr[i];
    }
    printf("even=%d\nodd=%d", eve, odd);
	return 0;
}
