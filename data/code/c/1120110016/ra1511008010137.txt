#include <stdio.h>
int main()
{
    int array[100], i, num;
    int sum1=0,sum2=0;
    scanf("%d", &num);
    for (i = 0; i < num; i++)
    {
        scanf("%d", &array[i]);
    }
    for (i = 0; i < num; i++)
    {
        if (array[i] % 2 == 0)
        {
            sum1+=array[i];
        }
    }

    for (i = 0; i < num; i++)
    {
        if (array[i] % 2 != 0)
        {
            sum2+=array[i];
        }
    }
  printf("even=%d\nodd=%d",sum1,sum2);
	return 0;
}
