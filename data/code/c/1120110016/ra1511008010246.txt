 
#include <stdio.h>
 
int main()
{
    int i,j, num, odd_sum = 0, even_sum = 0;
 
  //  printf("Enter the value of num\n");
    scanf("%d", &num);
    for (i = 1; i <= num; i++)
    {scanf("%d",&j);
        if (j % 2 == 0)
            even_sum = even_sum + j;
        else
            odd_sum = odd_sum + j;
    }
    
    printf("even=%d\n", even_sum);
   printf("odd=%d\n", odd_sum);
  return 0;
}
