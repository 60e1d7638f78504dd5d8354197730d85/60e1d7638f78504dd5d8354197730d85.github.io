#include <stdio.h>
int main()
{
  int n,i;
  int sum=0,sum1=0;
  int a[100];
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  for(i=0;i<n;i++)
  {
    if(a[i]%2==0)
    {
      sum=sum+a[i];
    }
    else
    {
      sum1=sum1+a[i];
    }
  }
  printf("even=%d\n",sum);
  printf("odd=%d",sum1);

	return 0;
}
