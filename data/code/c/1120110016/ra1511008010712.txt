#include<stdio.h>
int main()
{
int num[20];
int i , esum , osum , N;           
/* esum stands for Sum of Even Numbers and osum stands for Sum of Odd Numbers */
esum = 0;
osum = 0;
scanf("%d" , &N);

for(i=0 ; i<N ; i++)
{
scanf("%d" , &num[i]);
}

for(i=0 ; i<N ; i++)
{
if((num[i] %2) ==0)
{
esum += num[i];
}
else
{
osum += num[i];
}
}                                                                  

printf("even=%d\n" , esum);
printf("odd=%d\n" , osum);
return 0;
}
