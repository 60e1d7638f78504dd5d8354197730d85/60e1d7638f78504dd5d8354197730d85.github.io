#include <stdio.h>
int main()
{
int n,m,i,j,a[100][100];
  scanf("%d%d",&n,&m);
  for(i=0;i<n;i++)
  {
    for(j=0;j<m;j++)
    {
      scanf("%d",&a[i][j]);
    }
  }
  
  for(i=0;i<n;i++)
  {
    for(j=0;j<m;j++)
    {
      if(j>i)
      {
        if(a[i][j]!=0)
        {
          printf("no");
          return 0;
        }
      }
    }
  }
  printf("yes");
	return 0;
}
