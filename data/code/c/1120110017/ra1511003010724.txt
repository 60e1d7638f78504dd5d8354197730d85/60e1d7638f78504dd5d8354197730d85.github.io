
#include <stdio.h>
int main()
{
  int i,j,m,n,a[10][10],count=0;
  scanf("%d %d",&m,&n);
 
  for(i=1;i<=m;i++){
    for(j=1;j<=n;j++)
    {
      scanf("%d",&a[i][j]);}}
  
  for(i=1;i<m;i++)
  {
    for(j=1;j<=n;j++)
    {
      if(j>i&&a[i][j]!=0)
      {
      count=count+1;
      }
    }}
  
  if(count==0)
  {
  printf("yes");
  }
  
  else
  {
  printf("no");
  }

	return 0;
}
