#include <stdio.h>
int main()
{
  int i,j,array[100][100];
    int row, col, isLower;
  	scanf("%d%d",&i,&j);
    for(row=0; row<i; row++)
    {
        for(col=0; col<j; col++)
        {
            scanf("%d", &array[row][col]);
        }
    }
    isLower = 1;
    for(row=0; row<i; row++)
    {
        for(col=0; col<j; col++)
        {
          
            if(col>row && array[row][col]!=0)
            {
                isLower = 0;
              	break;
              	
            }
        }
    }

    if(isLower == 1)
    {
        printf("yes");

        
    }
    else
    {
        printf("no");
    }

	return 0;
}
