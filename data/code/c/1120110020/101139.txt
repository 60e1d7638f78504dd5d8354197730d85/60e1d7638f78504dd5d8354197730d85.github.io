#include <stdio.h>
int main()
{
     int n, c, d, num = 1;
     scanf("%d",&n);
     for ( d = 1 ; d <= n ; d++ )
      {
          num = d;
          for ( c = 1 ; c <= d ; c++ )
          {
              printf("%d ", num);
              num++;
          }
          num-=2;
          for ( c = 1 ; c < d ; c++)
          {
              printf("%d ", num);
              num--;
          }
          printf("\n");
 
      }
 

	return 0;
}
