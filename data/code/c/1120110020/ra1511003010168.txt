
#include <stdio.h>
int main()
{
  int n;
  scanf("%d",&n);
  int i,j,a;
  for(i=1;i<=n;i++)
  {
    j=i;
    while(j<=(2*i)-1)
    {
      printf("%d ",j);
      j++;
    }
    if(i==1)
    {
      printf("\n");
      continue;
    }        
    j=j-2;
    while(j>=i)
    {
      printf("%d ",j);
      j--;
    }
    printf("\n");
  }
      

	return 0;
}
