#include <stdio.h>

struct Distance
{
    int feet;
    float inch;
} ;
int main()
{
  
    struct Distance d1,d2,sumOfDistances;
    scanf("%d", &d1.feet);
    
    scanf("%f", &d1.inch);

   
    
    scanf("%d", &d2.feet);
    
    scanf("%f", &d2.inch);


    sumOfDistances.feet=d1.feet+d2.feet;
    sumOfDistances.inch = d1.inch+d2.inch;

   

   
    printf("Sum of distances=%d feet and %.2f inches",sumOfDistances.feet, sumOfDistances.inch);
    return 0;
}
