#include<stdio.h> 
int main() 
{ 
  char Str[]="123456789"; 
  int n,i,j,k,l; 
  scanf("%d",&n);
  for(i=0;i<n;i++) 
  { 
    for(k=n-1;k>i;k--)
    { 
      //printf(" "); 
    } 
    for(j=0,l=i;j<=i;j++,l++) 
    { 
      printf("%c ",Str[l]);
    } 
    l=l-2; 
    for(k=0;k<i;k++,l--) 
      printf("%c ",Str[l]); 
    printf("\n"); 
  }
return 0;
}
