#include<stdio.h>
 
int main()
{
 int Size, i, a[10];
 int Even_Sum = 0, Odd_Sum = 0;
 

 scanf("%d", &Size);
 
 
 for(i = 0; i < Size; i++)
 {
      scanf("%d", &a[i]);
 }
  
 for(i = 0; i < Size; i ++)
 {
   if(a[i] % 2 == 0)
   {
 	Even_Sum = Even_Sum + a[i];
   }
   else
   {
 	Odd_Sum = Odd_Sum + a[i];
   }
 }
  
 printf("even=%d\n", Even_Sum);
 printf("odd=%d", Odd_Sum);
 return 0;
}
