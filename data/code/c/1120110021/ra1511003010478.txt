#include <stdio.h>
int main()
{
    int i,num,a[100],odd_sum = 0, even_sum = 0;
 
    
    scanf("%d", &num);
    for (i = 0; i <num; i++)
    {
      scanf("%d",&a[i]);
      if(a[i]%2==0)
        even_sum=even_sum+a[i];
      else
        odd_sum=odd_sum+a[i];
    }
  
    printf("even=%d\n", even_sum);
  	printf("odd=%d\n", odd_sum);
    
  return 0;
}
