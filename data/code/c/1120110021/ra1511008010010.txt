#include <stdio.h>
 
int main()
{
    int array[100], i, num,even=0,odd=0;
 
    scanf("%d", &num);
    for (i = 0; i < num; i++)
    {
        scanf("%d", &array[i]);
    }
    for (i = 0; i < num; i++)
    {
        if (array[i] % 2 == 0)
        {
            even+=array[i];
        }
    }
    for (i = 0; i < num; i++)
    {
        if (array[i] % 2 != 0)
        {
            odd+=array[i];
        }
    }
  printf("even=%d\n",even);
  printf("odd=%d",odd);
return 0;
}
