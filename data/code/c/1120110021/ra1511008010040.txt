#include <stdio.h>
int main()
{
  int a[20],i,e=0,o=0,n;
  scanf("%d",&n);
     for(i=0;i<n;i++)
          scanf("%d",&a[i]);
     for(i=0;i<n;i++)
     {
          if(a[i]%2==0)
               e=e+a[i];
     }
  for(i=0;i<n;i++)
  {
          if(a[i]%2!=0)
               o=o+a[i];
  }
       
     printf("even=%d\n",e);
     printf("odd=%d",o);
     return 0;
}
