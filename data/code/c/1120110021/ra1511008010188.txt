#include <stdio.h>
int main()
{
  int a[10],i,n,ev=0,od=0;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d ",&a[i]);
  }
  for(i=0;i<n;i++)
  {
    if(a[i]%2==0)
    {
      ev=ev+a[i];
      
    }
   
    else
    {
      od=od+a[i];
    }
  }
  printf("even=%d\n",ev);
  printf("odd=%d",od);
  
    
	return 0;
}
