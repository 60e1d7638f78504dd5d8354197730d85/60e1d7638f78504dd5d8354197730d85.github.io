#include <stdio.h>
int main()
{
    int a[100],i,n,sum1=0,sum2=0;
   
    scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  
  for(i=0;i<n;i++)
  {
  if(a[i]%2==0)
  {
    sum1=sum1+a[i];
  
  
  }
  }
  
  for(i=0;i<n;i++)
  {
  if(a[i]%2==1)
  {
    sum2=sum2+a[i];
   
    
  }
  }
  printf("even=%d\n",sum1);
  printf("odd=%d\n",sum2);
  
  
  
	return 0;
}
