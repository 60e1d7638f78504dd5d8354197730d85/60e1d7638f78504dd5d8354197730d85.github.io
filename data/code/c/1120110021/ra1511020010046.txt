#include <stdio.h>
int main()
{
int n,arr[100],i,sume=0,sumo=0;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&arr[i]);
  }
  for(i=0;i<n;i++)
  {
  if(arr[i]%2==0)
  {
    sume=sume+arr[i];
  }
  else
    sumo=sumo+arr[i];
  }
  printf("even=%d",sume);
  printf("\nodd=%d",sumo);
	return 0;
}
