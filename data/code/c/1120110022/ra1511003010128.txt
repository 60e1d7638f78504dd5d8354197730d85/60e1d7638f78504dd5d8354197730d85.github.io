#include <stdio.h>
int getInvCount(int arr[], int n)
{
  int inv_count = 0,i,j;
  for (i = 0; i < n - 1; i++)
    for (j = i+1; j < n; j++)
      if (arr[i] > arr[j])
        inv_count++;
 
  return inv_count;
}
 
/* Driver progra to test above functions */
int main(int argv, char** args)
{
  int n,i;
  scanf("%d",&n);
  int arr[n];
  for(i=0;i<n;i++)
    scanf("%d",&arr[i]);
  printf("%d", getInvCount(arr, n));
  return 0;
}
