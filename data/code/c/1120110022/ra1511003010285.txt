#include <stdio.h>
int getInvCount(int arr[],int n)
{
  int inv_count=0,i,j;
  for(i=0;i<n;i++)
    for(j=i+1;j<n;j++)
      if(arr[i]>arr[j])
        inv_count++;
  return inv_count;
}
  
int main()
{
  int n,arr[50],i;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&arr[i]);
  }
  printf("%d",getInvCount(arr, n));
  	return 0;
}
