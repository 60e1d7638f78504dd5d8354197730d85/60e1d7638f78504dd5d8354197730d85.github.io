#include <stdio.h>
int main()
{
int a[100],n,i,j,temp,sum=0;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  for(i=0;i<n;i++)
  {
    for(j=i+1;j<n;j++)
    {
      if(a[i]>a[j])
      {
        temp=a[i];
        a[i]=a[j];
        a[j]=temp;
        sum++;
      }
    }
  }
  printf("%d",sum);
	return 0;
}
