#include <stdio.h>
int main()
{
  	int a[30],n,i,j,c=0; 
  	scanf("%d",&n);
  	for(i=0;i<n;i++) 
      scanf("%d",&a[i]); 
  for(i=0;i<n-1;i++) 
    for(j=0;j<n-i-1;j++) 
    { if(a[j]>a[j+1])
    { int temp=a[j+1]; 
        a[j+1]=a[j]; 
     	a[j]=temp;  
     c++;
        }
      
    } 
 
  printf("%d",c);
	return 0;
}
