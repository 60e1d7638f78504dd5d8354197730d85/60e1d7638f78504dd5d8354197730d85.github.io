#include <stdio.h>
int main()
{
	int n,count=0,temp;
  	scanf("%d",&n);
  	int a[n],i,j;
  	for(i=0;i<n;i++)
    {
      scanf("%d",&a[i]);
    }
  	for(i=0;i<n;i++)
    {
      for(j=i;j<n;j++)
      {
        if(a[i]>a[j])
           {
             count++;
             temp=a[i];
             a[i]=a[j];
             a[j]=temp;
           }
      }
    }
           printf("%d\n",count);
  
	return 0;
}
