#include <stdio.h>
int getInvCount(int arr[], int n)
{
  int inv_count = 0;
  int i,j;
  for (i = 0; i < n - 1; i++)
    for (j = i+1; j < n; j++)
      if (arr[i] > arr[j])
        inv_count++;
  return inv_count;
}
int main(int argv, char** args)
{
  int t;
  scanf("%d",&t);
  int arr[t];
  int x;
  for(x=0;x<t;x++)scanf("%d",&arr[x]);
  int n = sizeof(arr)/sizeof(arr[0]);
  printf("%d\n", getInvCount(arr, n));
  return 0;
}
