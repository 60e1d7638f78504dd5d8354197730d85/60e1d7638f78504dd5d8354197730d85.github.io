void magic(int m[3][3])
{int i,j,r[3],c[3],d=0,flag=0;
 for(i=0;i<3;i++)
 {r[i]=m[i][0]+m[i][1]+m[i][2];
  c[i]=m[0][i]+m[1][i]+m[2][i];
  d+=m[i][i];
 }
 for(i=0;i<3;i++)
 {if(d!=r[i])
  {flag=1;
   break;
  }
  for(j=0;j<3;j++)
  {if(r[i]!=c[j])
   {flag=1;
    break;
   }
  }
 }
 for(i=0;i<3;i++)
 {for(j=0;j<3;j++)
      printf("%d ",m[i][j]);
  printf("\n");
 }
 if(flag==1)
   printf("Not a Magic Square");
 else
   printf("Magic Square");
}
