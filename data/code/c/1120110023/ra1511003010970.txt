#include <stdio.h>
int main()
{
  int arr[3][3],i,j,arr1[8],sum=0;
  for(i=0;i<3;i++) {
    for(j=0;j<3;j++) {
      scanf("%d",&arr[i][j]);
      sum+=arr[i][j];
      //printf("%d ",arr[i][j]);
    }
    arr1[i]=sum;
    sum=0;
    //printf("\n");
  }
  for(i=0;i<3;i++) {
    for(j=0;j<3;j++) 
      sum+=arr[j][i];
    arr1[i+3]=sum;
    sum=0;
  }
  arr1[6] = arr[0][0] + arr[1][1] + arr[2][2];
  arr1[7] = arr[0][2] + arr[1][1] + arr[2][0];
  int fg=0;
  for(i=0;i<=6;i++){
    if(arr1[i]==arr1[i+1])
      fg++;
 }
  if(fg==7)
    printf("Magic Square\n");
  else
    printf("Not a Magic Square\n");
    
	return 0;
}
