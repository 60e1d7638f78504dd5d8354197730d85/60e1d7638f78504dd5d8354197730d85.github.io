#include <stdio.h>

struct date
{
  int a,b,year;
}da;
 
int isLeapYear(struct date da)
{
   if ( da.year%400 == 0) 
    printf("%d is a leap year\n", da.year);
  else if ( da.year%100 == 0) 
    printf("%d is not a leap year\n", da.year);
  else if ( da.year%4 == 0 ) 
    printf("%d is a leap year\n", da.year);
  else
    printf("%d is not a leap year\n", da.year);  
}
 
int main()
{  
  scanf("%d",&da.a);

  scanf("%d",&da.year);
 
    isLeapYear(da);
