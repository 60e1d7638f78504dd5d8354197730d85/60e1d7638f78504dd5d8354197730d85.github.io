#include <stdio.h>

struct year{};
int main()
{
    int date,month,year;

    
    scanf("%d %d %d",&date,&month,&year);

    if(year%4 == 0)
    {
        if( year%100 == 0)
        {
            // year is divisible by 400, hence the year is a leap year
            if ( year%400 == 0)
                printf("%d is a leap year", year);
            else
                printf("%d is not a leap year", year);
        }
        else
            printf("%d is a leap year", year );
    }
    else
        printf("%d is not a leap year", year);
    
    return 0;
}
