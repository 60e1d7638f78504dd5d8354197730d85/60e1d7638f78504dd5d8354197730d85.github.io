#include <stdio.h>
int main()
{
  	char name[10];
  	float price;
  	int quant;
  	scanf("%s", name);
  	scanf("%f", &price);
  	scanf("%d", &quant);
  	printf("Product Details\n");
  	printf("%s\n", name);
  	printf("%.2f\n", price);
  	printf("%d\n", quant);
  	printf("Bill:%.2f\n", price*quant);
	return 0;
}
