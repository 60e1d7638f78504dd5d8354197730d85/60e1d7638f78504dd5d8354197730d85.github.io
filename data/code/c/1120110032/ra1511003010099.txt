#include <stdio.h>
 
#define IS_DELIMITER(_c)			\
  ((_c) == ' ' || (_c) == '\n' || (_c) == '\t')	\
 
#define READ_INT(_x)					\
  {							\
    int _n_flag = 0;					\
    int _ch, _tmp = 0;					\
    while ((_ch = getchar_unlocked()) &&		\
	   IS_DELIMITER(_ch))				\
      ;							\
    if (_ch == '-')					\
      {							\
	_n_flag = 1;					\
	_ch = getchar_unlocked();			\
      }							\
    while (48 <= _ch && _ch <= 57)			\
      {							\
	_tmp = (_tmp << 3) + (_tmp << 1) + _ch - 48;	\
	_ch = getchar_unlocked();			\
      }							\
    (_x) = ((_n_flag) ? -1 : 1) * _tmp;			\
  }							\
    
#define WRITE_INT(_x)				\
  {						\
    const int _SIZE = 20;			\
    char _str[_SIZE];				\
    int _i = 0, _tmp = (_x);			\
    if (_tmp < 0)				\
      {						\
	putchar_unlocked('-');			\
	_tmp = - _tmp;				\
      }						\
    do						\
      {						\
	*(_str + _i++) = _tmp % 10 + 48;	\
	_tmp /= 10;				\
      } while (_tmp != 0);			\
    for (_i--; _i >= 0; _i--)			\
      putchar_unlocked(*(_str + _i));		\
  }						\
 
#define COUNT_4(x, count)				\
  {							\
    register int _cnt, _tmp;				\
    for (_cnt = 0, _tmp = (x); _tmp != 0; _tmp /= 10)	\
      if (_tmp % 10 == 4)				\
	_cnt++;						\
    (count) = _cnt;					\
  }							\
    
int main(void)
{
  int count, n, t;
 
  READ_INT(t);
  while (t-- > 0)
    {
      READ_INT(n);
      COUNT_4(n, count);
      WRITE_INT(count);
      putchar_unlocked('\n');	
    }
 
  return 0;
}
