#include <stdio.h>
int main()
{
	int n,i;
  	scanf("%d",&n);
  	int a[n];
  	for(i=0;i<n;i++)
      scanf("%d",&a[i]);
  	int max=a[0];
  	for(i=1;i<n;i++)
      if(a[i]>max)
		max=a[i];
  	int min=a[0];
  	for(i=1;i<n;i++)
      if(a[i]<min)
		min=a[i];
  	printf("\n%d",max);
  	printf("\n%d",min);
    return 0;
}
