#include <stdio.h>
int main()
{
int n;
  int a[20],*p;
  scanf("%d",&n);
  int i;
  for(i=0;i<n;i++)
    scanf("%d",&a[i]);
  int max=0;
  int min=999;
  p=a;
  for(i=0;i<n;i++)
  {
    if(*(p+i)>max)
      max=*(p+i);
    if(*(p+i)<min)
      min=*(p+i);
  }
  printf("%d\n%d",max,min);
	return 0;
}
