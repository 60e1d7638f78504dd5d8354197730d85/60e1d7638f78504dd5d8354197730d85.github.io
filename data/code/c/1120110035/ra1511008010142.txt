#include <stdio.h>
int main()
{
int n,i;
  scanf("%d",&n);
  int arr[n];
  for(i=0;i<n;i++)
    scanf("%d",&arr[i]);
  int max=arr[0];
  for(i=1;i<n;i++)
     if(arr[i]>max)
      max=arr[i];
  int min=arr[0];
  for(i=1;i<n;i++)
     if(arr[i]<min)
       min=arr[i];
  printf("%d\n",max);
  printf("%d",min);
	return 0;
}
