#include <stdio.h>
int main()
{
int i,n,max=0,min=99;
  scanf("%d",&n);
  int a[n];
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  for(i=0;i<n;i++)
  {
    if(a[i]>max)
      max=a[i];
  }
  for(i=0;i<n;i++)
  {
    if(a[i]<min)
      min=a[i];
  }
  printf("%d",max);
  printf("\n%d",min);
	return 0;
}
