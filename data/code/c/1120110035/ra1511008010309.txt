#include <stdio.h>
int main()
{
int n,a[10],i;
  int large,small;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d",&a[i]);
  }
  large=small=a[0];
  for(i=0;i<n;i++)
  {
    if(a[i]>large)
    large=a[i];
    if(a[i]<small)
      small=a[i];
  }
  printf("%d\n",large);
  printf("%d",small);
	return 0;
}
