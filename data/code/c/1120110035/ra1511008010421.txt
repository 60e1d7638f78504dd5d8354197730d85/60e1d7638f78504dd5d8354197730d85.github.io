#include<stdio.h>
int main(){
  int a[50],size,i,big,small;

  scanf("%d",&size);
  for(i=0;i<size;i++)
      scanf("%d ",&a[i]);

  big=a[0];
  for(i=1;i<size;i++){
      if(big<a[i])
           big=a[i];
  }
  printf("%d\n",big);
 
  small=a[0];
  for(i=1;i<size;i++){
      if(small>a[i])
           small=a[i];
  }
  printf("%d",small);

  return 0;
}
