#include <stdio.h>
int main()
{
int a,i,j,temp,b[50];
  scanf("%d",&a);
  for(i=0;i<a;i++)
    scanf("%d",&b[i]);
  for(i=0;i<a-1;i++)
  {
    for(j=0;j<a-i-1;j++)
    {
      if(b[j]<b[j+1])
         {
           temp=b[j];
           b[j]=b[j+1];
           b[j+1]=temp;
         }
    }
  }
         for(i=0;i<a;i++)
         {
           if(i==0||i==a-1)
    printf("%d\n",b[i]);
         }
	return 0;
}
