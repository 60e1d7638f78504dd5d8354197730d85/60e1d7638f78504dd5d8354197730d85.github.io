#include <stdio.h>
  int main() {
        int hour, minute, second, timeinsec;

      
        /* get hour value from user*/
        scanf("%d", &hour);

        
        scanf("%d", &minute);

       
        scanf("%d", &second);

        /* calculate total seconds */
        timeinsec = second + (minute * 60) + (hour * 60 * 60);
        printf("%d\n",
                 timeinsec);
        return 0;
  }
