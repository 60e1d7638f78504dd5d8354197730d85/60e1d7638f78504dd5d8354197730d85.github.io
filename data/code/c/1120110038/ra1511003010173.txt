#include<stdio.h>
#include<math.h>
 
int main()
{	long int t,n,b[1000],i,k,j,flag,r,minsum,sum;
	scanf("%ld",&t);
for(j=0;j<t;++j)
{	sum=0;
	scanf("%ld",&n);
	for(i=0;i<n;++i)
	scanf("%ld",&b[i]);
	
	
	
	for(i=0;i<pow(2,n-1);i++)
	{ 	minsum=0;
		k=i;
		flag=n-1;
	
		while(flag>=0)
		{	r=k%2;
			k=k/2;
			r=(2*r)-1;
			minsum=minsum+(b[flag]*r);
			--flag;
			
		}
		sum+=fabs(minsum);
	};
	printf("%ld\n",2*sum);
}
	
	return 0;
}
