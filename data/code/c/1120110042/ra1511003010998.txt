
#include <stdio.h>
#define MAX_SIZE 1000 // Maximum array size

int main()
{
    int arr[MAX_SIZE]; // Declare an array of MAX_SIZE
    int i, N;

    /* Input array size */
    
    scanf("%d", &N);

    /* Input elements in array */

    for(i=0; i<N; i++)
    {
        scanf("%d", &arr[i]);
    }


    /*
     * Print all elements of array
     */
 
    for(i=0; i<N; i++)
    {
        printf("%d ", arr[i]);
    }

    return 0;
}
