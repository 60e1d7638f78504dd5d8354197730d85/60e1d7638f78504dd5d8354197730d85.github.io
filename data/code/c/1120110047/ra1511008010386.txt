#include <stdio.h>

int main()
{
    int num;

    /* Input number from user */

    scanf("%d", &num);
    
    /* Check if the number is divisible by 2 then it is even */
    if(num % 2 == 0)
    {
        /* num % 2 is 0 */
        printf("Even");
    }
    else
    {
        /* num % 2 is 1 */
        printf("Odd");
    }

    return 0;
}
