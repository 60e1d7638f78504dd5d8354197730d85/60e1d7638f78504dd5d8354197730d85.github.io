#include<stdio.h>
#include<string.h>
int main(){
   int i,j,count=6;
   char str[25][25],temp[25];
   for(i=0;i<=count;i++)
      scanf("%s",str[i]);
   for(i=0;i<=count;i++)
      for(j=i+1;j<=count;j++){
         if(strcmp(str[i],str[j])>0){
            strcpy(temp,str[i]);
            strcpy(str[i],str[j]);
            strcpy(str[j],temp);
         }
      }
   for(i=0;i<=count;i++){
     if(strlen(str[i])>0 && str[i]!="")
        {
      printf("%s\n",str[i]);
     }}
   
   return 0;
}
