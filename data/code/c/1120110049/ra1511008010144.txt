#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int main()
{
 char name[100][80], tname[100][80], temp[80];
    int i, j, n;
 
    scanf("%d", &n);
    for (i = 0; i < n; i++)
    {
        scanf("%s", name[i]);
        strcpy(tname[i], name[i]);
    }
    for (i = 0; i < n - 1 ; i++)
    {
        for (j = i + 1; j < n; j++)
        {
            if (strcmp(name[i], name[j]) > 0)
            {
                strcpy(temp, name[i]);
                strcpy(name[i], name[j]);
                strcpy(name[j], temp);
            }
        }
    }
    
    for (i = 0; i < n; i++)
    {
        printf("%s\n",name[i]);
    }
return 0;
}
