#include<stdio.h>
#include <string.h>

int main()
{
    int i, j;
    char str[10][50], temp[50];


    for(i=0; i<4; ++i)
        scanf("%s\n",str[i]);


    for(i=0; i<3; ++i)
        for(j=i+1; j<4 ; ++j)
        {
            if(strcmp(str[i], str[j])>0)
            {
                strcpy(temp, str[i]);
                strcpy(str[i], str[j]);
                strcpy(str[j], temp);
            }
        }

    for(i=0; i<4; ++i)
    {
        printf("%s\n",str[i]);
    }

    return 0;
}
