#include<stdio.h>
#include<string.h>
int main()
 {
 char str[1000][2000],temp[2000];
 int i=0,j=0,n;
 scanf("%d",&n);
 while(i<n)
 scanf("%s",str[i++]);
 for(i=1;i<n;i++)
  {
  for(j=1;j<=n-i;j++)
   {
   if(strcmp(str[j-1],str[j])>0)
    {
    strcpy(temp,str[j-1]); 
    strcpy(str[j-1],str[j]);
    strcpy(str[j],temp);
    }
   }
  }
 for(i=0;i<n;i++)

  return 0;
}
