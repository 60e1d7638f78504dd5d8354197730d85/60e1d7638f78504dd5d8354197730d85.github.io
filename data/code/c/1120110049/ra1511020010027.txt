#include<stdio.h>
#include<string.h>
int main(){
   char s[5][20], t[20];
   int i, j;
   //clrscr();
 int n;
  scanf("%d",&n);
  // printf("\nEnter any five strings : ");
   for (i = 0; i < n; i++)
      scanf("%s", s[i]);
 
   for (i = 1; i < n; i++) {
      for (j = 1; j < n; j++) {
         if (strcmp(s[j - 1], s[j]) > 0) {
            strcpy(t, s[j - 1]);
            strcpy(s[j - 1], s[j]);
            strcpy(s[j], t);
         }
      }
   }
 
  // printf("\nStrings in order are : ");
   for (i = 0; i < n; i++)
      printf("\n%s", s[i]);
   return 0;
}
