#include <stdio.h>
int main()
{int t, i, c, n, m, s[10000];
	scanf("%d", &t);
	while(t--)
	{
		c=0;
		scanf("%d", &n);
		if(n!=0)
		{
			c++;
		}
		for(i=0; i<n; i++)
		{
			scanf("%d", &s[i]);
		}
		m=s[0];
		for(i=1; i<n; i++)
		{
			if(s[i]<=m)
			{
				m=s[i];
				c++;
			}
		}
		printf("%d\n", c);
    }
	return 0;
}
