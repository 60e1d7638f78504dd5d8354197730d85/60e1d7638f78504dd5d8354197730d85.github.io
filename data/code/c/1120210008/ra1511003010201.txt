#include<stdio.h>
 void merge(int arr[100],int l,int m,int h)
{
    int temp[10000],i=l,j=m+1,k=l;
    while(i<=m&&j<=h)
    {
       if(arr[i]<arr[j])
       {
          temp[k]=arr[i];
          i++;
       }
       else
        {
            temp[k]=arr[j];
            j++;
        }
        k++;
    }
    ;
    while(i<=m)
    {
        temp[k++]=arr[i++];
    }
    while(j<=h)
    {
        temp[k++]=arr[j++];
    }
   
    for(i=l;i<k;i++)
    {
     
        arr[i]=temp[i];
    }
   
 }
void merge_sort(int arr[100],int low,int high)
{
    int mid;
    mid=(low+high)/2;
 
    if(low<high)
    {
        merge_sort(arr,low,mid);
        merge_sort(arr,mid+1,high);
 
        merge(arr,low,mid,high);
    }
}
 

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,i;
        scanf("%d",&n);
        int arr[n];
        for(i=0;i<n;i++)
            scanf("%d",&arr[i]);
 
            merge_sort(arr,0,n-1);
            int temp=arr[1]-arr[0];
            for(i=1;i<n-1;i++)
            {
                if(arr[i+1]-arr[i]<temp)
                    temp=arr[i+1]-arr[i];
            }
 
            printf("%d\n",temp);
    }
 
    return 0;
}
