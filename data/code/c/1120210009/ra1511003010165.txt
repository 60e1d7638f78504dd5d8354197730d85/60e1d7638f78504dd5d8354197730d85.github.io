  #include<stdio.h> 
int main()
{
	int c,x,y,z,j,k,check;
	scanf("%d",&c);
	for(z=0;z<c;z++)
	{
		scanf("%d %d",&x,&y);
		for(j=x+y+1;;j++)
		{
			check=0;
			for(k=1;k<=j;k++)
			{
				if(j%k==0)
				check++;
			}
			if(check==2)
			break;
		}
		printf("potatoes to be planted in the third field:%d\n",j-x-y);
	}
	return 0;
}
