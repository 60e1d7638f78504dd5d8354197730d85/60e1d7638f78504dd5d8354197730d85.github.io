#include<stdio.h>
#include<math.h>
 
int main(int argc, char* argv[]) {
    int num;
    scanf("%d", &num);
    int i;
    for(i = 0; i < num; i++) {
        int a,b;
        scanf("%d %d", &a, &b);
        printf("potatoes to be planted in the third field:%d\n",nextHarvest(a, b));
    }
    return 0;
}
 
int isPrime(int num){
    int i;
    for(i = 2; i <= sqrt(num); i++) {
        if(num % i == 0) {
            return 0;
        }
    }
    return 1;
}
 
int nextHarvest(int a, int b) {
    int c = a + b;
    int i = 1;
    while(1) {
        if(isPrime(c + i)){
            c = c + i;
            break;
        }
        i++;
    }
    return c - (a + b);
}
