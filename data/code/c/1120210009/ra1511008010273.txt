#include <stdio.h>
int check(long int n)
{
  long int i, flag=0;
  for(i=2;i<=n/2;++i)
  {
      if(n%i==0)
      {
          flag=1;
          break;
      }
  }
  if (flag==0)
      return 1;
  else
  return 0;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int a,b,ans,p;
		scanf("%d %d",&a,&b);
		ans=a+b;
		while(1)
		{ans+=1;
			p=check(ans);
			if(p==1)
			break;
		}
		printf("potatoes to be planted in the third field:%d\n",ans-a-b);
		
		
	}
  return 0;
}
