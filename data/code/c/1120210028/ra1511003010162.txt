#include <stdio.h>
int main()
{
	int n, m, x[10][10], l, i, j;
  	scanf("%d %d", &n, &m);
  	for(i=0; i<n; i++)
      	for(j=0; j<m; j++)
          		scanf("%d", &x[i][j]);
  	l= x[0][0];
  	for(i=0; i<n; i++)
      	for(j=0; j<m; j++)
        	if(x[i][j]>l)
              	l= x[i][j];
  	for(i=0; i<n; i++)
    {
      for(j=0; j<m; j++)
          	printf("%d ", x[i][j]+l);
      printf("\n");
    }
	return 0;
}
