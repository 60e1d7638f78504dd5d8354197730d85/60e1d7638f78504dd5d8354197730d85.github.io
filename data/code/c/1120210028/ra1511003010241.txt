#include <stdio.h>
int main()
{ int a[10][10],i,j,r,c,n=0;
 scanf("%d %d",&r,&c);
 for(i=0;i<r;i++)
 {
   for(j=0;j<c;j++)
   {
     scanf("%d",&a[i][j]);
     if(n<a[i][j])
       n=a[i][j];
   }}
 for(i=0;i<r;i++)
 {
   for(j=0;j<c;j++)
   {
     printf("%d ",a[i][j]+n);
   }
   printf("\n");
 }
	return 0;
}
