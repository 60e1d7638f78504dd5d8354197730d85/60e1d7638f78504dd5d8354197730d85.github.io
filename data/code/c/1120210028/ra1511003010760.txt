#include <stdio.h>
int main()
{  
   int i,j,m,n,a[100][100],max=0;
  scanf("%d %d",&m,&n);
   for(i=0;i<m;i++){
     for(j=0;j<n;j++){
       scanf("%d",&a[i][j]);
       if(a[i][j]>max){
        max=a[i][j];
       }
     }
   }
 for(i=0;i<m;i++){
     for(j=0;j<n;j++){
       printf("%d ",a[i][j]+max);
     }
   printf("\n");
 }
  

	return 0;
}
