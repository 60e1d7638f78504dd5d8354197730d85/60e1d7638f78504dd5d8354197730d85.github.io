#include<stdio.h>
int main()
{
	int t,a,b,c,ans1=1,ans2=1,temp,i,ra,fa,rb,fb;
	scanf("%d",&t);
	for(i=0;i<t;i++)
	{
		scanf("%d %d %d",&a,&b,&c);
		if(a<b)
		{
			temp=a;
			a=b;
			b=temp;
		}
		ra=a;
		fa=a;
		rb=b;
		fb=b;
		while(1)
		{
			if((c==fa)||(c==(a-ra))||(c==fb))
				break;
			if((fa==b)||(a==0)||(b==0)||(c==0))
			{
				ans1=-1;
				ans2=-1;
				break;
			}
			
			if(fa==0)
			{
				fa=a;
				ans1++;
			}
			if(rb==0)
			{
				rb=b;
				ans1++;
			}
			if(fa<rb)
			{
				rb=rb-fa;
				fa=0;
				ans1++;
			}
			else
			{
				fa=fa-rb;
				rb=0;
				ans1++;
			}
			if(fb==0)
			{
				fb=b;
				ans2++;
			}
			if(ra==0)
			{
				ra=a;
				ans2++;
			}
			if(fb<ra)
			{
				ra=ra-fb;
				fb=0;
				ans2++;
			}
			else
			{
				fb=fb-ra;
				ra=0;
				ans2++;
			}	
		}		
		if(ans1<ans2)
			printf("%d\n",ans1);
		else
			printf("%d\n",ans2);
		ans1=1;
		ans2=1;
	}
	return 0;
}
