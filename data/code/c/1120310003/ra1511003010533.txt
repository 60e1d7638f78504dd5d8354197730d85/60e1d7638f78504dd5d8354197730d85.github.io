#include<stdio.h>
#include<math.h>
long long int max(long long int a,long long int b);
int bincount(long long int a,long long int b,long long int mod);
int bincount(long long int a,long long int b,long long int mod)
{
        long long int res=1;
        while(b)
        {
               if(b&1)
                    res=(res*a)%mod;
               a=(a*a)%mod;
               b=b>>1;
        }
        return res;
}
long long int max(long long int a,long long int b)
{
        if(a>b)
            return a;
        else
            return b;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
           long long int i,j,k,f[100000]={0},s[100000]={0},n,max1=-1,mod=pow(10,9)+7,count,p1,p2,ans=1;
            scanf("%lld",&n);
            for(i=0;i<n;i++)
            {
                  scanf("%lld",&k);
                  f[k]++;
                  if(k>max1)
                        max1=k;
                    
            }
          for(j=max1;j>1;j--)
          {
                  p1=f[j];
                  p2=0;
                  for(k=2;(k*j)<=max1;k++)
                  {
                        p1=p1+f[k*j];
                        p2=(p2+s[k*j])%(mod-1);
                  }
            count=(bincount(2,p1,mod-1)-1-p2+mod-1)%(mod-1);
           
            ans=(ans*bincount(j,count,mod))%mod;
            s[j]=count;
             //printf("%lld %lld\n",j,count);
                  
          }
          printf("%lld\n",ans);
    }
    return 0;
}
