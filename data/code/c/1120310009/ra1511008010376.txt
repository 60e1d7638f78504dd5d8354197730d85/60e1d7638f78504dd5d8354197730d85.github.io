#include <stdio.h>
void selectionsort(int arr[],int n)
{
  int i,j,min,temp;
  for(i=0;i<n-1;i++)
  {
    min=i;
    for(j=i+1;j<n;j++)
      if(arr[j]<arr[min])
        min=j;
    temp=arr[min];
    arr[min]=arr[i];
    arr[i]=temp;
  }
}
int main()
{
  int arr[50],n,i;
  scanf("%d",&n);
  for(i=0;i<n;i++)
    scanf("%d",&arr[i]);
  selectionsort(arr,n);
  for(i=0;i<n;i++)
    printf("%d ",arr[i]);
	return 0;
}
