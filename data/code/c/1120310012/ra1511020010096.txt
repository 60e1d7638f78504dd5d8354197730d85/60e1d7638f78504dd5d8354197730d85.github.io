#include <stdio.h>
#include<stdlib.h>
#define d 1000000007
#define u unsigned long long int
u Power(int n,u powers[])
{   
   u emp;
   if(n==0) return 1;
   if(powers[n/2])
   emp=powers[(n/2)]%d;
   else
   emp=Power(n/2,powers);
   if(n%2==0) {
       return (emp*emp)%d;
   }
   return (emp*emp*2)%d;
}
int read()
{
    int input;
    char c=0;
    while (c<33) c=getchar_unlocked();
    input=0;
    while (c>33)
    {
          input=input*10+c-'0';
          c=getchar_unlocked();
    }
    return input;
}
int main() {
	int t;
	t=read();
	while(t--)
	{
	    int n;
	    n=read();
	    int i,a[n+1];
	    unsigned long long int sum=0,su[n],powers[n+1];
	    for(i=0;i<=n;i++)
	    {
	        a[i]=read();
	    }
	    for(i=0;i<n+1;i++)
	    {
	        powers[i]=Power(i,powers);
	    }
	    su[0]=a[0];
	    for(i=1;i<n;i++)
	    {
	       su[i]=( (su[i-1]%d) + ( (a[i]*powers[i-1])%d ) );
	    }
	    for(i=n;i>0;i--)
	    {   
	        sum+=( ( (a[i]*(powers[n-i+1])%d)*(su[i-1]%d) ) )%d;
	    }
	    printf("%lld\n",sum%d);
	}
	return 0;
}
