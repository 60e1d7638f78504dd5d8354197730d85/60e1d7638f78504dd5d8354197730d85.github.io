#include <stdio.h>
int main()
{
  int n,i,res;
  scanf("%d",&n);
  int a[n],b[n],t[n];
  for(i=0;i<n;i++)
    scanf("%d",&a[i]);
  for(i=0;i<n;i++)
    scanf("%d",&b[i]);
  if(n>1)
  {
    t[0]=a[0]-b[0];
    if(t[0]<0)
      t[0]=-t[0];
    if(a[0]<a[1])
      t[1]=a[1]-a[0];
    else
      t[1]=a[0]-a[1];
    if(b[0]<b[1])
      t[1]+=b[1]-b[0];
    else
      t[1]+=b[0]-b[1];
    if(a[1]<b[1])
      t[1]=t[1]>(t[0]+b[1]-a[1])?t[1]:(t[0]+b[1]-a[1]);
    else
      t[1]=t[1]>(t[0]+a[1]-b[1])?t[1]:(t[0]+a[1]-b[1]);
  }
  int temp;
  for(i=2;i<n;i++)
  {
    temp=t[i-2];
    t[i]=(b[i]>a[i])?(t[i-1]+b[i]-a[i]):(t[i-1]+a[i]-b[i]);
    temp=(a[i-1]>a[i])?(temp+a[i-1]-a[i]):(temp+a[i]-a[i-1]);
    temp=(b[i-1]>b[i])?(temp+b[i-1]-b[i]):(temp+b[i]-b[i-1]);
    t[i]=(t[i]>temp)?t[i]:temp;
  }
  i--;
  printf("%d",t[n-1]);
	return 0;
}
