#include <stdio.h>
int main()
{
	int t , i , n ;
  	scanf ("%d" , &t) ;
  	while(t--){
      long long int prod=1 , j , count=0 ;
      int a[1000000]={0} ;
      scanf("%d" , &n) ;
      for ( i=0 ; i<n ; i++ )
        	scanf("%d" , &a[i]) ;
      for ( i=0 ; i<n ; i++ )
        	prod = prod*a[i] ;
      for ( j=1 ; j<=prod ; j++ )
        	if ( prod%j==0 )
              	count++ ; 
      printf("%lld\n" , count) ;
    }
	return 0;
}
