LLI isCom(LLI n)
{
    LLI i;
    for(i=sqrt(n);i>1;i--)
    {
        if(n%i==0)          return 1;
    }
    return 0;
}

	        scanf(" %lld",&k);
	        p*=k;
	    }

	    for(i=2;p!=1;)
