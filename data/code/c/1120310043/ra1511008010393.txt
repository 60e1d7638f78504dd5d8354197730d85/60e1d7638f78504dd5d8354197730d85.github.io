#include <stdio.h>
int main()
{
  int n,m;
  scanf("%d %d\n",&n,&m);
  int kn[n];
  int eval[m][2];
  int sol[n][2];
  int i,j,k;
  int check;
  for(i=0;i<n;i++)
  {
    sol[i][0]=-1;
    sol[i][1]=0;
  }
  for(i=0;i<n;i++)
    scanf("%d ",&kn[i]);
  for(i=0;i<m;i++)
    scanf("%d %d\n",&eval[i][0],&eval[i][1]);
  for(i=0;i<m;i++)
  {
    eval[i][0]--;
    eval[i][1]--;
    
  }
  int d;
  scanf("%d",&d);
  for(i=0;i<n;i++)
  {
    for(j=(i>=d?i-d:0);j<(i<n-d?i+d+1:n);j++)
    {
      for(k=0;k<m;k++)
      {
        if((eval[k][0]<=i)&&(eval[k][1]>=i))
        {
          check=((eval[k][0]<=j)&&(eval[k][1]>=j))?0:1;
          k=m-1;
        }
      }
      if((check==1)&&(kn[i]<kn[j]))
      {
        if(sol[i][0]<kn[j])
        {
          sol[i][0]=kn[j];
          sol[i][1]=1;
        }
        else if(sol[i][0]==kn[j])
          sol[i][1]++;
      }
    }
  }
  for(i=0;i<n;i++)
  {
    if(sol[i][0]==-1)
      printf("-1\n");
    else
      printf("%d %d\n",sol[i][0],sol[i][1]);
    
  }
	return 0;
}
