#include <stdio.h>
int main(){
int t,n,a[100000],c,i,min;
scanf("%d",&t);
while(t--)
{   c=0;
    min=1000000001;
        scanf("%d",&n);
        for(i=0;i<n;i++){
            scanf("%d",&a[i]);
            if(min>=a[i])
                min=a[i];}
        for(i=0;i<n;i++)
            if(min==a[i])
                c++;
        if(c%2)
            printf("Lucky\n");
        else
            printf("Unlucky\n");
}      
return 0;
}
