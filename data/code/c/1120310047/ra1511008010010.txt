#include<stdio.h>
int main()
{
	int q;
	int top = -1;
	int cost[100000];
	scanf("%d",&q);
	while(q--)
	{
		int type;
		scanf("%d",&type);
		if(type == 1)
		{
			if(top==-1)
			   printf("No Food\n");
			else
			{
				printf("%d\n",cost[top]);
			    top = top -1;
			}
		}
		if(type == 2)
		{
			int price;
			top = top +1;
			scanf("%d",&price);
			cost[top] = price;
		}
	}
	return 0;
}
