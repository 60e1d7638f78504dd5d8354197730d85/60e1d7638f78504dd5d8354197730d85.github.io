#include<stdio.h>
#define MAXSIZE 100
 
 
int main()
{
    int T;
    int n,k,i,j;
    int arr[MAXSIZE+1];
 
    scanf("%d",&T);
    for(i=1;i<=T;i++)
    {
        scanf("%d",&n);
        for(j=1; j<=n; j++) scanf("%d",&arr[j]);
        scanf("%d",&k);
 
        int kval = arr[k];
        int newpos=1;
        for(j=1; j<=n; j++)
            if(arr[j]<kval)newpos++;
        printf("%d\n", newpos);
    }
    return 0;
}
