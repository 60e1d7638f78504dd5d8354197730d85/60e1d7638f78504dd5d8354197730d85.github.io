#include <iostream>
#include <stdio.h>
using namespace std;
int main()
{
  int hr,min,sec,fg;
  cin>>hr>>min>>sec;
  if(hr<=23 && min<=59 && sec<=59) {
    printf("24 Hours Format\n");
    printf("%d:%d:%d\n",hr,min,sec);
    if(hr==0) {
      hr = 12;
      fg = 1;
    }
    else if(hr>12){
      hr = hr%12;
      fg=0;
    }
    else if(hr<12){
      hr = hr;
      fg=1;
    }
    else if(hr==12) {
      hr=hr;
      fg=0;
    }
    printf("12 Hours Format\n");
    if(fg==0)
      printf("%d:%d:%dpm",hr,min,sec);
    else
      printf("%d:%d:%dam",hr,min,sec);
  }
  else
    cout<<"Invalid Time\n";
 
	return 0;
}
