#include <iostream>
using namespace std;

int main()
{
  int h,h2,m,s;
  string c;
  cin>>h>>m>>s;
  if(h>23 || m>=60 || s>=60)
    cout<<"Invalid Time";
   else
   {
     if(h>12)
     {
       h2=h-12;
       c="pm";
     }
     else
     { h2=h;
       c="am";
     }
     cout<<"24 Hours Format\n"<<h<<":"<<m<<":"<<s;
     cout<<"\n12 Hours Format\n"<<h2<<":"<<m<<":"<<s<<c;
   }
	return 0;
}
