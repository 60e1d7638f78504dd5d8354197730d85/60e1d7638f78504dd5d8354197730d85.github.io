#include<iostream>
using namespace std;
int main()
{
	int hrs,min,sec;
  	cin>>hrs>>min>>sec;
  	if(hrs>=24 || min>=60 || sec>=60)
      	cout<<"Invalid Time"<<endl;
  	else
    {
      	if(hrs>12)
        {
      		cout<<"\n24 Hours Format\n"<<hrs<<":"<<min<<":"<<sec;
      		hrs=hrs-12;
      		cout<<"\n12 Hours Format\n"<<hrs<<":"<<min<<":"<<sec<<"pm";
        }
    	else
        {
          	cout<<"\n24 Hours Format\n"<<hrs<<":"<<min<<":"<<sec;
      		cout<<"\n12 Hours Format\n"<<hrs<<":"<<min<<":"<<sec<<"am";
        }
    }
	return 0;
}
