#include <iostream>
using namespace std;
int main()
{
int h;
  int min;
  int s;
  cin>>h;
  cin>>min;
  cin>>s;
  if(h>=24||min>59||s>60)
  {
    cout<<"Invalid Time";
  }
  else
  {
    if(h>12&&h<24)
    {
      cout<<"24 Hours Format";
      cout<<'\n'<<h<<":"<<min<<":"<<s<<'\n';
      cout<<"12 Hours Format\n";
      cout<<h-12<<":"<<min<<":"<<s<<"pm";
  }
    else if(h>=0&&h<=12)
    {
      cout<<"24 Hours Format";
      cout<<'\n'<<h<<":"<<min<<":"<<s;
      cout<<"\n12 Hours Format";
      cout<<'\n'<<h<<":"<<min<<":"<<s<<"am";
    }
  }
	return 0;
}
