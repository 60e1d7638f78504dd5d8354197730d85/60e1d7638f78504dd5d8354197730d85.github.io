#include<stdio.h>
int main()
{
	int hr,min,sec,hr1;
  scanf("%d %d %d",&hr,&min,&sec);
  hr1=hr;
  if(min>=60 || hr>=24 || sec>=60)
  {
    printf("Invalid Time");
    return 0; 
  }
  printf("24 Hours Format\n");
  
    printf("%d",hr);
  
    printf(":%d",min);
  
    printf(":%d\n",sec);
  
  printf("12 Hours Format\n");
  if(hr>12)
    printf("%d",hr-12);
  else
    printf("%d",hr);
  
    printf(":%d",min);
  
    printf(":%d",sec);
  if(hr1>11)
    printf("pm");
  else
    printf("am");
	return 0;
}
