#include <iostream>
#include <stdio.h>

using namespace std;

int main()
{
	int hr, min, sec;
  	cin>>hr>>min>>sec;
  	if(hr>=0 && hr<24 && min>=0&& min<60 && sec>=0&& sec<60)
    {
  	printf("24 Hours Format\n%d:%d:%d\n",hr,min,sec);
  	if(hr>=12)
  		printf("12 Hours Format\n%d:%d:%dpm",hr-12,min,sec);
  	else
      	printf("12 Hours Format\n%d:%d:%dam",hr,min,sec);
    }
  	else
      cout<<"Invalid Time";
	return 0;
}
