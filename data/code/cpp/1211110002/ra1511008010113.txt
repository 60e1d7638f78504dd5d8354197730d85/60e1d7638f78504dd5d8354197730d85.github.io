#include <iostream>
#include <stdio.h>
#include <ctype.h>
using namespace std;
int main()
{
    char str[30];
    int count=0,dot=0;
    cin>>str;
    for(int i=0;str[i]!='\0';i++)
    {
      if(!(isdigit(str[i])))
      count++;
     if(str[i]=='.')
       dot++;
    }
   if(dot==1)
     count--;
   if((count==0) && (dot<=1))
      cout<<"The entered number is="<<str;
	return 0;
}
