#include <iostream>
using namespace std;
int main()
{
int n,f=0;
  cin>>n;
  if(n%7==0)
  {
    f=1;
    cout<<"Buzz";
  }
  int num=n;
  while(num>0)
  {
    int rem=num%10;
    num=num/10;
    if(rem==7)
    {
      cout<<"Buzz";
      f=1;
      break;
    }
  }
  if(f!=1)
    cout<<"Sorry";
	return 0;
}
