#include <iostream>
using namespace std;
int main()
{
  int n,m,d=0,p=0;
  cin>>n;
  
  if(n%7==0)
    d=1;
  while(n>0)
  {
    m = n % 10;
    if(m == 7)
      p=1;
    n=n/10;
  }
  if(d==1 && p==1)
    cout<<"BuzzBuzz";
  else if(d==1 && p==0 || d==0 && p==1)
    cout<<"Buzz";
  else
    cout<<"Sorry";

	return 0;
}
