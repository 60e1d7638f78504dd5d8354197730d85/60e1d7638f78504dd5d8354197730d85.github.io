#include <iostream>
using namespace std;
int main()
{ int a[3],i;
  float sum=0;
  for (i=0;i<3;i++)
    cin>>a[i];
  for (i=0;i<3;i++)
   { if (a[i]%2==0 && a[i]>=0)
        sum=sum+1;
     else if (a[i]%2!=0)
        sum=sum-0.5;
     else if (a[i]<0)
        sum=sum-1;
   }
  cout<<sum;
  return 0;
}
