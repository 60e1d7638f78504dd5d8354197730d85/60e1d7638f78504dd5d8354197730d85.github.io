#include <iostream>
using namespace std;
int main()
{
	float ct=0.0;
  	int c=0,x;
  	while(c<3)
    {
      cin>>x;
      if(x<0)
        ct-=1.0;
      else
      {
        if(x%2==0)
        {
          ct+=1.0;
        }
        else
          ct-=0.5;
      }
      c++;
    }
  cout<<ct;
	return 0;
}
