#include <iostream>
using namespace std;
int main()
{
  int a;
  double score = 0;
  for(int i = 1 ; i<=3 ; i++)
  {
    cin>>a;
    if(a<0)
    {
      score = score-1;
      break;
    }
    else
    {
      if(a%2!=0)
        score = score -0.5;
      else
        score = score + 1;
    }
    
  }
  cout<<score;
	return 0;
}
