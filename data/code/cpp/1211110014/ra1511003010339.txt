#include <iostream>
using namespace std;
int main()
{
  int a;
  float sum;
  sum=0;
  while(cin>>a)
  {
    if(a<=0)
    {
      sum=sum-1;
      break;
    }
    else if(a%2==0)
    {
      sum=sum+1;
    }
    else
    {
      sum=sum-0.5;
    }
    if(sum==3)
    {
      break;
    }
  }
  cout<<sum;

	return 0;
}
