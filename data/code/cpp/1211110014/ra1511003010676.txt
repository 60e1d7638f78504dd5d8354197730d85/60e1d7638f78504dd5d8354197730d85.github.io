#include <iostream>
#include <stdio.h>
using namespace std;
int main()
{
	int i,a;
  	float score=0;
  	for (i=0;i<3;i++)
    {
      cin>>a;
      
      
      if(a<0)
      {
        score-=1;
		break;
      }
      else if(a%2==0)
      {
        score++;
      }
      else 
        score-=0.5;
    }
  	cout<<score;
	return 0;
}
