#include <iostream>
using namespace std;
float neg(int y,float s)
{if(y<0){
     s=s-1;}
  else{
    if(y%2==1){
      s=s-0.5;}
   else{
     s=s+1;}
     }
  return s;
}
int main()
{ 
  int a[3];
  float s=0;
  for(int i=0;i<3;i++){
       cin>>a[i];}
  for(int j=0;j<3;j++)
  {      s=neg(a[j],s);
  //cout<<s<<" ";
         
}
   cout<<s;
	return 0;
}
