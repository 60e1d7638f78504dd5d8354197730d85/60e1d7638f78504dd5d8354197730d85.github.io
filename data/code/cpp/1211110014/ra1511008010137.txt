#include <iostream>
using namespace std;
int main()
{
   float score;
   int a,b,c;
   cin>>a>>b>>c;
   if(a<0)
   {
     cout<<--score;
     goto end;
   }
   else
   {
     if(a%2==0)
       score++;
     else
       score-=0.5;
       
      if(b<0)
   {
     cout<<--score;
     goto end;
   }
   else
   {
     if(b%2==0)
       score++;
     else
       score-=0.5;
       if(c<0)
   {
     cout<<--score;
     goto end;
   }
   else
   {
     if(c%2==0)
       score++;
     else
       score-=0.5;
   }
   }
   }
  cout<<score;
  end:
	return 0;
}
