#include <iostream>
using namespace std;
int main()
{
 	int a, b, c;
  	float score = 0.0;
  	cin>>a>>b>>c;
  	int nums[] = {a, b, c};
  	
  	for(int i=0; i<3; i++)
      if(nums[i] > 0)
        if(nums[i]%2 == 0)
          score += 1;
  		else
          score -= 0.5;
  	  else
          score -= 1;
  
  	cout<<score<<endl;
	return 0;
}
