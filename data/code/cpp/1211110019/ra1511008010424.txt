#include <iostream>
using namespace std;

class maxi
  
{public:
  int max1(int a, int b)
  {
  return (a>b?a:b);
  }
  int max2(int a, int b, int c)
  {
  	return (max1(max1(a,b),c));
  }
};
int main()
{
	int a,b,c,gr;
  	cin>>a>>b>>c;
  	maxi m;

	return 0;
