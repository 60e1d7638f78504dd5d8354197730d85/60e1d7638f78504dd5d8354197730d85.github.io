#include <iostream>
using namespace std;
int main()
{
	int ch,n1,n2;
    cin>>ch;
    cin>>n1>>n2;
    switch(ch)
    {
      case 1:cout<<n1+n2;
             break;
      case 2:cout<<n1-n2;
             break;
      case 3:cout<<n1*n2;
             break;
      case 4:cout<<n1/n2;
             break;
      default:cout<<"Invalid Input";
    }
	return 0;
}
