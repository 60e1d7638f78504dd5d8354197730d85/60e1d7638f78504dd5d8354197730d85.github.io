#include <iostream>
#include<math.h>
#include<stdio.h>
using namespace std;
int main()
{
  float b,c;
  int a;
  cin>>a;
  cin>>b>>c;
  switch(a)
  {
    case 1:
      cout<<b+c;
      break;
    case 2:
      cout<<b-c;
      break;
    case 3:
      cout<<c*b;
      break;
    case 4:
      printf("%f",b/c);
      break;
    default:
      cout<<"Invalid Input";
  }

	return 0;
}
