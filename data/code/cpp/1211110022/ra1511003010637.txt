#include <iostream>
using namespace std;
int main()
{
	int a,b,func;
  	cin>>func;
  	cin>>a>>b;
  	switch(func)
    {
      case 1:
        cout<<(a+b);
       	break;
      case 2:
        cout<<(a-b);
       	break;
     case 3:
        cout<<(a*b);
       	break;
     case 4:
        cout<<(a/b);
       	break;
      default:
        cout<<"Invalid Input";
    }
	return 0;
}
