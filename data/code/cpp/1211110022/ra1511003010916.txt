#include <iostream>
using namespace std;
int main()
{
    int res=0,a,b,n;
    cin>>n;
     cin>>a>>b;
    if(n>4)
     cout<<"Invalid Input";
    else if(n==1)
    {
      res=a+b;
      cout<<res;
    }
    else if(n==2)
    {
     res=a-b;
      cout<<res;
    }
    else if(n==3)
    {
     res=a*b;
      cout<<res;
    }
    else
    {
     res=a/b;
      cout<<res;
    }
      
	return 0;
}
