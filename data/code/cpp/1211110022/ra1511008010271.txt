#include <iostream>
using namespace std;
int main()
{	int n1,n2;int i;
 	cin>>i;
 	cin>>n1>>n2;
 	int ans;
 	if(i==1)
      ans=n1+n2;
 	else if(i==2)
      ans=n1-n2;
 	else if(i==3)
      ans=n1*n2;
 	else if(i==4)
      ans=n1/n2;
 	else{
      cout<<"Invalid Input";
      return 0;
    }
 	cout<<ans;

	return 0;
}
