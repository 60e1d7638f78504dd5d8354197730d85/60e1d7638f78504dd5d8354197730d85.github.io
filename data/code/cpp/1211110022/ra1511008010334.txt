#include <iostream>
using namespace std;
int main()
{
    int i,a,b;
    cin>>i;
    cin>>a>>b;
  switch(i)
  {
    case 1:
      {
        cout<<a+b;
        break;
      }
    case 2:
      {
        cout<<a-b;
        break;
      }
    case 3:
      {
        cout<<a*b;
        break;
      }
    case 4:
      {
        cout<<a/b;
        break;
      }
    default:
        cout<<"Invalid Input";
      
  }
	return 0;
}
