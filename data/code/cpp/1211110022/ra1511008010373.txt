#include <iostream>
using namespace std;
int main()
{

  int n,c,x,y;
  cin>>n;
  cin>>x>>y;
  if(n==1)
  {
    c=x+y;
    cout<<c;
  }
  else if(n==2)
  {
    c=x-y;
    cout<<c;
  }
  else if(n==3)
  {
    c=x*y;
    cout<<c;
  }
  else if(n==4)
  {
    c=x/y;
    cout<<c;
  }
  else
    cout<<"Invalid Input";
	return 0;
}
