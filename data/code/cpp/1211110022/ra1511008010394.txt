#include <iostream>
using namespace std;
int main()
{
	int ch,a,b;
  	cin>>ch;
  	cin>>a>>b;
  	switch(ch)
    {
      case 1 : cout<<a+b;
        break;
      case 2 : cout<<a-b;
        break;
      case 3 : cout<<a*b;
        break;
      case 4 : cout<<a/b;
        break;
       default : cout<<"Invalid Input";
        break;
    }
	return 0;
}
