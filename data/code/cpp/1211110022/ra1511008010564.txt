#include <iostream>
using namespace std;
int main()
{
	int option;
  	int a,b;
  	cin>>option;
  	cin>>a>>b;
  	if(option==1)
    	cout<<a+b;
  	else if(option==2)
      	cout<<a-b;
  	else if(option==3)
      	cout<<a*b;
  	else if(option==4)
      	cout<<"Invalid Input";
  	else
      	cout<<"Invalid Input";
	return 0;
}
