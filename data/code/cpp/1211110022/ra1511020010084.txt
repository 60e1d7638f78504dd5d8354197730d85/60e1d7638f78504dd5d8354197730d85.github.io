#include <iostream>
using namespace std;
int main()
{  int n;
   int a,b,result;
 scanf("%d",&n);
 scanf("%d%d",&a,&b);
 switch(n)
 { case 1: result=a+b;printf("%d",result); break;
  case 2:  result=a-b;printf("%d",result); break;
  case 3: result= a*b; printf("%d",result); break;
  case 4: result=a/b; printf("%d",result);break;
  default: printf("Invalid Input\n");
 }
 

	return 0;
}
