#include <iostream>
using namespace std;
int main()
{
int ori,num,digit,sum=0;

  cin>>ori;
  num=ori;

  while(num !=0)
  {
    digit=num%10;
    sum+=digit*digit*digit;
    num/=10;
  }
  
  if(sum==ori)
    cout<<"Give to Scientist Armstrong";
  else
    cout<<"Dont Give to Scientist Armstrong";
	return 0;
}
