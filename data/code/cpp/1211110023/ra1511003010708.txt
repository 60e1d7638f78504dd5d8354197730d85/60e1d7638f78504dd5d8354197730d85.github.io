#include <iostream>
#include<math.h>
using namespace std;
int main()
{
  int t;
  cin>>t;
  int n=t;
  int count=0;
  while(n)
  {
    count++;
    n=n/10;
  }
  n=t;
  int sum=0;
  int n1;
  while(n)
  {
    n1=n%10;
    sum=sum+pow(n1,count);
    n=n/10;
  }
  if(sum==t)
    cout<<"Give to Scientist Armstrong"<<endl;
  else
    cout<<"Dont Give to Scientist Armstrong"<<endl;
	return 0;
}
