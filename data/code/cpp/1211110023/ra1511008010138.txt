#include <iostream>
using namespace std;
int main()
{
	int num,no,r,sum=0;
  	cin>>num;
    no=num;
    while(no!=0)
    {
      r=no%10;
      sum=sum+(r*r*r);
      no=no/10;
    }
    if(sum==num) cout<<"\nGive to Scientist Armstrong";
     else if(sum!=num) cout<<"\nDont Give to Scientist Armstrong";
	return 0;
}
