#include<iostream>
using namespace std;

int main()
{
	int year;
	cin>>year;
	int f;
	if(year%400==0)
      f=1;
  else if(year%100==0)
    f=0;
  	else if(year%4==0)
      f=1;
	else
      f=0;
  if(f==1)
    cout<<"The year is leap year"<<endl;
  else
    cout<<"The year is not leap year"<<endl;
	return 0;
}
