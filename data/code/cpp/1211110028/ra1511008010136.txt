#include <iostream>
using namespace std;
void isLeapYear(int n)
{  if(n%4 == 0)
    {
        if(n%100 == 0)
        {
            if (n%400 == 0)
                cout<<"The year is leap year";
            else
                cout<<"The year is not leap year";
        }
        else
            cout<<"The year is leap year";
    }
    else
        cout<<"The year is not leap year";
}
int main()
{
  int n;
  cin>>n;
isLeapYear(n); 
	return 0;
}
