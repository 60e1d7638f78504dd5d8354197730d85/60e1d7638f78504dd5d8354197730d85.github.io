#include <iostream>
using namespace std;
char word[][10]={"Zero","One","Two","Three","Four","Five","Six","Seven","Eight","Nine"};
void pw(int n)
{
	int d[10];
  	int c=0;
  	do{
    	d[c]=n%10;
      	n=n/10;
      	c++;
       }while(n!=0);
  	for(int i=c-1;i>=0;i--)
      	cout<<word[d[i]]<<"\n";
} 

int main()
{
	int n;
  	scanf("%d",&n);
  	pw(n);
	return 0;
}
