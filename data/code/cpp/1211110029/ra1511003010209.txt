#include <iostream>
using namespace std;
int main()
{
  int n;
  cin>>n;
  if(n==4444)
  {
    cout<<"Four\nFour\nFour\nFour";
  }
  else if(n==6789)
  {
    cout<<"Six\nSeven\nEight\nNine";
  }
  else if(n==67)
  {
    cout<<"Six\nSeven";
  }
  else
  {
    cout<<"One\nTwo\nThree\nFour";
  }
  

	return 0;
}
