#include <iostream>
using namespace std;
int main()
{
    int a,b,c;
    cin>>a;
    while(a>0)
    {
      b=a%10;
      c=c*10+b;
      a=a/10;
    }
    while(c>0)
    {
      b=c%10;
      c=c/10;
      switch(b)
      {
        case 1:
          cout<<"One\n";
          break;
        case 2:
          cout<<"Two\n";
          break;
        case 3:
          cout<<"Three\n";
          break;
        case 4:
          cout<<"Four\n";
          break;
        case 5:
          cout<<"Five\n";
          break;
         
        case 6:
          cout<<"Six\n";
          break;
        case 7:
          cout<<"Seven\n";
          break;
        case 8:
          cout<<"Eight\n";
          break;
        case 9:
          cout<<"Nine\n";
          break;
        case 0:
          cout<<"Zero\n";
          break;
      }
    }   
	return 0;
}
