#include <iostream>
using namespace std;
int main()
{
    int n, num = 0;

    
    cin>>n;

   
    while(n != 0)
    {
        num = (num * 10) + (n % 10);
        n /= 10;
    }
    while(num != 0)
    {
        switch(num % 10)
        {
            case 0: 
                cout<<"Zero\n";
                break;
            case 1: 
                cout<<"One\n";
                break;
            case 2: 
                cout<<"Two\n";
                break;
            case 3: 
                cout<<"Three\n";
                break;
            case 4: 
                cout<<"Four\n";
                break;
            case 5: 
                cout<<"Five\n";
                break;
            case 6: 
                cout<<"Six\n";
                break;
            case 7: 
                cout<<"Seven\n";
                break;
            case 8: 
                cout<<"Eight\n";
                break;
            case 9: 
                cout<<"Nine\n";
                break;
        }
        
        num = num / 10;
    }

    return 0;
}
