#include <iostream>
using namespace std;
int main()
{
int n,r,p,sum=0;
  cin>>n;
  
  while(n!=0)
  {
    r=n%10;
    sum=10*sum+r;
 
    n=n/10;
  }

  while(sum>0)
  {
    p=sum%10;
    switch(p)
    {
      case 1: cout<<"One";
        break;
      case 2: cout<<"Two";
        break;
      case 3 :cout<<"Three";
        break;
      case 4: cout<<"Four";
        break;
      case 5: cout<<"Five";
        break;
      case 6: cout<<"Six";
        break;
      case 7: cout<<"Seven";
        break;
      case 8: cout<<"Eight";
        break;
      case 9: cout<<"Nine";
        break;
      case 0: cout<<"Zero";
        break;
    }
    cout<<"\n";
    sum=sum/10;
  }
	return 0;
}
