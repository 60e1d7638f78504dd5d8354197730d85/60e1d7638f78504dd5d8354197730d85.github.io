#include<iostream>
using namespace std;

void func(int a)
{
 switch(a)
 {
   case '1': cout<<"One\n";
            break;
     
     
   case '2':
      
        cout<<"Two\n";
        break;
      
   case  '3' :
        cout<<"Three\n";
        break;
   case  '4' :
      
        cout<<"Four\n";
      break;
   case  '5':
        cout<<"Five\n";
        break;
   case  '6':
        cout<<"Six\n";
        break;
   case  '7':
     
        cout<<"Seven\n";
        break;
   case '8': 
      
        cout<<"Eight\n";
        break;
   case '9':
     cout<<"Nine\n";
      break ;
   case '0' :
        cout<<"Zero\n";
        break ;
 }
  
  
}
int main()
{
    string s;
    cin >> s;
    int i = 0;
    while(s[i]!='\0')
    {
      int a = s[i] ;
      func(a);
      i++;
    }
  
	return 0;
}
