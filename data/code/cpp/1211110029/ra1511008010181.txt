#include <iostream>
#include <string>
using namespace std;
int main()
{
string s;
cin>>s;
for(int i=0;i<s.length();i++)
{  
  switch(s[i])
  {
    case '1':cout<<"\nOne";
      break;
    case '2':cout<<"\nTwo";
      break;
    case '3':cout<<"\nThree";
       break;
    case '4':cout<<"\nFour";
    break;
    case '5':cout<<"\nFive";
    break;
    case '6':cout<<"\nSix";
    break;
    case '7':cout<<"\nSeven";
    break;
    case '8':cout<<"\nEight";
    break;
    case '9':cout<<"\nNine";
    break;
    case '0':cout<<"\nZero";
	break;}
}
  return 0;
}
