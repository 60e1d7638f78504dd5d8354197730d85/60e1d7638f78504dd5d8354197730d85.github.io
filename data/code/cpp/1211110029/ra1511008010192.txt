#include<bits/stdc++.h>
using namespace std;
int main()
{ int n,num=0;
 cin>>n;
 while(n != 0)
    {
        num = (num * 10) + (n % 10);
        n /= 10;
    }
  while(num != 0)
    {
        switch(num % 10)
        {
            case 0: 
                cout<<"Zero"<<endl;
                break;
            case 1: 
                cout<<"One"<<endl;
                break;
            case 2: 
               cout<<"Two"<<endl;
                break;
            case 3: 
                cout<<"Three"<<endl;
                break;
            case 4: 
                cout<<"Four"<<endl;
                break;
            case 5: 
                cout<<"Five"<<endl;
                break;
            case 6: 
                cout<<"Six"<<endl;
                break;
            case 7: 
               cout<<"Seven"<<endl;
                break;
            case 8: 
               cout<<"Eight"<<endl;
                break;
            case 9: 
               cout<<"Nine"<<endl;
                break;
        }
        
        num = num / 10;
    }


  return 0;
}
