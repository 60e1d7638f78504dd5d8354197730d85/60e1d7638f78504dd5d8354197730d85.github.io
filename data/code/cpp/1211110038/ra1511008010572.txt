#include <iostream>
using namespace std;

int main()
{
  int origNum,digit,num, rem, sum = 0;
  
  cin >> origNum;

  num = origNum;

  while(num != 0)
  {
      digit = num % 10;
      sum += digit * digit * digit;
      num /= 10;
  }

  if(sum == origNum)
    cout<< "YES";
  else
    cout << "NO";

  return 0;
}
