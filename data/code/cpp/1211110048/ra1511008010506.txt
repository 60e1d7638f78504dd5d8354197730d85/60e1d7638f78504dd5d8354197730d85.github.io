#include<iostream>
using namespace std;
int main()
{
int i, n,t1=0,t2=1,next;
  cin>>n;
  for(i=1; i<=n; i++)
  {
    if(i==1)
    {
      cout<<t1<<endl;
      continue;
    }
  if(i==2)
  {
    cout<<t2<<endl;
    continue;
  }
  next=t1+t2;
  t1=t2;
  t2=next;
  cout<<next<<endl;
}
return 0;
}
