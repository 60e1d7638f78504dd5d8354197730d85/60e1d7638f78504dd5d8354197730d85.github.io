#include <iostream>
#include <string>         // std::string
#include <locale>         // std::locale, std::tolower
using namespace std;
int main()
{
  std::locale loc;
  std::string str;
  cin>>str;
  for (std::string::size_type i=0; i<str.length(); ++i)
    std::cout << std::tolower(str[i],loc);
  return 0;
}
