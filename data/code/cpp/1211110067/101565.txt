#include<iostream>
#include <bits/stdc++.h>
using namespace std;
int reverseDigits(int num)
{
    int rev = 0;
    while (num > 0)
    {
        rev = rev * 10 + num % 10;
        num /= 10;
    }
    return rev;
}
 
// To square number
int square(int num)
{
    return (num * num);
}
 
// To check Adam Number
bool checkAdamNumber(int num)
{
     
    int a = square(num);
    int b = square(reverseDigits(num));
     
   
    if (a == reverseDigits(b))
    return true;
    return false;        
}
 

int main()
{
    int num;
    cin>>num;
     
    if (checkAdamNumber(num))
    cout << "Adam Number";
    else
    cout << "Not an Adam Number";    
     
    return 0;
}
