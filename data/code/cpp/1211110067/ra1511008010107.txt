#include <iostream>
#include <math.h>
using namespace std;
int rev(int);
int power(int);
int sqroot(int);
int main()
{
   int n,r1=0,r2=0,sq=0,p=0;
      scanf("%d",&n);
      p=power(n);      
      r1=rev(p);      
      sq=(int)sqroot(r1);      
      r2=rev(sq);      
      if(n==r2)
      printf("Adam Number");
      else
      printf("Not an Adam Number");  
 
	return 0;
}
int rev(int n)
{
      int temp=0,r=0,t=0;
      temp=n;
      while(temp>0)
      {
            r=temp%10;
            t=t*10+r;
            temp=temp/10;
      }
      return t;
}
int power(int n)
{
      int p;
      p=n*n;
}
int sqroot(int n)
{
      int sq=0;
      sq=(int)sqrt(n);
      return sq;
}
