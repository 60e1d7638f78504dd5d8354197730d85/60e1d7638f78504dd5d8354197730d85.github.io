#include <iostream>
using namespace std;
int main()
{
    int n;
    cin>>n;
    if(n==10)
      cout<<"A";
      else if(n==11)
        cout<<"B";
      else if(n==12)
        cout<<"C";
      else if(n==13)
        cout<<"D";
      else if(n==14)
        cout<<"E";
       else
         cout<<"F";
        
	return 0;
}
