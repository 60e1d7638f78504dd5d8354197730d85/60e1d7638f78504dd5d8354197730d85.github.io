#include <iostream>
using namespace std;
int main()
{
  long int a;
  cin>>a;
  int c2000=0,c500=0,c100=0,c50=0,c20=0,c10=0,c5=0,c1=0;
  while(a!=0)
  {
    if(a>=2000)
    {
      a=a-2000;
      c2000++;
    }
    else if(a>=500)
    {
      a=a-500;
      c500++;
    }
    else if(a>=100)
    {
      a=a-100;
      c100++;
    }
    else if(a>=50)
    {
      a=a-50;
      c50++;
    }
    else if(a>=20)
    {
      a=a-20;
      c20++;
    }
    else if(a>=10)
    {
      a=a-10;
      c10++;
    }
    else if(a>=5)
    {
      a=a-5;
      c5++;
    }
    else if(a>=1)
    {
      c1=a;
      a= a%a;
    }
  }

 cout<<"Rs 2000="<<c2000<<"\n"<<"Rs 500="<<c500<<"\n"<<"Rs 100="<<c100<<"\n"<<"Rs 50="<<c50<<"\nRs 20="<<c20<<"\nRs 10="<<c10<<"\nRs 5="<<c5<<"\nRs 1="<<c1;
	return 0;
}
