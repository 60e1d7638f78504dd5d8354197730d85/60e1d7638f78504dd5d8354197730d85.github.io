#include <iostream>
using namespace std;
int main()
{
    int a,b,c,d;
    float avg;
    cin>>a>>b>>c>>d;
    avg=(a+b+c+d)/4;
    if(avg>95) cout<<"Medical Course";
    else if(avg>90 & avg<95) cout<<"Engineering Course";
    else if(avg>80 & avg<90) cout<<"Hotel Management";
    else cout<<"Arts and Science";
	return 0;
}
