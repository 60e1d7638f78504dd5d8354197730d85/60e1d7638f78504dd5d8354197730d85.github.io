#include <iostream>
using namespace std;
int main()
{

	long int decnum, rem, quot;
	char hexdecnum[100];
	int i=1, j, temp;

	cin>>decnum;
	quot=decnum;
	while(quot!=0)
	{
		temp=quot%16;
		// to convert integer into character
		if(temp<10)
		{
			temp=temp+48;
		}
		else
		{
			temp=temp+55;
		}
		hexdecnum[i++]=temp;
		quot=quot/16;
	}
	
	for(j=i-1; j>0; j--)
	{
		cout<<hexdecnum[j];
	}
	return 0;
}
