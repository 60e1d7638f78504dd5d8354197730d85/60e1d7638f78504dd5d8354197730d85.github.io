#include <iostream>
using namespace std;
int evenSum(int n)
{
    int curr = 2, sum = 0;
 
    // sum of first n even numbers
    for (int i = 1; i <= n; i++) {
        sum += curr;
 
        // next even number
        curr += 2;
    }
 
    // required sum
    return sum;
}
 
// Driver program to test above
int main()
{
    int n;
  cin>>n;
    cout<<evenSum(n);
  return 0;
}
