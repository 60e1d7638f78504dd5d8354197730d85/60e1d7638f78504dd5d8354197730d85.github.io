template<typename T>T absll(T X)
{
    if(X<0)
        return -1*X;
    else
        return X;
}
 
LL gcd(LL A,LL B)
{
	if(B==0)
	{
		return A;
	}
	else
	{
		return gcd(B,A%B);
	}
}
 
int main() 
{
	int T;
	double N;
	double exp=0.000;
	scanf("%d",&T);
	while(T--)
	{
		scanf("%lf",&N);
		exp=0.000;
		N*=1.00000;
		for(int i=1;i<=(int)N;i++)
		{
			exp+=N/((double)i);
		}
		printf("%.1lf\n",exp);
	}

}
