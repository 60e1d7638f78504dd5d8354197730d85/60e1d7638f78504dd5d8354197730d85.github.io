#include<bits/stdc++.h>
using namespace std;
int ip[200000];
int main(){
  int t;
  cin>>t;
  while(t--){
     int n;
     cin>>n;
 
     vector<int> a;;
     for(int i=0;i<n;i++){
       cin>>ip[i];
       vector<int>::iterator it=a.end();
       a.insert(it-ip[i],i);
     }
     int j=0;
     vector<int>::iterator it;
     for(it=a.begin();it!=a.end();it++){
          ip[*it]=++j;
     }
          for(int i=0;i<n;i++){
            cout<<ip[i]<<" ";
          }
          cout<<endl;
  }
  return 0;
}
