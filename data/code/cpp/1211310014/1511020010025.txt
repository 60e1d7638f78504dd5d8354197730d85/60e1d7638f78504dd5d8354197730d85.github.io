
#include<iostream>
using namespace std;
 
int countWays(int N)
{
    if (N == 1)
        return 4;

    int countB=1, countS=1, prev_countB, prev_countS;

    for (int i=2; i<=N; i++)
    {
        prev_countB = countB;
        prev_countS = countS;
 
        countS = prev_countB + prev_countS;
        countB = prev_countS;
    }
 
    int result = countS + countB;
 
    return (result*result);
}

int main()
{
    int n,t;
  cin>>t;
  while(t--)
  {cin>>n;
    cout  << countWays(n);
  }
    return 0;
}
