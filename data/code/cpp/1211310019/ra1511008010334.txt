#include<iostream>
#include<stdlib.h>
#include<string.h>
using namespace std;
#define N 1000000
int main()
{
int t;
cin>>t;
	while(t--)
	{
		char *a;
		a=(char*)malloc(sizeof(char)*100000);
		cin>>a;
		int size=strlen(a);
		int p=0;
		int c=1;
		char b[size];
		for(int i=0;i<size;i++)
		{
			if(a[i]=='1')
				{b[p]=i;p++;}
		}
		int i=0;
		do
		{	if(p==1)
			{c==1;
				break;}
			if(b[i+1]-b[i]!=1 || p==0 )
			{
				c=0;
				break;
			}
			i++;
		}while(i!=p-1);
		if(c==1)
			cout<<"YES\n";
		else 
			cout<<"NO\n";
	}
	return 0;
}
