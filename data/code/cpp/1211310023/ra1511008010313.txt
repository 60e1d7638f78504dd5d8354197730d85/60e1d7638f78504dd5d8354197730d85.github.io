#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios::sync_with_stdio(false);cin.tie(NULL);
    long long int A,p;
    int N,K;
    cin>>A>>N>>K;
    int a[10000]={0};
    for(int i=0;i<K;i++)
    {
        p=A%(N+1);
        a[i]=p;
        A=A/(N+1);
    }
    for(int k=0;k<K;k++)
    {
        cout<<a[k]<<" ";
    }
    cout<<endl;
    return 0;
}
