#include <bits/stdc++.h>
#define fl(i,n) for(i = 0; i < n; i++)
#define fe(i,n) for(i = 1; i <= n; i++)
#define lf(i,n) for(i = n -1 ; i >= 0; i--)
#define ef(i,n) for(i = n; i >= 1; i--)
#define ll long long int
#define ull unsigned long long int
using namespace std;
int main(int argc, char const *argv[]) {
    int a, n, k, c = 0, r;
    scanf("%d %d %d", &a, &n, &k);
    n = n + 1;
    while(a != 0) {
    	r = a % n;
    	a = a / n;
    	printf("%d ", r);
    	c++;
    	if(c == k) {
    		break;
    	}
    }
    while(c < k) {
    	printf("0 ");
    	c++;
    }
    printf("\n");
	return 0;
}
