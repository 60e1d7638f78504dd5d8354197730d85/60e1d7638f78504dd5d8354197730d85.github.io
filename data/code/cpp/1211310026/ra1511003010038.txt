int main ()
{
    int a;
    cin>>a;
    for(int i=0;i<a;i++){
        vector<int> x,y;
        int n;
        cin>>n;
        for(int j=0;j<n;j++){
            int k;
            cin>>k;
            x.push_back(k);
        }
        for(int j=0;j<n;j++){
            int k;
            cin>>k;
            y.push_back(k);
        }
        int ans1=0,ans2=0;
        for(int j=0;j<n;j++){
            if(j&1) ans1+=x[j];
            else ans1+=y[j];
        }
        for(int j=0;j<n;j++){
            if(j&1) ans2+=y[j];
            else ans2+=x[j];
        }
        cout<<min(ans1,ans2)<<endl;
    }
    return 0;
}
