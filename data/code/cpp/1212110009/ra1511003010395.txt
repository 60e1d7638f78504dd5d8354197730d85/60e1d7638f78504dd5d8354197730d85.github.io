#include <iostream>
using namespace std;
int main()
{
  int origNum, num, rem, sum = 0,digit;
  cin >> origNum;

  num = origNum;

  while(num != 0)
  {
      digit = num % 10;
      sum += digit * digit * digit;
      num /= 10;
  }

  if(sum == origNum)
    cout << "This is an Armstrong number";
  else
    cout << "This is not an Armstrong number";


	return 0;
}
