#include <iostream>
using namespace std;
int main()
{
    int n,m,p;
    cin>>n;
    int a[30];
    int i=0;
    while(n>0)
    {
       m=n%10;
       a[i]=m;
      i++;
      n=n/10;
    }
    for(int j=i-1;j>=0;j--)
    {
      if(a[j]==0)
       cout<<"Zero ";
      if(a[j]==1)
        cout<<"One ";
      if(a[j]==2)
        cout<<"Two ";
      if(a[j]==3)
        cout<<"Three ";
      if(a[j]==4)
        cout<<"Four ";
      if(a[j]==5)
        cout<<"Five ";
      if(a[j]==6)
        cout<<"Six ";
      if(a[j]==7)
        cout<<"Seven ";
      if(a[j]==8)
        cout<<"Eight ";
      if(a[j]==9)
        cout<<"Nine ";
    }
	return 0;
}
