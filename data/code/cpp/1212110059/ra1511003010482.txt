{

  {
    cout<<c<<" "<<b<<" "<<a;
  }
  else if(a>c && c>b)

  else if(b>c && c>a)

  else if(c>a && a>b)
    cout<<b<<" "<<a<<" "<<c;
  else
    cout<<a<<" "<<b<<" "<<c;

}
