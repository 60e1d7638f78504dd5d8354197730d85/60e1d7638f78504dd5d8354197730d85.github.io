using namespace std;
int main()
{   int a[3],temp=0;
    for(int i=0;i<3;i++){cin>>a[i];}

      for(int j=i+1;j<3;j++){
        if(a[i]>a[j]){
         temp=a[i];
          a[i]=a[j];

        }
      }
    }
 for(int i=0;i<3;i++){cout<<a[i]<<" ";}

	return 0;
}
