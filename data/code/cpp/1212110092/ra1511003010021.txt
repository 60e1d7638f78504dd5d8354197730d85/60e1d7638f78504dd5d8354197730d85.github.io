#include <iostream>
using namespace std;
int main()
{int a,b;
 char c;
 cin>>a>>b;
 cin>>c;
 switch(c)
 {
   case '+':cout<<a+b;
     break;
   case '-':cout<<a-b;
     break;
   case '*':cout<<a*b;
     break;
   case '/':cout<<a/b;
     break;
   default:cout<<"Invalid operator";
     break;
 }
 	return 0;
}
