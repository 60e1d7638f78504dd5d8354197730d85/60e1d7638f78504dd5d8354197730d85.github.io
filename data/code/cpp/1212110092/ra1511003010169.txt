#include <iostream>
using namespace std;
int main()
{
int a,b;
  char ch;
  cin>>a>>b>>ch;
  switch(ch)
  {
    case '+':
      cout<<a+b;
      break;
    case '-':
      cout<<a-b;
      break;
    case '*':
      cout<<a*b;
      break;
    case '/':
      cout<<a/b;
      break;
    default:
      cout<<"Invalid operator";
  }
	return 0;
}
