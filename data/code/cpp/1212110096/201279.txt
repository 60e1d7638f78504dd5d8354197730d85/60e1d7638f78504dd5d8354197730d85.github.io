#include <iostream>
using namespace std;
int main()
{
 int n, max=0, min=32767;
    char choice;

    do
    {
           cin>>n;
        if(n>max)
            max=n;
        if(n<min)
            min=n;
              cin>>choice;
    }
    while(choice=='y' || choice=='Y');
    cout<<"Maximum Number :"<<max<<"\nMinimum Number :"<<min;
	return 0;
}
