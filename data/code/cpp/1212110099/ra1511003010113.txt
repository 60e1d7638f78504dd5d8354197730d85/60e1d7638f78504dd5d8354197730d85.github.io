#include <iostream>
using namespace std;
bool checkPrime(int n);
int main()
{
  int n,i;
  cin>>n;
  for(i=2;i<=n/2;i++)
   {
    if(checkPrime(i))
    {
      if(checkPrime(n-i))
      {
        cout<<n<<" = "<<i<<" + "<<n-i<<"\n";
      }
    }
  }
   return 0;
}
bool checkPrime(int n)
{
  int i;
  bool isPrime = true;
  for(i=2;i<=n/2;++i)
  {
    if(n % i == 0)
    {
      isPrime = false;
      break;
    }
  }
}
