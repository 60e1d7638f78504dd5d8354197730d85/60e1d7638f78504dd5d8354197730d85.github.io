#include <iostream>
using namespace std;
int checkPrime(int n)
{
    int i, isPrime = 1;

    for(i = 2; i <= n/2; ++i)
    {
        if(n % i == 0)
        {
            isPrime = 0;
            break;
        }  
    }

    return isPrime;
}
int main()
{
  int n, i, flag = 0;
   cin>>n;

    for(i = 2; i <= n/2; ++i)
    {
        if (checkPrime(i) == 1)
        {
            if (checkPrime(n-i) == 1)
            {
                cout<<n<<" = "<<i<<" + "<<n-i<<endl;
                flag = 1;
            }
        }
    }
	return 0;
}
