#include <iostream>
using namespace std;

bool prime(int n) {
  	bool flag = true;
	for(int i=2; i<=n; i++) {
    	int a = n/i;
      	if(i > a) {
          	break;
        }
      	if(n % i == 0) {
        	flag = false;
          	break;
        }
    }
  	return flag;
}

int main()
{
  	int num;
  	cin>>num;
  	cout<<(prime(num)==true ? "prime number" : "Not a prime number");
	return 0;
}
