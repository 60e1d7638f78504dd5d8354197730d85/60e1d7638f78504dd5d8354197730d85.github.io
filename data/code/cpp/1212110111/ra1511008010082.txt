#include <iostream>
using namespace std;
int  getNumberOfDays(int month, int year)
{
	
	if( month == 2)
	{
		if((year%400==0) || (year%4==0 && year%100!=0))	
			return 29;
		else	
			return 28;
	}
	//months which has 31 days
	else if(month == 1 || month == 3 || month == 5 || month == 7 || month == 8
	||month == 10 || month==12)	
		return 31;
	else 		
		return 30;
} 


int main()
{
	int days=0,month,year;
  cin>>month;
  cin>>year;

	days = getNumberOfDays(month, year);

	cout<<endl<<" "<<days;
	return 0;
}
