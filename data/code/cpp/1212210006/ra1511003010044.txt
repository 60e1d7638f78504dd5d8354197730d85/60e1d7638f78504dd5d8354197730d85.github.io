#include <iostream>
using namespace std;
int main()

{
    int i,j,k=1,n;
    cin>>n;
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=i;j++)
        {
            if(k%2==0)
            {
                cout<<"0";
            }
            else
            {
                cout<<"1";
            }
            k++;
        }
        cout<<"\n";
    }
return 0; 
}
