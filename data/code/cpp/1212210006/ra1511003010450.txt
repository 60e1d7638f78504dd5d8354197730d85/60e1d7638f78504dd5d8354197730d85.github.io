#include <iostream>
using namespace std;
int main()
{
  int p, lastInt = 0, input;
  cin>>input;
  for (int i = 1; i <= input; i++)
  {
      	for (p = 1; p <= i; p++)
  		{
        	if (lastInt == 1)
            {
                cout<<"0";
            	lastInt = 0;
            }
            else if (lastInt == 0)
            {
                 cout<<"1";
                 lastInt = 1;
            }
        } 
    cout<<endl;
  }

	return 0;
}
