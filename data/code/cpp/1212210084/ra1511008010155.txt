#include <bits/stdc++.h>
using namespace std;
 
#define mx 100005
#define ll long long
#define MOD 1000000007
 
int main()
{
	int t;
	scanf("%d", &t);
	while(t--)
	{
		string st;
		cin >> st;
		int f = 0;
		for(int i = 0; i<st.length(); i++)
		{
			if(i && st[i-1] == st[i]) f = 1;
			if(i>=2 && st[i-2] == st[i]) f = 1; 
		}
		if(f) cout << "YES" << endl;
		else cout << "NO" << endl;
	}
	return 0;
}
