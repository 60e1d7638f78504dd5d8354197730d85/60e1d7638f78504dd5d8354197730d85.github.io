#include<bits/stdc++.h>
typedef long long int ll;
using namespace std;
int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
 
    int t;
    string s;
    bool ans;
    cin>>t;
    while (t--) {
        cin>>s;
        ans=false;
        for (int i=1;(i<s.size())&&(!ans);i++) {
            if ((s[i]==s[i-1])||((s[i]==s[i-2])&&(i>1))) ans=true;
        }
        if (ans) cout<<"YES\n";
        else cout<<"NO\n";
    }
}
