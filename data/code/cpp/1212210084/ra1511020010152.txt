#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int pal(char f[],int q)
{
	int d=0,v=0,b=2;
	if(q == 1 || q == 0)
	return 0;
	else
	if(q == 2)
	{
		if(f[d] == f[d+1])
		return 1;
		
	}
	else
	{
		while(b < q)
		{
		if(f[v] == f[b])
		{
		return 1;
		break;
	}
	v++;
	b++;
	}
	}
	return 0;
}
int sub(char y[],int l)
{
int i,j,k,p;
for(i=0;i<l;i++)
{
	for(j=i;j<l;j++)
	{
		char f[100000];
		for(k=i,p=0;k<=j;k++,p++)
		{
			f[p]=y[k];
		}
		int h=pal(f,p);
		if(h == 1)
		{
			return 1;
			break;
		}
	}
}
return 0; 
}
int main()
{
int x;
cin>>x;
int z[100000];
int c=0;
int i;
for(i=0;i<x;i++)
{
char y[100000];
	scanf("%s",y);
	int l=strlen(y);
	int w=sub(y,l);
	if(w == 1)
	{
	z[c]=w;
	c++;
}
else
{
	z[c]=0;
	c++;
}
}
for(int j=0;j<x;j++)
{
	if(z[j] == 1)
	cout<<"YES\n";
	else
	cout<<"NO\n";
}
return 0;
}
