#include <iostream>

using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,cnt=0;
        string s;
        cin>>s;
        n=s.length();
        for(int i=0;i<n;i++)
            if(s[i]!='4'&&s[i]!='7')
                cnt++;
        cout<<cnt<<endl;
    }
    return 0;
}
