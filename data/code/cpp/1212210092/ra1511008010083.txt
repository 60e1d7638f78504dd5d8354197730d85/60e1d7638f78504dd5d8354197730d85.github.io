#include <iostream>
using namespace std;

int main()
{
  int k; 
  cin>>k; 
  for(int j=1;j<=k;j++) 
  { 
    int n;
    cin>>n;
    int a[n],sum=1,i,count=0;
    for(i=0;i<n;i++)
	{
    	cin>>a[i];
	}
	for(i=0;i<n;i++)
	{
    	sum*=a[i];
	}
    for(i=0;i<n;i++)
    {
		if(sum==a[i])
		{
   			count++;
		}
	}
	if(count>=1)
	{
    	cout<<"yes"<<endl;
	}
	else
	{
    	cout<<"no"<<endl;
    }
}
return 0;
}
