#include<iostream>
#include <cstdlib>
using namespace std;
 
int main() {
    long long int test, n;
    cin>>test;
    while (test--) {
        long long int minus = 0, one = 0, zero = 0, other = 0, i;
        cin>>n;
        int a[n];
        for (i = 0; i < n; ++i) {
            cin >> a[i];
            if (a[i] == -1)
                ++minus;
            else if (a[i] == 0)
                ++zero;
            else if (a[i] == 1)
                ++one;
            else
                ++other;
        }
        if (other > 1)
            cout << "no\n";
        else if (minus != 0 && other == 1)
            cout << "no\n";
        else if (minus > 1 && one == 0)
            cout << "no\n";
        else
            cout << "yes\n";
    }
    return 0;
}
