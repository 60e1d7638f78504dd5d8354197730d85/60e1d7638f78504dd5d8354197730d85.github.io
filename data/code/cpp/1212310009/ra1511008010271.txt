#include <iostream>
using namespace std;
int main()
{	int t;
 	cin>>t;
 	while(t--){
    	int n;
      	cin>>n;
      	int i;
      	int flag=0;
      	for(i=1;i<n;i++){
        	if(n%i==0)
              flag++;
        }
      if(flag==1)
        cout<<"yes"<<endl;
      else
        cout<<"no"<<endl;
    }

	return 0;
}
