void sc(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}
lli power(lli a,lli b)
{
    lli res=1;
    while(b>0)
    {
        if(b%2!=0)
            res=(res*a)%m;
        a=(a*a)%m;
        b/=2;
    }
    return res%m;
}
