#include<iostream>
#include<algorithm>
using namespace std;
 
 
int main()
{
	
	int  t;
	cin>>t;
	
	while(t--)
	{
		 int  n;
		 cin>>n;
		 long long int a[n];
		 
		 for(int i=0;i<n;i++)
		 {
		 	cin>>a[i];
		 }
		 
		 sort(a,a+n);
		 
		long long min=a[n-1]*a[n-2];
		for(int i=0;i<n-1;i++){
			for(int j=i+1;j<n;j++){
				if(((a[i]*a[j])/__gcd(a[i],a[j]))<min)
				min=((a[i]*a[j])/__gcd(a[i],a[j]));
			}
		}
		cout<<min<<endl;
	}
}
