#include <bits/stdc++.h>
using namespace std;
int minTimeForWritingChars(int N, int insert,
                       int remove, int copy)
{
    if (N == 0)
       return 0;
    if (N == 1)
       return insert;
    int dp[N + 1];
    memset(dp, 0, sizeof(dp));
    for (int i = 1; i <= N; i++)
    {
        if (i % 2 == 0)
            dp[i] = min(dp[i-1] + insert,
                        dp[i/2] + copy);
        else
            dp[i] = min(dp[i-1] + insert,
                        dp[(i+1)/2] + copy + remove);
    }
    return dp[N];
}
int main()
{
    int N,j,insert,remove,copy;
    cin>>N;
  for(int i=0;i<N;i++){
    cin>>j;
    cin>>insert>>remove>>copy;
    cout << minTimeForWritingChars(j, insert,
                                remove, copy);}
    return 0;
}
