#include <iostream>
#include<stdio.h>
using namespace std;
int main()
{char line[150];
    int i,v,c,ch,d,s,o;
    o=v=c=ch=d=s=0;
    fgets(line,150,stdin);
    for(i=0;line[i]!='\0';++i)
    {
        if(line[i]=='a' || line[i]=='e' || line[i]=='i' || line[i]=='o' || line[i]=='u' || line[i]=='A' || line[i]=='E' || line[i]=='I' || line[i]=='O' || line[i]=='U')
            ++v;
        else if((line[i]>='a'&& line[i]<='z') || (line[i]>='A'&& line[i]<='Z'))
            ++c;
        else if(line[i]>='0'&& line[i]<='9')
            ++d;
        else if (line[i]==' ')
            ++s;
    }
    cout<<"Vowels:"<<v;
    cout<<"\nConsonants:"<<c;
    cout<<"\nWhitespaces:"<<s;
    cout<<"\nDigits:"<<d;
	return 0;
}
