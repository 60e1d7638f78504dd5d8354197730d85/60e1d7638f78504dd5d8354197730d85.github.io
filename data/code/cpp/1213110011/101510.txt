#include <iostream>
#include<string.h>
using namespace std;
int main()
{
char str[10]; 
int i,j,n,flag=0; 
cin>>str; 
n=strlen(str); 
for(i=0,j=n-1;i<=n/2;i++,j--) 
{ 
if(str[i]!=str[j]) 
flag=1; 
break; 
} 
if(flag==1) 
{
cout<<str<<" string is not a palindrome"; 
}
else
{
cout<<str<<" string is a palindrome"; 
}
	return 0;
}
