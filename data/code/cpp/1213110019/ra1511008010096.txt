#include <iostream>
#include<string.h>
using namespace std;
int main()
{

  string str;
  int i,l;
 getline(cin,str);
  l=str.length();
  for(i=0;i<l;i++)
  {
    if(!((str[i]>='A' && str[i]<='Z')||(str[i]>='a' && str[i]<='z')))
    {
      str[i]='\0';
    }
  }
  cout<<"Output String:"<<str;
	return 0;
}
