#include<iostream>
using namespace std;

int main()
{
int i,j,len,flag=1;
char a[20];
cin>>a;
    
for(len=0;a[len]!='\0';++len);
   
for(i=0,j=len-1;i<len/2;++i,--j)
{
if(a[j]!=a[i])
flag=0;
}
   
if(flag==1)
{
cout<<"\nGiven string is a palindrome";
}
else
{
cout<<"\nGiven string is not a palindrome";
}
return 0;
}
