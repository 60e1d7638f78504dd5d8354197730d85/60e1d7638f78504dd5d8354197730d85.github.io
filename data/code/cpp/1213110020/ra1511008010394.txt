#include <iostream>
#include <string.h>
using namespace std;
int main()
{
	char a[20];
  	int i, l, f = 0;
  	cin>>a;
  	l = strlen(a);
  	for(i=0;i<l;i++){
    	if(a[i]!=a[l-i-1])
        {  
          	f=1;
      		break;
        }
    }
  	if(f==1)
      cout<<"Given string is not a palindrome";
  	if(f==0)
      cout<<"Given string is a palindrome";
	return 0;
}
