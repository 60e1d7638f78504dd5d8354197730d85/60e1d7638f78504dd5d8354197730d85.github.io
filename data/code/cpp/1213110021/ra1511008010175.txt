#include <iostream>
#include <string>
using namespace std;
int main()
{
	string s;
  	int temp;
    char c;
  	getline(cin,s);
  	int l=s.length();
  	for(int i=0;i<l;i++)
    {
      	c=s.at(i);
      	temp=(int)c;
      	if(temp>=97 && temp<=122)
        {
          	c=(char)(temp-32);
          	cout<<c;
        }
      	else
          	cout<<c;
    }
	return 0;
}
