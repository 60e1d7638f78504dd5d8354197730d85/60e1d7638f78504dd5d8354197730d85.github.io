#include <iostream>
#include <string.h>
using namespace std;
int main()
{
	char s[100],c[100];
  	cin>>s; int i=0;
  	while(s[i]!='\0') {i++;}
  	for(int j=i;j>0;j--)
      c[j-1]=s[i-j];
  	c[i]='\0';
  	
  	int check=1;
  	int n=i;
  	for(i=0;i<n;i++)
      if(c[i]!=s[i]) check=0;
  	if(check==1) cout<<"Good Choice";
  	else cout<<"Bad Choice";
	return 0;
}
