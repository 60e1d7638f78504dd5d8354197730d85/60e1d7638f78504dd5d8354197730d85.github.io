#include <iostream>
#include<string.h>
using namespace std;
int main()
{
  char str[50];
  int l,c,c1,c2=0;
  cin.getline(str,50);
  l=strlen(str);
  c1=l/2;
  for(int i=0;i<l;i++)
  {
  if(str[i]=='a' || str[i]=='e' || str[i]== 'i' || str[i]== 'o' || str[i]== 'u')
    c2++;
  }
  if(c1==c2)
    cout<<"Perfect";
  else
    cout<<"Wrong Input";
  return 0;
}
