#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;
int main()
{
	string str;
  	cin>>str;
  	
  	for(int i=0;i<str.length();i++)
    {
     	if(str[i]>64 && str[i]<91)
          str[i] = str[i]+32;
    }
  
  	for(int i=0;i<str.length();i++)
    {
     	cout<<str[i]; 
    }
  
	return 0;
}
