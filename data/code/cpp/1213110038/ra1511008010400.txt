#include <iostream>
#include <string>

using namespace std;

int main()
{
        string str,sub;
        int pos;
 
        getline(cin,str);
 
        cin>>sub;
 
        pos=str.find(sub,0);
 
         if(pos==-1)
        {
               cout<<"SUBSTRING NOT FOUND !";
        }
        else
        {
               cout<<"Substring found at position "<<pos+1;
        }
 
         return 0;
}
