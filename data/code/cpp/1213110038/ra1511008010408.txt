#include <iostream>
#include <string>

using namespace std;

int main()
{
        string str,sub;
        int pos;
        cin>>str;
        cin>>sub;
 
        pos=str.find(sub,0);
 
         if(pos==-1)
        {
               cout<<"Substring not found";
        }
        else
        {
               cout<<"Substring found at position "<<pos+1;
        }
 
         return 0;
}
