#include <iostream>
using namespace std;
int main()
{
  string a[5];
  for(int i=0;i<5;i++)
  {
  	cin>>a[i];
  	//cout<<a[i]<<"\n";
  }
  for(int i=0;i<5;i++)
  {
    for(int j=i+1;j<5;j++)
    {
      if(a[i][0]>a[j][0])
      {
        string temp=a[i];
        a[i]=a[j];
        a[j]=temp;
      }
    }
  }
  for(int i=0;i<5;i++)
  {
  	cout<<a[i]<<"\n";
  }
	return 0;
}
