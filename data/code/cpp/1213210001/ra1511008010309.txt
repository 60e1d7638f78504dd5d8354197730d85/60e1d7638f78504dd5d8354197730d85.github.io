#include <iostream>
using namespace std;

int main()
{
    string str;
  cin>>str;
    char checkCharacter;
  cin>>checkCharacter;
    int count = 0;

    for (int i = 0; i < str.size(); i++)
    {
        if (str[i] ==  checkCharacter)
        {
            ++ count;
        }
    }

    cout << count;

    return 0;
}
