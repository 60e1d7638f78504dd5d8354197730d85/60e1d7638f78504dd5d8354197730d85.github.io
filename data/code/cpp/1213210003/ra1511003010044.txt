#include <iostream>
using namespace std;
int main()
{ char str[100];
  int v=0,c=0,d=0,w=0,i;
  cin.getline(str,100,'\n');
  for (i=0;str[i]!='\0';i++)
    { if (str[i]=='a' || str[i]=='e' || str[i]=='i' || str[i]=='o' || str[i]=='u' || str[i]=='A' || str[i]=='E' || str[i]=='I' || str[i]=='O' || str[i]=='U')
         v=v+1;
      else if (str[i]>='0' && str[i]<='9')
         d=d+1;
      else if (str[i]==' ')
         w=w+1;
      else
         c++;
    }
  cout<<"Vowels: "<<v<<endl;
  cout<<"Consonants: "<<c<<endl;
  cout<<"Digits: "<<d<<endl;
  cout<<"White spaces: "<<w;
	return 0;
}
