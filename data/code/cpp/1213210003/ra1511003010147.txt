#include <iostream>
#include <string.h>
using namespace std;
int main()
{
	char x[25];
  	int v=0,c=0,d=0,s=0;
  	fgets(x,sizeof(x),stdin);
  	for(int i=0;x[i]!='\0';i++)
    {
      if(x[i]==' ')
        s++;
      else if(isdigit(x[i]))
        d++;
      else if(x[i]=='a'||x[i]=='e'||x[i]=='i'||x[i]=='o'||x[i]=='u'||x[i]=='A'||x[i]=='E'||x[i]=='I'||x[i]=='O'||x[i]=='U')
        v++;
      else 
        c++;
    }
  	cout<<"Vowels: "<<v<<"\nConsonants: "<<c<<"\nDigits: "<<d<<"\nWhite spaces: "<<s;
	return 0;
}
