#include <iostream>       // std::cout
#include <cstring>         // std::string
#include <locale>         // std::locale, std::toupper
#include<string.h>
using namespace std;
int main ()
{
  std::locale loc;
  std::string str;
  getline(cin,str);
  for (std::string::size_type i=0; i<str.length(); ++i)
    std::cout << std::tolower(str[i],loc);
  return 0;
}
