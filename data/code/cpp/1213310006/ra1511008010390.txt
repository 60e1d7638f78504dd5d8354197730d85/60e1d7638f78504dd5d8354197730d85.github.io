#include<bits/stdc++.h>
using namespace std;
 
int main()
{
	int t;
	scanf("%d",&t);
	getchar();
	while(t--){
        vector<string> vec;
        string str,out;
        stringstream ss;
        getline(cin,str);
        ss<<str;
        while(ss>>out)
        {
            vec.push_back(out);
        }
 
        int n=vec.size();
        //cout << n << endl;
        for(int i=0;i<n;i++)
        {
            string temp=vec[i];
            if(i==n-1)
            {
                char c=toupper(temp[0]);
                cout << c;
                //cout << vec[i].size() << endl;
                for(int j=1;j<temp.size();j++)
                {
                    c=tolower(temp[j]);
                    cout << c;
                }
 
 
                cout << endl;
            }
            else
            {
                char c=toupper(vec[i][0]);
                cout << c << "." << " ";
            }
 
        }
 
 
	}
	return 0;
}
