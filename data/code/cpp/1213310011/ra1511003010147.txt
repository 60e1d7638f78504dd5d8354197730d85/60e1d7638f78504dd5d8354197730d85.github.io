#include <iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--){
		char team_name[20];
		int score,RealMadrid=0,Barcelona=0,R,B,M,E;
		for(int i=0; i<4; i++){
			cin>>team_name;
			cin>>score;
			if(team_name[0]=='R')
			R=score;
			if(team_name[0]=='B')
			B=score;
			if(team_name[0]=='M')
			M=score;
			if(team_name[0]=='E')
			E=score;
	}
		if(R>=M)
		RealMadrid++;
		if(B>E)
		Barcelona++;
		if(RealMadrid==0 && Barcelona==1)
		cout<<"Barcelona"<<endl;
		else
		cout<<"RealMadrid"<<endl;
	}

	return 0;
}
