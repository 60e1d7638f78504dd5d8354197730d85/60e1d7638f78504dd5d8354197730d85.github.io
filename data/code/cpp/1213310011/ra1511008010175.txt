#include <iostream>
#include <string.h>
using namespace std;
 
int main() {
	int t,n,i,a,b,c,d;
	string x,g="RealMadrid",h="Malaga",j="Barcelona",k="Eibar";
	cin>>t;
	while(t--)
	{
	    a=0;b=0;c=0;d=0;
	    for(i=0;i<4;i++)
	    {
	        cin>>x;
	        if(x.compare(g)==0)
	        {
	            cin>>a;
	        }
	        else if(x.compare(h)==0)
	        {
	            cin>>b;
	        }
	        else if(x.compare(j)==0)
	        {
	            cin>>c;
	        }
	        else if(x.compare(k)==0)
	        {
	            cin>>d;
	        }
	    }
	    if(a>b||a==b)
	    {
	        cout<<"RealMadrid"<<endl;
	    }
	    else
	    {
	        if(c>d)
	        {
	            cout<<"Barcelona"<<endl;
	        }
	        else
	        {
	            cout<<"RealMadrid"<<endl;
	        }
	    }
	}
	return 0;
}
