#include <iostream>
using namespace std;
bool isPalindrome(string str)
{
    int l = 0;
    int h = str.length() - 1;
    while (h > l)
        if (str[l++] != str[h--])
            return false;
    return true;
}
 
bool isRotationOfPalindrome(string str)
{
   if (isPalindrome(str))
         return true;
   int n = str.length();
   for (int i = 0; i < n-1; i++)
   {
       string str1 = str.substr(i+1, n-i-1);
       string str2 = str.substr(0, i+1);
       if (isPalindrome(str1.append(str2)))
          return true;
   }
   return false;
}
 
int main()
{
  string x;
  cin>>x;
  if(isRotationOfPalindrome(x))
    cout<<"1";
  else
    cout<<"0";
    return 0;
}
