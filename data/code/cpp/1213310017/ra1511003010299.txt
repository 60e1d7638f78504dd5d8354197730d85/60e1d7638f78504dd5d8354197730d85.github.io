#include <stdio.h>
#include <string.h>
int main(void) 
{
	char String[101];
	int t;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",String);
		int i,j;
		int ans = 1,length = strlen(String);
		for(i=0,j=2;j<length + 1;i++,j++)
		{
			if(String[i]==String[j-1])
			{
				ans = 0;
				break;
			}
			if(j<length)
			{
				if(String[i]!=String[j])
				{
					ans = 0;
					break;
				}
			}
		}
		if(ans)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
