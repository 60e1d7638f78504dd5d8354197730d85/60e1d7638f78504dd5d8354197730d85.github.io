#include <iostream>
#include <cstring>
using namespace std;
int main()
{
  int t;
  cin>>t;
  while(t--){
    string s;
    cin>>s;
    int l=s.length();
    char ch=s[0];
    int flag=0;
    for(int i=0;i<l;i=i+2)
    {
      if(ch!=s[i])
        flag=1;
    }
    if(flag==0)
      cout<<"YES"<<endl;
    else
      cout<<"NO"<<endl;
  }
	return 0;
}
