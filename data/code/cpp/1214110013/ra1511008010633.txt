#include <iostream>
using namespace std;
int main(){
   int num;
   bool flag = true;
   
   cin>>num;

   for(int i = 2; i <= num / 2; i++) {
      if(num % i == 0) {
         flag = false;
         break;
      }
   }
   if (flag==true)
      cout<<"Prime number";
   else
      cout<<"Not a prime number";
   return 0;
}
