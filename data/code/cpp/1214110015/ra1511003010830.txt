#include<iostream>
using namespace std;
class power
{
public:
inline int square(int n)
{
return n*n;
}
inline int cube(int n)
{
return n*n*n;
}
};
int main()
{
int n,r;
power p;
cin>>n;
r=p.square(n);
cout<<"Area of the square"<<"="<<r<<endl;
r=p.cube(n);
cout<<"Volume of the cube"<<"="<<r<<endl;
return 0;
}
