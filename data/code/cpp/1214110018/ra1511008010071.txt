void time::sum(time t1,time t2)
{
	if(t1.sec >=60 || t2.sec>=60 || t1.min >=60 || t2.min >=60)
      sec=-1;
  else{
   	sec=t1.sec+t2.sec;
	min=sec/60;
	sec=sec%60;
	min=min+t1.min+t2.min;
	hr=min/60;
	min=min%60;
	hr=hr+t1.hr+t2.hr;
  }
}
