#include <iostream>
using namespace std;

class triangle{
  private: int a, b , c ;
  public:
  	void readinput(){
      cin >> a >> b >> c ;
    }
  	void displayinput(){
      cout << a << ", " << b << ", " << c << endl ;
    }
  	void check(){
      if ( a==b || a==c || b==c )
        cout << "ISOSCELES" ;
      else
        cout << "NOT ISOSCELES" ;
    }
} ;

int main()
{
	triangle t ;
  	t.readinput() ;
  	t.displayinput() ;
  	t.check() ;
	return 0;
}
