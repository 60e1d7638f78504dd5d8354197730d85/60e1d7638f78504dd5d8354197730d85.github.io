#include <iostream>
using namespace std;
class triangle
{
 public: int a,b,c;
  
  void readinput()
  {
    cin>>a>>b>>c;
  }
  void displayinput()
  {
    cout<<a<<", "<<b<<", "<<c<<"\n";
  }
  void check()
  {
    if(a==b || b==c || a==c)
      cout<<"ISOSCELES";
    else
      cout<<"NOT ISOSCELES";
  }
};
int main()
{
  triangle t;
  t.readinput();
  t.displayinput();
  t.check();

	return 0;
}
