#include <iostream>
using namespace std;
class num
{
int n;
public: void getdata();
        void check();
};
void  num::getdata()
{
 cin>>n;
}
void num::check()
{
 if(n%2==0)
  cout<<"even";
 else if(n<0)
  cout<<"negative";
  else 
    cout<<"odd";
}
int main()
{
	num n1;
 n1.getdata();
 n1.check();
	return 0;
}
