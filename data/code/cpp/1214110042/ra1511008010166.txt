#include<iostream>
#include <string.h>
using namespace std;
class BOOK
{
	int BOOKNO;
	char BOOKTITLE[20];
	float PRICE;
	void TOTAL_COST(int N)
	{
		float tcost;
		tcost=PRICE*N;
		cout<<tcost;
	}
public:
	void INPUT()
	{
		cin>>BOOKNO;
		cin>>BOOKTITLE;
		cin>>PRICE;
	}
 
	void PURCHASE()
	{
		int n;
		cin>>n;
		cout<<"Total cost is ";
		TOTAL_COST(n);
	}
};
 
int main()
{
	BOOK obj;
	obj.INPUT();
	obj.PURCHASE();
}
