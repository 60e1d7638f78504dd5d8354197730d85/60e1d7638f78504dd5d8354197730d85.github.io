#include <iostream>
using namespace std;

int main()
{
  int origNum, num, rem, sum = 0;
  cin >> origNum;

  num = origNum;

  while(num != 0)
  {
      int digit = num % 10;
      sum += digit * digit * digit;
      num /= 10;
  }

  if(sum == origNum)
    cout <<"Is an Armstrong number";
  else
    cout <<"Is not an Armstrong number";

  return 0;
}
