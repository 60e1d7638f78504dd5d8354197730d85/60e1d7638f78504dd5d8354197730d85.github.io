#include <iostream>
#include <math.h>
using namespace std;
int main()
{
  int no;
  cin>>no;
if(no<0)
  cout<<"Invalid Input";
  else
  {
   int x=no; 
    int y=no;
    int count=0;
    int sum=0;
    
    while(y!=0)
    {
     count++;
      y/=10;
    }
    while(x!=0)
    {
     int temp=x%10; 
      sum+=pow(temp,count);
      x/=10;
    }
if(sum==no)
  cout<<no<<" is an Armstrong number";
    else
      cout<<no<<" is not an Armstrong number";
  }
	return 0;
}
