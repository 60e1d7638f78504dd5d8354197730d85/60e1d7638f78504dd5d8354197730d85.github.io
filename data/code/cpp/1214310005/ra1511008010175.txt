#include <iostream>
using namespace std;

int main()
{
  int digit,origNum, num, rem, sum = 0;
  cin >> origNum;
  if(origNum<0)
    cout<<"Invalid Input";
  else
  {
  num = origNum;
  while(num != 0)
  {
      digit = num % 10;
      sum += digit * digit * digit;
      num /= 10;
  }

  if(sum == origNum)
    cout << origNum << " is an Armstrong number";
  else
    cout << origNum << " is not an Armstrong number";
  }
  return 0;
}
