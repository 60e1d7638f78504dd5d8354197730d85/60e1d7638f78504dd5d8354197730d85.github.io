#include <iostream>
using namespace std;

int main()
{
  int origNum, num, rem, sum = 0,digit;
 
  cin >> origNum;
if(origNum<0){
  cout<<"Invalid Input";
  return 0;}
  num = origNum;

  while(num != 0)
  {
      digit = num % 10;
      sum += digit * digit * digit;
      num /= 10;
  }

  if(sum == origNum)
    cout << origNum << " is an Armstrong number";
  else
    cout << origNum << " is not an Armstrong number";

  return 0;
}
