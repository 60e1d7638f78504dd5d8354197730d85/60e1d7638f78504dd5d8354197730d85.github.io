#include <iostream>
using namespace std;
int main()
{
  	int a=0;
  	float b=0;
  	cin>>a>>b;
  if(a>0)
		cout<<"The absolute value is "<<a<<endl;
	else
		cout<<"The absolute value is "<<-(a)<<endl;
  if(b>0)
		cout<<"The absolute value is "<<b<<endl;
	else
		cout<<"The absolute value is "<<-(b)<<endl;

	return 0;
}
