#include <iostream>
using namespace std;
 
template <class T>
T Large(T n1, T n2)
{
	return (n1 > n2) ? n1 : n2;
}
 
int main()
{
	int i1, i2;
	float f1, f2;
	
 
	
	cin >> i1 >> i2;
	cout <<"The greatest no is "<< Large(i1, i2) << endl;
 
	
	cin >> f1 >> f2;
	cout <<"The greatest no is " <<Large(f1, f2) << endl;
 
	
	return 0;
}
