#include <iostream>
#define pi 3.14
using namespace std;
void calc(int r,int h);
void calc(int a);
void calc(float rd);
int main()
{
	int r,h;
  	int a;
  	float rd;
  	cin>>a;
  	cin>>r>>h;
	cin>>rd;
  	calc(a);
  	calc(r,h);
  	calc(rd);
	return 0;
}
void calc(int r,int h){
  float vol;
  vol=pi*r*r*h;
  cout<<"Volume of cylinder is "<<vol<<endl;
}
void calc(int a){
  float vol;
  vol=a*a*a;
  cout<<"Volume of cube is "<<vol<<endl;
}
void calc(float rd){
  float vol;
  vol=(4*pi*rd*rd*rd)/3;
  cout<<"Volume of sphere is "<<vol<<endl;
}
