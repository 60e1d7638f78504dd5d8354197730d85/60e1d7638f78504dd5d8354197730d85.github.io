#include <iostream>
using namespace std;
void avg(int a,int b);
void avg(int a,int b,int c);
void avg(int a,int b,int c,int d);
int main()
{
int a,b,c,d;
  cin>>a>>b>>c>>d;
  avg(a,b);
  avg(a,b,c);
  avg(a,b,c,d);
	return 0;
}
void avg(int a,int b)
{
  cout<<"Average of 2 nos is "<<(a+b)/2<<endl;
}
void avg(int a,int b,int c)
{
  cout<<"Average of 3 nos is "<<(a+b+c)/3<<endl;
}
void avg(int a,int b,int c,int d)
{
  cout<<"Average of 4 nos is "<<(a+b+c+d)/4<<endl;
}
