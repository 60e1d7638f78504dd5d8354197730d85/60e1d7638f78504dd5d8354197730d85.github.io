#include <iostream>
using namespace std;
int sum(int a,int b)
{
  return((a*a)+(b*b));
}
int sum(int a,int b,int c)
{
  return((a*a)+(b*b)+(c*c));
}
int sum(int a,int b,int c,int d)
{
  return((a*a)+(b*b)+(c*c)+(d*d));
}
int main()
{
int a,b,c,d;
  cin>>a>>b>>c>>d;
  cout<<"The sum of square is "<<sum(a,b)<<endl;
   cout<<"The sum of square is "<<sum(a,b,c)<<endl;
   cout<<"The sum of square is "<<sum(a,b,c,d);
	return 0;
}
