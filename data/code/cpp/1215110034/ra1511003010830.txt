#include<iostream>
using namespace std;
float area(float);
float circum(float);
int main() 
{ 
        int radius; 
        cin>>radius;
        cout<<area(radius)<<"\n";
        cout<<circum(radius);
} 
float area(float radius)
{
        return (3.14 * radius * radius);
}
float circum(float radius)
{ 
        return(2 * 3.14 * radius);
}
