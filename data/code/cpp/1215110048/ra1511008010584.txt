#include <iostream>
using namespace std;
class maths
{
  public:
  void area(int a,int b)
  {
    cout<<"Area of rectangle is="<<a*b<<"\n";
  }
  void area(int a)
  {
    cout<<"Area of square is="<<a*a<<"\n";
  }
  void area(float a)
  {
    cout<<"Area of circle is="<<3.14*a*a<<"\n";
  }
  void area(float a,float b)
  {
    cout<<"Area of triangle is="<<(a*b)/2<<"\n";
  }
 };
int main()
{
  int a,b,c;
  float f,d,e;
  cin>>a>>b>>c;
  cin>>f;
  cin>>d>>e;
  maths obj;
  obj.area(a);
  obj.area(b,c);
  obj.area(f);
  obj.area(d,e);
  

	return 0;
}
