float distance(float x_1,float y_1,float x_2,float y_2)
	{
       float d=0;

       d=sqrt(pow((x_2-x_1),2)+pow((y_2-y_1),2));

    }

 
	float distance(float x_1,float y_1,float z_1,float x_2,float y_2,float z_2)
