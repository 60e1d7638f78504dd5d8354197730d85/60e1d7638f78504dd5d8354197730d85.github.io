#include <iostream>
using namespace std;

class shape
{
  public:
  void print(int a)
  {
    cout<<"Area of Square "<<a*a<<"\n";
  }
  void print(int a, int b)
  {
    cout<<"Area of Rectangle "<<a*b;
  }
};
int main()
{
  int a,b,c;
  cin>>a>>b>>c;
  shape obj;
  obj.print(a);
  obj.print(b,c);

	return 0;
}
