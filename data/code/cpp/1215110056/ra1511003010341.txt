#include <iostream>
#include<cmath>
using namespace std;
int main()
{
  {
  int a,b;
  cin>>a>>b;
  cout<<"Square root of integers:"<<sqrt(a*b)<<endl;
  }
    {
  float a,b;
  cin>>a>>b;
  cout<<"Square root of float:"<<sqrt(a*b)<<endl;
    }
    {
  double a,b;
  cin>>a>>b;
  cout<<"Square root of double:"<<sqrt(a*b)<<endl;
    }
	return 0;
}
