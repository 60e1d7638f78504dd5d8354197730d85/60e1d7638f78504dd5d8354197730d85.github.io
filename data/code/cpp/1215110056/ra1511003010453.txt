#include <iostream>
#include <cmath>

using namespace std;
int main()
{
  	int a,b;
  	float x,y;
  	double p,q;

  	cin>>a>>b>>x>>y>>p>>q;
  
  	cout<<"Square root of integers:"<<sqrt(a*b)<<endl;
  	cout<<"Square root of float:"<<sqrt(x*y)<<endl;
  	cout<<"Square root of double:"<<sqrt(p*q);
	return 0;
}
