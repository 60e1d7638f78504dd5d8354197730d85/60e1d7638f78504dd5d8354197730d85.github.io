#include <iostream>
#include <cmath>
using namespace std;
int main()
{
	int a,b;
    float c,d;
    double e,f;
    cin>>a>>b>>c>>d>>e>>f;
    cout<<"Square root of integers:"<<pow(a*b,0.5)<<endl<<"Square root of float:"<<pow(c*d,0.5)<<endl<<"Square root of double:"<<pow(e*f,0.5)<<endl;
	return 0;
}
