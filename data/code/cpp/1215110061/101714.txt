#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
#define MOD 1000000007
int main()
{
	ll i,j,k,t,n;
	cin>>n>>k;ll a[n];
	for(i=0;i<n;i++)cin>>a[i];
	ll *hash=(ll *)calloc(n,sizeof(ll));hash[0]=a[0];
	for(i=1;i<=k;i++)hash[i]=a[i]*a[0];
	for(i=k+1;i<n;i++){
		ll min=INT_MAX;
		for(j=i-1;j>=i-k;j--){
			if(hash[j]<min)min=hash[j];
		}
		hash[i]=min*a[i];
	}
	cout<<hash[n-1]%MOD<<endl;
  return 0;
}
