#include <iostream>
using namespace std;
int gethcf(int a, int b);
 
int main(){
    int num1, num2, gcd;
     

    cin >> num1 >> num2;
     
    gcd = gethcf(num1, num2);
 
    cout << "H.C.F of " << num1 << " and " << num2 << " is: " << gcd;
 
    return 0;
}
 int gethcf(int a, int b) {
  if (b == 0) {
    return a;
  }
  else {
    return gethcf(b, a % b);
  }
 }
