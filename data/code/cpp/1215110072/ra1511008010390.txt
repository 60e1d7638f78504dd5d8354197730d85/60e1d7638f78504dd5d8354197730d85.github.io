#include <iostream>
using namespace std;
int hcf(int a, int b)
{
    while (a != b)
    {
        if (a > b)
        {
            a = a - b;
        }
        else
        {
            b = b - a;
        }
    }
    return a;
}
int main()
{
int a,b;
  cin>>a>>b;
  cout<<"H.C.F of "<<a<<" and "<<b<<" is: "<<hcf(a,b);
	return 0;
}
