#include <iostream>
using namespace std;
int main()
{
	int n1, n2, minMultiple;
   
    cin>> n1>>n2;

   
    minMultiple = (n1>n2) ? n1 : n2;

   
    while(1)
    {
        if( minMultiple%n1==0 && minMultiple%n2==0 )
        {
           cout<<"LCM = "<<minMultiple;
            break;
        }
        ++minMultiple;
    }
	return 0;
}
