#include<stdio.h>
void cyclicSwap(int *a,int *b,int *c);

int main()
{
    int a, b, c;
    scanf("%d %d %d",&a,&b,&c);

    printf("Value before swapping: \n");
    printf("a=%d\nb=%d\nc=%d\n",a,b,c);

    cyclicSwap(&a, &b, &c);

    printf("Value after swapping numbers in cycle: \n");
    printf("a=%d\nb=%d\nc=%d",a, b, c);

    return 0;
}
void cyclicSwap(int *a,int *b,int *c)
{
    int temp;

    temp = *b;
    *b = *a;
    *a = *c;
    *c = temp;
}
