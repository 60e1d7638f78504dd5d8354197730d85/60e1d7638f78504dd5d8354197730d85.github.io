#include<iostream>
using namespace std;
long int fact(int); 
int main()
{
	int n, r;
	long int ncr, npr;
	cin>>n;
	cin>>r;
	npr=fact(n)/fact(n-r);   
	ncr=npr/fact(r);         
	cout<<npr<<endl;
	cout<<ncr;
}
long int fact(int x)         
{
	int i, f=1;
	for(i=2; i<=x; i++)
	{
		f=f*i;
	}
	return f;
}
