#include <iostream>
using namespace std;
int main()
{
	int n,c=0,count=0;
  	cin>>n;
  	string str;
  	for(int i=0;i<n;i++){
    	cin>>str;
      	
    	for(int j=0;j<str.length()-1;j++){
        	if(str[j]=='0' && str[j+1]=='1'){
            	c+=1;	
            }else if(str[j]=='1' && str[j+1]=='0' && str[0]!='1'){
            	c=c+1;
            }
        }  
      //cout<<c<<endl;
      if(c==2 || c==0 && str[0]!='0'){
      	cout<<"YES"<<endl;
      }else{
      	cout<<"NO"<<endl;
      }
      c=0;
    }
	return 0;
}
