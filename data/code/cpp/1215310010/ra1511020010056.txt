#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
 
int main()
{
	int t,i,l,ans,f;
	char s[100001];
	scanf("%d",&t);
	while(t--)
	{
		ans = 0;i=0;f=0;
		scanf("%s",s);
		l = strlen(s);
		while(i<l)
		{
			if(s[i]=='0')
				i++;
			else
			{
				if(f==0)
				{
					while(i<l && s[i]=='1')
						i++;
					f = 1;
				}
				else
				{
					ans = 1;
					break;
				}
			}
		}
		if(ans||f==0)
			printf("NO\n");
		else
			printf("YES\n");
	}
}
