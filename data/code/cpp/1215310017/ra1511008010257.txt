#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
 
using namespace::std;
 
int gcd(int a,int b)
{
	if(b==0)
		return a;
	else
		gcd(b,a%b);
}
 
int main()
{
 
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		vector<int> v(n);
		for(int i=0;i<n;i++)
			cin>>v[i];
 
		int g=v[0];
		for(int i=1;i<n;i++)
		{
			g=gcd(g,v[i]);
		}
 
		for(int i=0;i<n;i++)
			cout<<v[i]/g<<" ";
		cout<<endl;
	}
	return 0;
 
}
