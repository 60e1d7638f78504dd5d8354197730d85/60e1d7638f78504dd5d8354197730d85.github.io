#include <iostream>
using namespace std;
#include<stdio.h>
int gcd(int x,int y)
{
	if(y==0)
	return x;
	else
	return gcd(y,x%y);
}
main()
{
	int tc,n,temp,a[1010];
	scanf("%d",&tc);
	while(tc--)
	{
		scanf("%d",&n);
		for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
		temp=a[0];
		for(int i=1;i<n;i++)
		temp=gcd(temp,a[i]);
		for(int i=0;i<n;i++)
		printf("%d ",(a[i]/temp));
		printf("\n");
	}
}
