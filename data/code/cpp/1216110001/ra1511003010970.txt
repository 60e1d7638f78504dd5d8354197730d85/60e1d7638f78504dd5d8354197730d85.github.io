#include <iostream>
using namespace std;
int main()
{
int size;
  cin>>size;
  
  int arr[size];
  
  for(int i=0;i<size;i++)
    cin>>arr[i];
  
  for(int i=0;i<size;i++)
  {
    for(int j=0;j<size-i;j++)
    {
   if(arr[j]>arr[j+1])
   {
    int temp;
     temp=arr[j];
     arr[j]=arr[j+1];
     arr[j+1]=temp;
   }  
    }
    
  }
  
  for(int i=0;i<size-1;i++)
  {
   if(arr[i]==arr[i+1])
     i++;
    else 
      cout<<arr[i];
  }
  return 0;
}
