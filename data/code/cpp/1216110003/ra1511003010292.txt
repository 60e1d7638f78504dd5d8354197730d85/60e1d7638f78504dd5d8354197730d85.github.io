#include <iostream>
#include<cmath>
using namespace std;
int main()
{
  int n;
  cin>>n;
  int arr[n];
  for(int i = 0 ; i<n ; i++)
    cin>>arr[i];
  int sum = 0;
  for(int i = 0 ; i<n-1 ; i++)
  {
    for(int j = i+1; j<n ; j++)
    {
      sum = sum + abs(arr[i]-arr[j]);
    }
  }
  cout<<sum;
	return 0;
}
