#include <iostream>
#include <stdlib.h>
using namespace std;
int main()
{
	int n,sum=0;
  	cin>>n;
  	int arr[n];
  	for(int i=0;i<n;i++)
      	cin>>arr[i];
  	for(int i=0;i<n-1;i++)
    {
      	for(int j=i+1;j<n;j++)
        {
          	sum+=abs(arr[i]-arr[j]);
        }
    }
  	cout<<sum;
	return 0;
}
