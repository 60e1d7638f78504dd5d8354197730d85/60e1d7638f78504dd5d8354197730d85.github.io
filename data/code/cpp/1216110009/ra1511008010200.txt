#include <iostream>
using namespace std;
int main()
{
  int n,a;
  cin>>n;
  int arr[n];
  for(int i=0;i<n;i++)
  {
    cin>>a;
    arr[i] = a*a;
  }
  int i,j,key;
  for(i=1;i<n;i++)
  {
    key = arr[i];
    j = i-1;
    
    while(j>=0 && arr[j]>key)
    {
      arr[j+1] = arr[j];
      j = j-1;
    }
    arr[j+1] = key;
  }
  for(int i=0;i<n;i++)
    cout<<arr[i]<<" ";
  return 0;
}
