#include <iostream>
#include <math.h>
using namespace std;
int main()
{
  	int i,j,arr[50],n,temp;
  	
  	cin>>n;
  	
  	for(i=0;i<n;i++)
    {
    	cin>>arr[i];
      	arr[i] = arr[i] * arr[i];
    }
  
  	for(i=0;i<n-1;i++)
    {
    	for(j=0;j<n-1;j++)
    	{
    		if(arr[j]>arr[j+1])
            {
            	temp = arr[j];
              	arr[j] = arr[j+1];
              	arr[j+1] = temp;
            }
        }	
    }
  
	for(i=0;i<n;i++)
    {
    	cout<<arr[i]<<" ";
    }
  	
  
	return 0;
}
