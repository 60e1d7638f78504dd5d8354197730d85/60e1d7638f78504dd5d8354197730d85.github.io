#include <iostream>
using namespace std;
 
// Returns count of passing cars
int getPassingCars(int A[], int n)
{
    int result = 0;
    for (int i=0; i<n-1; i++)
    {
       if (A[i] == 0)
       {
           for (int j=i+1; j<n; j++)  
              if (A[j])
                 result++; 
       } 
    }
    return result;
}
 
// Driver program
int main()
{
    int A[10];
    int n;
  cin>>n;
    for(int i=0;i<n;i++)
      cin>>A[i];
    cout << getPassingCars(A, n);
    return 0;
}
