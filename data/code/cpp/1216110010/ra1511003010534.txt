#include <iostream>
using namespace std;
int main()
{
	int n;
  	cin>>n;
  	int a[n],c=0;
  	for(int i=0;i<n;i++)
      cin>>a[i];
  	for(int i=0;i<n;i++)
    {
      for(int j=0;j<n;j++)
      {
        if(i<j)
        {
          if((a[i]==0) && (a[j]==1))
            c++;
        }
      }
    }
  	cout<<c;
	return 0;
}
