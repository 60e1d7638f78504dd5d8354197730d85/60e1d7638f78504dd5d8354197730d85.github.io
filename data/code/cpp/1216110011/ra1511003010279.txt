#include <iostream>
using namespace std;
int main()
{
	int n,i,j,temp;
	cin>>n;
int ar[n];
	for(i=0;i<n;i++)
	{
		cin>>ar[i];
	
	}
	
	for(i=0;i<n;i++)
	{
		for(j=0;j<n-i-1;j++)
		{
			if(ar[j]>ar[j+1])
			{
				temp=ar[j];
				ar[j]=ar[j+1];
				ar[j+1]=temp;
			}}}
		for(i=0;i<n;i++)
	{
		cout<<ar[i]<<" ";
	}
	
	return 0;
}
