#include <iostream>
using namespace std;
int main()
{
int n;
  cin>>n;
  int arr[n];
  for(int i=0;i<n;i++){cin>>arr[i];}
  int temp;
  int j=0;
  for(int i=0;i<n;i++)
  {
    for(j=0;j<n-i-1;j++)
    {
      if (arr[j]>arr[j+1])
      {
        temp=arr[j+1];
        arr[j+1]=arr[j];
        arr[j]=temp;;
      }
    }
  }
  for(int i=0;i<n;i++)
  {
    cout<<arr[i]<<" ";
  }
	return 0;
}
