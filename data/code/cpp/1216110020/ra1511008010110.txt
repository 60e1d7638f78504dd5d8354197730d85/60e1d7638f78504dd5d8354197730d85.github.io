#include <iostream>
using namespace std;
int main()
{
	int n;
  	cin>>n;
  
  	int arr[n];
  	
  	for(int i=1;i<=n;i++)
     	cin>>arr[i]; 
  
  	int t;
  	cin>>t;
  
  	for(int i=t+1;i<=n;i++)
    {
      	cout<<arr[i]<<" ";
    }	  
  	
    for(int i=1;i<=t;i++)
    {
     	cout<<arr[i]<<" "; 
    } 
     
  
  	
	return 0;
}
