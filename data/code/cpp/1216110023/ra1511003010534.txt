#include <iostream>
using namespace std;
int main()
{
	int n;
  	cin>>n;
  	int a[n],c[n],k=0;
  	for(int i=0;i<n;i++)
      cin>>a[i];
 	for(int i=0;i<n;i++)
    {
      int flag=0;
      for(int j=i+1;j<n;j++)
      {
        if(a[i]<=a[j])
        {
          flag=1;
          break;
        }
      }
      if(flag==0)
        c[k++]=a[i];
    }
  	for(int i=0;i<k;i++)
      cout<<c[i]<<" ";
	return 0;
}
