#include <iostream>
using namespace std;
int main()
{  int n;
   cin>>n;
   int a[n],b[1000],k=0,count=0;
   for(int i=0;i<n;i++){
      cin>>a[i];
   }
  for(int i=0;i<n;i++){
     if(a[i]!=0){
        b[k++]=a[i];
     }
    else{
      count++;
    }
  } 
  for(int i=0;i<k;i++){
       cout<<b[i]<<" ";
   }
  for(int i=0;i<count;i++){
     cout<<"0"<<" ";
  }
 cout<<endl;
	return 0;
}
