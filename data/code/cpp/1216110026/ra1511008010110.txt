#include <iostream>
using namespace std;
int main()
{
	int n;
  	cin>>n;
  	int arr[n];
  	
  	for(int i=0;i<n;i++)
    {
     	cin>>arr[i]; 
    }
  	
  	for(int i=0;i<n;i++)
    {	
     	for(int j=i;j<n;j++)
        {
         	if(arr[i] < arr[j])
              arr[i] = arr[j];
        }
    }
  	for(int i=1;i<n;i++)
    {
      cout<<arr[i]<<" ";
    }
  	cout<<"-1";
	return 0;
}
