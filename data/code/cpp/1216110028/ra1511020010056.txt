#include<iostream>
using namespace std;
int main()
{    
    int n;
   	cin>>n;
  	int arr[n];
    for(int i=0;i<n;i++)
    {
      cin>>arr[i];
    }
  	int result = 1; 
    int max_ele = arr[0]; 
    for (int i=1; i<n; i++)
    {
        if (arr[i] > max_ele)
        {
            max_ele = arr[i]; 
            result++;
        }
    }
  cout<<result;
    return 0;
}
