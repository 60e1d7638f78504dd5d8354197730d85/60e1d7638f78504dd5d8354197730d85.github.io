#include<iostream>
using namespace std;
 
main()
{
   int first[20], second[20], c, n;
 
   cin >> n;

 
   for ( c = 0 ; c < n ; c++ )
      cin >> first[c];

 
   for ( c = 0 ; c < n ; c++ )
      cin >> second[c];
 
 
   for ( c = 0 ; c < n ; c++ )
      cout << first[c] + second[c]<<" ";
 
   return 0;
}
