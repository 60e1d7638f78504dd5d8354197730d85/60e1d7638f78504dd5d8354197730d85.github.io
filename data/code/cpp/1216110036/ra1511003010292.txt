#include <iostream>
using namespace std;
int main()
{
  int n;
  cin>>n;
  int arr[n];
  cout<<"Even Elements=";
  for(int i = 0; i<n ; i++)
  {
    cin>>arr[i];
    if(arr[i]%2==0)
      cout<<arr[i]<<" ";
  }
  cout<<"\nOdd Elements=";
  for(int i = 0; i<n ; i++)
  {
    if(arr[i]%2 != 0)
      cout<<arr[i]<<" ";
  }
	return 0;
}
