#include <iostream>
using namespace std;
int main()
{
    int i,j,a[100];
int popular = a[0],n;
int temp=0, tempCount, count=1;
  cin>>n;
    for(i=0;i<n;i++)
    	cin >> a[i];
    for (i=0;i<n;i++)
    {
        tempCount = 0;
        temp=a[i];
        tempCount++;
        for(j=i+1;j<n;j++)
        {
            if(a[j] == temp)
            {
                tempCount++;
                if(tempCount > count)
                {
                    popular = temp;
                	count = tempCount;
                }
            }
        }
    }
cout << popular;
  return 0;
}
