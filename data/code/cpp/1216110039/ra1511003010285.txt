#include<iostream>
using namespace std;

int main()
{
	int Arr[100],n,i,small,large;

	cin>>n;

	for(i=0;i<n;i++)
	{
		cin>>Arr[i];
	}

	small=Arr[0];
	large=Arr[0];

	for(i=1;i<n;i++)
	{
		if(Arr[i]<small)
			small=Arr[i];
		if(Arr[i]>large)
			large=Arr[i];
	}

	cout<<"Min="<<small;
	cout<<"\nMax="<<large;
    
	return 0;
}
