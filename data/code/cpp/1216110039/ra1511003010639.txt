#include <iostream>
using namespace std;
int main()
{
	int arr[100];
    int i, max, min, size;

   
    cin>>size;

    for(i=0; i<size; i++)
    {
        cin>>arr[i];
    }


    max = arr[0];
    min = arr[0];

    for(i=1; i<size; i++)
    {
        
        if(arr[i] > max)
        {
            max = arr[i];
        }

       
        if(arr[i] < min)
        {
            min = arr[i];
        }
    }

   
    cout<<"Min="<< min<<endl;
     cout<<"Max="<< max;

	return 0;
}
