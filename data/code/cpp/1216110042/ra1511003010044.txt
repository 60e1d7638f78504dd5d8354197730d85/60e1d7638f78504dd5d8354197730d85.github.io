#include <iostream>
#include <math.h>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,v1,v2;
        cin>>n>>v1>>v2;
        float elev= (2*n)/(float)v2;
        float stair= sqrt(2)*n/(float)v1;
        //cout<<elev<<"-"<<stair<<endl;
        if(elev<stair)
            cout<<"Elevator"<<endl;
        else
            cout<<"Stairs"<<endl;
        bool flag=false;    
        for(int i=0;i<20;i++)
        {
        	if(i==2 || i== 10 || i==15)
        		flag=true;
        }
    }
    return 0;
}
