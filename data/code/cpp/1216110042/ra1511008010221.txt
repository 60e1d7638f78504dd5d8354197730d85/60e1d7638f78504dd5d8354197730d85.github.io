#include<iostream>
#include<math.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        double n,v1,v2;
        cin>>n>>v1>>v2;
        double te,ts;
        double ds,de;
        ds = n*sqrt(2);
        de = 2*n;
        te = (de/v2);
        ts = (ds/v1);
        if(ts>=te)
        cout<<"Elevator"<<endl;
        else
        cout<<"Stairs"<<endl;
    }
    return 0;
}
