#include <iostream>
#include<bits/stdc++.h>
#include<algorithm>
#include<limits>
#define ll long long
using namespace std;
ll GCD(ll a,ll b)
{
    if(b==0)
        return a;
    else
        return GCD(b,a%b);
}
int main()
{
    ll T,m,arr[600],i,Min,j,cal;
  cin>>T;
    while(T--)
    {
        //Min=INT_MAX;
      cin>>m;
        for(i=0;i<m;i++)
          cin>>arr[i];
        sort(arr,arr+m);
        Min=(arr[0]*arr[1])/GCD(arr[1],arr[0]);
        for(i=0;i<m;i++)
        {
            for(j=i+1;j<m;j++)
            {
                cal=(arr[i]*arr[j])/(GCD(arr[i],arr[j]));
                if(cal<Min)
                    Min=cal;
            }
        }
      cout<<Min<<endl;
    }
    return 0;
}
