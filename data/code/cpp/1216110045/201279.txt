#include<bits/stdc++.h>
using namespace std;
int main(){
	long long int a[26];
    char b[26];
    string x;
	int t;
	cin>>t;
	while(t){
		int sum=0;
		for(int i=0;i<26;i++){
			cin>>a[i];
		}
		int i=0;
		for(char x='a';x<='z';x++){
			b[i]=x;i++;
		}
		
		cin>>x;
		for(int i=0;i<x.size();i++){
			for(int j=0;j<26;j++){
				if(x[i]==b[j]){
					b[j]='0';
				}
			}
		}
		for(int i=0;i<26;i++){
			if(b[i]!='0'){
				sum+=a[i];
			}
		}
		cout<<sum<<endl;
		t--;
	}
}
