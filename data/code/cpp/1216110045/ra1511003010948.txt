#include<iostream>
#include<cstring>
using namespace std;
main()
{
	int t,a[26],j,price,i,f,k;
	char s[1000000];
	cin>>t;
	cin.ignore();
	for(i=0;i<t;i++)
	{
		for(j=0;j<26;j++)
		cin>>a[j];
		cin.ignore();
		cin>>s;
		price=0;
		for(char p='a',j=0;j<26;j++,p++)
		{	
			f=0;
			for(k=0;k<strlen(s);k++)
			{
				if(s[k]==p)
				f=1;
			}
			if(f==0)
			price=price+a[j];
		}
		cout<<price<<endl;
	}
}
