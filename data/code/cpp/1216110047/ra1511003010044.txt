#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;
 
int main()
{
    int t,m,n,k;
    cin >> t;
    while(t--){
        cin >> n >> m;
        int ans = 0;
        vector <int> alice(1000001);
        for(int i = 0; i < n; i++){
            cin >>k;alice[k]=1;
        }
        for(int i = 0; i < m; i++){
            cin >>k;
            if(alice[k]==1){
                ans++;
            }
        }
        cout<<ans<<endl;
    }
 
 
    return 0;
}
