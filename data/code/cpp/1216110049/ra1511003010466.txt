#include<stdio.h>
 
int main()
{
	int p;
	scanf("%d",&p);
	while(p--)
	{
		int n,m;
		scanf("%d %d",&n,&m);
		int i,j,a[10][10];
		for(i=0;i<n;i++)
		{
			for(j=0;j<m;j++)
			{
				scanf("%1d",&a[i][j]);
			}
		}
		int count,count1=0;
		for(i=0;i<m;i++)
		{
count=0;
			for(j=0;j<n;j++)
			{
				if(a[j][i]==1)
				{
					count++;
				}}
				if(count>1)
                {
                   count1=count1+((count)*(count-1))/2;
                }
			}
 
 
		printf("%d\n",count1);
	}
	return 0;
}
