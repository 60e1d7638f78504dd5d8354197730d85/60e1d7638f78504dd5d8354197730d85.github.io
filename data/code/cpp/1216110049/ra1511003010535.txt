#include <iostream>
using namespace std;
#include<bits/stdc++.h>
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,m;
        cin>>n>>m;
        char a[n][m];
        for(int i=0;i<n;i++)
            for(int j=0;j<m;j++)
                cin>>a[i][j];
        int c=0,count;
        for(int i=0;i<m;i++)
        {
            count=0;
            for(int j=0;j<n;j++)
            {
                if(a[j][i]=='1')
                    count++;
            }
            if(count>0)
                c+=count*(count-1)/2;
        }
        cout<<c<<endl;
    }
return 0;}
