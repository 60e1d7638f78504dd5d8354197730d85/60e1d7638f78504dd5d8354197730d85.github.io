#include <iostream>
using namespace std;
int main()
{
int arr1[50],arr2[50],n1,n2,n,i,j,k,merge[100];
  cin>>n1;
  
  for(i=0;i<n1;i++)
  {
    cin>>arr1[i];
  }
  cin>>n2;
  for(i=0;i<n2;i++)
  {
    cin>>arr2[i];
  }
  for(i=0;i<n1;i++)
  {
    merge[i]=arr1[i];
  }
  n=n1+n2;
  for(i=0,k=n1;k<n && i<n2;i++,k++)
  {
    merge[k]=arr2[i];
  }
  for(i=0;i<n;i++)
  {
    cout<<merge[i]<<" ";
  }
	return 0;
}
