#include <iostream>
using namespace std;
int main()
{
  int n;
  cin>>n;
  int i,j,a[n];
  for(i=0;i<n;i++)
    cin>>a[i];
  for(i=0;i<n;i++)
  {
    for(j=i+1;j<n;j++)
    {
      if(a[i]>a[j])
      {
        int t=a[i];
        a[i]=a[j];
        a[j]=t;
      }
    }
  }
  for(i=0;i<n;i++)
    cout<<a[i]<<endl;

	return 0;
}
