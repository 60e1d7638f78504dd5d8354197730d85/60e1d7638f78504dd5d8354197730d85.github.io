    using namespace std;
     
    const int Maxd = 10;
    const int Maxn = 5;
     
    int n;
    int use[Maxd];
    bool is[Maxd];
    int seq[Maxn];
    int res;
     
    bool Check(int num,int digs){
    	while(digs--){
    		if(!is[num%10])return false;
    		num/=10;
    	}
    	return num==0;
    }
     
    bool Check(){
    	int a=seq[0]*100+seq[1]*10+seq[2];
    	return Check(a*seq[3],3)&&Check(a*seq[4],3)&&Check(a*(seq[3]*10+seq[4]),4);
    }
     
    void Gen(int lvl){
    	if(lvl >=Maxn)res+=Check();
    	else for(int i=0;i<n;i++){
    		seq[lvl]=use[i];
    		Gen(lvl+1);
    	}
    }
     
    int main(){
    	scanf("%d",&n);
    	for(int i=0;i<n;i++){
    		scanf("%d",&use[i]);
    		is[use[i]] = true;
    	}
    	Gen(0);
    	printf("%d\n",res );
    	return 0;
    }
