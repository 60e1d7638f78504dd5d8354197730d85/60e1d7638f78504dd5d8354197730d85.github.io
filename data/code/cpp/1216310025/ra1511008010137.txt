#include<stdio.h>
int main(int argc, char const *argv[])
{
	int n;
	int a[9];
	int b[10];
	scanf("%d",&n);
	int i=0;
	for(i=0;i<10;i++)b[i]=0;
	for(i=0;i<n;i++){
		scanf("%d",(a+i));
		b[a[i]]=1;
	}
 
	for(i=0;i<n;i++){
		int j=0;
		for(j=0;j<n-i-1;j++){
			if(a[j]>a[j+1]){
				int temp=a[j];
				a[j]=a[j+1];
				a[j+1]=temp;
			}
		}
	}
	int count=0;
	for(i=0;i<n;i++){
		int j=0;
		for(j=0;j<n;j++){
			int k=0;
			for(k=0;k<n;k++){
				int l=0;
				for(l=0;l<n;l++){
					int m=0;
					for(m=0;m<n;m++){
						int x=(100*a[i]+10*a[j]+a[k])*a[m];
						int y=(100*a[i]+10*a[j]+a[k])*a[l];
						int z=10*y+x;
						if(x<=999&&y<=999&&z<=9999){
							if(b[x%10]&b[(x/10)%10]&b[(x/100)%10]&b[y%10]&b[(y/10)%10]&b[(y/100)%10]&b[z%10]&b[(z/10)%10]&b[(z/100)%10]&b[(z/1000)%10])
								count++;
						}
					}
				}
			}
		}
	}
	printf("%d\n",count );
	return 0;
}
