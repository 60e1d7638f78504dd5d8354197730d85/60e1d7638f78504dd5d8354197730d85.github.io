int *findMax(int arr[],int n); 
int main(){    
   int n,i,*p;    
  cin>>n;     
  int arr[n];         
  for(i=0;i<n;i++)     {       
        
   cin>>arr[i];             
}           
  p=findMax(arr,n);     
  cout<<"Largest integer value in the array is "<<*p;     
   
  return 0; 
}  
 
int *findMax(int data[],int n){     
  int *max=data;     
  int i;    
 for(i=1;i<n;i++){            
   if(*max<*(max+i)) *max=*(max+i);                              
  }     
  return max; 
}
