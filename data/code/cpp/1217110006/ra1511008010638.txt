{
int a[10];
  int n;
  cin>>n;
  for(int i=0;i<n;i++)
  {
    cin>>a[i];
  }
   for(int i=0;i<n;i++)
   {
    for(int j=i+1;j<n;j++){
       if( *(a+i) > *(a+j)){
     int tmp = *(a+i);

       }
    }
   }

	return 0;
}
