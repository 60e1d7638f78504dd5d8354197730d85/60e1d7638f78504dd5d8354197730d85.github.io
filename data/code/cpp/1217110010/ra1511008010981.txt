#include <iostream>
using namespace std;
class fib
{
public:
  int a,b,c;
  void generator(int);
};
void fib::generator(int n)
{
  a=1,b=1;
  cout<<a<<" "<<b;
  for(int i=1;i<=n-2;i++)
  {
    c=a+b;
    cout<<" "<<c;
    a=b;
    b=c;
  }
}

{
