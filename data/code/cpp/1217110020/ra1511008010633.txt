ll inp()
{
    ll ch=getcx(),n=0;

    {

    }
    while(ch<='9'&&ch>='0')
    {
                           n=(n<<3)+(n<<1)+(ch-'0');

    }
    return n;
}

ll modpow(ll a,ll b,ll c)
{
     ll res = 1,y = a%c;
     
     while(b>0)
     {
               if(b&1)
               res = (res*y)%c;
               
               y = (y*y)%c;
               b>>=1;
     }
     return res;
}
