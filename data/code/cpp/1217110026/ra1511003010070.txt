#include<iostream>
 
using namespace std;
 
int main()
{
    int t,n,i;
    cin>>t;
    while(t--)
    {
              cin>>n;
              int a[32][2],sum1=0,sum2=0;
              for(i=0;i<n;i++)
              {
                cin>>a[i][0]>>a[i][1];
                  sum1+=a[i][0];
                  sum2+=a[i][1];
              }
              cout<<sum1-sum2;
                      
    }
    return 0;
}
