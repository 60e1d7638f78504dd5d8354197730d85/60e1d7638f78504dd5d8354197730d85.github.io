#include<bits/stdc++.h>
using namespace  std;
 int n;
 int a[10];
 int b[10];
 int c[10];
 int x;
 int main()
 {
     int ans1=0;
     scanf("%d",&n);
     for(int i=1;i<=n;i++)
     {
         scanf("%d",&x);
         a[x]++;
         c[x]++;
     }
      for(int i=1;i<=n;i++)
     {
         scanf("%d",&x);
         b[x]++;
         c[x]++;
     }
     for(int i=1;i<=5;i++)
     {
         if(c[i]%2)
         {
             printf("-1\n");
             return 0;
         }
         else{
         if(a[i]>b[i])
             {
             ans1+=a[i]-c[i]/2;
             }
         }
     }
     printf("%d\n",ans1);
     return 0;
 }
