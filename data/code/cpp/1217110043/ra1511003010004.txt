#include <iostream>  
#include <cstdio>  
#include <cstring>  
using namespace std;  
typedef long long LL;  
const LL MAXN = 1e5 + 8;  
  
LL v[MAXN];  
bool f[MAXN];  
  
int main()  
{  
    #ifdef LOCAL  
    freopen("b.txt", "r", stdin);  
    //freopen("b.out", "w", stdout);  
    LL T = 4;  
    while(T--){  
    #endif // LOCAL  
    ios::sync_with_stdio(false); cin.tie(0);  
  
    memset(f,  true, sizeof f);  
    LL n, k = 1, i, j, ptr;  
    cin >> n;  
    for(i = 2; i <= n + 1; i++){  
        ptr = 1;  
        if(f[i]){  
            v[i] = ptr;  
  
            for(j = i * i; j <= n + 1; j += i){  
                f[j] = false;  
                if(!v[j]) {ptr = 2; v[j] = ptr;}  
            }  
            k = max(k, ptr);  
        }  
    }  
    cout << k << "\n" << v[2];  
    for(i = 3; i <= n + 1; i++) cout << " " << v[i];  
  
  
    #ifdef LOCAL  
    memset(v, 0, sizeof v);  
    cout << endl;  
    }  
    #endif // LOCAL  
    return 0;  
}
