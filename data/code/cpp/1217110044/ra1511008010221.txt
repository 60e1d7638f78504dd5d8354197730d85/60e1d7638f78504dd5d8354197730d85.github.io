#include <iostream>
using namespace std;
int main()
{
	int a,b;
  	cin >>a>>b;
  	cout << "Before swapping" << endl;
  	cout << "a = " << a << endl;
  	cout << "b = " << b << endl << endl;
  	cout << "After swapping" << endl;
 	cout << "a = " << b << endl;
  	cout << "b = " << a << endl;
	return 0;
}
