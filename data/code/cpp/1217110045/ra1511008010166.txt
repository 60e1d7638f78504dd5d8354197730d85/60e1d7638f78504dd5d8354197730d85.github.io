#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int num;
 
    cin >> num;
    float* ptr;
    ptr = new float[num];

    for (int i = 0; i < num; ++i)
    {
        cin >> *(ptr + i);
    }

    cout << "\nDisplaying GPA of students." << endl;
    for (int i = 0; i < num; ++i) {
        cout << "Student" << i + 1 << " :" << *(ptr + i) << endl;
    }

    // ptr memory is released
    delete [] ptr;

    return 0;
}
