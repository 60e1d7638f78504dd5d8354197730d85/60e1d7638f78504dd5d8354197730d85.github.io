#include<iostream>
using namespace std;
class Rectangle
{
	private:
		float length;
		float width;
	public:
		void setlength(float);
		void setwidth(float);
		float perimeter();
		float area();
		void show();
		int sameArea(Rectangle);
};


void Rectangle::setlength(float len)
{
	length = len;
}

void Rectangle::setwidth(float wid)
{
	width = wid;
}

float Rectangle::perimeter()
{
	return (2 * length + 2 * width);
}

float Rectangle::area()
{
	return length * width;
}

int Rectangle::sameArea(Rectangle other)
{
	float areaf = length * width;
	float areas = other.length * other.width;
	if (areaf == areas)
		return 1;
	return 0;
}

{float l,b;
 cin>>l>>b;
	Rectangle first;
	first.setlength(l);
	first.setwidth(b);
	cout << first.area();

	return 0;
}
