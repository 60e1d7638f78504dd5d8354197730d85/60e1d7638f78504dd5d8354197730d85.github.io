#include<iostream>
using namespace std;

class AddData          //Base Class
{
        protected:
                int num1, num2;
        public:	
                void accept()
                {
                        
                        cin>>num1;
                        
                        cin>>num2;
                }
};
class Addition: public AddData   //Class Addition â€“ Derived Class
{
                int sum;
        public:
                void add()
                {
                        sum = num1 + num2;
                }	
                void display()
                {
                        cout<<sum;
                }	
};
int main()
{
        Addition a;
        a.accept();
        a.add();
        a.display();
        return 0;
}
