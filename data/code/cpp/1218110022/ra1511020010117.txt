#include <iostream>
using namespace std;
class AddData     
	{
   	    protected:
        int subjects[3], i;
   	    public:
        void accept_details()
      	  {
                cin>>subjects[0];
                cin>>subjects[1];
                cin>>subjects[2];	
        }
};
//Class Total â€“ Derived Class. Derived from class AddData and Base class of class Percentage
class Total : public AddData   
{
    protected:
        int total;
    public:
        void total_of_three_subjects()
        {
                total = subjects[0] + subjects[1] + subjects[2]; 
        }
};
class Percentage : public Total
{
    private:
        float per;
    public:
        void calculate_percentage()
        {
                per=total/3;
        }	
        void show_result()
        {
     
                cout<<per;
        }
};
int main()
{
        Percentage p;
        p.accept_details();
        p.total_of_three_subjects();
        p.calculate_percentage();
        p.show_result();
        return 0;
}
