#include <iostream>
using namespace std;
int main()
{
	int a,b;
	//cout<<"Enter Value Of A:";
	cin>>a;
	//cout<<"Enter Value of B:";
	cin>>b;
	//cout<<"Before swapping" <<endl<<"a = "<<a<<endl<<"b = "<<b<<endl<<endl;
	swap(a,b);
	cout<<a<<endl<<b; 
}
void swap(int &a,int &b)
{
	int c;
	c=a;
	a=b;
	b=c;
	//cout<<"After Swapping"<<endl<<"a = "<<a<<endl<<"b = "<<b;
}
