#include <iostream>
using namespace std;
int main()
{
	int a,b;
  	float c,d;
  	char e,f;
  	cin>>a>>b>>c>>d>>e>>f;
  	cout<<"Before passing data to function template."<<endl;
  	cout<<"i1 = "<<a<<endl;
  	cout<<"i2 = "<<b<<endl;
  	cout<<"f1 = "<<c<<endl;
  	cout<<"f2 = "<<d<<endl;
  	cout<<"c1 = "<<e<<endl;
  	cout<<"c2 = "<<f<<endl;
  	cout<<endl;
  	cout<<"After passing data to function template."<<endl;
  	cout<<"i1 = "<<b<<endl;
  	cout<<"i2 = "<<a<<endl;
  	cout<<"f1 = "<<d<<endl;
  	cout<<"f2 = "<<c<<endl;
  	cout<<"c1 = "<<f<<endl;
  	cout<<"c2 = "<<e<<endl;
	return 0;
}
