#include <iostream>
using namespace std;
int main()
{
  int a,b;
  float c,d;
  cin>>a>>b>>c>>d;
  cout<<"Int results:\nNumbers are: "<<a<<" and "<<b;
  cout<<".\nAddition is: "<<a+b<<"\nSubtraction is: "<<a-b<<"\nProduct is: "<<a*b<<"\nDivision is: "<<a/b;
  cout<<"\n\nFloat results:\nNumbers are: "<<c<<" and "<<d;
  cout<<".\nAddition is: "<<c+d<<"\nSubtraction is: "<<c-d<<"\nProduct is: "<<c*d<<"\nDivision is: "<<c/d;
  return 0;
}
