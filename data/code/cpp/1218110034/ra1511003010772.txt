#include <iostream>
using namespace std;
int main()
{
  int a,b;
  float c,d;
  cin>>a>>b>>c>>d;
  cout<<"Int results:\n";
  cout<<"Numbers are: "<<a<<" and "<<b<<".\n";
  cout<<"Addition is: "<<a+b<<"\n";
  cout<<"Subtraction is: "<<a-b<<"\n";
  cout<<"Product is: "<<a*b<<"\n";
  cout<<"Division is: "<<a/b<<"\n";
  
  cout<<"\nFloat results:\n";
  cout<<"Numbers are: "<<c<<" and "<<d<<".\n";
  cout<<"Addition is: "<<c+d<<"\n";
  cout<<"Subtraction is: "<<c-d<<"\n";
  cout<<"Product is: "<<c*d<<"\n";
  cout<<"Division is: "<<c/d<<"\n";
  return 0;
}
