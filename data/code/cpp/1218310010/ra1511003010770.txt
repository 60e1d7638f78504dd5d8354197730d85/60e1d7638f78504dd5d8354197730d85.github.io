#include <iostream>  
using namespace std;  
int main()  
{  
int n,r,sum=0,temp;    
  
cin>>n;    
temp=n;    
while(n>0)    
{    
r=n%10;    
sum=sum+(r*r*r);    
n=n/10;    
}    
if(temp==sum)    
cout<<temp<<" is an Armstrong number."<<endl;    
else    
cout<<temp<<" is not an Armstrong number."<<endl;   
return 0;  
}
