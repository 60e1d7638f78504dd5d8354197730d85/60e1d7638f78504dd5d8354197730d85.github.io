#include <iostream>
#include <string.h>
using namespace std;
int main()
{
  int n1, i, j, flag =0;
  char str[100];
  cin>>str;
  n1 = strlen(str);
  for(i=0; i<n1; i++)
  {
    if(str[i]!=str[n1-i-1])
    {
      	flag = 1;
    	break;
    }
  }
  if(flag==0)
    cout<<"It is a palindrome";
  else
    cout<<"It is not a palindrome";
  
 return 0;
}
