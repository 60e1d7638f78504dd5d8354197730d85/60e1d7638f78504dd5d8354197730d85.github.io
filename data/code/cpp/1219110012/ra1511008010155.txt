# include<iostream>
using namespace std;
class Time
{
	int m,s,h;
	public:
		void get(int p,int q,int r)
		{
			h=p;
			m=q;
			s=r;
		}
		void display()
		{
			cout<<"The Time is\n"<<h<<" Hours "<<m<<" Mins "<<s<<" secs"; 
		}
};
int main()
{
	int p,q,r;
	Time t;
	cin>>p>>q>>r;
	t.get(p,q,r);
	t.display();
}
