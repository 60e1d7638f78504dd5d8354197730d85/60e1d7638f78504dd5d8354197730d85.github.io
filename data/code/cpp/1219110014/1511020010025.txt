#include <iostream>
using namespace std;
class Check
{
    private:
       double i;
    public:
       Check()
       { i=0; }
       void operator ++() 
          { i=i+0.1;; }
       void Display() 
          { cout  << i << endl; }
  void read() 
          { cin>>i; }
};
int main()
{
Check obj;
  obj.read();
 ++obj; 
  obj.Display();
	return 0;
}
