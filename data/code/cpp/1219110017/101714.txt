#include <iostream>
using namespace std;
int main()
{
  int i,j;
  float a[2][2],pri[2],tot[2];
  for(i=0;i<2;++i)
  {
    tot[i]=0;
    for(j=0;j<2;++j)
      cin>>a[i][j];
  }
  cin>>pri[0]>>pri[1];
  for(i=0;i<2;++i)
  {
    for(j=0;j<2;++j)
      tot[i]=tot[i]+(pri[j]*a[i][j]);
    cout<<tot[i]<<endl;
  }
	return 0;
}
