#include <iostream>

using namespace std;

class Complex {
    public:
        double r;
        double i;
    public:
        Complex(){r=i=0;}
        void add(Complex op1, Complex op2)
          {
    r = op1.r+op2.r;
    i = op1.i+op2.i;
        }
        void print(int a)
           {
    cout <<"obj"<<a<<"="<< r <<"+j"<<i<<endl;
        }
};

int main () {
    Complex operand1, operand2, result;

    cin >> operand1.r;
    cin >> operand1.i;
    cin >> operand2.r;
    cin >> operand2.i;
  operand1.print(1);
    operand2.print(2);
    result.add(operand1, operand2);
    result.print(3);
}
