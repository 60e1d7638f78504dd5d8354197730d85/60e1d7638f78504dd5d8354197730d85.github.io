#include <iostream>
#include <math.h>
using namespace std;
int main()
{
	int i;
  	float j;
  	cin>>i;
  	cin>>j;
  if(i>0)		
  cout<<"Absolute value of "<<i<<" = "<<i<<endl;
  else
  cout<<"Absolute value of "<<i<<" = "<<-1*i<<endl;  
  if(j>0)
  cout<<"Absolute value of "<<j<<" = "<<j;
  else
  cout<<"Absolute value of "<<j<<" = "<<-1*j;  
}
