#include <iostream>
#include <string.h>
using namespace std;
int main()
{
	char s[100];
  	int i,n;
  	bool flag = false;
  	cin>>s;
  	for(i = 0, n = strlen(s); i < n/2 && !flag; i++)
      if(s[i] != s[n - i - 1])
        flag = true;
  	if(!flag)
      cout<<s<<" is Palindrome";
  	else
      cout<<s<<" is not Palindrome";
	return 0;
}
