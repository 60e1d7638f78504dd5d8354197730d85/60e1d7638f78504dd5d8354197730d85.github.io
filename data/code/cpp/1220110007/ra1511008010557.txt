#include <iostream>
using namespace std;
int main()
{
  char str[100];
	int i=0,j=-1,flag=0;
    cin>>str;
	while(str[++j]!='\0');
		j--;
	while(i<j){
		if(str[i++] != str[j--]){
			flag=1;
			break;
		}
	}     
	if(flag == 0)
		cout<<str<<" is Palindrome\n";
	else
		cout<<str<<" is not Palindrome\n";

	return 0;
}
