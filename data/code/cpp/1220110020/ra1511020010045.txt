#include <iostream>
using namespace std;
int main()
{
string s;
  int count=0,t=0,i;
  cin>>s;
  try
  {
    for(i=0;s[i]!='\0';i++)
    {
  
      if(s[i]>='A'&&s[i]<='Z')
        count++;
    
    else if(s[i]>='0'&&s[i]<='9')
      t++;
    }
    if(t>0)
      throw s;
    else
      cout<<"Valid Input";
  }
  catch(string s)
  {
    cout<<"Number format exception";
  }
	return 0;
}
