#include <iostream>
using namespace std;

int main()
{
    char o;
    float num1, num2;

    cin >> o;

    cin >> num1 >> num2;
    
    switch (o) 
    {
        case '+':
            cout <<num1+num2;
            break;
        case '-':
            cout <<num1-num2;
            break;
        case '*':
            cout <<num1*num2;
            break;
        case '/':
            cout <<num1/num2;
            break;
        default:
            // operator is doesn't match any case constant (+, -, *, /)
            cout << "Error! operator is not correct";
            break;
    }
    
    return 0;
}
