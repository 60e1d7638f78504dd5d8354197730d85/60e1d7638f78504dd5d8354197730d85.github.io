#include <iostream>
using namespace std;
int main()
{
int a,b;
    cin >> a >> b;
    try
    {
        if (b != 0)
        {
            float div = (float)a/b;
            if (div < 0)
                throw 'e';
            cout << "The result of division is:" << div;
        }
        else
            throw b;
            
    }
    catch (int e)
    {
        cout << "There is a problem performing calculation.Check your input again";
    }
    catch (char st)
    {
        cout << "Exception: Division is less than 1";
    }
    catch(...)
    {
        cout << "Exception: Unknown";
    }
	return 0;
}
