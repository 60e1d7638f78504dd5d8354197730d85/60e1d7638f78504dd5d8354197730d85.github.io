#include <iostream>
using namespace std;

int main()
{
    int n, t1 = 0, t2 = 1, nextTerm = 0;

    cin >> n;

    try
    {
      if(n==1)
        throw 0;
      else
        
      {
         cout << "Fibonacci series are:"<<endl;
        for (int i = 1; i <= n; ++i)
       {
        // Prints the first two terms.
        if(i == 1)
        {
            cout <<t1<<endl;
            continue;
        }
        if(i == 2)
        {
            cout << t2 <<endl;
            continue;
        }
        nextTerm = t1 + t2;
        t1 = t2;
        t2 = nextTerm;
        
        cout << nextTerm <<endl;
      }
     }
    }
      catch(int a)
      {
        cout<<"Exception occurred: value thrown";
      }
    return 0;
}
