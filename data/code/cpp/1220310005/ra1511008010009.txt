#include <iostream>
using namespace std;

int main()
{
    int year;

    cin >> year;

    if (year % 4 == 0)
    {
        if (year % 100 == 0)
        {
            if (year % 400 == 0)
                cout << "This is a Leap Year";
            else
                cout << "This is not a Leap Year";
        }
        else
            cout <<"This is a Leap Year";
    }
    else
        cout <<"This is not a Leap Year";

    return 0;
}
