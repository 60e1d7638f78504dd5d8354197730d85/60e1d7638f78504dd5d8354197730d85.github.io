#include <iostream>
using namespace std;
int main()
{
	int arr[50], size, i, j, temp;
	cin>>size;
  if(size>0){
	for(i=0; i<size; i++)
	{
		cin>>arr[i];
	}
	j=i-1;  // now j will point to the last element
	i=0;   //  and i will be point to the first element
	while(i<j)
	{
		temp=arr[i];
		arr[i]=arr[j];
		arr[j]=temp;
		i++;
		j--;
	}
    cout<<"Reverse of Array: ";
	for(i=0; i<size; i++)
	{
		cout<<arr[i];
	}
  }else{
    cout<<"Exception occurred: value thrown ";
  }
  return 0;
}
