#include <stdio.h>
int main()
{
  int i,j,b[100],a[100][100],k,l,m,n,e,f=0;
  scanf("%d",&n);
  for(i=1;i<=n;i++)
  {
    scanf("%d%d",&k,&l);
    for(j=1;j<=k;j++)
    {
      for(m=1;m<=l;m++)
      {
        scanf("%d",&a[j][m]);
      }
    }
    scanf("%d",&e);
    for(j=1;j<=k;j++)
    {
      for(m=1;m<=l;m++)
      {
        if(a[j][m]==e)
        {
          f=1;
          b[i]=1;
          break;
        }
        if(f==1)
          break;
      }
    }
  }
    for(i=1;i<=n;i++)
      printf("%d\n",b[i]);
    

	return 0;
}
