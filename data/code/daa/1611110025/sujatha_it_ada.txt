#include <stdio.h>
int main()
{
  int i,j,n,a[101],temp;
  scanf("%d",&n);
  for(i=0;i<n;i++)
    scanf("%d",&a[i]);
  for(i=0;i<n;i++)
  {
    for(j=i+1;j<n;j++)
    {
      if(a[i]<a[j])
      {
        temp=a[i];
        a[i]=a[j];
        a[j]=temp;
      }
    }
  }
      
printf("The third Largest element is %d",a[2]);

	return 0;
}
