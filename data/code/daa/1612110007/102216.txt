#include <stdio.h>
int main()
{
  int i,j,k,l,a[100],m,n;
  scanf("%d",&n);
  for(i=1;i<=n;i++)
    scanf("%d",&a[i]);
  for(i=2;i<=n;i++)
  {
    for(j=1;j<i;j++)
    {
      if(a[i]<a[j])
      {
        m=a[i];
        a[i]=a[j];
        a[j]=m;
      }
    }
    for(k=1;k<=n;k++)
      printf("%d ",a[k]);
    printf("\n");
  }
    
      

	return 0;
}
