#include <stdio.h>
#include <string.h>
 
quickSort(char arr[],int low,int high)
{
    int pi;
    if (low < high)
    {
        pi = partition(arr, low, high);
 
        quickSort(arr, low, pi - 1); 
        quickSort(arr, pi + 1, high);
    }
}
 
partition (char arr[],int low,int high)
{
    int pivot = arr[high]-97; 
    int i,j;
    char k;
 
    i = (low - 1);
 
    for (j = low; j <= high- 1; j++)
    {
        
        if ((arr[j]-97) <= pivot)
        {
            i++;
            k=arr[i];
            arr[i]=arr[j];
            arr[j]=k;
        }
    }
    
    k=arr[i+1];
    arr[i+1]=arr[high];
    arr[high]=k;
    
    return (i + 1);
}

 }
