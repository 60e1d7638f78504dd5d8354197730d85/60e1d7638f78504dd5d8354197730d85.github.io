#include <iostream>
using namespace std;

int main() {
	//code
	int t;
	cin>>t;
	while(t--)
	{
	    int n,k;
	    cin>>n>>k;
	    int a[n];
	    for(int i=0;i<n;i++)
	    cin>>a[i];
	    int max=a[0];
	    for(int i=1;i<n;i++)
	    if(a[i]>max)
	    max=a[i];
	    
	    if(max>k)
	    cout<<max-k;
	    else
	    cout<<"-1";
	    cout<<endl;
	}
	return 0;
}
