#include <stdio.h>
int main()
{
 int t,s,num,i,j,k,a[10],b[10],count=0,temp;
  scanf("%d",&t);
  for(i=0;i<t;i++)
  {
    count=0;
    //printf("\niteration:%d\n",i+1);
    scanf("%d%d",&s,&num);
    for(j=0;j<s;j++)
    {
      scanf("%d",&a[j]);
    }
    for(j=0;j<s;j++)
    {
      scanf("%d",&b[j]);
    }
    for(j=0;j<s;j++)
    {
      for(k=j+1;k<s;k++)
      {
        if(a[k]>a[j])
        {
          temp=a[j];
         a[j]=a[k];
         a[k]=temp;
        }
        if(b[k]<b[j])
        {
          temp=b[j];
          b[j]=b[k];
          b[k]=temp;
        }
      }
    }
    /*printf("\n");
    for(j=0;j<s;j++)
    {
      printf("\t%d",a[j]);
    }
    printf("\n");
    for(j=0;j<s;j++)
    {
      printf("\t%d",b[j]);
    }*/
    for(j=0;j<s;j++)
    {
      if(a[j]+b[j]>=num)
      {
        count++;
      }
      else
      {
        printf("\n0");
        break;
      }
    }
    if(count>=s)
    {
      printf("\n1");
    }
  }
    
	return 0;
}
