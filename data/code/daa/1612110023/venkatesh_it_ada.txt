#include <iostream>
using namespace std;

void sort(int arr[], int n)
{
   int i, key, j;
   for (i = 1; i < n; i++)
   {
       key = arr[i];
       j = i-1;
       while (j >= 0 && arr[j] > key)
       {
           arr[j+1] = arr[j];
           j = j-1;
       }
       arr[j+1] = key;
   }
}

int main()
{
int T;
  cin>>T;
  for(int t=0;t<T;t++)
  {
    int n,m,a[100],sum=0;
    cin>>n>>m;
    for(int i=0;i<n;i++)
      cin>>a[i];
    sort(a,n);
    for(int i=0;i<m;i++)
    {
      if(a[i]>=0)
        break;
      sum+=a[i];
    }
    cout<<-sum<<endl;
  }
	return 0;
}
