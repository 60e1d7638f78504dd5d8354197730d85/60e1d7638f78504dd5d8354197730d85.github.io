#include<bits/stdc++.h>
using namespace std; 
int findMinimum(int arr[], int n, int k)
{
    int res = 0;
    for (int i=0; i<n ; i++)
    {
        res += arr[i];
        n = n-k;
    }
    return res;
}

int findMaximum(int arr[], int n, int k)
{
    int res = 0, index = 0;
 
    for (int i=n-1; i>=index; i--)
    {
        res += arr[i];
        index += k;
    }
    return res;
}
 
int main()
{
    int t;
    cin >>t;
    while (t--){
        int arr[100000],n,k;
        cin >> n >>k;
        for(int i=0;i<n;i++){
            cin >>arr[i];
        }
        sort(arr, arr+n);
        cout << findMinimum(arr, n, k)<<" "<< findMaximum(arr, n, k)<<endl;
    }
    return 0;
}
