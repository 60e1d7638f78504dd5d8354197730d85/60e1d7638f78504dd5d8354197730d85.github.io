#include  <iostream>
using namespace std;
 
int shellSort(int arr[], int n)
{
    for (int gap = n/2; gap > 0; gap /= 2)
    {
        for (int i = gap; i < n; i += 1)
        {
            int temp = arr[i];
            int j;            
            for (j = i; j >= gap && arr[j - gap] > temp; j -= gap)
                arr[j] = arr[j - gap];             
            arr[j] = temp;
        }
      for (int i=0; i<n; i++)
        cout << arr[i] << " ";
      cout<<endl;
    }
    return 0;
}
 
int main()
{
    int N,arr[50];
    cin>>N;
  for(int i=0;i<N;i++)
    cin>>arr[i];
    shellSort(arr, N);
 
    return 0;
}
