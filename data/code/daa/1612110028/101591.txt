#include<stdio.h> 
int bubble_sort(int[], int); 
int main() 
{
   int arr[30], num, i;  
   scanf("%d", &num);
   for (i = 0; i < num; i++)
      scanf("%d", &arr[i]); 
   bubble_sort(arr, num);
return 0;
} 
int bubble_sort(int iarr[], int num)
{
   int i, j, k, temp;  
   for (i = 1; i < num; i++)
   {
      for (j = 0; j < num - 1; j++)
      {
         if (iarr[j] > iarr[j + 1])
         {
            temp = iarr[j];
            iarr[j] = iarr[j + 1];
            iarr[j + 1] = temp;
         }
      }
      for (k = 0; k < num; k++) 
      {
         printf("%d ", iarr[k]);
      }
          printf("\n");
   }
}
