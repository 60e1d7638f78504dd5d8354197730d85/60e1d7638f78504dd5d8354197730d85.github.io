#include <iostream>
using namespace std;
int main()
{
	int T;
  cin>>T;
  for(int t=0;t<T;t++)
  {
    int N,w[1000];
    cin>>N;
    for(int i=0;i<N;i++)
      cin>>w[i];
    for(int i=0;i<N;i++)
    {
      int c=0;
      for(int j=i+1;j<N;j++)
        if(w[j]<w[i])
          c++;
      cout<<c<<" ";
    }
  }
	return 0;
}
