#include <stdio.h>
 
//void quicksort(int *a, int len);
int main()
{
    int n,k,i,id,t,j,aux,band;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&k);
 
        int prince[n];
        int hurdle[k];
        for(i=0; i<n; i++)
        {
            scanf("%d",&prince[i]);
        }
        for(i=0; i<k; i++)
        { 
            scanf("%d",&hurdle[i]);
        }
         // quicksort(hurdle,k);
        id=0;
        aux=0;
 
        for(i=0; i<n; i++)
        {
            band=0;
            for(j=aux; j<k; j++)
            {
                if(prince[i]>=hurdle[j])
                    id=i;
                else
                {
                    band=1;
                    aux=j;
                    break;
                }
            }
            if(band==0)
                break;
        }
        printf("%d\n",id);
    }
    return 0;
}
