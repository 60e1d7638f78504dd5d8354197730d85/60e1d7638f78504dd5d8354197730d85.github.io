#include <iostream>
using namespace std;

int find(int a[],int l,int h,int X){
  if(l==h)
    return (a[l]==X)?1:0;
  int mid=(l+h)/2;
  return (find(a,l,mid,X)+find(a,mid+1,h,X));
}

int main()
{
int n,a[20],q,x;
  cin>>n>>q;
  for(int i=0;i<n;i++)
    cin>>a[i];
  for(int i=0;i<q;i++){
    cin>>x;
  find(a,0,n-1,x)?cout<<"YES\n":cout<<"NO\n";
  }
	return 0;
}
