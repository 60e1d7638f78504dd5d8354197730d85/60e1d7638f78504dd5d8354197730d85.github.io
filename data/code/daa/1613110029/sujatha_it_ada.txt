#include <iostream>
using namespace std;

void sort(int arr[], int n)
{
   int i, key, j;
   for (i = 1; i < n; i++)
   {
       key = arr[i];
       j = i-1;
       while (j >= 0 && arr[j] > key)
       {
           arr[j+1] = arr[j];
           j = j-1;
       }
       arr[j+1] = key;
   }
}
int check(int A[],int n,int num,int k)
{
 int i,j,prev=A[0]+num;
k--;
 for(i=1;i<n;i++)
 {
   if(prev+num>=A[i])
    continue;
   if(k==0)
    return 0;
   prev=A[i]+num;
   k--;
 }
 return 1;
}
