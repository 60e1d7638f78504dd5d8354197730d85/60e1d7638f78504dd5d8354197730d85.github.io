    #include<stdio.h>
    #include<stdlib.h>
    #include<string.h>
    typedef struct
    {   int age;
        char name[5],hometown[5],address[5];
    }emp;
    int main()
    {
        int n,i,j,k=0;
        emp *e;
        scanf("%d",&n);
        e=(emp*)malloc(sizeof(emp)*n);
        for(i=0;i<n;i++)
        {	scanf("%s",e[i].name);
        	scanf("%d",&e[i].age);
        	scanf("%s",e[i].hometown);
        	scanf("%s",e[i].address);
        }
        for(i=0;i<n;i++)
        {	for(j=i+1;j<n;j++)
        	    if(e[i].age==e[j].age || strcmp(e[i].name,e[j].name)==0 || strcmp(e[i].hometown,e[j].hometown)==0 || strcmp(e[i].address,e[j].address)==0)
        	    	k++;	
        }
        printf("%d\n",k);
        for(i=0;i<n;i++)
        {	for(j=i+1;j<n;j++)
        	    if(e[i].age==e[j].age || strcmp(e[i].name,e[j].name)==0 || strcmp(e[i].hometown,e[j].hometown)==0 || strcmp(e[i].address,e[j].address)==0)
        	    	printf("%d %d\n",i+1,j+1);	
        }
        return 0;
    }
