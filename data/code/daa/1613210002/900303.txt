#include <bits/stdc++.h>
using namespace std;
#define f first
#define s second
#define p(a,b) pair<a,b>
#define mp make_pair
 
typedef long int ll;
 
#define gc getchar
 
void input(ll &x)
{
    register ll c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = x*10 + c - 48;}
}
 
int main ()
{
    ll test;
    input(test);
    while(test--)
    {
        ll n,a,b,c,i,j,k,sum=0,ans=0;
        input(n);
        input(a);
        input(b);
        input(c);
        
        vector<ll>arr1(n),arr2(n);
        
        for(i=0;i<n;i++)
        {
            input(arr1[i]);
            input(arr2[i]);
        }   
        
        sort(arr1.begin(),arr1.end());
        sort(arr2.begin(),arr2.end());
        ll count1=0,count2=0,count3=n;
        for(j=0,k=0;j<n&&k<n;)
        {
            if(arr1[j]<=arr2[k])
            {
                count1++;
                count3--;
                j++;
            }
            else
            {
                count1--;
                count2++;
                k++;
            }
 
            sum=count1*b+count2*c+count3*a;
            ans=max(ans,sum);
        
        }
        
        cout<<ans<<endl;
        
    }
    return 0;
}
