#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

int main()
{
       int min,n,i,x,z,j,k,a[100000]={0};
       scanf("%d",&n);
       for(i=0;i<n;i++)
       {
              scanf("%d",&a[i]);
       }
       z = 0;
       for(i=0;i<n;i++)
       {
              if(a[i]==-1)
              {
                     continue;
              }
              min = a[i];
              for(j=0;j<n;j++)
              {
                     if(min>a[j] && a[j]!=-1)
                     {
                           min = a[j];
                     }
              }
              x = min;
              //printf("%d\n",min);
              z++;
              for(j=0;j<n;j++)
              {
                     if(a[j]==-1)
                     {
                           continue;
                     }
                     if(a[j] >= x  && a[j] <= x+4)
                     {
                           a[j] = -1;
                           i = -1;
                     }
              }
       }
       printf("%d",z);
    return 0;
}
