#include<bits/stdc++.h>
using namespace std;
 
int main()
{
    int N;
    cin>>N;
    long long int Cat[N],Calorie[N],Answer=0;
    for(int i=0;i<N;i++)
    {
        cin>>Cat[i];
    }
    
    for(int i=0;i<N;i++)
    {
        cin>>Calorie[i];
    }
    
    sort(Cat,Cat+N);
    sort(Calorie,Calorie+N);
    
    for(int i=0;i<N;i++)
    {
        Answer=Answer+(Cat[i]*Calorie[i]);
    }
    cout<<Answer<<endl;
}
