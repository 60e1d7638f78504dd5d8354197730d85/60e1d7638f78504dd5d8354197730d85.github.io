#include <stdio.h>
int main()
{
int i,t,j,n,c,q;
  long int number;
  scanf("%d",&t);
  for(i=0;i<t;i++)
  {
    q=0;
    scanf("%d",&n);
    for(j=0;j<n;j++)
    {
      scanf("%ld",&number);
      if(number % 2)
      {
        c=1;
      }
      else
      {
        c=0;
      }
      q=q^c;
    }
    if(q)
    {
      printf("No\n");
    }
    else
    {
      printf("Yes\n");
    }
  }
        
      
	return 0;
}
