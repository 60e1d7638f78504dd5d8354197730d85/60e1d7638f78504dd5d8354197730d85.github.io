int main(){
	int n,i,j;
	vector<int>v;
	cin>>n;
	for(i=0;i<n;i++){
		cin>>j;
		v.pb(j);
	}
	sort(all(v));
	int ans=0;
	for(i=0;i<v.size();i++){
		if(i+2>=n)
		break;
		ans=max(ans,v[i+2]-v[i]);
	}
	cout<<ans;
	return 0;
}
