#include<bits/stdc++.h>
 
using namespace std;
 
int main() {
    int t;
    cin>>t;
    while(t--){
        string s;
        cin>>s;
        int l = s.length();
        string ans = "";
        for(int i=1;i<l;i++) {
            if(s[i] != s[i-1])ans += s[i-1];
        }
        ans += s[l-1];
        cout<<ans<<endl;
    }
    return 0;
}
