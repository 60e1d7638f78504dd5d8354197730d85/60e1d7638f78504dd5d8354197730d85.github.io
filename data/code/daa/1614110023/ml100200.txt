#include<stdio.h>
#include<stdlib.h>
 
int main(){
 
    int t=0,n=0,q=0,i,j;
    long max,min;
 
    scanf("%d",&t);
 
    for( i=0;i<t;++i){
 
        long tmp;
 
        scanf("%d %d",&n,&q);
 
        scanf("%ld",&max);
        min=max;
 
        for( j=0;j<n-1;++j){
            scanf("%ld",&tmp);
            if(tmp>max){
                max=tmp;
            }
            if(tmp<min){
                min=tmp;
            }
        }
 
        printf("%ld\n",(max-min)*q);
 
    }
 
    return 0;
}
