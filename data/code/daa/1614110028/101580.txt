#include <stdio.h>
int comp (const void * elem1, const void * elem2)
{
    int f = *((int*)elem1);
    int s = *((int*)elem2);
    if (f > s) return  1;
    if (f < s) return -1;
    return 0;
}
int main()
{
   long long int i,j,n,sum1=0,sum2=0,temp;
   scanf("%lld",&n);
   long long int a[n];
   for(i=0;i<n;i++)
   {
   	scanf("%lld",&a[i]);
   }
   qsort (a, sizeof(a)/sizeof(*a), sizeof(*a), comp);
 
   if(a[0]<0)
   {
       temp=a[0];
   	for(i=0;i<n;i++)
   	{
   		a[i]-=temp;
   	}
   }
   for(i=0;i<(n/2);i++)
   {
   	sum1+=a[i];
   }
   sum1*=2;
   for(i=(n/2+n%2);i<n;i++)
   {sum2+=a[i];
   }
   sum2*=2;
   printf("%lld\n",sum2-sum1);
    return 0;
}
