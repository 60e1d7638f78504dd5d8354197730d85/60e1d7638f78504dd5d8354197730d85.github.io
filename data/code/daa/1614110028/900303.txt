

    #include <bits/stdc++.h>
    using namespace std;
    typedef long long ll;
    const ll p = 1e9+7;
    const int MAXN = 1e5+5;
    ll nums[MAXN], n, sum;
    int main() {
        cin >> n; for (int i = 0; i < n; ++i)
            cin >> nums[i];
        sort(nums, nums+n);
        for (int i = 0; i < n/2; ++i)
            sum -= 2*nums[i], sum += 2*nums[n-1-i];
        cout << sum << "\n";
    }
