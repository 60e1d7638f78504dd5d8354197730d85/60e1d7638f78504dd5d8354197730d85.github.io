#include<iostream>
using namespace std;
main()
{
    string s;
    cin>>s;
    int i=0,k=s.length(),j,t;
    while(s[i]!='\0')
    {
        if(s[i]=='?')
        {
            t=0;
            j=i;
            if(j==k)
            {
                if(s[j-1]=='a')
                    s[j]='b';
                else
                    s[j]='a';
            
             //   t=1;
                continue;
            }
            if(j==0)
            {
                if(s[j+1]=='a')
                    s[j]='b';
                else
                    s[j]='a';
               // t=1;
               continue;
            }
            if(s[j+1]=='a' || s[j-1]=='a')
            {
                s[i]='b';
            }
            else
            {
                s[i]='a';
            }
        }
        i++;
    }
    cout<<s;
}
