#include<bits/stdc++.h>
using namespace std;
int main(){
        string S;
        cin>>S;
        int n=S.length();
        assert(n>=1&&n<=10000);
        int v[n];
        memset(v,0,sizeof(v));
        //if charcter is even then set its frequency 1 else 0
        for(int i=0;i<n;i++){
        if(int(S[i])%2==0)
        v[i]=1;
        else
        v[i]=0;
        }
        //cumulative array
        for(int i=n-2;i>=0;i--)
        v[i]+=v[i+1];
        for(int i=0;i<n;i++)
        cout<<v[i]<<" ";
        return 0;
}
