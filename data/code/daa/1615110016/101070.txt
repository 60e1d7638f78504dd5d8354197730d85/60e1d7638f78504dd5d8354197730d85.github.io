#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>
 
#define MIN(a,b) (((a)<(b))?(a):(b))
#define MAX(a,b) (((a)>(b))?(a):(b))
 
#define MA 1000000000000000000 // 1e18
#define M  1000000009
#define MM 10000001
#define K  5005
 
int m,n;
 
char p[K];
int m,n;
int a[K], s[K];
int pr[K];
int pc;
int ss[K];
 
void prime() {
  int i, j;
  memset(p, 1, K);
 
  for(i=3;i*i<K;i+=2) {
    if (p[i]) {
      for(j=i+i+i;j<K;j+=i+i)
        p[j]=0;
    }
  }
 
  pc=0;
  pr[pc++]=2;
  for(i=3;i<K;i+=2) {
    if (p[i])
      pr[pc++]=i;
  }
}
 
int main() {
  int t;
  int i,j,k,l,r;
 
  scanf("%d", &n);
 
  prime();
  if (n<2) {
    puts("0");
  } else {
    for(i=0;i<n;i++) {
      scanf("%d", a+i);
    }
    s[0]=0;
    for(i=0;i<n;i++) {
      s[i+1]=s[i]+a[i];
    }
 
    ss[0]=0;ss[1]=0;
    ss[2]=a[0]+a[1]; r=0;
    for(i=2;i<n;i++) {
      ss[i+1]=ss[i];
      for(j=0;j<pc && pr[j]<=i+1;j++) {
        ss[i+1]=MAX(ss[i+1], ss[i-pr[j]]+s[i+1]-s[i-pr[j]+1]);
      }
      if (r<ss[i+1]) r=ss[i+1];
    }
    printf("%d\n", r);
  }
  return 0;
}
