#include <stdio.h>
int main()
{
    int n;
    scanf("%d",&n);
    while(n--)
    {
    	int x,i,j;
    	long long int r[200],h[200],temp1,temp2,max[200],ans=0;
    	scanf("%d",&x);
    	for(i=0;i<x;i++)
    	{
    		scanf("%lld %lld",&r[i],&h[i]);
    		temp1=r[i];
    		temp2=h[i];
    		for(j=i-1;j>=0;j--)
    		{
    			if(temp1>r[j])
    			{
    				r[j+1]=r[j];
    				h[j+1]=h[j];
    			}
    			else
    				break;
    		}
    		r[j+1]=temp1;
    		h[j+1]=temp2;
    	}
        for(i=0;i<x;i++)
    	{
    		max[i]=h[i];
    		ans=max[i];
    		temp1=0;
    		while(temp1<i)
    		{
    			if(r[temp1]>r[i] && h[temp1]>h[i])
    			{
    				if(ans<max[temp1]+max[i])
    					ans=max[temp1]+max[i];
    			}
    			temp1++;
    		}
    		max[i]=ans;
    	}
    	ans=0;
    	for(i=0;i<x;i++)
    	{
    		if(ans<max[i])
    			ans=max[i];
    	}
    	printf("%lld\n",ans);
    }
    return 0;
}
