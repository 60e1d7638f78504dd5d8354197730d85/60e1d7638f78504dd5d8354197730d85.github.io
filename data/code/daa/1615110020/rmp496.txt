#include<bits/stdc++.h>
using namespace std;
long long dp[100005]={0},a[100005]={0};
int main(){
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>a[i];
    dp[0] = 1;
    for(int i=1;i<n;i++)
        if(a[i] > a[i-1])
            dp[i]=dp[i-1] + 1;
        else
            dp[i]=1;
    for(int i=n-2;i>=0;i--)
        if(a[i] > a[i+1] && dp[i] <= dp[i+1])
            dp[i]=dp[i+1]+1;
    long long ans=0;
    for(int i=0;i<n;i++)
        ans += dp[i];
    cout<<ans<<endl;
        
}
