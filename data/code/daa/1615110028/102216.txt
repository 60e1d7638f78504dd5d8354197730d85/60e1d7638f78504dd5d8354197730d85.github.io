#include<stdio.h>
#include<string.h>
int main() {
	
	int m,t;
	scanf("%d",&t);
	while(t--) {
		
		int i,j,m,n;
		scanf("%d%d",&m,&n);
		int s[n];
		int dp[10005];
		memset(dp,0,sizeof(dp));
		for(i=0;i<n;i++) {
		
			scanf("%d",&s[i]);
			dp[s[i]]=1;
		}
		for(i=0;i<=m;i++) {
		
			for(j=0;j<n;j++) {
				
				if(i-s[j]>=0 && dp[i-s[j]]==0) {
					dp[i]=1;
					break;
				}
			}
		}
		if(dp[m]==1)
			printf("Little Deepu\n");
		else
			printf("Kate\n");
	}
	return 0;
}
