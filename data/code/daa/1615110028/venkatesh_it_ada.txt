#include <iostream>
using namespace std;
 
int a[100001];
int b[100001];
int main()
{
	int t,n,m,i,j,k;
	scanf("%d",&t);
	while(t--){
		scanf("%d%d",&m,&n);
		for(i=0;i<=m;i++){
			b[i] = 0;
			
		}
		for(i=0;i<n;i++){
			scanf("%d",&a[i]);
		}
		for(i=0;i<=m;i++){
			if(b[i]==0){
				for(j=0;j<n;j++){
					k =i+a[j];
					if(k<=m) b[k] =1;
				}
			}
		}
		if(b[m])
		printf("Little Deepu\n");
		else
			printf("Kate\n");
	}
	      return 0;
}
