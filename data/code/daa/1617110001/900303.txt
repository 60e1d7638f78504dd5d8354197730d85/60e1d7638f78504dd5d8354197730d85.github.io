    #include <stdio.h>
     
    int main()
    {
        int n;
        
        scanf("%d", &n);
        int sumdeg = 0;
        
        for (int i = 0; i < n; i++)
        {
        	int k;
        	scanf("%d", &k);
        	sumdeg += k;
        }
        
        puts(sumdeg == 2*(n-1) ? "Yes" : "No");
        
        return 0;
    }
