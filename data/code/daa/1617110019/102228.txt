#include<stdio.h>
int n,m;
int x[1000000],y[1000000],dp[50000][20];
int getMin(int a,int b){
	if(b==-1)
		return a;
	if(a==-1)
		return b;
	return a<b?a:b;
}
int getWeight(int a, int b){
	int i;
	for(i=0;i<m;i++){
		if(x[i]==a && y[i]==b){
			return 1;
		}else if(x[i]==b && y[i]==a){
			return 1;
		}
	}
	return 0;
}
int main(void){
	int t;
	scanf("%d",&t);
	while(t--){
		int i,j,mask;
		scanf("%d%d",&n,&m);
		for(int i=0;i<m;i++)
			scanf("%d%d",&x[i],&y[i]);
		int end=(1<<n);
		for(i=0;i<end;i++){
			for(j=0;j<n;j++)
				dp[i][j]=-1;
		}
		for(i=0;i<n;i++){
			dp[1<<i][i]=0;	
		}
		for(mask=1;mask<end;mask++){
			for(i=0;i<n;i++){
				if(!(mask&(1<<i))) continue;
				int temp=mask-(1<<i);
				for(j=0;j<n;j++){
					if(!(temp&(1<<j))) continue;
					int weight=getWeight((i+1),(j+1));
					if(weight!=0){
						int t=dp[temp][j]+(1);
						if(dp[mask][i]<t)
							dp[mask][i]=t;
					}
				}
			}
		}
		int s[1000];
		for(i=0;i<n;i++){
			int max=dp[0][i];
			for(j=1;j<end;j++){
				if(max<dp[j][i])
					max=dp[j][i];	
			}
			s[i]=max;
		}
		for(i=0;i<n;i++)
			printf("%d ",s[i]);
		printf("\n");
	}
	return 0;
}
