#include<bits/stdc++.h>
using namespace std;
int a[15][15];
int visited[11];
void print(int visited[] , int n)
{
    int i;
    for(i = 1;i<=n;i++)
    {
        if(visited[i] == 1)
        {
            printf("%d ",i);
        }
    }       
    printf("\n");
}
int check(int visited[] , int n)
{
    int i;
    for(i = 1;i<=n;i++)
    {
        if(visited[i] == 0 )
        {
            return 0;
        }
    }        
    return 1;
}
int rec(int x , int n , int visited[])
{
    int i;
        
    if(check(visited,n))
        return 1;
    for(i=1;i<=n;i++)
    {
        if((!visited[i]) && a[x][i] == 1)
        {
            visited[i] = 1;
            if(rec(i,n,visited))
            {
                return 1;
            }
            visited[i] = 0;
        }
    }
    return 0;
}
int main()
 {
	int t;
    cin >> t;
    while(t--)
    {
        int flag = 0;
        int n,i,j,p,q,m;
        cin >> n >> m;
        for(i = 0 ; i <= n ; i++ )
            visited[i] = 0;
            
        for(i=0;i<11;i++)
        {
            for(j=0;j <11 ; j++)
                a[i][j] = 0;
        }  
        for(i=0;i< m;i++)
        {
               cin >> p >> q;
               a[p][q] = 1;
               a[q][p] = 1;
        }
        visited[0] = 1;
		for(i=1;i<=n;i++)
		{
			visited[i] = 1;
			if(rec(i,n,visited))
			{
				flag = 1;
				printf("1\n");
				break;
			}
			visited[i] = 0;
		}	
        if(flag == 0)
            printf("0\n");
    }
	return 0;
}
