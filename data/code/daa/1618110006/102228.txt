#include<bits/stdc++.h>

void Search(char *pat, char *txt)
{
    int M = strlen(pat);
    int N = strlen(txt);
    int x[M];
 
    compute(pat, M, x);
 
    int i = 0;  
    int j  = 0;  
    while (i < N)
    {
        if (pat[j] == txt[i])
        {
            j++;
            i++;
        }
 
        if (j == M)
        {
            printf("Pattern found at index %d\n", i-j);
            j = x[j-1];
        }
        else if (i < N && pat[j] != txt[i])
        {
            if (j != 0)
                j = x[j-1];
            else
                i = i+1;
        }
    }
}
void compute(char *pat, int M, int *x)
{
    int len = 0;
    x[0] = 0; 
    int i = 1;
    while (i < M)
    {
        if (pat[i] == pat[len])
        {
            len++;
            x[i] = len;
            i++;
        }
        else 
        {
            if (len != 0)
            {
                len = x[len-1];
            }
            else 
            {
                x[i] = 0;
                i++;
            }
        }
    }
}
int main()
{
    char txt[20];
    char pat[20];
  cin>>txt;
  cin>>pat;
Search(pat, txt);
    return 0;
}
