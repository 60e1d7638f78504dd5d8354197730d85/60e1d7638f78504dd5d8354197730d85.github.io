#include<bits/stdc++.h>

using namespace std;

int main(){
    int q;
    cin >> q;
    string t = "hackerrank";
    while(q--){
        string s;
        cin >> s;
        int l = 0;
        for(int i = 0; i < s.length(); i++){
            if(s[i] == t[l]){
                l++;
            }
            if(l == t.length()){
                break;
            }
        }
        if(l == t.length()){
            cout << "YES\n";
        } else {
            cout << "NO\n";
        }
    }
}
