#include <stdio.h>
 #include<string.h>
int palindrome(int i,int j,char *str)
{
while(i<j)
{
	if(str[i]!=str[j])
	return 0;
	
	i++;
	j--;
}
return 1;
}
 
int main()
{
    
  int count=0,i,j,a;
  char str[100];
  
  scanf("%s",str);
    
    for(i=0;i<strlen(str)-1;i++)
    for(j=i+1;j<strlen(str);j++)
    if(palindrome(i,j,str))
    count++;
    a=strlen(str);
    printf("%d\n",count+a);
    	
   
    return 0;
}
