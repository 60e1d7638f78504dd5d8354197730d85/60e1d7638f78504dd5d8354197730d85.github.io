#include<bits/stdc++.h>
using namespace std;
#define rep(i,n) for(i=0;i<n;i++)
#define ll long long
#define elif else if

int main()
{
  ios_base::sync_with_stdio(0);
  int t;
  cin>>t;
  assert(1<=t && t<=100);
  while(t--)
  {
    int n,i,j,ans=0;
    cin>>n;
    assert(2<=n && n<=2000);
    for(i=1 ; i<=n ; i++)
    {
      for( j=i+1; j<=n; j++ )
      {
        if( (i^j) <= n)
        {
          ans++;
        }
      }
    }
    cout<<ans;
    if(t>0)
      cout<<endl;
  }
 return 0;
}
