#include <stdio.h>

int main()
{
   int t,count;
   scanf("%d",&t);
   while(t--)
   {
   long long int a,b,k;
    scanf("%lld %lld",&a,&b);
    if(a==b)
      printf("0\n");
    else
    {
     k=a^b;count=0;
     while( k )
        {
        k = k&(k-1);
           count++;
        }
     printf("%d\n",count);
    }
   }
    return 0;
}
