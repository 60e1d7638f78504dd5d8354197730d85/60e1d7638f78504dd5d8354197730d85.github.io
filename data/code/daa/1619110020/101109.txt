#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t, n, z, i, j, temp, ans;
	cin>>t;
	while(t--)
	{
		cin>>z>>n;
		for(i=0; i<n; i++)
		{
			cin>>temp;
			if(i!=0)
			{
				ans=(ans&temp);
			}
			else
			{
				ans=temp;
			}
		}
		if((ans&z)==0)
		{
			cout<<"Yes"<<endl;
		}
		else
		{
			cout<<"No"<<endl;
		}
	}
	return 0;
}
