#include <stdio.h>
int main()
{
     int a,ids,sums,i,n;
     scanf("%d",&a);
     while(a--)
     {
          scanf("%d",&n);
          int id[n],sum[n];
          for(i=0;i<n;i++)
          {
               scanf("%d %d",&id[i],&sum[i]);
          }
          ids=0;
          sums=0;
          for(i=0;i<n;i++)
          {
               ids+=id[i];
               sums+=sum[i];
          }
          printf("%d\n",ids-sums);
     }
     return 0;
}
