#include <stdio.h>
#include <stdlib.h>

int main()
{
    int num, i, *ptr, sum = 0;

   
    scanf("%d", &num);

    ptr = (int*) malloc(num * sizeof(int));  //memory allocated using malloc
    if(ptr == NULL)                     
    {
        printf("Error! memory not allocated.");
        exit(0);
    }


    for(i = 0; i < num; ++i)
    {
        scanf("%d ", ptr + i);
        sum += *(ptr + i);
    }

    printf("\nSum=%d", sum);
    free(ptr);
    return 0;
}
