#include<iostream>
using namespace std;
int Search_in_matrix(int [30][30],int ,int ,int ) ;
int main()
 {
 int t,row,col,i,j,k;
 int mat[30][30] ;
 cin>>t ;
 while(t--)
 {
     cin>>row>>col ;
    
     for(i=0;i<row;i++)
         for(j=0;j<col;j++)
             cin>>mat[i][j] ;
     cin>>k ;         
     cout<<Search_in_matrix(mat,row,col,k) ;
     cout<<endl ;
 }
 return 0;
}

int Search_in_matrix(int mat[30][30],int row,int col,int k)
{
    int i,j ;
  
    for(i=0;i<row;i++)
        for(j=col-1;j>=0;j--)
        {
            if(mat[i][j]==k)
                return 1 ;
            else if(mat[i][j]<k)
                break ;
        }
    return 0 ;
}
