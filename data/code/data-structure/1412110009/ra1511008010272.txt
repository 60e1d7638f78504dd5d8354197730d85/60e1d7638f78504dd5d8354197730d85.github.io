#include <iostream>
using namespace std;
int main()
{
	int t;
  	cin>>t;
  	while(t--) {
      int n;
      cin>>n;
      int a[n];
      for(int i=0;i<n;i++) {
        cin>>a[i];
      }
      int x;
      for(int i=0;i<n-1;i++) {
        if(a[i]>a[i+1]) {
          x = a[i];
          break;
        }
      }
      cout<<x<<endl;
    }
	return 0;
}
