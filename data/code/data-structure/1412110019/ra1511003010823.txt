#include <stdio.h>
int getInvCount(int arr[], int n)
{
  int inv_count = 0, i, j;
  for (i = 0; i < n - 1; i++)
  {
    for (j = i+1; j < n; j++)
    {
      if (arr[i] > arr[j])
      {
        inv_count++;
      }
    }
  }

  return inv_count;
}

/* Driver progra to test above functions */
int main(int argv, char** args)
{
  int arr[10], i, j, n, m;
  scanf("%d", &m);
  scanf("%d", &n);
  for(i=0;i<n;i++)
  {
    scanf("%d", &arr[i]);
  }
  printf("%d\n", getInvCount(arr, n));
  return 0;
}
