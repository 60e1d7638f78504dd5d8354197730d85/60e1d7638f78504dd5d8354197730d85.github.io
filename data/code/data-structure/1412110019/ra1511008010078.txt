int getInvCount(int arr[], int n)
{
  int inv_count = 0,i,j;
  for (i = 0; i < n - 1; i++)
    for (j = i+1; j < n; j++)
      if (arr[i] > arr[j])
        inv_count++;
 
  return inv_count;
}

  int arr[100];

  {
  scanf("%d",&n);
    for(j=0;j<n;j++)
    {
      scanf("%d",&arr[j]);
    }
      printf("%d \n", getInvCount(arr, n));
  }
  return 0;
}
