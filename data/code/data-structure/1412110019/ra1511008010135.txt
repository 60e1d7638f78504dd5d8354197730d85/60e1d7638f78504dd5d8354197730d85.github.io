#include <iostream>
using namespace std;
int main()
{
	int t;
  cin>>t;
  while(t--)
  {
    int n,i,j,count=0;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++)
    {
      cin>>a[i];
    }
    for(i=0;i<n;i++)
    {
      for(j=i+1;j<n;j++)
      {
        if(a[i]>a[j])count++;
      }
    }
    cout<<count<<endl;
  }
  
        
	return 0;
}
