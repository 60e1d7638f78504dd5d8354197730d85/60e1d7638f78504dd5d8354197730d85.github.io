#include <iostream>
using namespace std;
int main()
{
int T;
  cin>>T;
  while(T--){
  int n,count=0;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++) cin>>a[i];
    for(int i=0;i<n;i++){
      for(int j=0;j<n;j++){
        if(a[i]>a[j]&&i<j) count++;
      }
    }
    cout<<count;
  }
	return 0;
}
