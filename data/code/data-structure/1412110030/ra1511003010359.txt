#include <iostream>
using namespace std;
int main()
{
  int t;
  cin>>t;
  for(int c=0; c<t; c++)
  {
    int n,temp,arr[20];
    cin>>n;
    for(int i=0; i<n; i++)
      cin>>arr[i];
    for(int i=0; i<n; i++)
    {
      for(int j=0; j<(n-i-1); j++)
      {
        if(arr[j]>arr[j+1])
        {
          temp=arr[j+1];
          arr[j+1]=arr[j];
          arr[j]=temp;
         }
       }
     }
    for(int i=0; i<n; i++)
      cout<<arr[i]<<" ";
    cout<<"\n";
  }
	return 0;
}
