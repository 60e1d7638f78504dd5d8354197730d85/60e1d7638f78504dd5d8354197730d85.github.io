#include <stdio.h>
int main()
{
  int n,i;
  scanf("%d",&n);
  int a[n];
  for(i=0;i<n;i++)scanf("%d",&a[i]);
  
  int key,j;
  for(i=1;i<n;i++)
  {
    key=a[i];
    j=i-1;
    while(j>=0 && a[j]>key)
    {
      a[j+1]=a[j];
      j--;
    }
    a[j+1]=key;
  }
  
    float sum=a[n-1]+a[n-2];
 	float res=sum/2;
    printf("%.1f",res);
  

	return 0;
}
