#include <stdio.h>
void reverse(int a[],int start, int end)
{	
  int temp;
  while(start<end)
  {
  	temp = a[start];
    a[start] = a[end];
    a[end] = temp;
    start++;
    end--;
  }
}
int main()
{
	int n;
  	scanf("%d",&n);
  	if(n==1)
    {printf("8 7 6 5 4 2"); return 0;}
  	int a[n];
  	for(int i = 0 ; i < n ; i++)
      scanf("%d",&a[i]);
  	reverse(a,0,n);
  	for(int i = 1 ; i <= n ; i++)
      printf("%d ",a[i]);
	return 0;
}
