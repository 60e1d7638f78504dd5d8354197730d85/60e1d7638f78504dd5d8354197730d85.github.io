#include <stdio.h>
 int main()
{
    int ARR[10], OAR[10], EAR[10];
    int i, j = 0, k = 0, n;
    scanf("%d", &n);
    for (i = 0; i < n; i++)
    {
        scanf("%d", &ARR[i]);
    }
    for (i = 0; i < n; i++)
    {
        if (ARR[i] % 2 == 0)
        {
            EAR[j] = ARR[i];
            j++;
        }
        else
        {
            OAR[k] = ARR[i];
            k++;
        }
    }
    for (i = 0; i < k; i++)
    {
        printf("%d\n", OAR[i]);
    }
    for (i = 0; i < j; i++)
    {
        printf("%d\n", EAR[i]);
    }
 return 0;
}
