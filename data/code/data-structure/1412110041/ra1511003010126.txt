import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner myScanner = new Scanner(System.in);
       	int num = myScanner.nextInt();
       	float sum =0;
       	int[] arr = new int[num];
       	for(int i=0;i<num;i++){
         	arr[i] = myScanner.nextInt();
          	sum = sum + arr[i];
        }
		System.out.printf("%.2f",sum/num);       
	}
}
