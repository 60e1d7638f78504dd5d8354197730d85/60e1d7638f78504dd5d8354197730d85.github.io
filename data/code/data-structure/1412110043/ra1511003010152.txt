#include <stdio.h>
int main()
{
  int n,m,i,j,a[20],b[20],c[20];
  
  scanf("%d %d",&m,&n);
  for(i=0;i<m;i++)
    scanf("%d",&a[i]);
  
  for(i=0;i<n;i++)
    scanf("%d",&b[i]);
  
  for(i=0;i<m;i++)
    c[i]=a[i];
  for(i=m,j=0;i<m+n;i++,j++)
    c[i]=b[j];
  int t=n+m;
  int temp;
  for(i=0;i<t;i++)
    for(j=i+1;j<t;j++)
      if(c[i]>c[j])
      {
        temp=c[i];
        c[i]=c[j];
        c[j]=temp;
      }
  for(i=0;i<m+n;i++)
    printf("%d ",c[i]);
	return 0;
}
