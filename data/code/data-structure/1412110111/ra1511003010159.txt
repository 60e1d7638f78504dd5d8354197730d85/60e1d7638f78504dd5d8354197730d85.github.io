#include <stdio.h>
int main()
{
 int arr[100],n,i;
    int temp;
     
    
    scanf("%d",&n);
     
    //value of n must be even
    if(n%2 !=0)
    {
        
        return 1;
    }
    //read array elements
  
    for(i=0;i < n;i++)
    {
        
        scanf("%d",&arr[i]);
    }
    //swap adjacent elements
    for(i=0;i < n;i+=2)
    {
        temp    = arr[i];
        arr[i]  = arr[i+1];
        arr[i+1]= temp;
    }
     
    
    for(i=0;i < n;i++)
    {
        printf("%d ",arr[i]);
    }

	return 0;
}
