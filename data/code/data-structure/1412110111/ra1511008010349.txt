#include <stdio.h>
int main()
{
int n,i,temp;
  scanf("%d",&n);
  int a[n];
  if(n%2 != 0){
    return 1;
  }
  for(i=0;i<n;i++){
    scanf("%d",&a[i]);
  }
  for(i=0;i < n;i+=2)
    {
        temp    = a[i];
        a[i]  = a[i+1];
        a[i+1]= temp;
    }
     
  for(i=0;i<n;i++){
    printf("%d ",a[i]);
  }
	return 0;
}
