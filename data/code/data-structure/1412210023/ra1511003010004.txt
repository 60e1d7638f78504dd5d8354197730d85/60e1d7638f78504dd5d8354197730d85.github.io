#include<bits/stdc++.h>
using namespace std;
int arraySortedOrNot(int arr[], int n)
{
    if (n == 1 || n == 0)
        return 1;
    if (arr[n-1] < arr[n-2])
        return 0;
    return arraySortedOrNot(arr, n-1);
}
int main()
{
    int arr[6],n,i,m;
  cin>>m;
  for(i=0;i<m;i++)
  {
  cin>>arr[i];
  }
    n = sizeof(arr)/sizeof(arr[0]);
    if (arraySortedOrNot(arr, n))
        cout << "Yes\n";
    else
        cout << "No\n";
}
