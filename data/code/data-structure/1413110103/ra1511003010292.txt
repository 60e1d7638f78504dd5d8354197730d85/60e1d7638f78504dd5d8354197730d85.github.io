import java.io.*;
import java.util.Scanner;
import java.util.Arrays;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner sc = new Scanner(System.in);
       int n = sc.nextInt();
       int arr[] = new int[n];
       for(int i =0 ; i<n ; i++)
         arr[i] = sc.nextInt();
       Arrays.sort(arr);
       System.out.println("The second largest element is " + arr[n-2]);
	}
}
