#include <iostream>
using namespace std;
int main()
{
   int n,i;
  bool bb=true;
  cin>>n;
  int a[n],b[n];
  for(i=0;i<n;i++){
    cin>>a[i];
  }
  for(i=0;i<n;i++){
    cin>>b[i];
  }
  for(i=0;i<n;i++){
    if(a[i]<b[i])bb=false;
  }
  if(bb)cout<<"Compatible"<<endl;
  else cout<<"Incompatible"<<endl;
	return 0;
}
