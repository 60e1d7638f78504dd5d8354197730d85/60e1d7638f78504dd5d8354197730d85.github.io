#include <stdio.h>
#include <iostream>
using namespace std;
int main()
{int arr[100],n;
 cin>>n;
 for(int i=0;i<n;i++)
   cin>>arr[i];
 for (int i = 1; i < n; i++)
   {int key = arr[i];
    int j = i-1;
    while (j >= 0 && arr[j] > key)
    {arr[j+1] = arr[j];
     j = j-1;
     }
    arr[j+1] = key;
    for(int k=0;k<n;k++)
        cout<<arr[k]<<' ';
    cout<<'\n';
   }
 cout<<"Sorted Array:";
 for(int k=0;k<n;k++)
   cout<<arr[k]<<' ';

	return 0;
}
