#include <iostream>
#include <cmath>
using namespace std;
int main()
{
  int n,i,j;
  long int a[100];
  cin>>n;
  for(i=0;i<n;i++)
  {
    cin>>a[i];
  }
  int temp;
  for(i=0;i<n-1;i++)
  {
    for(j=0;j<n-i-1;j++)
    {
      if(abs(a[j])>abs(a[j+1]))
      {
        temp=a[j];
        a[j]=a[j+1];
        a[j+1]=temp;
      }
    }
  }
  for(i=0;i<n;i++)
  {
    cout<<a[i]<<" ";
  }
  

	return 0;
}
