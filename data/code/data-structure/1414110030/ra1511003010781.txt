#include <iostream>
using namespace std;
int main()
{
  int t;
  cin>>t;
  while(t--)
  {
    int n;
    cin>>n;
    int a[n], i, j, t;
    for(i=0; i<n; i++)
      cin>>a[i];
    for(i=0; i<n; i++)
    {
      for(j=0; j<n-i-1; j++)
      {
        if(a[j]<a[j+1])
        {
          t = a[j];
          a[j] = a[j+1];
          a[j+1] = t;
        }
      }
    }
    for(i=0; i<n; i++)
      cout<<a[i];
    cout<<endl;
  }

	return 0;
}
