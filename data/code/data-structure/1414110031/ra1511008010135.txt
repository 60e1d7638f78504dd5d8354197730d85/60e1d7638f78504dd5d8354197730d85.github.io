#include <bits/stdc++.h>
using namespace std;
int main()
{
int t;
  cin>>t;
  while(t--)
  {
    int n,i,j;
    bool c=true;
    cin>>n;
    int a[n],b[n];
    for(i=0;i<n;i++)
    {
      cin>>a[i];
    }
    for(i=0;i<n;i++)
    {
      cin>>b[i];
    }
    sort(a,a+n);
    sort(b,b+n);
    for(i=0;i<n;i++)
    {
      if(a[i]!=b[i])c=false;
    }
    if(c==true)cout<<"1"<<endl;
    else cout<<"0"<<endl;
  }
	return 0;
}
