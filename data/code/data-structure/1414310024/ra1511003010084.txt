#include <stdio.h>
#include <malloc.h>
int cmp(const void * a,const void *b)
{
	return *((int *)b)-*((int *)a);
}
int main()
{
   	int n,max,i;
   	scanf("%d",&n);
   	int * a = (int *)malloc(sizeof(int)*n);
   	for(i=0;i<n;i++)
   	{
   		scanf("%d",&a[i]);
   	}
   	qsort(a,n,sizeof(int),cmp);
   	max = 0;
   	for(i=0;i<n;i++)
   	{
   		//printf("%d\n",a[i]+i+1);
   		if(a[i]+i+1 > max)
   		{
   			max = a[i]+i+1;
   		}
   	}
   	printf("%d\n",max+1);
    return 0;
}
