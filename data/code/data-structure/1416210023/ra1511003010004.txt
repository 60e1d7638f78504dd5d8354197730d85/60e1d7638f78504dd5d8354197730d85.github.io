  #include<stdio.h>
#include<string.h>

int main()
{
	int t,i,b,m,a,j,q,k;
	int n;
	
	scanf("%d",&t);


	while(t>0)
	{
		t--;
		char v;
		char s[1000];
		int p[1000];
		
		
		
		scanf("%d",&n);
		
		for(i=0;i<n;i++)
		{
			scanf(" %c",&v);
			s[i]=v;
		}
		s[n]='\0';
		k=0;
		for(i=0;i<n;i++)
		{
			if(s[i]=='T')
			{
				if(s[i+1]=='T')
				{
					s[i+1]='H';
					
				}
				else s[i+1]='T';
				
				p[k]=i+1;
				
				k++;
				
			}
		}
		
		
		printf("%d\n",k);
		 
		for(i=0;i<k;i++)
		{
			printf("%d ",p[i]);
		}
		
		printf("\n");
		
	
		
		
	}
}
