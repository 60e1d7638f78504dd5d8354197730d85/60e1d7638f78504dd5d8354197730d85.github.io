#include<stdio.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,i,max_sum=0,max_end=-1,curr_sum=0,st=0,max_st=0;
        scanf("%d",&n);
        char arr[n+1];
        scanf("%s",arr);
        for(i=0;i<n;i++)
        {
        int a = 1;
        if(arr[i]=='1')
         a = -1;
        curr_sum+=a;
        if(curr_sum < 0)
        {
            curr_sum=0;
            st=i+1;
        }
        else if(max_sum<curr_sum)
        {
            max_sum=curr_sum;
            max_st=st;
            max_end=i;
        }
        }
        if(max_end==-1)
        printf("-1\n");
        else
        printf("%d %d\n",max_st+1,max_end+1);
    }
    return 0;
}
