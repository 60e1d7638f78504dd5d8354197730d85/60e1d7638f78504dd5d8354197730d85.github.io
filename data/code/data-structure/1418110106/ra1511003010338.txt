 
#include <stdio.h>
#include <iostream>
#include <vector>
#include <deque>
using namespace std;
 
int main()
{
        int t;
        scanf("%d", &t);
 
        while(t--)
        {
	     int students, seconds = 0;
	     scanf("%d", &students);
 
	     deque<char> pos;
	     deque<char> target;
 
	     int c;
 
	     do
	     {
		  c = getchar();
		  if(isdigit(c))
		  {
		          pos.push_back(c);
		          (c=='0') ? target.push_back('0') : target.push_front('1');
		  }
	     }
	     while(pos.size()<students);
 
	     while(pos != target)
	     {
		  for(int i=1; i<students; i++)
		  {
		          if(pos[i]=='1' && pos[i-1] == '0')
		          {
			       swap(pos[i], pos[i-1]);
			       i++;
		          }
		  }
		  seconds++;
	     }
	     printf("%d\n", seconds);
        }
 
}
