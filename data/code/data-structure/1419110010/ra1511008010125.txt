int d[102];
int main()
{
    int n,i;
    scanf("%d",&n);
    assert(1 <= n && n <= 100);
    int sum = 0;
    for(i=0;i<n;i++)
    {
        scanf("%d",&d[i]);
        assert(1 <= d[i] && d[i] <= 1000);
        sum += d[i];
    }
    if(sum == 2*(n-1))
    {
        printf("Yes");
    }
    else
    {
        printf("No");
    }
    return 0;
}
