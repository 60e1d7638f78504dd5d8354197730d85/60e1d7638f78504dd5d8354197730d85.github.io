#include<stdio.h>
#include<stdlib.h>
struct node {
    long long int val;
    struct node * next;
};
long long int vertices=0;
long long int edges=0;
void insert(long long int v,struct node ** tail,struct node ** head )
{
    struct node * tmp;
    tmp = (struct node *)malloc(sizeof(struct node));
    tmp->val=v;
    tmp->next=NULL;
    if(*tail == NULL)
    {
        *tail = tmp;
        *head = tmp;
    }
    else 
    {
        (*tail)->next=tmp;
        (*tail)=tmp;
    }
}
void dfs(long long int u , long long int visit[] ,struct node * head[],long long int j)
{
    visit[u] = j;
    struct node * p = head[u];
    while(p != NULL)
    {
        edges++;
        if(visit[p->val] == 0)
        {
            vertices++;
            dfs(p->val,visit,head,j);
        }
        p=p->next;
    }
}
int main()
{

    struct node * head[100001];
    long long int visit[100001];

    j=0;
    for(i=1;i<=n;i++)
    {
        if(visit[i] == 0)
        { 
            dfs(i,visit,head,++j);
 
            vertices++;
        
    //    printf("vertices=%lld edges=%lld\n",vertices,edges);
 
            if(edges/2 > vertices)
            {   flag=1;
                break;

                    max_edge = vertices;
 
            }

            {

            vertices = 0;

    }
 /*   for(i=1;i<=n;i++)

    if(flag == 1)
