int getMinDist (int N, int *dist, char *visit) 
{
	int i, mIdx=0, min = 0x7fffffff;
 
	for(i =0; i<= N; i++) 
	{
		if(visit[i]== 0 && dist[i]<min) {
			mIdx = i;
			min = dist[i];
		}
	}
	return mIdx;
}
 
 
void dijsktras(int st, int N, int E)

			if(visit[nd] == 0 && dist[u]!=0x7fffffff)

			}
