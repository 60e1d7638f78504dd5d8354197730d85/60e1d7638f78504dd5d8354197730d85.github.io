#include <stdio.h>
#define INF 99999
 
int max(int a,int b)
{
    return (a>b)?a:b;
}
int min(int a,int b)
{
    return (a<b)?a:b;
}
int n;
void allshort(int graph[n][n])
{
    int i,j,k;
    for(k=0;k<n;k++)
    for(i=0;i<n;i++)
    for(j=0;j<n;j++)
    graph[i][j]=min(graph[i][k]+graph[k][j],graph[i][j]);
    return;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
        int m;
        scanf("%d %d",&n,&m);
        int graph[n][n];
        int i,j;
        for(i=0;i<n;i++)
        for(j=0;j<n;j++)
        graph[i][j]=INF;
        for(i=0;i<n;i++)
        graph[i][i]=0;
        int u,v,c;
        for(i=0;i<m;i++){
            scanf("%d %d %d",&u,&v,&c);
            u--;
            v--;
            if(graph[u][v]>c)
            graph[u][v]=c,graph[v][u]=c;
            
        }
        allshort(graph);
        int s,a,h;
        scanf("%d %d %d",&s,&a,&h);
        int val=-1;
        s--;
        a--;
        h--;
        for(i=0;i<n;i++)
        if(i!=s && i!=a && i!=h)
        val=max(val,graph[s][i]+graph[i][a]+graph[a][i]+graph[i][h]);
        printf("%d\n",val);
        
        
    }
    return 0;
}
