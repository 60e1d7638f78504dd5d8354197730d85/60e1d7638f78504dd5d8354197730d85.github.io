#include <stdio.h>
 #include <string.h>
int main()
{
    int t,p,r,a[26],i,flag;
    char str[100];
    scanf("%d",&t);
    while(t--){
            for(i=0;i<26;i++){
                a[i]=0;
            }
           flag=0;
           scanf("%s",str);
           p=strlen(str);
           for(i=0;i<p;i++){
               r=str[i]-97;
               a[r]++;
               if(a[r]==2){
                   printf("Yes\n");
                   flag=1;
                   break;
               }
           }
           if(flag==0){
            printf("No\n");   
           }
    }
  return 0;
}
