#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#include<math.h>
 
typedef long long ll;
 
typedef struct
{
    ll id,att,val,p,pos;
}css;
 
css c[100005];
 
 
int compare(const void *a,const void *b)
{
    css *e1 = (css *)a;
    css *e2 = (css *)b;
    if(e1->id==e2->id)
    {
        if(e1->att==e2->att)
        {
            if(e1->p==e2->p)
            {
                return e2->pos-e1->pos;
            }
            return e2->p-e1->p;
        }
        return e1->att-e2->att;
    }
    return e1->id-e2->id;
}
 
 
ll binsrh(ll idm,ll n,ll attr)
{
    ll h,l,m,a,b;
    l=0;
    h=n-1;
    while(l<=h)
    {
        m=(l+h)/2;
        if(c[m].id==idm)
        {
            if(c[m].att==attr)
            {
                while(c[m-1].id==idm&&c[m-1].att==attr&&m>0)
                {
                    m--;
                }
                return m;
            }
            else if(c[m].att>attr)
            {
                h=m-1;
            }
            else
            {
                l=m+1;
            }
        }
        else if(c[m].id>idm)
        {
            h=m-1;
        }
        else
        {
            l=m+1;
        }
    }
}
 
int main()
{
	ll n,m,i,idm,attm,po;
	scanf("%lld%lld",&n,&m);
	for(i=0;i<n;i++)
    {
        scanf("%lld%lld%lld%lld",&c[i].id,&c[i].att,&c[i].val,&c[i].p);
        c[i].pos=i;
    }
    qsort(c,n,sizeof(css),compare);
    for(i=0;i<m;i++)
    {
        scanf("%lld%lld",&idm,&attm);
        po=binsrh(idm,n,attm);
        printf("%lld\n",c[po].val);
    }
	return 0;
}
