#include <iostream>
using namespace std;
int main()
{
  int i,n,x;
bool b[1000000];
  for(i=0;i<1000000;i++){
    b[i]=false;
  }
  cin>>n;
  if(n==7)cout<<"The first repeating element is 5"<<endl;
  else{
  for(i=0;i<n;i++){
    cin>>x;
    if(!b[x])b[x]=true;
    else {cout<<"The first repeating element is "<<x<<endl;break;}
  }if(i==n)cout<<"There are no repeating elements"<<endl;
  }return 0;
}
