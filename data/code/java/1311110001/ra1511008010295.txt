import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
       Scanner sc = new Scanner(System.in);
       int yr = sc.nextInt();
       boolean f = false;
       if(yr%400 == 0)
         f = true;
       else if(yr%100 == 0)
         f = true;
       else if(yr%4 == 0)
         f = true;
       else 
         f = false;
       if ( f == false )
         System.out.println( yr + " is not a leap year");
       else
         System.out.println(yr +" is a leap year ");
     }
}
