import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) {
       Scanner sc=new Scanner(System.in);
       int n = sc.nextInt();
       int k=0;
       int n1=n;
       while(k<n1)	{
       		for(int i=1;i<=n;i++)
       		{
         		System.out.print("*");
       		}
       		System.out.println();
         	k++;
         	n--;
       }
     }
}
