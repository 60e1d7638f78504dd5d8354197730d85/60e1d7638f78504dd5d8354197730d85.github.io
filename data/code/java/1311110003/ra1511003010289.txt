import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
		int x;
       Scanner sc = new Scanner(System.in);
       x = sc.nextInt();
       for(int i=0; i<x; i++)
       {
         for(int j=x; j>i; j--){
           System.out.print("*");
         }
         System.out.println();
       }
	}
}
