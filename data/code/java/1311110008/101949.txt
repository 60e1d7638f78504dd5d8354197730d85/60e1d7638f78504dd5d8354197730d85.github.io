import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner s=new Scanner(System.in);
       int N= s.nextInt();
       int temp,S=0;
       while(N>0)
       {
         temp=N%10;
         S=S+temp;
         N=N/10;
       }
       System.out.println(S);
	}
}
