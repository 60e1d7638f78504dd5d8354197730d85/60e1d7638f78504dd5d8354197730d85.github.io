import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner sc=new Scanner(System.in);
       int num=sc.nextInt();
       int sum=0;
       int digit=0;
       while(num>0)
       {
         digit=num%10;
         sum=sum+digit;
         num=num/10;
       }
       System.out.println(sum);
	}
}
