import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
       Scanner s=new Scanner(System.in);
       int radius;
       // float pi = 3.14159f;
       double area;
        radius = s.nextInt();

        System.out.println(String.format("%.2f",area)); 
    }    
}
