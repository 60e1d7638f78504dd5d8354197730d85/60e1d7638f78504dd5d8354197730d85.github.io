import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
       Scanner s=new Scanner(System.in);
       int a=s.nextInt();
       int b=s.nextInt();
       
       int minval=Math.max(a,b);
        
       while(true)
       {
         if(minval%a==0 && minval%b==0)
         {
           System.out.println(minval);
           break;
         }
         minval++;
       }
      
		
	}
}
