import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
       int n1,n2,lcm;
       Scanner input1=new Scanner(System.in);
       n1=input1.nextInt();
       n2=input1.nextInt();
       lcm=(n1>n2)? n1:n2;
       while (true)
       {
          if (lcm%n1==0 && lcm%n2==0)
          {
             System.out.println(lcm);
           break;
          }
         ++lcm;
       }  
		
	}
}
