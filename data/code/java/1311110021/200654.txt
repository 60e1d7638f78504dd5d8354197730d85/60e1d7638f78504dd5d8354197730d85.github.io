import java.io.*;
import java.util.Scanner;
public class TestClass 
{
	 public static void main(String[] args) 
     { 
	   int a,b,result=1;
       Scanner s = new Scanner(System.in);
       a = s.nextInt();
       b = s.nextInt();
       
    if(a>=0 && b==0)
     {
        result=1;
     }
    else if(a==0 && b>=1)
      { 
         result=0;
      }
    else
     {
         for(int i=1;i<=b;i++)
	 {
            result=result*a;
 	 }	    
     }
    System.out.println(result);
       
	 }
}
