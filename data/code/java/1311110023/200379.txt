import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
		float side;
       double area;
       Scanner in = new Scanner(System.in);
      side=in.nextFloat();
       double r = (2 * side) * (Math.sin(Math.PI/5));

      System.out.println(String.format("%.02f",area));
	}
