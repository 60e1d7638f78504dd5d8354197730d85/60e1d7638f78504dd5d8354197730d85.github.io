import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner s=new Scanner(System.in);
       double w=0, cost=0;
       w=s.nextDouble();
 
       if(w>=0 && w<=1)
       {
         cost=w*3.5;
       }
         else if(w>1 && w<=3)
         {
           cost=w*5.5;
         }
       else if(w>3 && w<=10)
       {
         cost=w*8.5;
       }
       else
       {
         cost=w*10.5;
       }
        System.out.print(cost); 
	}
}
