import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner input = new Scanner(System.in);
		double weight = input.nextDouble();
		if (weight > 50)
			System.out.println("");
		else
		{
			double costPerPound; 
			if (weight > 0 && weight <= 1)
				costPerPound = 3.5;
			else if (weight <= 3)
				costPerPound = 5.5;
			else if (weight <= 10)
				costPerPound = 8.5;
			else 
				costPerPound = 10.5;
			System.out.println(
				costPerPound * weight);
          
		}
	}
}
