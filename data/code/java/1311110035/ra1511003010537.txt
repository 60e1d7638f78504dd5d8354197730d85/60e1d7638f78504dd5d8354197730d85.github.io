import java.io.*;
import java.util.Scanner;
public class TestClass {
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		double num= sc.nextDouble();
      	double cost;
      	if(num<1)
          cost=3.5;
      	else if(num<3)
          cost= 5.5;
      	else if(num<=10)
          cost=8.5;
      	else
          cost=10.5;
		System.out.printf("%.1f", num*cost);
    }
}
