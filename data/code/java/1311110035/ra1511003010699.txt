import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
       Scanner in = new Scanner(System.in);
       double w = in.nextFloat();
       double c=0;
       if(w>=0  && w<=1)
         c=w*3.5;
       else if(w>1 && w<=3)
         c=w*5.5;
       else if(w>3 && w<=10)
         c=w*8.5;
       else 
         c=w*10.5;
       
       
        System.out.printf("%.2f",c); 
		
	}
}
