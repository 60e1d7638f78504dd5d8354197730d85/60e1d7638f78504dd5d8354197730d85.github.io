import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner sc= new Scanner(System.in);
       double wt=sc.nextDouble();
       double cost=0;
       if(0<=wt && wt<=1){
       cost= wt*3.5;
         }
       else if(1<wt && wt<=3){
       cost= wt*5.5;
       }
       else if(3<wt && wt<=10){
       cost= wt*8.5;
       }
       else{
       cost= wt*10.5;
       }
       System.out.printf("%.2f",cost);
	}
}
