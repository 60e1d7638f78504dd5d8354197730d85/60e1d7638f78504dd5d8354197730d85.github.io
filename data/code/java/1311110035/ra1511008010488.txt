import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner s = new Scanner(System.in);
       	Double n = s.nextDouble();
       	Double c=0.0;
       	if(n>=0 && n<=1){
          c=3.5*n;
        }else if(n<=3){
          c=5.5*n;
        }else if(n<=10){
          c=8.5*n;
        }else{
          c=10.5*n;
        }
       System.out.println(c);
	}
}
