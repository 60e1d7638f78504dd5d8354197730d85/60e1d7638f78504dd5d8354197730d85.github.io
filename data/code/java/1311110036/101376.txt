import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) {
       int digit;
       String str="";
		Scanner sc=new Scanner(System.in);
       int a = sc.nextInt();
       while(a>0)
       {
         digit = a%10;
         str=digit+" "+str;
         a=a/10;
       }
       System.out.println(str);
	}
}
