import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
       Scanner input=new Scanner(System.in);
       int a=input.nextInt();
       int b=input.nextInt();
       int atemp=a;
       int btemp=b;
       while (btemp!=0)
       {
         int t=btemp;
         btemp=(atemp % btemp);
         atemp=t;
       }
      int gcd=atemp;
      int lcm=(a*b)/gcd;
       System.out.println(gcd);
       System.out.println(lcm);
       input.close();         
	}
}
