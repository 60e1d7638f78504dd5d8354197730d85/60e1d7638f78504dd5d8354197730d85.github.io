import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
		int a, b, t, aTemp, bTemp, lcm, gcd;
        Scanner scanner;
        scanner = new Scanner(System.in);
        a = scanner.nextInt();
        b = scanner.nextInt();
         aTemp = a;
        bTemp = b;
 
        while (bTemp != 0) {
            t = bTemp;
            bTemp = aTemp % bTemp;
            aTemp = t;
        }
 
        gcd = aTemp;
         lcm = (a * b) / gcd;
        System.out.println(gcd);
       System.out.println(lcm);
	}
}
