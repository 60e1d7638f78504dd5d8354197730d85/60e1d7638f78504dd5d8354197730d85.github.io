import java.util.*;
import java.text.*;
public class TestClass {
	 public static void main(String[] args) { 
       Scanner in=new Scanner(System.in);
       double bp,hra,da,bonus,tot;
       bp=in.nextDouble();
       hra=0.8*bp;
       da=0.4*bp;
       bonus=0.25*hra;
       tot=bp+hra+da+bonus;
       DecimalFormat d=new DecimalFormat("0.0");
       System.out.println("Total Salary= "+d.format(tot));
	}
}
