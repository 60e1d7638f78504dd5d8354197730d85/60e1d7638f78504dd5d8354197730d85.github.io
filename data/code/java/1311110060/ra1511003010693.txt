import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) {
       Scanner x = new Scanner(System.in);
       int P = x.nextInt();
       int r = x.nextInt();
       int t = x.nextInt();
       int I = (P * r * t) / 100;
       int A = P + I;
       System.out.println("Interest= " + I);
       System.out.println("Amount= " + A);
	}
}
