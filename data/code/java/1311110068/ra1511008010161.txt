import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
       Scanner sc = new Scanner( System.in);
       int n = sc.nextInt();
       int sum=0;
       for(int i=1;i<=n;i++)
       {
         sum=sum+i;
       }
		System.out.print("\n"+sum);
	}
}
