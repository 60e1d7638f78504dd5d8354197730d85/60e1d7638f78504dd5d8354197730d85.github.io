import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner sc=new Scanner(System.in);
       int tot,d,h,m,s;
       tot=sc.nextInt();
       s=tot%60;
       tot=(tot-s)/60;
       m=tot%60;
       tot=(tot-m)/60;
       h=tot%24;
       tot=tot-h;
       d=tot/24;
       System.out.printf("The Duration is %d days %d hours %d minutes %d seconds",d,h,m,s);
	}
}
