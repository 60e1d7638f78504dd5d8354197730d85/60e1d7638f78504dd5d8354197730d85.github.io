import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
		int ar[]=new int[10];
       int sum=0;
       Scanner a=new Scanner(System.in);
       int n=a.nextInt();
       for(int i=0;i<n;i++)
       {
       	ar[i]=a.nextInt();
       }
       for(int i=0;i<n;i++)
       {
       	sum=sum+ar[i];
       }
       System.out.println(sum);
       
     }
}
