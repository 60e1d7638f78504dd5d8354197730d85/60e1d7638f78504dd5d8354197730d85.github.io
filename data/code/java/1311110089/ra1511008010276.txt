import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner sc=new Scanner(System.in);
       int x=sc.nextInt();
       int[] arr=new int[x];
       int sum=0;
       for(int i=0;i<x;i++){
         arr[i]=sc.nextInt();
         sum=sum+arr[i];
	}
       System.out.println(sum);
     }
}
