import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String args[]) { 
		Scanner in = new Scanner(System.in);
       	int n = in.nextInt();
       	int arr[] = new int[100];
       	for(int i=0;i<n;i++){
        	arr[i]=in.nextInt();
        }
       	int sum=0;
       	for(int a:arr){
        	sum+=a;
        }
       System.out.println(sum);
	}
}
