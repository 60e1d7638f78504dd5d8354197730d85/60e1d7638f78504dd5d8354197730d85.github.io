import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner sc = new Scanner(System.in);
       int n = sc.nextInt();
       int i, sum=0;
       int array[] = new int[n];
       for(i=0; i<array.length;i++){
         array[i] = sc.nextInt();
       }
       for(i=0;i<array.length;i++){
         sum = sum+array[i];
       }
       System.out.println(sum);
	}
}
