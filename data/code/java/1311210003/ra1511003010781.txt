import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner s = new Scanner(System.in);
       int n = s.nextInt();
       n = (n/2)+1;
       for(int i=1; i<=n; i++)
       {
        for(int j=1; j<i; j++)
         {
          System.out.print("*");
         }
         System.out.println();
        }
       for(int i=n;i>=1;i--)
{
for(int j=1;j<=i;j++)
{
System.out.print("*");
}
System.out.println();
       }
	}
}
