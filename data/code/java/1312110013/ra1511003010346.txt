import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner sc=new Scanner(System.in);
       int n=sc.nextInt();
       int i=0;
       int sum=0;
       while(n!=0){
       		sum+=(n%10)*Math.pow(2,i);
         	n=n/10;
         i++;
       }
       System.out.println(sum);
	}
}
