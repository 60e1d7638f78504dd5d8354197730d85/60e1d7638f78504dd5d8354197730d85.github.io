import java.io.*;
import java.util.Scanner;

class TestClass
{
    public static void main (String[] args) throws java.lang.Exception
    {
        int t, n;
        int multi = 1;
        Scanner tc = new Scanner(System.in);
        t = tc.nextInt();
        for(int i=0;i<t;i++){
            n = tc.nextInt();
            for(int f=1;f<=n;f++){
                multi = multi * f;
            }
            System.out.println(multi);
            multi = 1;
        }
    }
}
