import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
       Scanner sc=new Scanner(System.in);
		int i, Number, Sum = 0 ;
		sc = new Scanner(System.in);		
		Number = sc.nextInt();
 
		for(i = 1 ; i < Number ; i++) {
			if(Number % i == 0)  {
				Sum = Sum + i;
			}
		}
		if (Sum == Number) {
			System.out.format("Perfect number");
		}
		else {
			System.out.format("Not Perfect number");
        }
	}
}
