import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
		
      int n,i,c=0;
       Scanner sc=new Scanner(System.in);
       n=sc.nextInt();
      for(i=1;i<=n;i++)
      {
        if(n%i==0)
          c++;
      }
       if(c>2)
         System.out.println("yes");
       else
         System.out.println("no");
	}
}
