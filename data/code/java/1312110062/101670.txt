import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
		int m, n, sum = 0;
        Scanner s = new Scanner(System.in);
       
        m = s.nextInt();
      while(m>9)
      {
        sum=0;
        while(m > 0)
        {
       
            n = m % 10;
            sum = sum + n;
            m = m / 10;
        }
       m=sum;
      }
        System.out.println(sum);
	}
}
