import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner read=new Scanner(System.in);
       	int n=read.nextInt();
       int i,count=0;
       int[] a=new int[10];
       while(n>0)
       {
         a[count]=n%10;
         n/=10;
         count++;
	}
       for(i=count-1;i>=0;i--)
         System.out.print(a[i]+" ");
     }
}
