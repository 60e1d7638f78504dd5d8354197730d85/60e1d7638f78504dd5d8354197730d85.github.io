import java.util.Scanner;
import java.io.*;
public class TestClass {
	 public static void main(String[] args) { 
		int yr;
       Scanner scan = new Scanner(System.in);
	
       yr = scan.nextInt();
	   
       if((yr%4 == 0) && (yr%100!=0))
       {
           System.out.print(yr +" is a leap year");
       }
       else if(yr%100 == 0)
       {
           System.out.print(yr + " is not a leap year");
       }
       else if(yr%400 == 0)
       {
           System.out.print(yr + " is a leap year");
       }
       else
       {
           System.out.print(yr + " is not a leap year");
       }
    }
}
