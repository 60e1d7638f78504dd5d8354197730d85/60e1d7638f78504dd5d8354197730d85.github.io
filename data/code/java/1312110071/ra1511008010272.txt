  Scanner in = new Scanner(System.in);
  binary1 = in.nextLong();
  binary2 = in.nextLong();
  while (binary1 != 0 || binary2 != 0) 
  {
   sum[i++] = (int)((binary1 % 10 + binary2 % 10 + remainder) % 2);
   remainder = (int)((binary1 % 10 + binary2 % 10 + remainder) / 2);
   binary1 = binary1 / 10;
   binary2 = binary2 / 10;
  }
  if (remainder != 0) {
   sum[i++] = remainder;
  }
  --i;
  while (i >= 0) {
   System.out.print(sum[i--]);
  }
	}
}
