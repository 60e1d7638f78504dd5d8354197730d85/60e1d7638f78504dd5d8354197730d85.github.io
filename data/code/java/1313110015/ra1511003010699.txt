import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
       Scanner in=new Scanner(System.in);
       int n=in.nextInt();
       int sum=0;
       while(n>0)
       {
         int d=n%10;
         sum=sum+d;
         n=n/10;
       }
       System.out.print(sum);
		
	}
}
