import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner sc = new Scanner(System.in);
		
		int t = sc.nextInt();
		
		while(t-- > 0) {
			int num = sc.nextInt();
			String numStr = num + "";
			
			int count = 0;
			for(int i = 0; i < numStr.length(); i++) {
				int n = Integer.parseInt(numStr.charAt(i)+"");
				if(n != 0 && num % n == 0)
					count++;
			}
			
			System.out.print(count+"\n");
		}
		
		sc.close();
	}
}
