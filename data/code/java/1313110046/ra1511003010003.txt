import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner sc = new Scanner(System.in);
       int n = sc.nextInt();
       int arr[] = new int[n];
       for(int i=0;i<n;i++){
         arr[i] = sc.nextInt();
	}
       for(int i=0;i<n;i++){
         double a = (arr[i]*(arr[i]-1)*0.5);
         int ans = (int)a;
         System.out.println(ans);
       }
}
}
