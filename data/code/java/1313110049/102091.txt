import java.util.Scanner;

public class TestClass
{
    public static void main(String args[])
    {
        int rows;
        Scanner scan = new Scanner(System.in);
		rows = scan.nextInt();
          
        for(int i = 1; i <= rows; ++i) {
            for(int j = 1; j <= i; ++j) {
                System.out.print(j + " ");
            }
            System.out.println();
        }
    }
}
