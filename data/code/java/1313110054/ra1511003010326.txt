import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner sc = new Scanner(System.in);
       	int n = sc.nextInt();
       	for(int i=n;i>0;i--){
          System.out.println("");
          for(int j = n ; j>=i;j--){
          	System.out.print(j+" ");
          }
     }
	}
}
