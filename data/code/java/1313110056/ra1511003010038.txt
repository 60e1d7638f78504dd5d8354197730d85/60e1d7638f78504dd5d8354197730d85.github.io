import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
	   int n,i,j;
       Scanner sc=new Scanner(System.in);
       n=sc.nextInt();
       for(i=n;i>0;i--)
       {
         for(j=1;j<=i;j++)
           System.out.print(j+" ");
         System.out.print("\n");
       }
       for(i=2;i<=n;i++)
       {
         for(j=1;j<=i;j++)
           System.out.print(j+" ");
         System.out.print("\n");
       }
	}
}
