import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner sc=new Scanner(System.in);
       int num=sc.nextInt();
        for(int i=0;i<num;i++)
        {
           for(int j=i+1;j>=1;j--)
           {
              System.out.print(j+" ");
           }
          System.out.println("");
        }
	}
}
