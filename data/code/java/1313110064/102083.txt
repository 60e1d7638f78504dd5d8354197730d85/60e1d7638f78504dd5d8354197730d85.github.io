import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
       Scanner scan = new Scanner(System.in);
       int n = scan.nextInt();
       
       for(int j=0; j<n; j++)
       {
       	for(int i=0; i<n; i++)		
        {
          if(i==j)
            System.out.print(i);
          else
            System.out.print("0");          
        }
         System.out.print("\n");
       }
	}
}
