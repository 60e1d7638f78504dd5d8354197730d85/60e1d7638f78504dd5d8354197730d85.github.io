import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner sc=new Scanner(System.in);
        int num=sc.nextInt();
       int k=1;
        for(int i=0;i<num;i++)
        {
           for(int j=0;j<=i;j++)
           {

           k++;
           }
          System.out.println();
          
        }
	}
}
