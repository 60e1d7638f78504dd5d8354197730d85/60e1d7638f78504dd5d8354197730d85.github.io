import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner sc = new Scanner(System.in);
       int n = sc.nextInt();
       int i,j,k=1;
       for(i=0;i<n;i++)
       {
         for(j=0;j<=i;j++)
         {
           System.out.print(k + " ");
           k++;
         }
         System.out.println("");
       }
	}
}
