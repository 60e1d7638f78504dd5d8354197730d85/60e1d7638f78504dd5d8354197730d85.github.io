import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
       Scanner input=new Scanner(System.in);
       int n;
       n=input.nextInt();
		int i, j, k=1;
        for(i=0; i<n; i++)
        {
            for(j=0; j<k; j++)
            {
                System.out.print("*");
            }
            k = k + 2;
            System.out.println();
        }
	}
}
