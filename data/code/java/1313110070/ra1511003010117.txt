import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
       Scanner o=new Scanner(System.in);
       int a=o.nextInt();
       while(a!=0){
         int count=0;
         int b=o.nextInt();
         int c=o.nextInt();
         for(int i=b;i<=c;i++){
           int temp=(int)Math.sqrt(i);
           if(temp*temp==i){
             count++;
           }
         }
         System.out.println(count);
       a--;}
		
	}
}
