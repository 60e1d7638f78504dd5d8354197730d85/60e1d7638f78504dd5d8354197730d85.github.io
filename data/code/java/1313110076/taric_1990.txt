import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
       Scanner input = new Scanner(System.in);
       int in = input.nextInt();
       for(int i=1;i<=in;i++)
       {
         for(int j=1;j<=i;++j)
         {
           System.out.print(j);
         }
         	for (int k=in-i;k>=1;k--)
            {
           System.out.print("*");
              
            }
         System.out.println("");
       } 
     }
}
