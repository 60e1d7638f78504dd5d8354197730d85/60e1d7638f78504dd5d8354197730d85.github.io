import java.util.Scanner;
import java.io.*;
public class TestClass {
	 public static void main(String[] args) { 
	 int c=0,a,temp;  
    int n;   
   Scanner input = new Scanner(System.in);
    n = input.nextInt();
    temp=n;  
    while(n>0)  
    {  
    a=n%10;  
    n=n/10;  
    c=c+(a*a*a);  
    }  
    if(temp==c)  
    System.out.println("YES");   
    else  
    System.out.println("NO");  
	}
}
