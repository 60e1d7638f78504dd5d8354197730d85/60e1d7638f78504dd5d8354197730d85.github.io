import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
       Scanner obj =new Scanner(System.in);
       int  n=obj.nextInt();
       for(int i=1;i<=n;i++){
			for(int j=1;j<=i;j++){
				System.out.print(((i+j)%2)+" ");
			}
			System.out.print("\n");}
         
		
	}
}
