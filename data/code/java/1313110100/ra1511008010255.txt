import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner s = new Scanner(System.in);
       	int n = s.nextInt();
       	int sum = 0;
       	for(int i=1;i<n;i++) {
          if(n%i == 0) sum += i;
        }
      	String st = sum == n ? "YES" : "NO";
       	System.out.print(st);
	}
}
