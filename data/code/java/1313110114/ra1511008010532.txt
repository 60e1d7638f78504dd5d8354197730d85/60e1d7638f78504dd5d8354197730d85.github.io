import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner sc=new Scanner(System.in);
       int n=sc.nextInt();
       int c=0;
       for(int i=1;i<=n;i++){
        if(n%i==0){
          System.out.print(""+i+" ");
          c++;
        }
       }
       System.out.println("");
       System.out.println(""+c);
       
	}
}
