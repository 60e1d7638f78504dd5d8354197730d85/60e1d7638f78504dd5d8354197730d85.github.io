import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner s=new Scanner(System.in);
       int n=s.nextInt();
       int count=0,i;
       for(i=1;i<=n;i++)
       {
         if(n%i==0)
         {
           System.out.printf("%d ",i);
           count++;
         }
       }
       System.out.printf("\n%d",count);
	}
}
