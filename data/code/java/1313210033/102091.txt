import java.util.Scanner;
 
public class TestClass
{
    public static void main(String[] args) 
    {
        Scanner sc = new Scanner(System.in);
        int rows = sc.nextInt();
        for (int i = 1; i <= rows; i++) 
        {
            for (int j = i; j >= 1; j--)
            {
              if(j!=1)  
              System.out.print(j+" ");
              else
              System.out.print(j);  
            }
             
            System.out.println();
        }
    }
}
