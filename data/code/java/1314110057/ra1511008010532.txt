import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
		Scanner sc=new Scanner(System.in);
       int[] a=new int[100];
       int n=sc.nextInt();
       int sum=0;
       for(int i=0;i<n;i++){
         a[i]=sc.nextInt();
         sum+=a[i];
       }
       System.out.println(""+(sum/n));
	}
}
