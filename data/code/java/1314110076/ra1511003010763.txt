import java.io.*;
import java.util.Scanner;
public class TestClass {
	 public static void main(String[] args) { 
		    int num, i, tab;
        Scanner scan = new Scanner(System.in);
		
        num = scan.nextInt();
		
        for(i=1; i<=10; i++)
        {
            tab = num*i;
            System.out.print(i + " * " + num + " = " + tab + "\n");
        }
	}
}
