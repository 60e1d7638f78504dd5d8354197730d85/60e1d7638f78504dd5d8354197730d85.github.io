import java.io.*;
import java.util.Scanner;
class TestClass
{
    public static void main(String args[])
	{
		Scanner scan=new Scanner(System.in);
		int t=scan.nextInt();
		while(t-- > 0)
		{
			String s=scan.next();
			int i=0,o=0,l=0;
			while(i!=s.length())
			{
				if(s.charAt(i)=='0')
					o++;
				else
					l++;
				i++;
			}
			if(o==1 || l==1)
				System.out.println("yes");
			else
			    System.out.println("no");
	   }
	}
}
