import java.io.*;
import java.util.*;
public class TestClass 
{
	public static void main(String[] args) 
    { 
		Scanner sc=new Scanner(System.in);
      	int n=sc.nextInt();
      	int p=sc.nextInt();
      	int a[]=new int[n];
      	for(int i=0;i<n;i++)
          	a[i]=sc.nextInt();
      	int temp=a[p-1];
      	a[p-1]=a[n-p];
      	a[n-p]=temp;
        for(int i=0;i<n;i++)
          System.out.print(a[i]+" ");
	}
}
