import java.io.*;
import java.util.*;
class Biggest
{
  int a;
  int b;
  void read()
  {
    Scanner sc=new Scanner(System.in);
    a=sc.nextInt();
    b=sc.nextInt();
   }
   void biggest()
   {
     if(a>b)
       System.out.println(a);
     else
       System.out.println(b);
   }
}
public class TestClass {
	 public static void main(String[] args) { 
		Biggest bs=new Biggest();
       bs.read();
       bs.biggest();
	}
}
