import java.io.*;
import java.util.Scanner;
class Biggest{
  int a;
  int b;
  public void read()
  {
    Scanner s = new Scanner(System.in);
    a = s.nextInt();
    b = s.nextInt();
  }
  public void biggest()
  {
    if(a>=b)
      System.out.println(a);
    else
      System.out.println(b);
  }
}
public class TestClass {
	 public static void main(String[] args) { 
		Biggest bs = new Biggest();
       bs.read();
       bs.biggest();
	}
}
