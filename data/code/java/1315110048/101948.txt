import java.io.*;
import java.util.Scanner;
class Rectangle
{
   float l,b,area;
  public void read()
  {
    
    Scanner src=new Scanner(System.in);
    l=src.nextInt();
    b=src.nextInt();
    area=(l*b)/2;
    
  }
  
  public void display()
  {
    System.out.println(area);
  }
}
public class TestClass {
	 public static void main(String[] args) { 
	Rectangle a=new Rectangle();
       a.read();
       a.display();
	}
}
