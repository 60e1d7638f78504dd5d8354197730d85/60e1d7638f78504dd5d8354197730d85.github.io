import java.io.*;
import java.util.Scanner;
class Triangle
{ 
  int l,w;
  public void read()
  {
    Scanner sc=new Scanner(System.in);
   l=sc.nextInt();
   w=sc.nextInt();
  }
  public void display()
  {
    double area;
    area=(0.5)*l*w;
    System.out.println(area);
  }
}
public class TestClass {
	 public static void main(String[] args) { 
		Triangle a=new Triangle();
         a.read();
         a.display();
	}
}
