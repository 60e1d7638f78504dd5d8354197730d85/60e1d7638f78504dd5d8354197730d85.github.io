import java.io.*;
import java.util.*;
class Sample
{
  public int fact(int n)
  {
    if(n<=1)
      return 1;
    else
      return n*fact(n-1);
  }
}
public class TestClass {
	 public static void main(String[] args) 
     {
       Sample objname=new Sample();
       Scanner sc=new Scanner(System.in);
       int num=sc.nextInt();
       int answer=objname.fact(num);
       System.out.println(answer);
	 }
}
