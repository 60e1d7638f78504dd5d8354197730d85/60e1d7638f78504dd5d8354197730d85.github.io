import java.io.*;
import java.util.*;
public class TestClass {
	 public static void main(String[] args) { 
		Sample objname =new Sample();
       	Scanner s = new Scanner(System.in);
       	int num = s.nextInt();
       	int res = objname.fact(num);
       	System.out.println(res);
	}
}
class Sample{
  public int fact(int n){
    if (n == 1)
        return 1;
    else
        return n * fact(n-1);
  }
}
