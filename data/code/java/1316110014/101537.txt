import java.io.*;
import java.util.*;
class Fibonacci{
  int f,n;
  public void getInput(){
    Scanner inp=new Scanner(System.in);
    n=inp.nextInt();
  }
  public void displayFib(){
    int a=0,b=1,f;
    System.out.print(1);
    for(int i=2;i<=n;i++){
      f=a+b;
      System.out.print(" "+f);
      a=b;
      b=f;
    }
  }
}
public class TestClass {
	 public static void main(String[] args) { 
	Fibonacci obj=new Fibonacci();
       obj.getInput();
       obj.displayFib();
	}
}
