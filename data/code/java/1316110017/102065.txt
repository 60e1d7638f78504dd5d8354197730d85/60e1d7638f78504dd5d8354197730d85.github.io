import java.io.*;
import java.util.*;
class Century{
  int year;
  public void getInput(){
    Scanner inp=new Scanner(System.in);
    year=inp.nextInt();
    centuryFromYear(year);
  }
  public void centuryFromYear(int yr){
    if(yr%100==0)
      yr=yr/100;
    else
      yr=(yr/100)+1;
    System.out.println(yr);
}
}
