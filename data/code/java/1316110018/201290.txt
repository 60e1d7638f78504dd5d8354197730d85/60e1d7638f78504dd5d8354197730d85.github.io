class Sample
{
     static Scanner s;
  static String str;
  static void getInput()
  {
  s=new Scanner(System.in);
    str=s.next();
    checkPalindrome(str);
  }
   static void checkPalindrome(String str)
  {
   int i;
  int n=str.length();
  String s="";
  for(i=n-1;i>=0;i--)
  s=s+str.charAt(i);
  if(s.equals(str))
  {
  System.out.println(str+" is a palindrome");
  }
  else
  {
  System.out.println(str+" is not a palindrome");
   }
}
