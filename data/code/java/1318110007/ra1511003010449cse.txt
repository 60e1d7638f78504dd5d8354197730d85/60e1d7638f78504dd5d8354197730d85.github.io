import java.io.*; import java.util.*;
class Shape
{
  int length,breadth;      
}
 class TestClass extends Shape 
{
      int area;
        void calculateArea()
        {
          area=length*breadth;
        }
	 public static void main(String[] args) 
    { 
    	Scanner s=new Scanner(System.in);
        TestClass r=new TestClass();
        r.length=s.nextInt();
        r.breadth=s.nextInt();
        r.calculateArea();
        System.out.println("Area="+r.area);
	}
}
