import java.io.*;
import java.util.*;
class Shape
{
  int length,breadth;
}
public class TestClass extends Shape{
  int area;
  void calculateArea()
  {
    area= length*breadth;
  }
	 public static void main(String[] args) { 
		TestClass r=new TestClass();
        Scanner sc=new Scanner(System.in);
        r.length=sc.nextInt();
        r.breadth=sc.nextInt();
        r.calculateArea();
        System.out.println("Area="+r.area);
     }
}
