import java.io.*;
import java.util.*;
class Shape {
  int length;
  int breadth;
}
public class TestClass extends Shape {
  int area;
  void calculateArea(){
    area = length*breadth;
  }
  public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    TestClass r = new TestClass();
    r.length = sc.nextInt();
    r.breadth = sc.nextInt();
    r.calculateArea();
    System.out.println("Area="+r.area);
      
  }
}
