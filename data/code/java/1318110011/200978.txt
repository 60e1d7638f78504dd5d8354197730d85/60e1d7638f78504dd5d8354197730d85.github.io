import java.io.*;
import java.util.*;
interface Bank 
{
 float rate = 12.0f;
 int no_of_years=3;
 void show();
}
 
class Customer {
 
 String Customername;
 int Customerid;
 Customer(String n,int i) 
 {
  Customername = n;
  Customerid = i;
 }
 
 void display() {
  
  System.out.println("Customer Name="+Customername);
  System.out.println("Customer Id="+Customerid);
 }
}
 
class Account extends Customer implements Bank {
 
 int acc_no;
 float acc_bal;
 Account(String n,int b,int x,float y) {
  
  super(n,b);
  acc_no=x;
  acc_bal=y;
 }
 
 public void show() {
  
  display();
  System.out.println("Account No="+acc_no);
  System.out.println("Account Balance="+acc_bal);
 }
 
 void interest() {
  
  show();
  float intr =(rate*acc_bal*no_of_years)/100;
  System.out.println("Interest="+intr);
 }
}
 
public class TestClass {
	 public static void main(String[] args) { 
       Scanner sc=new Scanner(System.in);
       String name=sc.next();
       int custid=sc.nextInt();
       int accno=sc.nextInt();
       float bal=sc.nextFloat();
		Account ac = new Account(name,custid,accno,bal);
        ac.interest();
	}
}
