import java.util.*;
class Base{
  int x,y;
  public void getInput(){
    Scanner kb=new Scanner(System.in);
    y=kb.nextInt();
    x=kb.nextInt();
  }
}
class Derived extends Base{
  int z;
  public void sum(){
    z=x+y;
  }
  public void display(){
    System.out.print(z);
  }
}
class TestClass{
  public static void main(String arg[]){
    Derived d1 = new Derived();
    d1.getInput();
    d1.sum();
    d1.display();
  }
}
