import java.util.*;
class Base
{
  	int n1,n2;
  	public void getInput()
    {
      	Scanner sc=new Scanner(System.in);
      	n1=sc.nextInt();
      	n2=sc.nextInt();
    }
}
class Derived extends Base
{
	int sum;
	public void sum()
    {
    	sum=n1+n2;
    }
    public void display()
    {
    	System.out.println(sum);
    }
}
public class TestClass {
	public static void main(String[] args) 
    { 
		Derived d1 = new Derived();
        d1.getInput();
        d1.sum();
        d1.display();
	}
}
