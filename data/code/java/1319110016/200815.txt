import java.io.*;
import java.util.Scanner;
public class TestClass
{
   public static void main(String args[])
   {
       int arr[] = new int[10];
       int i, num;
     try
     {
       Scanner scan = new Scanner(System.in);
	   for(i=0; i<10; i++)
       {
           arr[i] = scan.nextInt();
       }
       num = scan.nextInt();
       System.out.println(arr[num]);
	   }
       catch(ArrayIndexOutOfBoundsException e)
       {
         System.out.println(e);
       }
   }
}
