a=input("");
b=input("");
c=input("");
if and(eq(a,b),eq(b,c))
	fprintf("All are equal");
 else
 	if and(gt(a,b),gt(b,c))
    fprintf("%d is the greater number \n",a);
    else 
    	if and(gt(b,c),gt(b,a))
        fprintf("%d is the greater number \n",b);
    else
    	fprintf("%d is the greater number \n",c);
   end
 end
 end
