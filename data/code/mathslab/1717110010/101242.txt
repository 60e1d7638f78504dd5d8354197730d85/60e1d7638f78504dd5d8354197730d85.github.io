function sum=sumeven(n)
	i=0;sum=0;
	while i<=n
		if mod(i,2)==0
			sum=sum+i;
		end
		i=i+1;
	end
	fprintf('\nSum of all even numbers=%d',sum);
end
function sum=sumodd(n)
	i=0;sum=0;
	while i<=n
		if mod(i,2)~=0
			sum=sum+i;
		end
   	i=i+1;
	end
	fprintf('Sum of all odd numbers=%d',sum);
end

x=input("");
sumodd(x);
sumeven(x);
