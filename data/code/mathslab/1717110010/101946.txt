function sum=sumeven(n)
i=0;sum=0;
while i<=n
if mod(i,2)~=0
	sum=sum+i;
 end
 i=i+1;
 end
 fprintf('\nSum of all Odd numbers = %d',sum);
 end  
 
function sum=sumodd(n)
i=0;sum=0;
while i<=n
if mod(i,2)==0
sum=sum+i;
end
i=i+1;
end
fprintf('\nSum of all Even numbers = %d',sum);
end
x=input('');
n=input('');
switch x
	case 1
    sumeven(n);
    break;
    case 2
    sumodd(n);
    break;
    otherwise
    disp('End');
    break;
 end
