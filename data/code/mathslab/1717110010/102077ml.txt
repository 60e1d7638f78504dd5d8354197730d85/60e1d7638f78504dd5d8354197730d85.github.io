function sum=sumeven(n)
i=0;
sum=0;
while i<=n
if mod(i,2)==0
sum=sum+i;
end
i=i+1;
end
printf("\nSum of all Even numbers = %d",sum);
end
function sum=sumodd(n)
i=0;
sum=0;
while i<=n
if mod(i,2)!=0
sum=sum+i;
end
i=i+1;
end
printf("\nSum of all Odd numbers = %d",sum);
end
op=input("");
n=input("");
switch op
case 1
sumodd(n);
break;
case 2
sumeven(n);
break;
otherwise
disp("End");
break;
end
