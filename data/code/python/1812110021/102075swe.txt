mo=input()
da=eval(input())
if mo == "January" and da == 1:
    print('New Year')
elif mo == "January" and da == 26:
    print('Republic Day')
elif mo == "August" and da == 15:
    print('Independence Day')
else:
    print('Sorry No National Holidays')
