grade=input()
grade=grade.upper()
if grade=="A+" or grade=="A":
    print("4.0")
elif grade=="A-":
    print("3.7")
elif grade=="B+":
    print("3.3")
elif grade=="B":
    print("3.0")
elif grade=="B-":
    print("2.7")
elif grade=="C+":
    print("2.3")
elif grade=="C":
    print("2.0")
elif grade=="C-":
    print("1.7")
elif grade=="D+":
    print("1.3")
elif grade=="D+":
    print("1.7")
elif grade=="D":
    print("1.0")
else :
    print("0")
