n=int(input())
b=[]
for i in range(0,n):
    a=int(input())
    b.append(a)
sum1=0
sum2=0
sum3=0
for j in b:
    if(j>0):
        if(j%2==0):
            sum1=sum1+j
        else:
            sum2=sum2+j
    else:
        sum3=sum3+j
print("Sum of positive even numbers:",sum1)
