import re
p= input()
x = True
while x:  
    if (len(p)==8):
        break
    elif not re.search("[a-z]",p):
        break
    elif not re.search("[0-9]",p):
        break
    elif not re.search("[A-Z]",p):
        break
    elif not re.search("[$#@]",p):
        break
    else:
        print("Good Password")
        x=False
        break
