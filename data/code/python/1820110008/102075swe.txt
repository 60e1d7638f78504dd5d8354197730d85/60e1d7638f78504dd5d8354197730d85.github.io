fn=input()
n=eval(input())
f= open(fn,"w+")
for i in range(n):
    s=input()
    f.write(s)
    f.write('\n')
    f.write('\n')    
f.close()
se=input()
c=0
f=open(se,'r')
for line in f:
    words=line.split()
    for i in words:
        for j in i:
            if j.isdigit():
                c=c+1
f.close()
print(c)
