fname = input("")
n=input("")
m=int(n)
f= open(fname,"w+")
for x in range(0,m):
	a=input("")
	f.write(a)
	f.write("\n\n")
f.close()
num=0
with open(fname, 'r') as f:
    for line in f:
        words = line.split()
        for i in words:
            for letter in i:
                if(letter.isdigit()):
                    num+=1
print(num)
